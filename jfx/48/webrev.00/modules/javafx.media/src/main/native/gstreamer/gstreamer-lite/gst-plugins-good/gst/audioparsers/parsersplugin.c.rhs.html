42 #ifdef GSTREAMER_LITE
43 gboolean
44 plugin_init_audioparsers (GstPlugin * plugin)
45 #else // GSTREAMER_LITE
46 static gboolean
47 plugin_init (GstPlugin * plugin)
48 #endif // GSTREAMER_LITE
49 {
50 #ifndef GSTREAMER_LITE
51   gboolean ret;
52 
53   ret = gst_element_register (plugin, "aacparse",
54       GST_RANK_PRIMARY + 1, GST_TYPE_AAC_PARSE);
55   ret &= gst_element_register (plugin, "amrparse",
56       GST_RANK_PRIMARY + 1, GST_TYPE_AMR_PARSE);
57   ret &= gst_element_register (plugin, "ac3parse",
58       GST_RANK_PRIMARY + 1, GST_TYPE_AC3_PARSE);
59   ret &= gst_element_register (plugin, "dcaparse",
60       GST_RANK_PRIMARY + 1, GST_TYPE_DCA_PARSE);
61   ret &= gst_element_register (plugin, "flacparse",
62       GST_RANK_PRIMARY + 1, GST_TYPE_FLAC_PARSE);
63   ret &= gst_element_register (plugin, "mpegaudioparse",
64       GST_RANK_PRIMARY + 2, GST_TYPE_MPEG_AUDIO_PARSE);
65   ret &= gst_element_register (plugin, "sbcparse",
66       GST_RANK_PRIMARY + 1, GST_TYPE_SBC_PARSE);
67   ret &= gst_element_register (plugin, "wavpackparse",
68       GST_RANK_PRIMARY + 1, GST_TYPE_WAVPACK_PARSE);
69 
70   return ret;
71 #else // GSTREAMER_LITE
72   return gst_element_register (plugin, "mpegaudioparse",
73       GST_RANK_PRIMARY + 2, GST_TYPE_MPEG_AUDIO_PARSE);

74 #endif // GSTREAMER_LITE
75 }

82     plugin_init, VERSION, "LGPL", GST_PACKAGE_NAME, GST_PACKAGE_ORIGIN);
83 #endif // GSTREAMER_LITE
