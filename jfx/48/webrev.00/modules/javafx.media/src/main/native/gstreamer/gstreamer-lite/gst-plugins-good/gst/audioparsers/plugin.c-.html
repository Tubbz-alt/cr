33 static gboolean
34 plugin_init (GstPlugin * plugin)
35 {
36   gboolean ret;
37 
38   ret = gst_element_register (plugin, "aacparse",
39       GST_RANK_PRIMARY + 1, GST_TYPE_AAC_PARSE);
40   ret &= gst_element_register (plugin, "amrparse",
41       GST_RANK_PRIMARY + 1, GST_TYPE_AMR_PARSE);
42   ret &= gst_element_register (plugin, "ac3parse",
43       GST_RANK_PRIMARY + 1, GST_TYPE_AC3_PARSE);
44   ret &= gst_element_register (plugin, "dcaparse",
45       GST_RANK_PRIMARY + 1, GST_TYPE_DCA_PARSE);
46   ret &= gst_element_register (plugin, "flacparse",
47       GST_RANK_PRIMARY + 1, GST_TYPE_FLAC_PARSE);
48   ret &= gst_element_register (plugin, "mpegaudioparse",
49       GST_RANK_PRIMARY + 2, GST_TYPE_MPEG_AUDIO_PARSE);
50   ret &= gst_element_register (plugin, "sbcparse",
51       GST_RANK_PRIMARY + 1, GST_TYPE_SBC_PARSE);
52   ret &= gst_element_register (plugin, "wavpackparse",
53       GST_RANK_PRIMARY + 1, GST_TYPE_WAVPACK_PARSE);
54 
55   return ret;
56 }
57 
58 
59 GST_PLUGIN_DEFINE (GST_VERSION_MAJOR,
60     GST_VERSION_MINOR,
61     audioparsers,
62     "Parsers for various audio formats",
63     plugin_init, VERSION, "LGPL", GST_PACKAGE_NAME, GST_PACKAGE_ORIGIN);
