20 #ifndef __GST_PROTECTION_H__

23 #include <gst/gst.h>

25 G_BEGIN_DECLS
26 
27 /**
28  * GST_PROTECTION_SYSTEM_ID_CAPS_FIELD:
29  *
30  * The field name in a GstCaps that is used to signal the UUID of the protection
31  * system.
32  *
33  * Since: 1.6
34  */
35 #define GST_PROTECTION_SYSTEM_ID_CAPS_FIELD "protection-system"
36 
37 typedef struct _GstProtectionMeta GstProtectionMeta;
38 /**
39  * GstProtectionMeta:
40  * @meta: the parent #GstMeta.
41  * @info: the cryptographic information needed to decrypt the sample.
42  *
43  * Metadata type that holds information about a sample from a protection-protected
44  * track, including the information needed to decrypt it (if it is encrypted).
45  *
46  * Since: 1.6
47  */
48 struct _GstProtectionMeta
49 {
50   GstMeta meta;
51 
52   GstStructure *info;
53 };
