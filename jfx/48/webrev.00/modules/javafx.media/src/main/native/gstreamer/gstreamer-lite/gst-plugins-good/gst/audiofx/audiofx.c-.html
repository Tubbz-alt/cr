43 /* entry point to initialize the plug-in
44  * initialize the plug-in itself
45  * register the element factories and pad templates
46  * register the features
47  */
48 #ifdef GSTREAMER_LITE
49 gboolean
50 plugin_init_audiofx (GstPlugin * plugin)
51 #else // GSTREAMER_LITE
52 static gboolean
53 plugin_init (GstPlugin * plugin)
54 #endif // GSTREAMER_LITE
55 {
56 #ifdef GSTREAMER_LITE
57   return (gst_element_register (plugin, "audiopanorama", GST_RANK_NONE,
58           GST_TYPE_AUDIO_PANORAMA));
59 #else // GSTREAMER_LITE
60   return (gst_element_register (plugin, "audiopanorama", GST_RANK_NONE,
61           GST_TYPE_AUDIO_PANORAMA) &&
62       gst_element_register (plugin, "audioinvert", GST_RANK_NONE,
63           GST_TYPE_AUDIO_INVERT) &&
64       gst_element_register (plugin, "audiokaraoke", GST_RANK_NONE,
65           GST_TYPE_AUDIO_KARAOKE) &&
66       gst_element_register (plugin, "audioamplify", GST_RANK_NONE,
67           GST_TYPE_AUDIO_AMPLIFY) &&
68       gst_element_register (plugin, "audiodynamic", GST_RANK_NONE,
69           GST_TYPE_AUDIO_DYNAMIC) &&
70       gst_element_register (plugin, "audiocheblimit", GST_RANK_NONE,
71           GST_TYPE_AUDIO_CHEB_LIMIT) &&
72       gst_element_register (plugin, "audiochebband", GST_RANK_NONE,
73           GST_TYPE_AUDIO_CHEB_BAND) &&
74       gst_element_register (plugin, "audioiirfilter", GST_RANK_NONE,
75           GST_TYPE_AUDIO_IIR_FILTER) &&
76       gst_element_register (plugin, "audiowsinclimit", GST_RANK_NONE,
77           GST_TYPE_AUDIO_WSINC_LIMIT) &&
78       gst_element_register (plugin, "audiowsincband", GST_RANK_NONE,
79           GST_TYPE_AUDIO_WSINC_BAND) &&
80       gst_element_register (plugin, "audiofirfilter", GST_RANK_NONE,
81           GST_TYPE_AUDIO_FIR_FILTER) &&
82       gst_element_register (plugin, "audioecho", GST_RANK_NONE,
83           GST_TYPE_AUDIO_ECHO) &&
84       gst_element_register (plugin, "scaletempo", GST_RANK_NONE,
85           GST_TYPE_SCALETEMPO));
86 #endif // GSTREAMER_LITE
87 }
