41 /**
42  * GstTypeFindElement:
43  *
44  * Opaque #GstTypeFindElement data structure
45  */
46 struct _GstTypeFindElement {
47   GstElement    element;
48 
49   GstPad *    sink;
50   GstPad *    src;
51 
52   guint     min_probability;
53   GstCaps *   caps;
54 
55   guint     mode;
56   GstAdapter *    adapter;
57 
58   GList *               cached_events;
59   GstCaps *             force_caps;
60 
61   guint64   initial_offset;
62 
63   /* Only used when driving the pipeline */
64   gboolean need_segment;
65   gboolean need_stream_start;
66   GstSegment segment;
67   guint64 offset;
68 };
69 
70 struct _GstTypeFindElementClass {
71   GstElementClass   parent_class;
72 
73   /* signals */
74   void      (*have_type)  (GstTypeFindElement *element,
75            guint         probability,
76            GstCaps            *caps);
77 };
