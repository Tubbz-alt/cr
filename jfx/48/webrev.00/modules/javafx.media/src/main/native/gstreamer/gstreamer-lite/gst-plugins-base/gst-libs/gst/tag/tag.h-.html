367 /**
368  * GstTagImageType:
369  * @GST_TAG_IMAGE_TYPE_NONE                  : No image type. Can be used to
370  *      tell functions such as gst_tag_image_data_to_image_sample() that no
371  *      image type should be set.
372  * @GST_TAG_IMAGE_TYPE_UNDEFINED             : Undefined/other image type
373  * @GST_TAG_IMAGE_TYPE_FRONT_COVER           : Cover (front)
374  * @GST_TAG_IMAGE_TYPE_BACK_COVER            : Cover (back)
375  * @GST_TAG_IMAGE_TYPE_LEAFLET_PAGE          : Leaflet page
376  * @GST_TAG_IMAGE_TYPE_MEDIUM                : Medium (e.g. label side of CD)
377  * @GST_TAG_IMAGE_TYPE_LEAD_ARTIST           : Lead artist/lead performer/soloist
378  * @GST_TAG_IMAGE_TYPE_ARTIST                : Artist/performer
379  * @GST_TAG_IMAGE_TYPE_CONDUCTOR             : Conductor
380  * @GST_TAG_IMAGE_TYPE_BAND_ORCHESTRA        : Band/orchestra
381  * @GST_TAG_IMAGE_TYPE_COMPOSER              : Composer
382  * @GST_TAG_IMAGE_TYPE_LYRICIST              : Lyricist/text writer
383  * @GST_TAG_IMAGE_TYPE_RECORDING_LOCATION    : Recording location
384  * @GST_TAG_IMAGE_TYPE_DURING_RECORDING      : During recording
385  * @GST_TAG_IMAGE_TYPE_DURING_PERFORMANCE    : During performance
386  * @GST_TAG_IMAGE_TYPE_VIDEO_CAPTURE         : Movie/video screen capture
387  * @GST_TAG_IMAGE_TYPE_FISH                  : A fish as funny as the ID3v2 spec
388  * @GST_TAG_IMAGE_TYPE_ILLUSTRATION          : Illustration
389  * @GST_TAG_IMAGE_TYPE_BAND_ARTIST_LOGO      : Band/artist logotype
390  * @GST_TAG_IMAGE_TYPE_PUBLISHER_STUDIO_LOGO : Publisher/studio logotype
391  *
392  * Type of image contained in an image tag (specified as "image-type" field in
393  * the info structure in the image's #GstSample)
394  */
395 /* Note: keep in sync with register_tag_image_type_enum() */
396 typedef enum {
397   GST_TAG_IMAGE_TYPE_NONE = -1,
398   GST_TAG_IMAGE_TYPE_UNDEFINED = 0,
399   GST_TAG_IMAGE_TYPE_FRONT_COVER,
400   GST_TAG_IMAGE_TYPE_BACK_COVER,
401   GST_TAG_IMAGE_TYPE_LEAFLET_PAGE,
402   GST_TAG_IMAGE_TYPE_MEDIUM,
403   GST_TAG_IMAGE_TYPE_LEAD_ARTIST,
404   GST_TAG_IMAGE_TYPE_ARTIST,
405   GST_TAG_IMAGE_TYPE_CONDUCTOR,
406   GST_TAG_IMAGE_TYPE_BAND_ORCHESTRA,
407   GST_TAG_IMAGE_TYPE_COMPOSER,
408   GST_TAG_IMAGE_TYPE_LYRICIST,
409   GST_TAG_IMAGE_TYPE_RECORDING_LOCATION,
410   GST_TAG_IMAGE_TYPE_DURING_RECORDING,
411   GST_TAG_IMAGE_TYPE_DURING_PERFORMANCE,
412   GST_TAG_IMAGE_TYPE_VIDEO_CAPTURE,
413   GST_TAG_IMAGE_TYPE_FISH,
414   GST_TAG_IMAGE_TYPE_ILLUSTRATION,
415   GST_TAG_IMAGE_TYPE_BAND_ARTIST_LOGO,
416   GST_TAG_IMAGE_TYPE_PUBLISHER_STUDIO_LOGO
417 } GstTagImageType;
