20 /**
21  * SECTION:gstpluginsbaseversion
22  * @title: Version
23  * @short_description: GStreamer gst-plugins-base libraries version macros.
24  *
25  * Use the GST_PLUGINS_BASE_VERSION_* macros e.g. to check what version of
26  * gst-plugins-base you are building against, and gst_plugins_base_version()
27  * if you need to check at runtime what version of the gst-plugins-base
28  * libraries are being used / you are currently linked against.
29  *
30  * The version macros get defined by including &lt;gst/pbutils/pbutils.h&gt;.
31  */



32 
33 #include "gstpluginsbaseversion.h"
34 
35 /**
36  * gst_plugins_base_version:
37  * @major: (out): pointer to a guint to store the major version number, or %NULL
38  * @minor: (out): pointer to a guint to store the minor version number, or %NULL
39  * @micro: (out): pointer to a guint to store the micro version number, or %NULL
40  * @nano:  (out): pointer to a guint to store the nano version number, or %NULL
41  *
42  * Gets the version number of the GStreamer Plugins Base libraries.
43  */
44 void
45 gst_plugins_base_version (guint * major, guint * minor, guint * micro,
46     guint * nano)
47 {
48   if (major)
49     *major = GST_PLUGINS_BASE_VERSION_MAJOR;
50   if (minor)
51     *minor = GST_PLUGINS_BASE_VERSION_MINOR;
52   if (micro)
53     *micro = GST_PLUGINS_BASE_VERSION_MICRO;
54   if (nano)
55     *nano = GST_PLUGINS_BASE_VERSION_NANO;
56 }
57 
58 /**
59  * gst_plugins_base_version_string:
60  *
61  * This function returns a string that is useful for describing this version
62  * of GStreamer's gst-plugins-base libraries to the outside world: user agent
63  * strings, logging, about dialogs ...
64  *
65  * Returns: a newly allocated string describing this version of gst-plugins-base
66  */
67 gchar *
68 gst_plugins_base_version_string (void)
69 {
70   return g_strdup_printf ("GStreamer Base Plugins %d.%d.%d%s",
71       GST_PLUGINS_BASE_VERSION_MAJOR, GST_PLUGINS_BASE_VERSION_MINOR,
72       GST_PLUGINS_BASE_VERSION_MICRO,
73       ((GST_PLUGINS_BASE_VERSION_NANO == 0) ? "" :
74           ((GST_PLUGINS_BASE_VERSION_NANO == 1) ? " (GIT)" : " (prerelease)")));
75 }
