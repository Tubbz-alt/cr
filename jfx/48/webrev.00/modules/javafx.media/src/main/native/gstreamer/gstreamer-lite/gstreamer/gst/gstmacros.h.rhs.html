19 #ifndef __GST_MACROS_H__
20 #define __GST_MACROS_H__
21 
22 #include <glib.h>
23 
24 G_BEGIN_DECLS
25 
26 #if __GNUC__ > 2 || (__GNUC__ == 2 && __GNUC_MINOR__ > 4)
27 # define GST_GNUC_CONSTRUCTOR \
28   __attribute__ ((constructor))
29 #else  /* !__GNUC__ */
30 # define GST_GNUC_CONSTRUCTOR
31 #endif /* !__GNUC__ */
32 
33 #if defined (__GNUC__) && !defined (GST_IMPLEMENT_INLINES)
34 # define GST_INLINE_FUNC extern __inline__
35 # define GST_CAN_INLINE 1
36 #elif defined(_MSC_VER)
37 # define GST_INLINE_FUNC extern __inline
38 # define GST_CAN_INLINE 1
39 #else
40 # define GST_INLINE_FUNC extern
41 # undef GST_CAN_INLINE
42 #endif
43 
44 /* MSVC defines 'restrict' as a keyword and not a define */
45 #if (!defined(__STDC_VERSION__) || __STDC_VERSION__ < 199901L) && !defined(restrict)
46 #  if defined(__GNUC__) && __GNUC__ >= 4
47 #    define restrict __restrict__
48 #  elif defined(_MSC_VER)
49 #    define restrict __restrict
50 #  else
51 #    define restrict
52 #  endif
53 #endif
54 
55 G_END_DECLS
56 
57 #endif /* __GST_MACROS_H__ */
