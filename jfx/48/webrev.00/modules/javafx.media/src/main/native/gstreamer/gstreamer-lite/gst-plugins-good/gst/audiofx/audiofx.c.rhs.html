44 /* entry point to initialize the plug-in
45  * initialize the plug-in itself
46  * register the element factories and pad templates
47  * register the features
48  */
49 #ifdef GSTREAMER_LITE
50 gboolean
51 plugin_init_audiofx (GstPlugin * plugin)
52 #else // GSTREAMER_LITE
53 static gboolean
54 plugin_init (GstPlugin * plugin)
55 #endif // GSTREAMER_LITE
56 {
57 #ifdef GSTREAMER_LITE
58   return (gst_element_register (plugin, "audiopanorama", GST_RANK_NONE,
59           GST_TYPE_AUDIO_PANORAMA));
60 #else // GSTREAMER_LITE
61   return (gst_element_register (plugin, "audiopanorama", GST_RANK_NONE,
62           GST_TYPE_AUDIO_PANORAMA) &&
63       gst_element_register (plugin, "audioinvert", GST_RANK_NONE,
64           GST_TYPE_AUDIO_INVERT) &&
65       gst_element_register (plugin, "audiokaraoke", GST_RANK_NONE,
66           GST_TYPE_AUDIO_KARAOKE) &&
67       gst_element_register (plugin, "audioamplify", GST_RANK_NONE,
68           GST_TYPE_AUDIO_AMPLIFY) &&
69       gst_element_register (plugin, "audiodynamic", GST_RANK_NONE,
70           GST_TYPE_AUDIO_DYNAMIC) &&
71       gst_element_register (plugin, "audiocheblimit", GST_RANK_NONE,
72           GST_TYPE_AUDIO_CHEB_LIMIT) &&
73       gst_element_register (plugin, "audiochebband", GST_RANK_NONE,
74           GST_TYPE_AUDIO_CHEB_BAND) &&
75       gst_element_register (plugin, "audioiirfilter", GST_RANK_NONE,
76           GST_TYPE_AUDIO_IIR_FILTER) &&
77       gst_element_register (plugin, "audiowsinclimit", GST_RANK_NONE,
78           GST_TYPE_AUDIO_WSINC_LIMIT) &&
79       gst_element_register (plugin, "audiowsincband", GST_RANK_NONE,
80           GST_TYPE_AUDIO_WSINC_BAND) &&
81       gst_element_register (plugin, "audiofirfilter", GST_RANK_NONE,
82           GST_TYPE_AUDIO_FIR_FILTER) &&
83       gst_element_register (plugin, "audioecho", GST_RANK_NONE,
84           GST_TYPE_AUDIO_ECHO) &&
85       gst_element_register (plugin, "scaletempo", GST_RANK_NONE,
86           GST_TYPE_SCALETEMPO) &&
87       gst_element_register (plugin, "stereo", GST_RANK_NONE, GST_TYPE_STEREO));
88 #endif // GSTREAMER_LITE
89 }

94     audiofx,
95     "Audio effects plugin",
