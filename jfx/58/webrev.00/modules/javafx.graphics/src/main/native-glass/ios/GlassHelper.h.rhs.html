36 /*
37  * Function to find a glass class using the context class loader. All glass
38  * classes must be looked up using this function rather than FindClass so that
39  * the correct ClassLoader is used.
40  *
41  * Note that the className passed to this function must use "." rather than "/"
42  * as a package separator.
43  */
44 + (jclass)ClassForName:(char*)className withEnv:(JNIEnv*)env;
45 + (jmethodID)ApplicationNotifyWillBecomeActiveMethod;
46 + (jmethodID)ApplicationNotifyDidBecomeActiveMethod;
47 + (jmethodID)ApplicationNotifyWillResignActiveMethod;
48 + (jmethodID)ApplicationNotifyDidResignActiveMethod;
49 + (jmethodID)ApplicationNotifyWillQuitMethod;
50 + (jmethodID)ApplicationNotifyDidReceiveMemoryWarningMethod;
