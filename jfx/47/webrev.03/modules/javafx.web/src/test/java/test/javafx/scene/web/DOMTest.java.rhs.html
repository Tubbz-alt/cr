250     @Test public void testNodeTypes() {
251         final Document doc = getDocumentFor("src/test/resources/test/html/dom.html");
252         submit(() -> {
253             Element p = doc.getElementById("showcase-paragraph");
254             assertEquals("P element's node type", Node.ELEMENT_NODE, p.getNodeType());
255             assertEquals("P element's tag name", "P", p.getTagName());
256 
257             NodeList children = p.getChildNodes();
258             assertEquals("Paragraph child count", 3, children.getLength());
259             Node text = children.item(0);
260             assertEquals("Text node type", Node.TEXT_NODE, text.getNodeType());
261             Node comment = children.item(1);
262             assertEquals("Comment node type", Node.COMMENT_NODE, comment.getNodeType());
263             Node element = children.item(2);
264             assertEquals("SPAN element's node type", Node.ELEMENT_NODE, element.getNodeType());
265 
266             Element span = (Element) element;
267             assertEquals("SPAN element's tag name", "SPAN", span.getTagName());
268             assertTrue("SPAN has 'class' attribute", span.hasAttribute("class"));
269             assertTrue("SPAN has 'CLASS' attribute", span.hasAttribute("CLASS"));
270             assertEquals("SPAN attributes count", 1, span.getAttributes().getLength());
271 
272             Attr attr = span.getAttributeNode("class");
273             assertEquals("Attr node type", Node.ATTRIBUTE_NODE, attr.getNodeType());
274             children = span.getChildNodes();
275             assertEquals("SPAN element child count", 1, children.getLength());
276             text = children.item(0);
277             assertEquals("SPAN text node type", Node.TEXT_NODE, text.getNodeType());

440     // helper methods

442     private void verifyChildRemoved(Node parent,
443             int oldChildrenCount, Node leftSibling, Node rightSibling) {
444         assertSame("Children count",
445                 oldChildrenCount - 1, parent.getChildNodes().getLength());
446         assertSame("Left sibling's next sibling",
447                 rightSibling, leftSibling.getNextSibling());
448         assertSame("Right sibling's previous sibling",
449                 leftSibling, rightSibling.getPreviousSibling());
450     }
451 
452     private void verifyChildAdded(Node n, Node parent, int oldChildrenCount) {
453         assertEquals("Children count",
454                 oldChildrenCount + 1, parent.getChildNodes().getLength());
455         assertEquals("Added node's parent",
456                 parent, n.getParentNode());
457     }
458 
459     private void verifySiblings(Node n, Node leftSibling, Node rightSibling) {
460         assertSame("Added node's previous sibling",
461                 leftSibling, n.getPreviousSibling());
462         assertSame("Added node's next sibling",
463                 rightSibling, n.getNextSibling());
464 
465         if (leftSibling != null)
466             assertSame("Previous sibling's next sibling",
467                     n, leftSibling.getNextSibling());
468 
469         if (rightSibling != null)
470             assertSame("Next sibling's previous sibling",
471                     n, rightSibling.getPreviousSibling());
472     }
473 
474     private void verifyNodeRemoved(Node n) {
475         assertNull("Removed node's parent", n.getParentNode());
476         assertNull("Removed node's previous sibling", n.getPreviousSibling());
477         assertNull("Removed node's next sibling", n.getNextSibling());
478     }
