 26 package com.sun.javafx.sg.prism;
 27 
 28 /**
 29  * The peer of the {@code PointLight} class. Holds the default values of {@code PointLight}'s
 30  * properties and updates the visuals via {@link NGNode#visualsChanged} when one of the current
 31  * values changes. The peer receives its changes by {@code PointLight.doUpdatePeer} calls.
 32  */
 33 public class NGPointLight extends NGLightBase {
 34 
 35     /** Constant attenuation factor default value */
 36     private static final double DEFAULT_CA = 1;
 37     /** Linear attenuation factor default value */
 38     private static final double DEFAULT_LA = 0;
 39     /** Quadratic attenuation factor default value */
 40     private static final double DEFAULT_QA = 0;
 41     /** Max range default value */
 42     private static final double DEFAULT_MAX_RANGE = Double.POSITIVE_INFINITY;
 43 
 44     public NGPointLight() {
 45     }
 46 
 47     public static double getDefaultCa() {
 48         return DEFAULT_CA;
 49     }
 50 
 51     public static double getDefaultLa() {
 52         return DEFAULT_LA;
 53     }
 54 
 55     public static double getDefaultQa() {
 56         return DEFAULT_QA;
 57     }
 58 
 59     public static double getDefaultMaxRange() {
 60         return DEFAULT_MAX_RANGE;
 61     }
 62 
 63 
 64     private double ca = DEFAULT_CA;
 65 
 66     public double getCa() {
 67         return ca;
 68     }
 69 
 70     public void setCa(double ca) {
 71         this.ca = ca;
 72         visualsChanged();


 73     }
 74 
 75 
 76     private double la = DEFAULT_LA;
 77 
 78     public double getLa() {
 79         return la;
 80     }
 81 
 82     public void setLa(double la) {
 83         this.la = la;
 84         visualsChanged();


 85     }
 86 
 87 
 88     private double qa = DEFAULT_QA;
 89 
 90     public double getQa() {
 91         return qa;
 92     }
 93 
 94     public void setQa(double qa) {
 95         this.qa = qa;
 96         visualsChanged();


 97     }
 98 
 99 
100     private double maxRange = DEFAULT_MAX_RANGE;
101 
102     public double getMaxRange() {
103         return maxRange;
104     }
105 
106     public void setMaxRange(double maxRange) {
107         this.maxRange = maxRange < 0 ? 0 : maxRange;
108         visualsChanged();


109     }
110 }
