26 package com.sun.webkit.network;
27 
28 import java.lang.annotation.Native;
29 import java.nio.ByteBuffer;
30 
31 abstract class URLLoaderBase {
32     @Native public static final int ALLOW_UNASSIGNED = java.net.IDN.ALLOW_UNASSIGNED;
33 
34     /**
35      * Cancels the loader.
36      */
37     protected abstract void fwkCancel();
38 
39     protected static native void twkDidSendData(long totalBytesSent,
40                                               long totalBytesToBeSent,
41                                               long data);
42 
43     protected static native void twkWillSendRequest(int status,
44                                                      String contentType,
45                                                      String contentEncoding,
46                                                      long contentLength,
47                                                      String headers,
48                                                      String url,
49                                                      long data);
50 
51     protected static native void twkDidReceiveResponse(int status,
52                                                      String contentType,
53                                                      String contentEncoding,
54                                                      long contentLength,
55                                                      String headers,
56                                                      String url,
57                                                      long data);
58 
59     protected static native void twkDidReceiveData(ByteBuffer byteBuffer,
60                                                  int position,
61                                                  int remaining,
62                                                  long data);
63 
64     protected static native void twkDidFinishLoading(long data);
65 
66     protected static native void twkDidFail(int errorCode,
67                                           String url,
68                                           String message,
69                                           long data);
70 
71 }
