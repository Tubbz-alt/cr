  40 static jint translate_gdk_action_to_glass(GdkDragAction action)
  41 {
  42     jint result = 0;
  43     result |= (action & GDK_ACTION_COPY)? com_sun_glass_ui_gtk_GtkDnDClipboard_ACTION_COPY : 0;
  44     result |= (action & GDK_ACTION_MOVE)? com_sun_glass_ui_gtk_GtkDnDClipboard_ACTION_MOVE : 0;
  45     result |= (action & GDK_ACTION_LINK)? com_sun_glass_ui_gtk_GtkDnDClipboard_ACTION_REFERENCE : 0;
  46     return result;
  47 }
  48 
  49 static GdkDragAction translate_glass_action_to_gdk(jint action)
  50 {
  51     int result = 0;
  52     result |= (action & com_sun_glass_ui_gtk_GtkDnDClipboard_ACTION_COPY)? GDK_ACTION_COPY : 0;
  53     result |= (action & com_sun_glass_ui_gtk_GtkDnDClipboard_ACTION_MOVE)? GDK_ACTION_MOVE : 0;
  54     result |= (action & com_sun_glass_ui_gtk_GtkDnDClipboard_ACTION_REFERENCE)? GDK_ACTION_LINK : 0;
  55     return static_cast<GdkDragAction>(result);
  56 }

  71 static void init_target_atoms()
  72 {
  73     if (target_atoms_initialized) {
  74         return;
  75     }
  76     TARGET_UTF8_STRING_ATOM = gdk_atom_intern_static_string("UTF8_STRING");
  77     TARGET_MIME_TEXT_PLAIN_ATOM = gdk_atom_intern_static_string("text/plain");
  78     TARGET_COMPOUND_TEXT_ATOM = gdk_atom_intern_static_string("COMPOUND_TEXT");
  79     TARGET_STRING_ATOM = gdk_atom_intern_static_string("STRING");
  80 
  81     TARGET_MIME_URI_LIST_ATOM = gdk_atom_intern_static_string("text/uri-list");
  82 
  83     TARGET_MIME_PNG_ATOM = gdk_atom_intern_static_string("image/png");
  84     TARGET_MIME_JPEG_ATOM = gdk_atom_intern_static_string("image/jpeg");
  85     TARGET_MIME_TIFF_ATOM = gdk_atom_intern_static_string("image/tiff");
  86     TARGET_MIME_BMP_ATOM = gdk_atom_intern_static_string("image/bmp");
  87 
  88     target_atoms_initialized = TRUE;
  89 }

  91 static gboolean target_is_text(GdkAtom target)
  92 {
  93     init_target_atoms();
  94 
  95     return (target == TARGET_UTF8_STRING_ATOM ||
  96             target == TARGET_STRING_ATOM ||
  97             target == TARGET_MIME_TEXT_PLAIN_ATOM/* ||
  98             target == TARGET_COMPOUND_TEXT_ATOM*/);
  99 }

 150 static void reset_enter_ctx() {
 151     if (enter_ctx.mimes != NULL) {
 152         mainEnv->DeleteGlobalRef(enter_ctx.mimes);
 153     }
 154 
 155     memset(&enter_ctx, 0, sizeof(enter_ctx));
 156 }

 158 static void process_dnd_target_drag_enter(WindowContext *ctx, GdkEventDND *event)
 159 {
 160     reset_enter_ctx();
 161     enter_ctx.ctx = event->context;
 162     enter_ctx.just_entered = TRUE;
 163     gdk_window_get_origin(ctx->get_gdk_window(), &enter_ctx.dx, &enter_ctx.dy);
 164     is_dnd_owner = is_in_drag();
 165 }

 167 static void process_dnd_target_drag_motion(WindowContext *ctx, GdkEventDND *event)
 168 {
 169     if (!enter_ctx.ctx) {
 170         gdk_drag_status(event->context, static_cast<GdkDragAction>(0), GDK_CURRENT_TIME);
 171         return; // Do not process motion events if no enter event was received
 172     }
 173     jmethodID method = enter_ctx.just_entered ? jViewNotifyDragEnter : jViewNotifyDragOver;
 174     GdkDragAction suggested = gdk_drag_context_get_suggested_action(event->context);
 175     GdkDragAction result = translate_glass_action_to_gdk(mainEnv->CallIntMethod(ctx->get_jview(), method,
 176             (jint)event->x_root - enter_ctx.dx, (jint)event->y_root - enter_ctx.dy,
 177             (jint)event->x_root, (jint)event->y_root,
 178             translate_gdk_action_to_glass(suggested)));
 179     CHECK_JNI_EXCEPTION(mainEnv)
 180 
 181     if (enter_ctx.just_entered) {
 182         enter_ctx.just_entered = FALSE;
 183     }
 184     gdk_drag_status(event->context, result, GDK_CURRENT_TIME);
 185 }

 187 static void process_dnd_target_drag_leave(WindowContext *ctx, GdkEventDND *event)
 188 {
 189     (void)event;
 190 
 191     mainEnv->CallVoidMethod(ctx->get_jview(), jViewNotifyDragLeave, NULL);
 192     CHECK_JNI_EXCEPTION(mainEnv)
 193 }

 195 static void process_dnd_target_drop_start(WindowContext *ctx, GdkEventDND *event)
 196 {
 197     if (!enter_ctx.ctx || enter_ctx.just_entered) {
 198         gdk_drop_finish(event->context, FALSE, GDK_CURRENT_TIME);
 199         gdk_drop_reply(event->context, FALSE, GDK_CURRENT_TIME);
 200         return; // Do not process drop events if no enter event and subsequent motion event were received
 201     }
 202     GdkDragAction selected = gdk_drag_context_get_selected_action(event->context);
 203 
 204     mainEnv->CallIntMethod(ctx->get_jview(), jViewNotifyDragDrop,
 205             (jint)event->x_root - enter_ctx.dx, (jint)event->y_root - enter_ctx.dy,
 206             (jint)event->x_root, (jint)event->y_root,
 207             translate_gdk_action_to_glass(selected));
 208     LOG_EXCEPTION(mainEnv)
 209 
 210     gdk_drop_finish(event->context, TRUE, GDK_CURRENT_TIME);
 211     gdk_drop_reply(event->context, TRUE, GDK_CURRENT_TIME);
 212 }

 214 static gboolean check_state_in_drag(JNIEnv *env)
 215 {
 216     if (!enter_ctx.ctx) {
 217         jclass jc = env->FindClass("java/lang/IllegalStateException");
 218         if (!env->ExceptionCheck()) {
 219             env->ThrowNew(jc,
 220                     "Cannot get supported actions. Drag pointer haven't entered the application window");
 221         }
 222         return TRUE;
 223     }
 224     return FALSE;
 225 }

 227 // Events coming from application that are related to us being a DnD target
 228 void process_dnd_target(WindowContext *ctx, GdkEventDND *event)
 229 {
 230     switch (event->type) {
 231         case GDK_DRAG_ENTER:
 232             process_dnd_target_drag_enter(ctx, event);
 233             break;
 234         case GDK_DRAG_MOTION:
 235             process_dnd_target_drag_motion(ctx, event);
 236             break;
 237         case GDK_DRAG_LEAVE:
 238             process_dnd_target_drag_leave(ctx, event);
 239             break;
 240         case GDK_DROP_START:
 241             process_dnd_target_drop_start(ctx, event);
 242             break;
 243         default:
 244             break;
 245     }
 246 }

 248 jobjectArray dnd_target_get_mimes(JNIEnv *env)
 249 {
 250     if (check_state_in_drag(env)) {
 251         return NULL;
 252     }
 253     if (!enter_ctx.mimes) {
 254         GList* targets = gdk_drag_context_list_targets(enter_ctx.ctx);
 255         jobject set = env->NewObject(jHashSetCls, jHashSetInit, NULL);
 256         EXCEPTION_OCCURED(env);
 257 
 258         while (targets) {
 259             GdkAtom target = GDK_POINTER_TO_ATOM(targets->data);
 260             gchar *name = gdk_atom_name(target);
 261 
 262             if (target_is_text(target)) {
 263                 jstring jStr = env->NewStringUTF("text/plain");
 264                 EXCEPTION_OCCURED(env);
 265                 env->CallBooleanMethod(set, jSetAdd, jStr, NULL);
 266                 EXCEPTION_OCCURED(env);
 267             }
 268 
 269             if (target_is_image(target)) {
 270                 jstring jStr = env->NewStringUTF("application/x-java-rawimage");
 271                 EXCEPTION_OCCURED(env);
 272                 env->CallBooleanMethod(set, jSetAdd, jStr, NULL);
 273                 EXCEPTION_OCCURED(env);
 274             }
 275 
 276             if (target_is_uri(target)) {
 277                 selection_data_ctx ctx;
 278                 if (dnd_target_receive_data(env, TARGET_MIME_URI_LIST_ATOM, &ctx)) {
 279                     gchar** uris = g_uri_list_extract_uris((gchar *) ctx.data);
 280                     guint size = g_strv_length(uris);
 281                     guint files_cnt = get_files_count(uris);
 282                     if (files_cnt) {
 283                         jstring jStr = env->NewStringUTF("application/x-java-file-list");
 284                         EXCEPTION_OCCURED(env);
 285                         env->CallBooleanMethod(set, jSetAdd, jStr, NULL);
 286                         EXCEPTION_OCCURED(env);
 287                     }
 288                     if (size - files_cnt) {
 289                         jstring jStr = env->NewStringUTF("text/uri-list");
 290                         EXCEPTION_OCCURED(env);
 291                         env->CallBooleanMethod(set, jSetAdd, jStr, NULL);
 292                         EXCEPTION_OCCURED(env);
 293                     }
 294                     g_strfreev(uris);
 295                 }
 296                 g_free(ctx.data);
 297             } else {
 298                 jstring jStr = env->NewStringUTF(name);
 299                 EXCEPTION_OCCURED(env);
 300                 env->CallBooleanMethod(set, jSetAdd, jStr, NULL);
 301                 EXCEPTION_OCCURED(env);
 302             }
 303 
 304             g_free(name);
 305             targets = targets->next;
 306         }
 307         enter_ctx.mimes = env->NewObjectArray(env->CallIntMethod(set, jSetSize, NULL),
 308                 jStringCls, NULL);
 309         EXCEPTION_OCCURED(env);
 310         enter_ctx.mimes = (jobjectArray)env->CallObjectMethod(set, jSetToArray, enter_ctx.mimes, NULL);
 311         enter_ctx.mimes = (jobjectArray)env->NewGlobalRef(enter_ctx.mimes);
 312     }
 313     return enter_ctx.mimes;
 314 }

 316 jint dnd_target_get_supported_actions(JNIEnv *env)
 317 {
 318     if (check_state_in_drag(env)) {
 319         return 0;
 320     }
 321     return translate_gdk_action_to_glass(gdk_drag_context_get_actions(enter_ctx.ctx));

 324 static void wait_for_selection_data_hook(GdkEvent * event, void * data)
 325 {
 326     selection_data_ctx *ctx = (selection_data_ctx*)data;
 327     GdkWindow *dest = glass_gdk_drag_context_get_dest_window(enter_ctx.ctx);
 328     if (event->type == GDK_SELECTION_NOTIFY &&
 329             event->selection.window == dest) {
 330         if (event->selection.property) { // if 0, that we received negative response
 331             ctx->length = gdk_selection_property_get(dest, &(ctx->data), &(ctx->type), &(ctx->format));
 332         }
 333         ctx->received = TRUE;
 334     }
 335 }

 337 static gboolean dnd_target_receive_data(JNIEnv *env, GdkAtom target, selection_data_ctx *selection_ctx)
 338 {
 339     GevlHookRegistration hookReg;
 340 
 341     memset(selection_ctx, 0, sizeof(selection_data_ctx));
 342 
 343     gdk_selection_convert(glass_gdk_drag_context_get_dest_window(enter_ctx.ctx), gdk_drag_get_selection(enter_ctx.ctx), target,
 344                           GDK_CURRENT_TIME);
 345 
 346     hookReg =
 347             glass_evloop_hook_add(
 348                     (GevlHookFunction) wait_for_selection_data_hook,
 349                     selection_ctx);
 350     if (HANDLE_MEM_ALLOC_ERROR(env, hookReg,
 351                                "Failed to allocate event hook")) {
 352         return TRUE;
 353     }
 354 
 355     do {
 356         gtk_main_iteration();
 357     } while (!(selection_ctx->received));
 358 
 359 
 360     glass_evloop_hook_remove(hookReg);
 361     return selection_ctx->data != NULL;
 362 }

 364 static jobject dnd_target_get_string(JNIEnv *env)
 365 {
 366     jobject result = NULL;
 367     selection_data_ctx ctx;
 368 
 369     if (dnd_target_receive_data(env, TARGET_UTF8_STRING_ATOM, &ctx)) {
 370         result = env->NewStringUTF((char *)ctx.data);
 371         EXCEPTION_OCCURED(env);
 372         g_free(ctx.data);
 373     }
 374     if (!result && dnd_target_receive_data(env, TARGET_MIME_TEXT_PLAIN_ATOM, &ctx)) {
 375         result = env->NewStringUTF((char *)ctx.data);
 376         EXCEPTION_OCCURED(env);
 377         g_free(ctx.data);
 378     }
 379     // TODO find out how to convert from compound text
 380     // if (!result && dnd_target_receive_data(env, TARGET_COMPOUND_TEXT_ATOM, &ctx)) {
 381     // }
 382     if (!result && dnd_target_receive_data(env, TARGET_STRING_ATOM, &ctx)) {
 383         gchar *str;
 384         str = g_convert( (gchar *)ctx.data, -1, "UTF-8", "ISO-8859-1", NULL, NULL, NULL);
 385         if (str != NULL) {
 386             result = env->NewStringUTF(str);
 387             EXCEPTION_OCCURED(env);
 388             g_free(str);
 389         }
 390         g_free(ctx.data);
 391     }
 392     return result;
 393 }

 408 static jobject dnd_target_get_image(JNIEnv *env)
 409 {
 410     GdkPixbuf *buf;
 411     GInputStream *stream;
 412     jobject result = NULL;
 413     GdkAtom targets[] = {
 414         TARGET_MIME_PNG_ATOM,
 415         TARGET_MIME_JPEG_ATOM,
 416         TARGET_MIME_TIFF_ATOM,
 417         TARGET_MIME_BMP_ATOM,
 418         0};
 419     GdkAtom *cur_target = targets;
 420     selection_data_ctx ctx;
 421 
 422     while(*cur_target != 0 && result == NULL) {
 423         if (dnd_target_receive_data(env, *cur_target, &ctx)) {
 424             stream = g_memory_input_stream_new_from_data(ctx.data, ctx.length * (ctx.format / 8),
 425                     (GDestroyNotify)g_free);
 426             buf = gdk_pixbuf_new_from_stream(stream, NULL, NULL);
 427             if (buf) {
 428                 int w;
 429                 int h;
 430                 int stride;
 431                 guchar *data;
 432                 jbyteArray data_array;
 433                 jobject buffer;
 434 
 435                 if (!gdk_pixbuf_get_has_alpha(buf)) {
 436                     GdkPixbuf *tmp_buf = gdk_pixbuf_add_alpha(buf, FALSE, 0, 0, 0);
 437                     g_object_unref(buf);
 438                     buf = tmp_buf;
 439                 }
 440 
 441                 w = gdk_pixbuf_get_width(buf);
 442                 h = gdk_pixbuf_get_height(buf);
 443                 stride = gdk_pixbuf_get_rowstride(buf);
 444                 data = gdk_pixbuf_get_pixels(buf);
 445 
 446                 //Actually, we are converting RGBA to BGRA, but that's the same operation
 447                 data = (guchar*) convert_BGRA_to_RGBA((int*) data, stride, h);
 448                 data_array = env->NewByteArray(stride * h);
 449                 EXCEPTION_OCCURED(env);
 450                 env->SetByteArrayRegion(data_array, 0, stride*h, (jbyte*) data);
 451                 EXCEPTION_OCCURED(env);
 452 
 453                 buffer = env->CallStaticObjectMethod(jByteBufferCls, jByteBufferWrap, data_array);
 454                 EXCEPTION_OCCURED(env);
 455                 result = env->NewObject(jGtkPixelsCls, jGtkPixelsInit, w, h, buffer);
 456                 EXCEPTION_OCCURED(env);
 457 
 458                 g_object_unref(buf);
 459                 g_free(data); // data from convert_BGRA_to_RGBA
 460             }
 461             g_object_unref(stream);
 462         }
 463         ++cur_target;
 464     }
 465     return result;
 466 }

 468 static jobject dnd_target_get_raw(JNIEnv *env, GdkAtom target, gboolean string_data)
 469 {
 470     selection_data_ctx ctx;
 471     jobject result = NULL;
 472     if (dnd_target_receive_data(env, target, &ctx)) {
 473         if (string_data) {
 474              result = env->NewStringUTF((char *)ctx.data);
 475              EXCEPTION_OCCURED(env);
 476         } else {
 477             jsize length = ctx.length * (ctx.format / 8);
 478             jbyteArray array = env->NewByteArray(length);
 479             EXCEPTION_OCCURED(env);
 480             env->SetByteArrayRegion(array, 0, length, (const jbyte*)ctx.data);
 481             EXCEPTION_OCCURED(env);
 482             result = env->CallStaticObjectMethod(jByteBufferCls, jByteBufferWrap, array);
 483             EXCEPTION_OCCURED(env);
 484         }
 485     }
 486     g_free(ctx.data);
 487     return result;
 488 }

 490 jobject dnd_target_get_data(JNIEnv *env, jstring mime)
 491 {
 492     if (check_state_in_drag(env)) {
 493         return NULL;
 494     }
 495     const char *cmime = env->GetStringUTFChars(mime, NULL);
 496     jobject ret = NULL;
 497 
 498     init_target_atoms();
 499 
 500     if (g_strcmp0(cmime, "text/plain") == 0) {
 501         ret = dnd_target_get_string(env);
 502     } else if (g_strcmp0(cmime, "text/uri-list") == 0) {
 503         ret = dnd_target_get_list(env, FALSE);
 504     } else if (g_str_has_prefix(cmime, "text/")) {
 505         ret = dnd_target_get_raw(env, gdk_atom_intern(cmime, FALSE), TRUE);
 506     } else if (g_strcmp0(cmime, "application/x-java-file-list") == 0) {
 507         ret = dnd_target_get_list(env, TRUE);
 508     } else if (g_strcmp0(cmime, "application/x-java-rawimage") == 0 ) {
 509         ret = dnd_target_get_image(env);
 510     } else {
 511         ret = dnd_target_get_raw(env, gdk_atom_intern(cmime, FALSE), FALSE);
 512     }
 513     LOG_EXCEPTION(env)
 514     env->ReleaseStringUTFChars(mime, cmime);
 515 
 516     return ret;
 517 }

 553 static void add_gtk_target_from_jstring(JNIEnv *env, GtkTargetList **list, jstring string, guint flags)
 554 {
 555     const char *gstring = env->GetStringUTFChars(string, NULL);
 556 
 557     if (g_strcmp0(gstring, "text/plain") == 0) {
 558         gtk_target_list_add(*list, TARGET_UTF8_STRING_ATOM, flags, 0);
 559         gtk_target_list_add(*list, TARGET_MIME_TEXT_PLAIN_ATOM, flags, 0);
 560         gtk_target_list_add(*list, TARGET_STRING_ATOM, flags, 0);
 561         //gtk_target_list_add(*list, TARGET_COMPOUND_TEXT_ATOM, flags, ??);
 562     } else if (g_strcmp0(gstring, "application/x-java-rawimage") == 0) {
 563         gtk_target_list_add(*list, TARGET_MIME_PNG_ATOM, flags, 0);
 564         gtk_target_list_add(*list, TARGET_MIME_JPEG_ATOM, flags, 0);
 565         gtk_target_list_add(*list, TARGET_MIME_TIFF_ATOM, flags, 0);
 566         gtk_target_list_add(*list, TARGET_MIME_BMP_ATOM, flags, 0);
 567     } else if (g_strcmp0(gstring, "application/x-java-file-list") == 0) {
 568         gtk_target_list_add(*list, TARGET_MIME_URI_LIST_ATOM, flags, 0);
 569     } else if (g_strcmp0(gstring, "application/x-java-drag-image") == 0
 570         || g_strcmp0(gstring, "application/x-java-drag-image-offset") == 0) {
 571         // do nothing - those are DragView information
 572     } else {
 573         GdkAtom atom = gdk_atom_intern(gstring, FALSE);
 574         gtk_target_list_add(*list, atom, flags, 0);
 575     }
 576 
 577     env->ReleaseStringUTFChars(string, gstring);
 578 }

 581 {
 582     guint flags = GTK_TARGET_OTHER_APP | GTK_TARGET_SAME_APP;
 583 
 584     jobject keys;
 585     jobject keysIterator;
 586     jstring next;
 587 
 588     GtkTargetList *tlist = gtk_target_list_new (NULL, 0);
 589 
 590     init_target_atoms();
 591 
 592     gint added_count = 0;
 593 
 594     keys = env->CallObjectMethod(data, jMapKeySet, NULL);
 595     JNI_EXCEPTION_TO_CPP(env)
 596     keysIterator = env->CallObjectMethod(keys, jIterableIterator, NULL);
 597     JNI_EXCEPTION_TO_CPP(env)
 598     while (env->CallBooleanMethod(keysIterator, jIteratorHasNext) == JNI_TRUE) {
 599         next = (jstring)env->CallObjectMethod(keysIterator, jIteratorNext, NULL);
 600         JNI_EXCEPTION_TO_CPP(env)
 601         add_gtk_target_from_jstring(env, &tlist, next, flags);
 602     }
 603 
 604     return tlist;
 605 }

 659 static gboolean dnd_source_set_uri(GtkWidget *widget, GtkSelectionData *data, GdkAtom atom)
 660 {
 661     const gchar* url = NULL;
 662     jstring jurl = NULL;
 663 
 664     jobjectArray files_array = NULL;
 665     gsize files_cnt = 0;
 666 
 667     if (jurl = (jstring) dnd_source_get_data(widget, "text/uri-list")) {
 668         url = mainEnv->GetStringUTFChars(jurl, NULL);
 669     }
 670 
 671     if (files_array = (jobjectArray) dnd_source_get_data(widget, "application/x-java-file-list")) {
 672         files_cnt = mainEnv->GetArrayLength(files_array);
 673     }
 674 
 675     if (!url && !files_cnt) {
 676         return FALSE;
 677     }
 678 
 679     gboolean is_data_set;
 680     GString* res = g_string_new (NULL); //http://www.ietf.org/rfc/rfc2483.txt
 681 
 682     if (files_cnt > 0) {
 683         for (gsize i = 0; i < files_cnt; ++i) {
 684             jstring string = (jstring) mainEnv->GetObjectArrayElement(files_array, i);
 685             EXCEPTION_OCCURED(mainEnv);
 686             const gchar* file = mainEnv->GetStringUTFChars(string, NULL);
 687             gchar* uri = g_filename_to_uri(file, NULL, NULL);
 688 
 689             g_string_append(res, uri);
 690             g_string_append(res, URI_LIST_LINE_BREAK);
 691 
 692             g_free(uri);
 693             mainEnv->ReleaseStringUTFChars(string, file);
 694         }
 695     }
 696     if (url) {
 697         g_string_append(res, url);
 698         g_string_append(res, URI_LIST_LINE_BREAK);
 699         mainEnv->ReleaseStringUTFChars(jurl, url);
 700     }
 701 
 702     gchar *uri[2];
 703     uri[0] = g_string_free(res, FALSE);
 704     uri[1] = NULL;
 705 
 706     is_data_set = gtk_selection_data_set_uris(data, uri);
 707 
 708     g_free(uri[0]);
 709 
 710     return is_data_set;

 713 static gboolean dnd_source_set_raw(GtkWidget *widget, GtkSelectionData *sel_data, GdkAtom atom)
 714 {
 715     gchar *target_name = gdk_atom_name(atom);
 716     jobject data = dnd_source_get_data(widget, target_name);
 717     gboolean is_data_set = FALSE;
 718     if (data) {
 719         if (mainEnv->IsInstanceOf(data, jStringCls)) {
 720             const char *cstring = mainEnv->GetStringUTFChars((jstring)data, NULL);
 721             if (cstring) {
 722                 is_data_set = gtk_selection_data_set_text(sel_data, (gchar *) cstring, strlen(cstring));
 723                 mainEnv->ReleaseStringUTFChars((jstring)data, cstring);
 724             }
 725         } else if (mainEnv->IsInstanceOf(data, jByteBufferCls)) {
 726             jbyteArray byteArray = (jbyteArray)mainEnv->CallObjectMethod(data, jByteBufferArray);
 727             if (!EXCEPTION_OCCURED(mainEnv)) {
 728                 jbyte* raw = mainEnv->GetByteArrayElements(byteArray, NULL);
 729                 if (raw) {
 730                     jsize nraw = mainEnv->GetArrayLength(byteArray);
 731                     gtk_selection_data_set(sel_data, atom, 8, (guchar *) raw, nraw);
 732                     mainEnv->ReleaseByteArrayElements(byteArray, raw, JNI_ABORT);
 733                     is_data_set = TRUE;
 734                 }
 735             }
 736         }
 737     }
 738 
 739     g_free(target_name);
 740     return is_data_set;
 741 }

 794 static void dnd_source_push_data(JNIEnv *env, jobject data, jint supported)
 795 {
 796     if (supported == 0) {
 797         return; // No supported actions, do nothing
 798     }
 799 
 800     data = env->NewGlobalRef(data);
 801 
 802     GdkDragAction actions = translate_glass_action_to_gdk(supported);
 803 
 804     // this widget is used only to pass events and will
 805     // be destroyed on drag end
 806     drag_widget = gtk_window_new(GTK_WINDOW_POPUP);
 807     gtk_window_resize(GTK_WINDOW(drag_widget), 1, 1);
 808     gtk_window_move(GTK_WINDOW(drag_widget), -200, -200);
 809     gtk_widget_show(drag_widget);
 810 
 811     g_object_set_data_full(G_OBJECT(drag_widget), SOURCE_DND_DATA, data, clear_global_ref);
 812 
 813     g_signal_connect(drag_widget, "drag-begin",
 814         G_CALLBACK(dnd_drag_begin_callback), NULL);
 815 
 816     g_signal_connect(drag_widget, "drag-failed",
 817         G_CALLBACK(dnd_drag_failed_callback), NULL);
 818 
 819     g_signal_connect(drag_widget, "drag-data-get",
 820         G_CALLBACK(dnd_data_get_callback), NULL);
 821 
 822     g_signal_connect(drag_widget, "drag-end",
 823         G_CALLBACK(dnd_end_callback), NULL);
 824 
 825     GtkTargetList *tlist = data_to_gtk_target_list(env, data);
 826 
 827     GdkDragContext *context;
 828 
 829     gint x, y;
 830     glass_gdk_master_pointer_get_position(&x, &y);
 831 
 832     is_dnd_owner = TRUE;
 833 
 834 #if GTK_CHECK_VERSION(3, 10, 0)
 835     context = gtk_drag_begin_with_coordinates(drag_widget, tlist, actions, 1, NULL, x, y);
 836 #else
 837     context = gtk_drag_begin(drag_widget, tlist, actions, 1, NULL);
 838 #endif
 839 
 840     gtk_target_list_unref(tlist);
 841 }

 843 jint execute_dnd(JNIEnv *env, jobject data, jint supported)
 844 {
 845     try {
 846         dnd_source_push_data(env, data, supported);
 847     } catch (jni_exception&) {
 848         return com_sun_glass_ui_gtk_GtkDnDClipboard_ACTION_NONE;
 849     }
 850 
 851     while (is_in_drag()) {
 852         gtk_main_iteration();
 853     }
 854 
 855     return dnd_get_performed_action();
 856 }

 861  gboolean DragView::get_drag_image_offset(GtkWidget *widget, int* x, int* y)
 862  {
 863     gboolean offset_set = FALSE;
 864     jobject bb = dnd_source_get_data(widget, "application/x-java-drag-image-offset");
 865     if (bb) {
 866         jbyteArray byteArray = (jbyteArray)mainEnv->CallObjectMethod(bb, jByteBufferArray);
 867         if (!EXCEPTION_OCCURED(mainEnv)) {
 868             jbyte* raw = mainEnv->GetByteArrayElements(byteArray, NULL);
 869             jsize nraw = mainEnv->GetArrayLength(byteArray);
 870 
 871             if ((size_t) nraw >= sizeof(jint) * 2) {
 872                 jint* r = (jint*) raw;
 873                 *x = BSWAP_32(r[0]);
 874                 *y = BSWAP_32(r[1]);
 875                 offset_set = TRUE;
 876             }
 877 
 878             mainEnv->ReleaseByteArrayElements(byteArray, raw, JNI_ABORT);
 879         }
 880     }
 881     return offset_set;
 882 }

 884 GdkPixbuf* DragView::get_drag_image(GtkWidget *widget, gboolean* is_raw_image, gint* width, gint* height)
 885 {
 886     GdkPixbuf *pixbuf = NULL;
 887     gboolean is_raw = FALSE;
 888 
 889     jobject drag_image = dnd_source_get_data(widget, "application/x-java-drag-image");
 890 
 891     if (drag_image) {
 892         jbyteArray byteArray = (jbyteArray) mainEnv->CallObjectMethod(drag_image, jByteBufferArray);
 893         if (!EXCEPTION_OCCURED(mainEnv)) {
 894 
 895             jbyte* raw = mainEnv->GetByteArrayElements(byteArray, NULL);
 896             jsize nraw = mainEnv->GetArrayLength(byteArray);
 897 
 898             int w = 0, h = 0;
 899             int whsz = sizeof(jint) * 2; // Pixels are stored right after two ints
 900             // in this byteArray: width and height
 901             if (nraw > whsz) {
 902                 jint* int_raw = (jint*) raw;
 903                 w = BSWAP_32(int_raw[0]);
 904                 h = BSWAP_32(int_raw[1]);
 905 
 906                 // We should have enough pixels for requested width and height
 907                 if ((nraw - whsz) / 4 - w * h >= 0 ) {
 908                     guchar* data = (guchar*) g_try_malloc0(nraw - whsz);
 909                     if (data) {
 910                         memcpy(data, (raw + whsz), nraw - whsz);
 911                         pixbuf = gdk_pixbuf_new_from_data(data, GDK_COLORSPACE_RGB, TRUE, 8,
 912                                                           w, h, w * 4, pixbufDestroyNotifyFunc, NULL);
 913                     }
 914                 }
 915             }
 916             mainEnv->ReleaseByteArrayElements(byteArray, raw, JNI_ABORT);
 917         }
 918     }
 919 
 920     if (!GDK_IS_PIXBUF(pixbuf)) {
 921         jobject pixels = dnd_source_get_data(widget, "application/x-java-rawimage");
 922         if (pixels) {
 923             is_raw = TRUE;
 924             mainEnv->CallVoidMethod(pixels, jPixelsAttachData, PTR_TO_JLONG(&pixbuf));
 925             CHECK_JNI_EXCEPTION_RET(mainEnv, NULL)
 926         }
 927     }
 928 
 929     if (!GDK_IS_PIXBUF(pixbuf)) {
 930         return NULL;
 931     }
 932 
 933     int w = gdk_pixbuf_get_width(pixbuf);
 934     int h = gdk_pixbuf_get_height(pixbuf);
 935 
 936     if (w > DRAG_IMAGE_MAX_WIDTH || h > DRAG_IMAGE_MAX_HEIGH) {
 937         double rw = DRAG_IMAGE_MAX_WIDTH / (double)w;
 938         double rh =  DRAG_IMAGE_MAX_HEIGH / (double)h;
 939         double r = MIN(rw, rh);
 940 
 941         int new_w = w * r;
 942         int new_h = h * r;
 943 
 944         w = new_w;
 945         h = new_h;
 946 
 947         GdkPixbuf *tmp_pixbuf = gdk_pixbuf_scale_simple(pixbuf, new_w, new_h, GDK_INTERP_TILES);
 948         g_object_unref(pixbuf);
 949         if (!GDK_IS_PIXBUF(tmp_pixbuf)) {
 950             return NULL;
 951         }
 952         pixbuf = tmp_pixbuf;
 953     }
 954 
 955     *is_raw_image = is_raw;
 956     *width = w;
 957     *height = h;
 958 
 959     return pixbuf;
 960 }

 962 void DragView::set_drag_view(GtkWidget *widget, GdkDragContext *context)
 963 {
 964     gboolean is_raw_image = FALSE;
 965     gint w = 0, h = 0;
 966     GdkPixbuf* pixbuf = get_drag_image(widget, &is_raw_image, &w, &h);
 967

 969         gint offset_x = w / 2;
 970         gint offset_y = h / 2;

 973 
 974         DragView::view = new DragView::View(context, pixbuf, w, h, is_raw_image,
 975             is_offset_set, offset_x, offset_y);
 976     }
