 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 173 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 174                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 175         screen(_screen),
 176         frame_type(_frame_type),
 177         window_type(type),
 178         owner(_owner),
 179         jview(NULL),
 180         map_received(false),
 181         visible_received(false),
 182         on_top(false),
 183         is_fullscreen(false),
 184         is_iconified(false),
 185         is_maximized(false),
 186         is_mouse_entered(false),
 187         can_be_deleted(false),
 188         events_processing_cnt(0),
 189         win_group(NULL),
 190         pointer_device(NULL) {
 191 
 192     jwindow = mainEnv->NewGlobalRef(_jwindow);
 193 
 194     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 195 
 196     if (type != POPUP && owner == NULL) {
 197         win_group = gtk_window_group_new();
 198     }
 199 
 200     if (gchar * app_name = get_application_name()) {
 201         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 202         g_free(app_name);
 203     }
 204 
 205     if (owner) {
 206         owner->add_child(this);
 207         if (on_top_inherited()) {
 208             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 209         }
 210     }
 211 
 212     if (type == UTILITY) {
 213         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 214     }
 215 
 216     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 217 
 218     if (xvisualID != 0) {
 219         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 220         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 221     }
 222 
 223     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 224     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 225 
 226     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 227     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 228 
 229     gdk_window = gtk_widget_get_window(gtk_widget);
 230     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 231 
 232     glass_dnd_attach_context(this);
 233 
 234     gdk_windowManagerFunctions = wmf;
 235     if (wmf) {
 236         gdk_window_set_functions(gdk_window, wmf);
 237     }
 238 
 239     if (frame_type != TITLED) {
 240         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 241     }
 242 
 243     connect_signals(gtk_widget, this);
 244 }

 246 void WindowContext::paint(void *data, jint width, jint height) {
 247 #if GTK_CHECK_VERSION(3, 0, 0)
 248     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 249 #if GTK_CHECK_VERSION(3, 22, 0)
 250     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 251     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 252 #else
 253     gdk_window_begin_paint_region(gdk_window, region);
 254     cairo_t* context = gdk_cairo_create(gdk_window);
 255 #endif
 256 #else
 257     cairo_t *context = gdk_cairo_create(gdk_window);
 258 #endif
 259 
 260     if (bg_color.is_set) {
 261         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 262                                 (frame_type == TRANSPARENT) ? 0 : 1);
 263         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 264         cairo_paint(context);
 265     }
 266 
 267     cairo_surface_t *cairo_surface;
 268     cairo_surface = cairo_image_surface_create_for_data(
 269             (unsigned char *) data,
 270             CAIRO_FORMAT_ARGB32,
 271             width, height, width * 4);
 272 
 273     cairo_set_source_surface(context, cairo_surface, 0, 0);
 274 
 275     applyShapeMask(data, width, height);
 276     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 277     cairo_paint(context);
 278 
 279 #if GTK_CHECK_VERSION(3, 0, 0)
 280 #if GTK_CHECK_VERSION(3, 22, 0)
 281     gdk_window_end_draw_frame(gdk_window, dcontext);
 282     cairo_region_destroy(region);
 283 #else
 284     gdk_window_end_paint(gdk_window);
 285     cairo_region_destroy(region);
 286     cairo_destroy(context);
 287 #endif
 288 #else
 289     cairo_destroy(context);
 290 #endif
 291 
 292     cairo_surface_destroy(cairo_surface);
 293 }

 334 void WindowContext::process_focus(GdkEventFocus *event) {
 335     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 336         ungrab_mouse_drag_focus();
 337     }
 338 
 339     if (!event->in) {
 340         ungrab_focus();
 341     }
 342 
 343     if (xim.enabled && xim.ic) {
 344         if (event->in) {
 345             XSetICFocus(xim.ic);
 346         } else {
 347             XUnsetICFocus(xim.ic);
 348         }
 349     }
 350 
 351     if (jwindow) {
 352         if (!event->in || isEnabled()) {
 353             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 354                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 355                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 356             CHECK_JNI_EXCEPTION(mainEnv)
 357         } else {
 358             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 359             CHECK_JNI_EXCEPTION(mainEnv)
 360         }
 361     }
 362 }

 364 void WindowContext::process_property_notify(GdkEventProperty *event) {
 365     if (event->window == gdk_window) {
 366         if (event->atom == atom_net_wm_state) {
 367             process_net_wm_property();
 368         } else if (event->atom == atom_net_wm_frame_extents) {
 369             if (frame_type != TITLED) {
 370                 return;
 371             }
 372 
 373             int top, left, bottom, right;
 374 
 375             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 376                 if (top + left + bottom + right > 0) {
 377                     geometry.frame_extents_received = true;
 378                     geometry.adjust_w = left + right;
 379                     geometry.adjust_h = top + bottom;
 380                     geometry.view_x = left;
 381                     geometry.view_y = top;
 382 
 383                     save_cached_extents();
 384 
 385                     // set bounds again to set to correct window size that must
 386                     // be the total width and height accounting extents
 387                     set_bounds(0, 0, false, false, -1, -1, -1, -1);
 388                 }
 389             }
 390         }
 391     }
 392 }

 394 void WindowContext::process_configure() {
 395     calculate_adjustments();
 396 
 397     gint x, y, w, h, gtk_w, gtk_h;
 398 
 399     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 400     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 401     w = gtk_w + geometry.adjust_w;
 402     h = gtk_h + geometry.adjust_h;
 403 
 404     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 405     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 406                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 407 
 408     geometry.current_x = x;
 409     geometry.current_y = y;
 410     geometry.current_w = w;
 411     geometry.current_h = h;
 412     geometry.current_cw = gtk_w;
 413     geometry.current_ch = gtk_h;
 414 
 415     size_position_notify(size_changed, pos_changed);
 416 }

 418 void WindowContext::process_destroy() {
 419     if (owner) {
 420         owner->remove_child(this);
 421     }
 422 
 423     if (WindowContext::sm_mouse_drag_window == this) {
 424         ungrab_mouse_drag_focus();
 425     }
 426 
 427     ungrab_focus();
 428 
 429     std::set<WindowContext *>::iterator it;
 430     for (it = children.begin(); it != children.end(); ++it) {
 431         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 432         // WindowContext::process_destroy() to call remove_child() (because children
 433         // is being iterated here) but also prevents gtk_window_set_transient_for from
 434         // being called - this causes the crash on gnome.
 435         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 436         (*it)->set_owner(NULL);
 437         destroy_and_delete_ctx(*it);
 438     }
 439     children.clear();
 440 
 441     if (jwindow) {
 442         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 443         EXCEPTION_OCCURED(mainEnv);
 444     }
 445 
 446     if (jview) {
 447         mainEnv->DeleteGlobalRef(jview);
 448         jview = NULL;
 449     }
 450 
 451     if (jwindow) {
 452         mainEnv->DeleteGlobalRef(jwindow);
 453         jwindow = NULL;
 454     }
 455 
 456     can_be_deleted = true;
 457 }

 462         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 463         CHECK_JNI_EXCEPTION(mainEnv)

 475 void WindowContext::process_mouse_button(GdkEventButton *event) {
 476     bool press = event->type == GDK_BUTTON_PRESS;
 477     bool release = event->type == GDK_BUTTON_RELEASE;
 478 
 479     guint state = event->state;
 480     guint mask = 0;
 481 
 482     // We need to add/remove current mouse button from the modifier flags
 483     // as X lib state represents the state just prior to the event and
 484     // glass needs the state just after the event
 485     switch (event->button) {
 486         case 1:
 487             mask = GDK_BUTTON1_MASK;
 488             break;
 489         case 2:
 490             mask = GDK_BUTTON2_MASK;
 491             break;
 492         case 3:
 493             mask = GDK_BUTTON3_MASK;
 494             break;
 495         case MOUSE_BACK_BTN:
 496             mask = GDK_BUTTON4_MASK;
 497             break;
 498         case MOUSE_FORWARD_BTN:
 499             mask = GDK_BUTTON5_MASK;
 500             break;
 501     }
 502 
 503     if (press) {
 504         state |= mask;
 505     } else {
 506         state &= ~mask;
 507     }
 508 
 509     // Upper layers expects from us Windows behavior:
 510     // all mouse events should be delivered to window where drag begins
 511     // and no exit/enter event should be reported during this drag.
 512     // We can grab mouse pointer for these needs.
 513     if (press) {
 514         grab_mouse_drag_focus();
 515     } else {
 516         if ((event->state & MOUSE_BUTTONS_MASK)
 517             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 518             ungrab_mouse_drag_focus();
 519         } else if (event->button == 8 || event->button == 9) {
 520             // GDK X backend interprets button press events for buttons 4-7 as
 521             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 522             // be set on the event->state from GDK. Thus we cannot check if all
 523             // buttons have been released in the usual way (as above).
 524             ungrab_mouse_drag_focus();
 525         }
 526     }
 527 
 528     jint button = gtk_button_number_to_mouse_button(event->button);
 529 
 530     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 531         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 532                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 533                                 button,
 534                                 (jint) event->x, (jint) event->y,
 535                                 (jint) event->x_root, (jint) event->y_root,
 536                                 gdk_modifier_mask_to_glass(state),
 537                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 538                                 JNI_FALSE);
 539         CHECK_JNI_EXCEPTION(mainEnv)
 540 
 541         if (jview && event->button == 3 && press) {
 542             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 543                                     (jint) event->x, (jint) event->y,
 544                                     (jint) event->x_root, (jint) event->y_root,
 545                                     JNI_FALSE);
 546             CHECK_JNI_EXCEPTION(mainEnv)
 547         }
 548     }
 549 }

 551 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 552     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 553     jint isDrag = glass_modifier & (
 554             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 555             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 556             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 557             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 558             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 559     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 560 
 561     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 562         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 563     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 564         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 565     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 566         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 567     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 568         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 569     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 570         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 571     }
 572 
 573     if (jview) {
 574         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 575                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 576                                 button,
 577                                 (jint) event->x, (jint) event->y,
 578                                 (jint) event->x_root, (jint) event->y_root,
 579                                 glass_modifier,
 580                                 JNI_FALSE,
 581                                 JNI_FALSE);
 582         CHECK_JNI_EXCEPTION(mainEnv)
 583     }
 584 }

 586 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 587     jdouble dx = 0;
 588     jdouble dy = 0;
 589 
 590     // converting direction to change in pixels
 591     switch (event->direction) {
 592 #if GTK_CHECK_VERSION(3, 4, 0)
 593         case GDK_SCROLL_SMOOTH:
 594             //FIXME 3.4 ???
 595             break;
 596 #endif
 597         case GDK_SCROLL_UP:
 598             dy = 1;
 599             break;
 600         case GDK_SCROLL_DOWN:
 601             dy = -1;
 602             break;
 603         case GDK_SCROLL_LEFT:
 604             dx = 1;
 605             break;
 606         case GDK_SCROLL_RIGHT:
 607             dx = -1;
 608             break;
 609     }
 610     if (event->state & GDK_SHIFT_MASK) {
 611         jdouble t = dy;
 612         dy = dx;
 613         dx = t;
 614     }
 615     if (jview) {
 616         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 617                                 (jint) event->x, (jint) event->y,
 618                                 (jint) event->x_root, (jint) event->y_root,
 619                                 dx, dy,
 620                                 gdk_modifier_mask_to_glass(event->state),
 621                                 (jint) 0, (jint) 0,
 622                                 (jint) 0, (jint) 0,
 623                                 (jdouble) 40.0, (jdouble) 40.0);
 624         CHECK_JNI_EXCEPTION(mainEnv)
 625     }
 626 }

 653 void WindowContext::process_key(GdkEventKey *event) {
 654     bool press = event->type == GDK_KEY_PRESS;
 655     jint glassKey = get_glass_key(event);
 656     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 657     if (press) {
 658         glassModifier |= glass_key_to_modifier(glassKey);
 659     } else {
 660         glassModifier &= ~glass_key_to_modifier(glassKey);
 661     }
 662     jcharArray jChars = NULL;
 663     jchar key = gdk_keyval_to_unicode(event->keyval);
 664     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 665         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 666     } else {
 667 #ifdef GLASS_GTK2
 668         if (key == 0) {
 669             // Work around "bug" fixed in gtk-3.0:
 670             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 671             switch (event->keyval) {
 672             case 0xFF08 /* Backspace */: key =  '\b';
 673             case 0xFF09 /* Tab       */: key =  '\t';
 674             case 0xFF0A /* Linefeed  */: key =  '\n';
 675             case 0xFF0B /* Vert. Tab */: key =  '\v';
 676             case 0xFF0D /* Return    */: key =  '\r';
 677             case 0xFF1B /* Escape    */: key =  '\033';
 678             case 0xFFFF /* Delete    */: key =  '\177';
 679             }
 680         }
 681 #endif
 682     }
 683 
 684     if (key > 0) {
 685         jChars = mainEnv->NewCharArray(1);
 686         if (jChars) {
 687             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 688             CHECK_JNI_EXCEPTION(mainEnv)
 689         }
 690     } else {
 691         jChars = mainEnv->NewCharArray(0);
 692     }
 693     if (jview) {
 694         if (press) {
 695             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 696                                     com_sun_glass_events_KeyEvent_PRESS,
 697                                     glassKey,
 698                                     jChars,
 699                                     glassModifier);
 700             CHECK_JNI_EXCEPTION(mainEnv)
 701 
 702             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 703                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 704                                         com_sun_glass_events_KeyEvent_TYPED,
 705                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 706                                         jChars,
 707                                         glassModifier);
 708                 CHECK_JNI_EXCEPTION(mainEnv)
 709             }
 710         } else {
 711             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 712                                     com_sun_glass_events_KeyEvent_RELEASE,
 713                                     glassKey,
 714                                     jChars,
 715                                     glassModifier);
 716             CHECK_JNI_EXCEPTION(mainEnv)
 717         }
 718     }
 719 }

 721 void WindowContext::process_state(GdkEventWindowState *event) {
 722     if (event->changed_mask &
 723         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 724 
 725         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 726             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 727         }
 728         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 729             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 730         }
 731 
 732         jint stateChangeEvent;
 733 
 734         if (is_iconified) {
 735             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 736         } else if (is_maximized) {
 737             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 738         } else {
 739             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 740             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 741                 // in this case - the window manager will not support the programatic
 742                 // request to iconify - so we need to restore it now.
 743                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 744             }
 745         }
 746 
 747         notify_state(stateChangeEvent);
 748     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 749         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 750     }
 751 }

 753 void WindowContext::process_net_wm_property() {
 754     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 755 
 756     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 757     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 758     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 759     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 760 
 761     gint length;
 762     glong *atoms = NULL;
 763 
 764     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 765                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 766 
 767         bool is_hidden = false;
 768         bool is_above = false;
 769         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 770             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 771                 is_hidden = true;
 772             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 773                 is_above = true;
 774             }
 775         }
 776 
 777         g_free(atoms);
 778 
 779         if (is_iconified != is_hidden) {
 780             is_iconified = is_hidden;
 781 
 782             notify_state((is_hidden)
 783                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 784                          : com_sun_glass_events_WindowEvent_RESTORE);
 785         }
 786 
 787         notify_on_top(is_above);
 788     }
 789 }

 963 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 964     calculate_adjustments();
 965 
 966     // newW / newH always content sizes compatible with GTK+
 967     // if window has no decoration, adjustments will be ZERO
 968     int newW = w > 0 ? w - geometry.adjust_w : cw;
 969     int newH = h > 0 ? h - geometry.adjust_h : ch;
 970 
 971     gboolean size_changed = FALSE;
 972     gboolean pos_changed = FALSE;
 973 
 974     if (newW > 0 && newH > 0) {
 975         size_changed = TRUE;
 976 
 977         geometry.current_cw = newW;
 978         geometry.current_ch = newH;
 979         geometry.current_w = newW + geometry.adjust_w;
 980         geometry.current_h = newH + geometry.adjust_h;
 981 
 982         if (visible_received) {
 983             // update constraints if not resized by the user interface so it will
 984             // let gtk_window_resize succeed, because it's bound to geometry constraints
 985             apply_geometry();
 986             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
 987         } else {
 988             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
 989         }
 990     }
 991 
 992     if (xSet || ySet) {
 993         int newX = (xSet) ? x : geometry.current_x;
 994         int newY = (ySet) ? y : geometry.current_y;
 995 
 996         if (newX != geometry.current_x || newY != geometry.current_y) {
 997             pos_changed = TRUE;
 998             geometry.current_x = newX;
 999             geometry.current_y = newY;
1000             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1001         }
1002     }
1003 
1004     size_position_notify(size_changed, pos_changed);
1005 }

1208 
1209 // This function calculate the deltas between window and window + decoration (titleblar, borders).
1210 // Only useful if the window manager does not support the frame extents extension - in this case
1211 // it uses GDK calculation, that is primary based on _NET_FRAME_EXTENTS but will try harder
1212 // if that's not available.
1213 void WindowContext::calculate_adjustments() {
1214     if (frame_type != TITLED || geometry.frame_extents_received) {
1215         return;
1216     }
1217 
1218     gint x, y;
1219     gdk_window_get_origin(gdk_window, &x, &y);
1220 
1221     gint rx, ry;
1222     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1223 
1224     if (rx != x || ry != y) {
1225         // the left extends are correct - the right one is guessed to be the same
1226         geometry.adjust_w = (rx - x) * 2;
1227         // guess that bottom size is the same as left and right
1228         geometry.adjust_h = (ry - y) + (rx - x);
1229 
1230         // those will be correct
1231         geometry.view_x = (rx - x);
1232         geometry.view_y = (ry - y);
1233 
1234         save_cached_extents();
1235     } else {
1236         CachedExtents c = (window_type == NORMAL) ? normal_extents : utility_extents;
1237         geometry.adjust_w = c.adjust_w;
1238         geometry.adjust_h = c.adjust_h;
1239         geometry.view_x = c.view_x;
1240         geometry.view_y = c.view_y;
1241     }
1242 
1243     apply_geometry();
1244 }

1246 void WindowContext::save_cached_extents() {
1247     if (frame_type != TITLED) {
1248         return;
1249     }
1250 
1251     if (window_type == NORMAL) {
1252         normal_extents.adjust_w = geometry.adjust_w;
1253         normal_extents.adjust_h = geometry.adjust_h;
1254         normal_extents.view_x = geometry.view_x;
1255         normal_extents.view_y = geometry.view_y;
1256     } else {
1257         utility_extents.adjust_w = geometry.adjust_w;
1258         utility_extents.adjust_h = geometry.adjust_h;
1259         utility_extents.view_x = geometry.view_x;
1260         utility_extents.view_y = geometry.view_y;
1261     }

1264 void WindowContext::apply_geometry() {
1265     if (!map_received) {
1266         return;
1267     }
1268 
1269     GdkGeometry gdk_geometry;
1270     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1271 
1272     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1273         // not resizeable
1274         int w = (geometry.current_w - geometry.adjust_w) > 0
1275                 ? geometry.current_w - geometry.adjust_w
1276                 : geometry.current_cw;
1277 
1278         int h = (geometry.current_h - geometry.adjust_h) > 0
1279                 ? geometry.current_h - geometry.adjust_h
1280                 : geometry.current_ch;
1281 
1282         gdk_geometry.min_width = gdk_geometry.max_width = w;
1283         gdk_geometry.min_height = gdk_geometry.max_height = h;
1284     } else {
1285         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1286                                  ? geometry.minw - geometry.adjust_w : 1;
1287         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1288                                   ? geometry.minh - geometry.adjust_h : 1;
1289 
1290         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1291                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1292         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1293                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1294     }
1295 
1296     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1297                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1298 }

1326 void WindowContext::activate_window() {
1327     Display *display = GDK_DISPLAY_XDISPLAY(gdk_window_get_display(gdk_window));
1328     Atom navAtom = XInternAtom(display, "_NET_ACTIVE_WINDOW", True);
1329     if (navAtom != None) {
1330         XClientMessageEvent clientMessage;
1331         memset(&clientMessage, 0, sizeof(clientMessage));
1332 
1333         clientMessage.type = ClientMessage;
1334         clientMessage.window = GDK_WINDOW_XID(gdk_window);
1335         clientMessage.message_type = navAtom;
1336         clientMessage.format = 32;
1337         clientMessage.data.l[0] = 1;
1338         clientMessage.data.l[1] = gdk_x11_get_server_time(gdk_window);
1339         clientMessage.data.l[2] = 0;
1340 
1341         XSendEvent(display, XDefaultRootWindow(display), False,
1342                    SubstructureRedirectMask | SubstructureNotifyMask,
1343                    (XEvent * ) & clientMessage);
1344         XFlush(display);
1345     }

1370 
1371         if (pos_changed) {
1372             mainEnv->CallVoidMethod(jwindow, jWindowNotifyMove, geometry.current_x, geometry.current_y);
1373             CHECK_JNI_EXCEPTION(mainEnv)
1374         }
1375     }

1386 bool WindowContext::on_top_inherited() {
1387     WindowContext *o = owner;
1388     while (o) {
1389         WindowContext *topO = dynamic_cast<WindowContext *>(o);
1390         if (!topO) break;
1391         if (topO->on_top) {
1392             return true;
1393         }
1394         o = topO->owner;
1395     }
1396     return false;
1397 }

1401         WindowContext *topO = dynamic_cast<WindowContext *>(owner);

1403     }
1404     return on_top;
