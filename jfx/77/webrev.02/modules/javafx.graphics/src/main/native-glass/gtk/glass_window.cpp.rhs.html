 183 bool WindowContextBase::isEnabled() {
 184     if (jwindow) {
 185         bool result = (JNI_TRUE == mainEnv->CallBooleanMethod(jwindow, jWindowIsEnabled));
 186         LOG_EXCEPTION(mainEnv)
 187         return result;
 188     } else {
 189         return false;
 190     }

 193 void WindowContextBase::notify_state(jint glass_state) {
 194     if (glass_state == com_sun_glass_events_WindowEvent_RESTORE) {
 195         if (is_maximized) {
 196             glass_state = com_sun_glass_events_WindowEvent_MAXIMIZE;
 197         }
 198 
 199         notify_repaint();







 200     }
 201 
 202     if (jwindow) {
 203        mainEnv->CallVoidMethod(jwindow,
 204                jGtkWindowNotifyStateChanged,
 205                glass_state);
 206        CHECK_JNI_EXCEPTION(mainEnv);
 207     }
 208 }

 210 void WindowContextBase::notify_repaint() {
 211     int w, h;
 212     glass_gdk_window_get_size(gdk_window, &w, &h);
 213     if (jview) {
 214         mainEnv->CallVoidMethod(jview,
 215                 jViewNotifyRepaint,
 216                 0, 0, w, h);
 217         CHECK_JNI_EXCEPTION(mainEnv);
 218     }
 219 }

 221 void WindowContextBase::process_state(GdkEventWindowState* event) {
 222     if (event->changed_mask &
 223             (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 224 
 225         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 226             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 227         }
 228         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 229             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 230         }
 231 
 232         jint stateChangeEvent;
 233 
 234         if (is_iconified) {
 235             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 236         } else if (is_maximized) {
 237             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 238         } else {
 239             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 240             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 241                 // in this case - the window manager will not support the programatic
 242                 // request to iconify - so we need to restore it now.
 243                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 244             }
 245         }
 246 
 247         notify_state(stateChangeEvent);
 248     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 249         notify_on_top( event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 250     }
 251 }

 253 void WindowContextBase::process_focus(GdkEventFocus* event) {
 254     if (!event->in) {



 255         ungrab_focus();
 256     }
 257 
 258     if (xim.enabled && xim.ic) {
 259         if (event->in) {
 260             XSetICFocus(xim.ic);
 261         } else {
 262             XUnsetICFocus(xim.ic);
 263         }
 264     }
 265 
 266     if (jwindow) {
 267         if (!event->in || isEnabled()) {
 268             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 269                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 270             CHECK_JNI_EXCEPTION(mainEnv)
 271         } else {
 272             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 273             CHECK_JNI_EXCEPTION(mainEnv)
 274         }
 275     }
 276 }

 294 void destroy_and_delete_ctx(WindowContext* ctx) {
 295     if (ctx) {
 296         ctx->process_destroy();
 297 
 298         if (!ctx->get_events_count()) {
 299             delete ctx;
 300         }
 301         // else: ctx will be deleted in EventsCounterHelper after completing
 302         // an event processing
 303     }
 304 }

 306 void WindowContextBase::process_destroy() {
 307     ungrab_focus();






 308 
 309     std::set<WindowContextTop*>::iterator it;
 310     for (it = children.begin(); it != children.end(); ++it) {
 311         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 312         // WindowContextTop::process_destroy() to call remove_child() (because children
 313         // is being iterated here) but also prevents gtk_window_set_transient_for from
 314         // being called - this causes the crash on gnome.
 315         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 316         (*it)->set_owner(NULL);
 317         destroy_and_delete_ctx(*it);
 318     }
 319     children.clear();
 320 
 321     if (jwindow) {
 322         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 323         EXCEPTION_OCCURED(mainEnv);
 324     }
 325 
 326     if (jview) {
 327         mainEnv->DeleteGlobalRef(jview);
 328         jview = NULL;
 329     }
 330 
 331     if (jwindow) {
 332         mainEnv->DeleteGlobalRef(jwindow);
 333         jwindow = NULL;
 334     }
 335 
 336     can_be_deleted = true;
 337 }

 339 void WindowContextBase::process_delete() {
 340     if (jwindow && isEnabled()) {
 341         gtk_widget_hide_on_delete(gtk_widget);
 342         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 343         CHECK_JNI_EXCEPTION(mainEnv)
 344     }

 366 static inline jint gtk_button_number_to_mouse_button(guint button) {
 367     switch (button) {
 368         case 1:
 369             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 370         case 2:
 371             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 372         case 3:
 373             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 374         case MOUSE_BACK_BTN:
 375             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 376         case MOUSE_FORWARD_BTN:
 377             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 378         default:
 379             // Other buttons are not supported by quantum and are not reported by other platforms
 380             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 381     }
 382 }

 384 void WindowContextBase::process_mouse_button(GdkEventButton* event) {
 385     bool press = event->type == GDK_BUTTON_PRESS;
 386     guint state = event->state;
 387     guint mask = 0;
 388 
 389     // We need to add/remove current mouse button from the modifier flags
 390     // as X lib state represents the state just prior to the event and
 391     // glass needs the state just after the event
 392     switch (event->button) {
 393         case 1:
 394             mask = GDK_BUTTON1_MASK;
 395             break;
 396         case 2:
 397             mask = GDK_BUTTON2_MASK;
 398             break;
 399         case 3:
 400             mask = GDK_BUTTON3_MASK;
 401             break;
 402         case MOUSE_BACK_BTN:
 403             mask = GDK_BUTTON4_MASK;
 404             break;
 405         case MOUSE_FORWARD_BTN:
 406             mask = GDK_BUTTON5_MASK;
 407             break;
 408     }
 409 
 410     if (press) {
 411         state |= mask;
 412     } else {
 413         state &= ~mask;
 414     }
 415 






























 416     jint button = gtk_button_number_to_mouse_button(event->button);
 417 
 418     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 419         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 420                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 421                 button,
 422                 (jint) event->x, (jint) event->y,
 423                 (jint) event->x_root, (jint) event->y_root,
 424                 gdk_modifier_mask_to_glass(state),
 425                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 426                 JNI_FALSE);
 427         CHECK_JNI_EXCEPTION(mainEnv)
 428 
 429         if (jview && event->button == 3 && press) {
 430             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 431                     (jint)event->x, (jint)event->y,
 432                     (jint)event->x_root, (jint)event->y_root,
 433                     JNI_FALSE);
 434             CHECK_JNI_EXCEPTION(mainEnv)
 435         }
 436     }
 437 }

 439 void WindowContextBase::process_mouse_motion(GdkEventMotion* event) {
 440     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 441     jint isDrag = glass_modifier & (
 442             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 443             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 444             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 445             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 446             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 447     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 448 
 449     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 450         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 451     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 452         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 453     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 454         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 455     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 456         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 457     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 458         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 459     }
 460 
 461     if (jview) {
 462         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 463                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 464                 button,
 465                 (jint) event->x, (jint) event->y,
 466                 (jint) event->x_root, (jint) event->y_root,
 467                 glass_modifier,
 468                 JNI_FALSE,
 469                 JNI_FALSE);
 470         CHECK_JNI_EXCEPTION(mainEnv)
 471     }
 472 }

 474 void WindowContextBase::process_mouse_scroll(GdkEventScroll* event) {
 475     jdouble dx = 0;
 476     jdouble dy = 0;
 477 
 478     // converting direction to change in pixels
 479     switch (event->direction) {
 480 #if GTK_CHECK_VERSION(3, 4, 0)
 481         case GDK_SCROLL_SMOOTH:
 482             //FIXME 3.4 ???
 483             break;
 484 #endif
 485         case GDK_SCROLL_UP:
 486             dy = 1;
 487             break;
 488         case GDK_SCROLL_DOWN:
 489             dy = -1;
 490             break;
 491         case GDK_SCROLL_LEFT:
 492             dx = 1;
 493             break;
 494         case GDK_SCROLL_RIGHT:
 495             dx = -1;
 496             break;
 497     }
 498     if (event->state & GDK_SHIFT_MASK) {
 499         jdouble t = dy;
 500         dy = dx;
 501         dx = t;
 502     }
 503     if (jview) {
 504         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 505                 (jint) event->x, (jint) event->y,
 506                 (jint) event->x_root, (jint) event->y_root,
 507                 dx, dy,
 508                 gdk_modifier_mask_to_glass(event->state),
 509                 (jint) 0, (jint) 0,
 510                 (jint) 0, (jint) 0,
 511                 (jdouble) 40.0, (jdouble) 40.0);
 512         CHECK_JNI_EXCEPTION(mainEnv)
 513     }

 514 }

 516 void WindowContextBase::process_mouse_cross(GdkEventCrossing* event) {
 517     bool enter = event->type == GDK_ENTER_NOTIFY;
 518 
 519     if (jview) {
 520         guint state = event->state;
 521         if (enter) { // workaround for RT-21590
 522             state &= ~MOUSE_BUTTONS_MASK;
 523         }
 524 
 525         if (enter != is_mouse_entered) {
 526             is_mouse_entered = enter;
 527             mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 528                     enter ? com_sun_glass_events_MouseEvent_ENTER : com_sun_glass_events_MouseEvent_EXIT,
 529                     com_sun_glass_events_MouseEvent_BUTTON_NONE,
 530                     (jint) event->x, (jint) event->y,
 531                     (jint) event->x_root, (jint) event->y_root,
 532                     gdk_modifier_mask_to_glass(state),
 533                     JNI_FALSE,
 534                     JNI_FALSE);
 535             CHECK_JNI_EXCEPTION(mainEnv)
 536         }
 537     }
 538 }

 540 void WindowContextBase::process_key(GdkEventKey* event) {
 541     bool press = event->type == GDK_KEY_PRESS;
 542     jint glassKey = get_glass_key(event);
 543     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 544     if (press) {
 545         glassModifier |= glass_key_to_modifier(glassKey);
 546     } else {
 547         glassModifier &= ~glass_key_to_modifier(glassKey);
 548     }
 549     jcharArray jChars = NULL;
 550     jchar key = gdk_keyval_to_unicode(event->keyval);
 551     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 552         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 553     } else {
 554 #ifdef GLASS_GTK2
 555         if (key == 0) {
 556             // Work around "bug" fixed in gtk-3.0:
 557             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 558             switch (event->keyval) {
 559             case 0xFF08 /* Backspace */: key =  '\b';
 560             case 0xFF09 /* Tab       */: key =  '\t';
 561             case 0xFF0A /* Linefeed  */: key =  '\n';
 562             case 0xFF0B /* Vert. Tab */: key =  '\v';
 563             case 0xFF0D /* Return    */: key =  '\r';
 564             case 0xFF1B /* Escape    */: key =  '\033';
 565             case 0xFFFF /* Delete    */: key =  '\177';
 566             }
 567         }
 568 #endif
 569     }
 570 
 571     if (key > 0) {
 572         jChars = mainEnv->NewCharArray(1);
 573         if (jChars) {
 574             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 575             CHECK_JNI_EXCEPTION(mainEnv)
 576         }
 577     } else {
 578         jChars = mainEnv->NewCharArray(0);
 579     }
 580     if (jview) {
 581         if (press) {
 582             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 583                     com_sun_glass_events_KeyEvent_PRESS,
 584                     glassKey,
 585                     jChars,
 586                     glassModifier);
 587             CHECK_JNI_EXCEPTION(mainEnv)
 588 
 589             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 590                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 591                         com_sun_glass_events_KeyEvent_TYPED,
 592                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 593                         jChars,
 594                         glassModifier);

 866 // This function calculate the deltas between window and window + decoration (titleblar, borders)
 867 // Only useful if the window manager does not support the frame extents extension
 868 void WindowContextTop::calculate_adjustments() {
 869     if (frame_type == TITLED && !geometry.frame_extents_received) {
 870         GdkRectangle er;
 871         gdk_window_get_frame_extents(gdk_window, &er);
 872 
 873         int w, h;
 874         gtk_window_get_size(GTK_WINDOW(gtk_widget), &w, &h);
 875 
 876         int px, py;
 877         gdk_window_get_origin(gdk_window, &px, &py);
 878 
 879         geometry.adjust_w = (er.width - w > 0) ? er.width - w : 0;
 880         geometry.adjust_h = (er.height - h > 0) ? er.height - h : 0;
 881         geometry.view_x = (px - er.x > 0) ? px - er.x : 0;
 882         geometry.view_y = (py - er.y > 0) ? py - er.y : 0;
 883 
 884         apply_geometry();
 885     }
 886 }

 888 void WindowContextTop::apply_geometry() {
 889     GdkGeometry gdk_geometry;
 890     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;




 891 
 892     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
 893         gdk_geometry.min_width = geometry.current_w;
 894         gdk_geometry.min_height = geometry.current_h;
 895         gdk_geometry.max_width = geometry.current_w;
 896         gdk_geometry.max_height = geometry.current_h;
 897     } else {
 898         gdk_geometry.min_width = (geometry.minw > 0) ? geometry.minw - geometry.adjust_w : 1;
 899         gdk_geometry.min_height = (geometry.minh > 0) ? geometry.minh - geometry.adjust_h : 1;
 900         gdk_geometry.max_width = (geometry.maxw > 0) ? geometry.maxw  - geometry.adjust_w : G_MAXINT;
 901         gdk_geometry.max_height = (geometry.maxh > 0) ? geometry.maxh - geometry.adjust_h : G_MAXINT;
 902     }
 903 
 904     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), gtk_widget, &gdk_geometry,
 905         (GdkWindowHints) (GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
 906 }

 953         int *bottom, int *right) {

 978 void WindowContextTop::process_net_wm_property() {
 979     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 980 
 981     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 982     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 983     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 984     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 985 
 986     gint length;

 987     glong* atoms = NULL;
 988 
 989     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 990             0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar**) &atoms)) {
 991 
 992         bool is_hidden = false;
 993         bool is_above = false;
 994         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 995             if (atom_net_wm_state_hidden == (GdkAtom)atoms[i]) {
 996                 is_hidden = true;
 997             } else if (atom_net_wm_state_above == (GdkAtom)atoms[i]) {
 998                 is_above = true;
 999             }
1000         }
1001 
1002         g_free(atoms);
1003 
1004         if (is_iconified != is_hidden) {
1005             is_iconified = is_hidden;
1006 
1007             notify_state((is_hidden)
1008                     ? com_sun_glass_events_WindowEvent_MINIMIZE
1009                     : com_sun_glass_events_WindowEvent_RESTORE);
1010         }
1011 
1012         notify_on_top(is_above);
1013     }
1014 }

1016 void WindowContextTop::process_property_notify(GdkEventProperty* event) {
1017     if (event->window == gdk_window) {
1018         if (event->atom == atom_net_wm_state) {
1019             process_net_wm_property();
1020         } else if (event->atom == atom_net_wm_frame_extents) {
1021             if (frame_type != TITLED) {
1022                 return;
1023             }
1024 
1025             int top, left, bottom, right;
1026 
1027             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
1028                 if (top + left + bottom + right > 0) {
1029                     geometry.frame_extents_received = true;
1030                     geometry.adjust_w = left + right;
1031                     geometry.adjust_h = top + bottom;
1032                     geometry.view_x = left;
1033                     geometry.view_y = top;
1034 
1035                     // set bounds again to set to correct window size that must
1036                     // be the total width and height accounting extents
1037                     set_bounds(0, 0,
1038                                false, false,
1039                                geometry.current_w, geometry.current_h,
1040                                -1, -1);
1041                 }
1042             }
1043         }
1044     }

1049     // if the window is not visible, there will be no changes
1050     if (!map_received && !is_maximized) {
1051         return;

1061     h = gtk_h + geometry.adjust_h;

1069         geometry.current_y = y;

1114 void WindowContextTop::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
1115     calculate_adjustments();














1116 
1117     // newW / newH always content sizes compatible with GTK+
1118     int newW = w > 0 ? w - geometry.adjust_w : cw;
1119     int newH = h > 0 ? h - geometry.adjust_h : ch;


























1120 
1121     gboolean changed = FALSE;
1122     if (newW > 0 && newH > 0) {
1123         changed = TRUE;

1124 
1125         geometry.current_cw = newW;
1126         geometry.current_ch = newH;
1127         geometry.current_w = newW + geometry.adjust_w;
1128         geometry.current_h = newH + geometry.adjust_h;
1129 
1130         if (map_received) {
1131             apply_geometry(); // update constraints if not resized by the user interface
1132         }
1133 
1134         gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);



1135     }
1136 
1137     if (xSet || ySet) {
1138         int newX = (xSet) ? x : geometry.current_x;
1139         int newY = (ySet) ? y : geometry.current_y;
1140 
1141         if (newX != geometry.current_x || newY != geometry.current_y) {
1142             changed = TRUE;
1143             geometry.current_x = newX;
1144             geometry.current_y = newY;
1145             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1146         }
1147     }
1148 
1149     if (changed) {
1150         size_position_notify();
1151     }
1152 }

1305 void WindowContextTop::update_ontop_tree(bool on_top) {
1306     bool effective_on_top = on_top || this->on_top;
1307     gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), effective_on_top ? TRUE : FALSE);
1308     for (std::set<WindowContextTop*>::iterator it = children.begin(); it != children.end(); ++it) {
1309         (*it)->update_ontop_tree(effective_on_top);

1313 bool WindowContextTop::on_top_inherited() {
1314     WindowContext* o = owner;
1315     while (o) {
1316         WindowContextTop* topO = dynamic_cast<WindowContextTop*>(o);
1317         if (!topO) break;
1318         if (topO->on_top) {
1319             return true;
1320         }
1321         o = topO->owner;
1322     }
1323     return false;
1324 }

1326 bool WindowContextTop::effective_on_top() {
1327     if (owner) {
1328         WindowContextTop* topO = dynamic_cast<WindowContextTop*>(owner);
1329         return (topO && topO->effective_on_top()) || on_top;
1330     }
1331     return on_top;
1332 }

1334 void WindowContextTop::notify_on_top(bool top) {
1335     // Do not report effective (i.e. native) values to the FX, only if the user sets it manually
1336     if (top != effective_on_top() && jwindow) {
1337         if (on_top_inherited() && !top) {
1338             // Disallow user's "on top" handling on windows that inherited the property
1339             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
1340         } else {
1341             on_top = top;
1342             update_ontop_tree(top);
1343             mainEnv->CallVoidMethod(jwindow,
1344                     jWindowNotifyLevelChanged,
1345                     top ? com_sun_glass_ui_Window_Level_FLOATING :  com_sun_glass_ui_Window_Level_NORMAL);
1346             CHECK_JNI_EXCEPTION(mainEnv);
1347         }
1348     }
1349 }

1351 void WindowContextTop::set_level(int level) {
1352     if (level == com_sun_glass_ui_Window_Level_NORMAL) {
1353         on_top = false;
1354     } else if (level == com_sun_glass_ui_Window_Level_FLOATING
1355             || level == com_sun_glass_ui_Window_Level_TOPMOST) {
1356         on_top = true;
1357     }
1358     // We need to emulate always on top behaviour on child windows
1359 
1360     if (!on_top_inherited()) {
1361         update_ontop_tree(on_top);
1362     }
1363 }
