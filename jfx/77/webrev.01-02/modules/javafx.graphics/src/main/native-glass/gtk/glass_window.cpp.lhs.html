 184 bool WindowContextBase::isEnabled() {
 185     if (jwindow) {
 186         bool result = (JNI_TRUE == mainEnv->CallBooleanMethod(jwindow, jWindowIsEnabled));
 187         LOG_EXCEPTION(mainEnv)
 188         return result;
 189     } else {
 190         return false;
 191     }

 194 void WindowContextBase::notify_state(jint glass_state) {
 195     if (glass_state == com_sun_glass_events_WindowEvent_RESTORE) {
 196         if (is_maximized) {
 197             glass_state = com_sun_glass_events_WindowEvent_MAXIMIZE;
 198         }
 199 
 200         int w, h;
 201         glass_gdk_window_get_size(gdk_window, &w, &h);
 202         if (jview) {
 203             mainEnv->CallVoidMethod(jview,
 204                     jViewNotifyRepaint,
 205                     0, 0, w, h);
 206             CHECK_JNI_EXCEPTION(mainEnv);
 207         }
 208     }
 209 
 210     if (jwindow) {
 211        mainEnv->CallVoidMethod(jwindow,
 212                jGtkWindowNotifyStateChanged,
 213                glass_state);
 214        CHECK_JNI_EXCEPTION(mainEnv);
 215     }
 216 }

 218 void WindowContextBase::process_state(GdkEventWindowState* event) {
 219     if (event->changed_mask &
 220             (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 221 
 222         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 223             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 224         }
 225         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 226             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 227         }
 228 
 229         jint stateChangeEvent;
 230 
 231         if (is_iconified) {
 232             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 233         } else if (is_maximized) {
 234             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 235         } else {
 236             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 237             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 238                 // in this case - the window manager will not support the programatic
 239                 // request to iconify - so we need to restore it now.
 240                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 241             }
 242         }
 243 
 244         notify_state(stateChangeEvent);
 245     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 246         notify_on_top( event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 247     }
 248 }

 250 void WindowContextBase::process_focus(GdkEventFocus* event) {
 251     if (!event->in) {
 252         ungrab_focus();
 253     }
 254 
 255     if (xim.enabled && xim.ic) {
 256         if (event->in) {
 257             XSetICFocus(xim.ic);
 258         } else {
 259             XUnsetICFocus(xim.ic);
 260         }
 261     }
 262 
 263     if (jwindow) {
 264         if (!event->in || isEnabled()) {
 265             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 266                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 267             CHECK_JNI_EXCEPTION(mainEnv)
 268         } else {
 269             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 270             CHECK_JNI_EXCEPTION(mainEnv)
 271         }
 272     }
 273 }

 291 void destroy_and_delete_ctx(WindowContext* ctx) {
 292     if (ctx) {
 293         ctx->process_destroy();
 294 
 295         if (!ctx->get_events_count()) {
 296             delete ctx;
 297         }
 298         // else: ctx will be deleted in EventsCounterHelper after completing
 299         // an event processing
 300     }
 301 }

 303 void WindowContextBase::process_destroy() {
 304     ungrab_focus();
 305 
 306     std::set<WindowContextTop*>::iterator it;
 307     for (it = children.begin(); it != children.end(); ++it) {
 308         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 309         // WindowContextTop::process_destroy() to call remove_child() (because children
 310         // is being iterated here) but also prevents gtk_window_set_transient_for from
 311         // being called - this causes the crash on gnome.
 312         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 313         (*it)->set_owner(NULL);
 314         destroy_and_delete_ctx(*it);
 315     }
 316     children.clear();
 317 
 318     if (jwindow) {
 319         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 320         EXCEPTION_OCCURED(mainEnv);
 321     }
 322 
 323     if (jview) {
 324         mainEnv->DeleteGlobalRef(jview);
 325         jview = NULL;
 326     }
 327 
 328     if (jwindow) {
 329         mainEnv->DeleteGlobalRef(jwindow);
 330         jwindow = NULL;
 331     }
 332 
 333     can_be_deleted = true;
 334 }

 336 void WindowContextBase::process_delete() {
 337     if (jwindow && isEnabled()) {
 338         gtk_widget_hide_on_delete(gtk_widget);
 339         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 340         CHECK_JNI_EXCEPTION(mainEnv)
 341     }
 342 }

 344 #ifdef GLASS_GTK3
 345 void WindowContextBase::process_draw(cairo_t* cr) {
 346     if (jview) {
 347         if (!paint_buffer(cr)) {
 348             GdkRectangle r;
 349             if (gdk_cairo_get_clip_rectangle (cr, &r)) {
 350                 mainEnv->CallVoidMethod(jview, jViewNotifyRepaint,r.x, r.y, r.width, r.height);
 351                 CHECK_JNI_EXCEPTION(mainEnv)

 353         }
 354     }
 355 }

 357 void WindowContextBase::process_expose(GdkEventExpose* event) {
 358    if (jview) {
 359         if (!buffer.pending) {
 360             mainEnv->CallVoidMethod(jview, jViewNotifyRepaint, event->area.x, event->area.y,
 361                                      event->area.width, event->area.height);
 362             CHECK_JNI_EXCEPTION(mainEnv)
 363         } else {
 364             cairo_t* context = gdk_cairo_create(gdk_window);
 365             paint_buffer(context);
 366             cairo_destroy(context);
 367         }
 368     }

 372 static inline jint gtk_button_number_to_mouse_button(guint button) {
 373     switch (button) {
 374         case 1:
 375             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 376         case 2:
 377             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 378         case 3:
 379             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 380         case MOUSE_BACK_BTN:
 381             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 382         case MOUSE_FORWARD_BTN:
 383             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 384         default:
 385             // Other buttons are not supported by quantum and are not reported by other platforms
 386             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 387     }
 388 }

 390 void WindowContextBase::process_mouse_button(GdkEventButton* event) {
 391     bool press = event->type == GDK_BUTTON_PRESS;
 392     guint state = event->state;
 393     guint mask = 0;
 394 
 395     // We need to add/remove current mouse button from the modifier flags
 396     // as X lib state represents the state just prior to the event and
 397     // glass needs the state just after the event
 398     switch (event->button) {
 399         case 1:
 400             mask = GDK_BUTTON1_MASK;
 401             break;
 402         case 2:
 403             mask = GDK_BUTTON2_MASK;
 404             break;
 405         case 3:
 406             mask = GDK_BUTTON3_MASK;
 407             break;
 408         case MOUSE_BACK_BTN:
 409             mask = GDK_BUTTON4_MASK;
 410             break;
 411         case MOUSE_FORWARD_BTN:
 412             mask = GDK_BUTTON5_MASK;
 413             break;
 414     }
 415 
 416     if (press) {
 417         state |= mask;
 418     } else {
 419         state &= ~mask;
 420     }
 421 
 422     jint button = gtk_button_number_to_mouse_button(event->button);
 423 
 424     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 425         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 426                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 427                 button,
 428                 (jint) event->x, (jint) event->y,
 429                 (jint) event->x_root, (jint) event->y_root,
 430                 gdk_modifier_mask_to_glass(state),
 431                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 432                 JNI_FALSE);
 433         CHECK_JNI_EXCEPTION(mainEnv)
 434 
 435         if (jview && event->button == 3 && press) {
 436             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 437                     (jint)event->x, (jint)event->y,
 438                     (jint)event->x_root, (jint)event->y_root,
 439                     JNI_FALSE);
 440             CHECK_JNI_EXCEPTION(mainEnv)
 441         }
 442     }
 443 }

 445 void WindowContextBase::process_mouse_motion(GdkEventMotion* event) {
 446     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 447     jint isDrag = glass_modifier & (
 448             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 449             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 450             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 451             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 452             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 453     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 454 
 455     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 456         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 457     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 458         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 459     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 460         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 461     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 462         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 463     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 464         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 465     }
 466 
 467     if (jview) {
 468         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 469                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 470                 button,
 471                 (jint) event->x, (jint) event->y,
 472                 (jint) event->x_root, (jint) event->y_root,
 473                 glass_modifier,
 474                 JNI_FALSE,
 475                 JNI_FALSE);
 476         CHECK_JNI_EXCEPTION(mainEnv)
 477     }
 478 }

 480 void WindowContextBase::process_mouse_scroll(GdkEventScroll* event) {
 481     jdouble dx = 0;
 482     jdouble dy = 0;
 483 
 484     // converting direction to change in pixels
 485     switch (event->direction) {
 486 #if GTK_CHECK_VERSION(3, 4, 0)
 487         case GDK_SCROLL_SMOOTH:
 488             //FIXME 3.4 ???
 489             break;
 490 #endif
 491         case GDK_SCROLL_UP:
 492             dy = 1;
 493             break;
 494         case GDK_SCROLL_DOWN:
 495             dy = -1;
 496             break;
 497         case GDK_SCROLL_LEFT:
 498             dx = 1;
 499             break;
 500         case GDK_SCROLL_RIGHT:
 501             dx = -1;
 502             break;
 503     }
 504     if (event->state & GDK_SHIFT_MASK) {
 505         jdouble t = dy;
 506         dy = dx;
 507         dx = t;
 508     }
 509     if (jview) {
 510         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 511                 (jint) event->x, (jint) event->y,
 512                 (jint) event->x_root, (jint) event->y_root,
 513                 dx, dy,
 514                 gdk_modifier_mask_to_glass(event->state),
 515                 (jint) 0, (jint) 0,
 516                 (jint) 0, (jint) 0,
 517                 (jdouble) 40.0, (jdouble) 40.0);
 518         CHECK_JNI_EXCEPTION(mainEnv)
 519     }
 520 }

 522 void WindowContextBase::process_mouse_cross(GdkEventCrossing* event) {
 523     bool enter = event->type == GDK_ENTER_NOTIFY;
 524 
 525     if (jview) {
 526         guint state = event->state;
 527         if (enter) { // workaround for RT-21590
 528             state &= ~MOUSE_BUTTONS_MASK;
 529         }
 530 
 531         if (enter != is_mouse_entered) {
 532             is_mouse_entered = enter;
 533             mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 534                     enter ? com_sun_glass_events_MouseEvent_ENTER : com_sun_glass_events_MouseEvent_EXIT,
 535                     com_sun_glass_events_MouseEvent_BUTTON_NONE,
 536                     (jint) event->x, (jint) event->y,
 537                     (jint) event->x_root, (jint) event->y_root,
 538                     gdk_modifier_mask_to_glass(state),
 539                     JNI_FALSE,
 540                     JNI_FALSE);
 541             CHECK_JNI_EXCEPTION(mainEnv)
 542         }
 543     }
 544 }

 546 void WindowContextBase::process_key(GdkEventKey* event) {
 547     bool press = event->type == GDK_KEY_PRESS;
 548     jint glassKey = get_glass_key(event);
 549     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 550     if (press) {
 551         glassModifier |= glass_key_to_modifier(glassKey);
 552     } else {
 553         glassModifier &= ~glass_key_to_modifier(glassKey);
 554     }
 555     jcharArray jChars = NULL;
 556     jchar key = gdk_keyval_to_unicode(event->keyval);
 557     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 558         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 559     } else {
 560 #ifdef GLASS_GTK2
 561         if (key == 0) {
 562             // Work around "bug" fixed in gtk-3.0:
 563             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 564             switch (event->keyval) {
 565             case 0xFF08 /* Backspace */: key =  '\b';
 566             case 0xFF09 /* Tab       */: key =  '\t';
 567             case 0xFF0A /* Linefeed  */: key =  '\n';
 568             case 0xFF0B /* Vert. Tab */: key =  '\v';
 569             case 0xFF0D /* Return    */: key =  '\r';
 570             case 0xFF1B /* Escape    */: key =  '\033';
 571             case 0xFFFF /* Delete    */: key =  '\177';
 572             }
 573         }
 574 #endif
 575     }
 576 
 577     if (key > 0) {
 578         jChars = mainEnv->NewCharArray(1);
 579         if (jChars) {
 580             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 581             CHECK_JNI_EXCEPTION(mainEnv)
 582         }
 583     } else {
 584         jChars = mainEnv->NewCharArray(0);
 585     }
 586     if (jview) {
 587         if (press) {
 588             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 589                     com_sun_glass_events_KeyEvent_PRESS,
 590                     glassKey,
 591                     jChars,
 592                     glassModifier);
 593             CHECK_JNI_EXCEPTION(mainEnv)
 594 
 595             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 596                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 597                         com_sun_glass_events_KeyEvent_TYPED,

 864 // This function calculate the deltas between window and window + decoration (titleblar, borders)
 865 // Only useful if the window manager does not support the frame extents extension
 866 void WindowContextTop::calculate_adjustments() {
 867     if (frame_type == TITLED && !geometry.frame_extents_received) {
 868         GdkRectangle er;
 869         gdk_window_get_frame_extents(gdk_window, &er);
 870 
 871         int w, h;
 872         gtk_window_get_size(GTK_WINDOW(gtk_widget), &w, &h);
 873 
 874         int px, py;
 875         gdk_window_get_origin(gdk_window, &px, &py);
 876 
 877         geometry.adjust_w = (er.width - w > 0) ? er.width - w : 0;
 878         geometry.adjust_h = (er.height - h > 0) ? er.height - h : 0;
 879         geometry.view_x = (px - er.x > 0) ? px - er.x : 0;
 880         geometry.view_y = (py - er.y > 0) ? py - er.y : 0;
 881 
 882         apply_geometry();
 883     }
 884 }

 886 void WindowContextTop::apply_geometry() {
 887     GdkGeometry gdk_geometry;
 888     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
 889 
 890     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
 891         gdk_geometry.min_width = geometry.current_w;
 892         gdk_geometry.min_height = geometry.current_h;
 893         gdk_geometry.max_width = geometry.current_w;
 894         gdk_geometry.max_height = geometry.current_h;
 895     } else {
 896         gdk_geometry.min_width = (geometry.minw > 0) ? geometry.minw - geometry.adjust_w : 1;
 897         gdk_geometry.min_height = (geometry.minh > 0) ? geometry.minh - geometry.adjust_h : 1;
 898         gdk_geometry.max_width = (geometry.maxw > 0) ? geometry.maxw  - geometry.adjust_w : G_MAXINT;
 899         gdk_geometry.max_height = (geometry.maxh > 0) ? geometry.maxh - geometry.adjust_h : G_MAXINT;
 900     }
 901 
 902     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), gtk_widget, &gdk_geometry,
 903         (GdkWindowHints) (GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
 904 }

 950 bool WindowContextTop::get_frame_extents_property(int *top, int *left,
 951         int *bottom, int *right) {
 952     unsigned long *extents;
 953 
 954     if (gdk_property_get(gdk_window,
 955             atom_net_wm_frame_extents,
 956             gdk_atom_intern("CARDINAL", FALSE),

 976 void WindowContextTop::process_net_wm_property() {
 977     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 978 
 979     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 980     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 981     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 982     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 983 
 984     gint length;
 985     glong* atoms = NULL;
 986 
 987     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 988             0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar**) &atoms)) {
 989 
 990         bool is_hidden = false;
 991         bool is_above = false;
 992         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 993             if (atom_net_wm_state_hidden == (GdkAtom)atoms[i]) {
 994                 is_hidden = true;
 995             } else if (atom_net_wm_state_above == (GdkAtom)atoms[i]) {
 996                 is_above = true;
 997             }
 998         }
 999 
1000         g_free(atoms);
1001 
1002         if (is_iconified != is_hidden) {
1003             is_iconified = is_hidden;
1004 
1005             notify_state((is_hidden)
1006                     ? com_sun_glass_events_WindowEvent_MINIMIZE
1007                     : com_sun_glass_events_WindowEvent_RESTORE);
1008         }
1009 
1010         notify_on_top(is_above);
1011     }
1012 }

1014 void WindowContextTop::process_property_notify(GdkEventProperty* event) {
1015     if (event->window == gdk_window) {
1016         if (event->atom == atom_net_wm_state) {
1017             process_net_wm_property();
1018         } else if (event->atom == atom_net_wm_frame_extents) {
1019             if (frame_type != TITLED) {
1020                 return;
1021             }
1022 
1023             int top, left, bottom, right;
1024 
1025             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
1026                 if (top + left + bottom + right > 0) {
1027                     geometry.frame_extents_received = true;
1028                     geometry.adjust_w = left + right;
1029                     geometry.adjust_h = top + bottom;
1030                     geometry.view_x = left;
1031                     geometry.view_y = top;
1032 
1033                     // set bounds again to set to correct window size that must
1034                     // be the total width and height accounting extents
1035                     set_bounds(0, 0,
1036                                false, false,
1037                                geometry.current_w, geometry.current_h,
1038                                -1, -1);
1039                 }
1040             }
1041         }
1042     }
1043 }

1045 void WindowContextTop::process_configure(GdkEventConfigure* event) {
1046     // this is to let java know about user changes on the window
1047     // if the window is not visible, there will be no changes
1048     if (!map_received && !is_maximized) {
1049         return;
1050     }
1051 
1052     gint x, y, w, h, gtk_w, gtk_h;
1053 
1054     calculate_adjustments();
1055 
1056     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
1057     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
1058     w = gtk_w + geometry.adjust_w;
1059     h = gtk_h + geometry.adjust_h;
1060 
1061     gboolean changed = geometry.current_x != x || geometry.current_y != y
1062                         || geometry.current_w != w || geometry.current_h != h
1063                         || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
1064 
1065     if (changed) {
1066         geometry.current_x = x;
1067         geometry.current_y = y;
1068         geometry.current_w = w;
1069         geometry.current_h = h;
1070         geometry.current_cw = gtk_w;
1071         geometry.current_ch = gtk_h;
1072 
1073         size_position_notify();
1074     }
1075 }

1112 void WindowContextTop::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
1113     calculate_adjustments();
1114 
1115     // newW / newH always content sizes compatible with GTK+
1116     int newW = w > 0 ? w - geometry.adjust_w : cw;
1117     int newH = h > 0 ? h - geometry.adjust_h : ch;
1118 
1119     gboolean changed = FALSE;
1120     if (newW > 0 && newH > 0) {
1121         changed = TRUE;
1122 
1123         geometry.current_cw = newW;
1124         geometry.current_ch = newH;
1125         geometry.current_w = newW + geometry.adjust_w;
1126         geometry.current_h = newH + geometry.adjust_h;
1127 
1128         if (map_received) {
1129             apply_geometry(); // update constraints if not resized by the user interface
1130         }
1131 
1132         gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
1133     }
1134 
1135     if (xSet || ySet) {
1136         int newX = (xSet) ? x : geometry.current_x;
1137         int newY = (ySet) ? y : geometry.current_y;
1138 
1139         if (newX != geometry.current_x || newY != geometry.current_y) {
1140             changed = TRUE;
1141             geometry.current_x = newX;
1142             geometry.current_y = newY;
1143             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1144         }
1145     }
1146 
1147     if (changed) {
1148         size_position_notify();
1149     }
1150 }

1311 bool WindowContextTop::on_top_inherited() {
1312     WindowContext* o = owner;
1313     while (o) {
1314         WindowContextTop* topO = dynamic_cast<WindowContextTop*>(o);
1315         if (!topO) break;
1316         if (topO->on_top) {
1317             return true;
1318         }
1319         o = topO->owner;
1320     }
1321     return false;
1322 }

1324 bool WindowContextTop::effective_on_top() {
1325     if (owner) {
1326         WindowContextTop* topO = dynamic_cast<WindowContextTop*>(owner);
1327         return (topO && topO->effective_on_top()) || on_top;
1328     }
1329     return on_top;
1330 }

1332 void WindowContextTop::notify_on_top(bool top) {
1333     // Do not report effective (i.e. native) values to the FX, only if the user sets it manually
1334     if (top != effective_on_top() && jwindow) {
1335         if (on_top_inherited() && !top) {
1336             // Disallow user's "on top" handling on windows that inherited the property
1337             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
1338         } else {
1339             on_top = top;
1340             update_ontop_tree(top);
1341             mainEnv->CallVoidMethod(jwindow,
1342                     jWindowNotifyLevelChanged,
1343                     top ? com_sun_glass_ui_Window_Level_FLOATING :  com_sun_glass_ui_Window_Level_NORMAL);
1344             CHECK_JNI_EXCEPTION(mainEnv);
1345         }
1346     }
1347 }

1349 void WindowContextTop::set_level(int level) {
1350     if (level == com_sun_glass_ui_Window_Level_NORMAL) {
1351         on_top = false;
1352     } else if (level == com_sun_glass_ui_Window_Level_FLOATING
1353             || level == com_sun_glass_ui_Window_Level_TOPMOST) {
1354         on_top = true;
1355     }
1356     // We need to emulate always on top behaviour on child windows
1357 
1358     if (!on_top_inherited()) {
1359         update_ontop_tree(on_top);
1360     }
1361 }
