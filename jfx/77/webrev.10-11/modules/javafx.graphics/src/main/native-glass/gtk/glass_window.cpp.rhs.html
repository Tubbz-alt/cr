 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 171 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 172                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 173         screen(_screen),
 174         frame_type(_frame_type),
 175         window_type(type),
 176         owner(_owner),
 177         jview(NULL),
 178         map_received(false),
 179         visible_received(false),
 180         on_top(false),
 181         is_fullscreen(false),
 182         is_iconified(false),
 183         is_maximized(false),
 184         is_mouse_entered(false),
 185         can_be_deleted(false),
 186         events_processing_cnt(0),
 187         win_group(NULL) {
 188 
 189     jwindow = mainEnv->NewGlobalRef(_jwindow);
 190 
 191     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 192 
 193     if (type != POPUP && owner == NULL) {
 194         win_group = gtk_window_group_new();
 195     }
 196 
 197     if (gchar * app_name = get_application_name()) {
 198         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 199         g_free(app_name);
 200     }
 201 
 202     if (owner) {
 203         owner->add_child(this);
 204         if (on_top_inherited()) {
 205             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 206         }
 207     }
 208 
 209     if (type == UTILITY) {
 210         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 211     }
 212 
 213     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 214 
 215     if (xvisualID != 0) {
 216         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 217         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 218     }
 219 
 220     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 221     gtk_widget_set_app_paintable(gtk_widget, TRUE);



 222 
 223     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 224     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 225 
 226     gdk_window = gtk_widget_get_window(gtk_widget);
 227 
 228     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 229 
 230     glass_dnd_attach_context(this);
 231 
 232     gdk_windowManagerFunctions = wmf;
 233     if (wmf) {
 234         gdk_window_set_functions(gdk_window, wmf);
 235     }
 236 
 237     if (frame_type != TITLED) {
 238         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 239     }
 240 
 241     connect_signals(gtk_widget, this);
 242 }

 244 void WindowContext::paint(void *data, jint width, jint height) {
 245 #if GTK_CHECK_VERSION(3, 0, 0)
 246     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 247 #if GTK_CHECK_VERSION(3, 22, 0)
 248     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 249     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 250 #else
 251     gdk_window_begin_paint_region(gdk_window, region);
 252     cairo_t* context = gdk_cairo_create(gdk_window);
 253 #endif
 254 #else
 255     cairo_t *context = gdk_cairo_create(gdk_window);
 256 #endif
 257 
 258     if (bg_color.is_set) {
 259         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 260                                 (frame_type == TRANSPARENT) ? 0 : 1);
 261         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 262         cairo_paint(context);
 263     }
 264 
 265     cairo_surface_t *cairo_surface;
 266     cairo_surface = cairo_image_surface_create_for_data(
 267             (unsigned char *) data,
 268             CAIRO_FORMAT_ARGB32,
 269             width, height, width * 4);
 270 
 271     cairo_set_source_surface(context, cairo_surface, 0, 0);
 272 
 273     applyShapeMask(data, width, height);
 274     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 275     cairo_paint(context);
 276 
 277 #if GTK_CHECK_VERSION(3, 0, 0)
 278 #if GTK_CHECK_VERSION(3, 22, 0)
 279     gdk_window_end_draw_frame(gdk_window, dcontext);
 280     cairo_region_destroy(region);
 281 #else
 282     gdk_window_end_paint(gdk_window);
 283     cairo_region_destroy(region);
 284     cairo_destroy(context);
 285 #endif
 286 #else
 287     cairo_destroy(context);
 288 #endif
 289 
 290     cairo_surface_destroy(cairo_surface);
 291 }

 332 void WindowContext::process_focus(GdkEventFocus *event) {
 333     if (!event->in) {
 334         ungrab_focus();
 335     }
 336 
 337     if (xim.enabled && xim.ic) {
 338         if (event->in) {
 339             XSetICFocus(xim.ic);
 340         } else {
 341             XUnsetICFocus(xim.ic);
 342         }
 343     }
 344 
 345     if (jwindow) {
 346         if (!event->in || isEnabled()) {
 347             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 348                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 349                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 350             CHECK_JNI_EXCEPTION(mainEnv)
 351         } else {
 352             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 353             CHECK_JNI_EXCEPTION(mainEnv)
 354         }
 355     }
 356 }

 358 void WindowContext::process_property_notify(GdkEventProperty *event) {
 359     if (event->window == gdk_window) {
 360         if (event->atom == atom_net_wm_state) {
 361             process_net_wm_property();
 362         } else if (event->atom == atom_net_wm_frame_extents) {
 363             if (frame_type != TITLED) {
 364                 return;
 365             }
 366 
 367             int top, left, bottom, right;
 368 
 369             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 370                 if (top + left + bottom + right > 0) {
 371                     geometry.frame_extents_received = true;
 372                     geometry.adjust_w = left + right;
 373                     geometry.adjust_h = top + bottom;
 374                     geometry.view_x = left;
 375                     geometry.view_y = top;
 376 
 377                     save_cached_extents();
 378 
 379                     // set bounds again to set to correct window size that must
 380                     // be the total width and height accounting extents
 381                     set_bounds(0, 0, false, false, -1, -1, -1, -1);
 382                 }
 383             }
 384         }
 385     }
 386 }

 388 void WindowContext::process_configure() {
 389     calculate_adjustments();
 390 
 391     gint x, y, w, h, gtk_w, gtk_h;
 392 
 393     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 394     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 395     w = gtk_w + geometry.adjust_w;
 396     h = gtk_h + geometry.adjust_h;
 397 
 398     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 399     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 400                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 401 
 402     geometry.current_x = x;
 403     geometry.current_y = y;
 404     geometry.current_w = w;
 405     geometry.current_h = h;
 406     geometry.current_cw = gtk_w;
 407     geometry.current_ch = gtk_h;
 408 
 409     size_position_notify(size_changed, pos_changed);
 410 }

 412 void WindowContext::process_destroy() {
 413     if (owner) {
 414         owner->remove_child(this);
 415     }
 416 
 417     ungrab_focus();
 418 
 419     std::set<WindowContext *>::iterator it;
 420     for (it = children.begin(); it != children.end(); ++it) {
 421         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 422         // WindowContext::process_destroy() to call remove_child() (because children
 423         // is being iterated here) but also prevents gtk_window_set_transient_for from
 424         // being called - this causes the crash on gnome.
 425         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 426         (*it)->set_owner(NULL);
 427         destroy_and_delete_ctx(*it);
 428     }
 429     children.clear();
 430 
 431     if (jwindow) {
 432         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 433         EXCEPTION_OCCURED(mainEnv);
 434     }
 435 
 436     if (jview) {
 437         mainEnv->DeleteGlobalRef(jview);
 438         jview = NULL;
 439     }
 440 
 441     if (jwindow) {
 442         mainEnv->DeleteGlobalRef(jwindow);
 443         jwindow = NULL;
 444     }
 445 
 446     can_be_deleted = true;
 447 }

 449 void WindowContext::process_delete() {

 451         gtk_widget_hide_on_delete(gtk_widget);
 452         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 453         CHECK_JNI_EXCEPTION(mainEnv)
 454     }

 465 void WindowContext::process_mouse_button(GdkEventButton *event) {
 466     bool press = event->type == GDK_BUTTON_PRESS;
 467     guint state = event->state;
 468     guint mask = 0;
 469 
 470     // We need to add/remove current mouse button from the modifier flags
 471     // as X lib state represents the state just prior to the event and
 472     // glass needs the state just after the event
 473     switch (event->button) {
 474         case 1:
 475             mask = GDK_BUTTON1_MASK;
 476             break;
 477         case 2:
 478             mask = GDK_BUTTON2_MASK;
 479             break;
 480         case 3:
 481             mask = GDK_BUTTON3_MASK;
 482             break;
 483         case MOUSE_BACK_BTN:
 484             mask = GDK_BUTTON4_MASK;
 485             break;
 486         case MOUSE_FORWARD_BTN:
 487             mask = GDK_BUTTON5_MASK;
 488             break;
 489     }
 490 
 491     if (press) {
 492         state |= mask;
 493     } else {
 494         state &= ~mask;
 495     }
 496 
 497     jint button = gtk_button_number_to_mouse_button(event->button);
 498 
 499     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 500         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 501                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 502                                 button,
 503                                 (jint) event->x, (jint) event->y,
 504                                 (jint) event->x_root, (jint) event->y_root,
 505                                 gdk_modifier_mask_to_glass(state),
 506                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 507                                 JNI_FALSE);
 508         CHECK_JNI_EXCEPTION(mainEnv)
 509 
 510         if (jview && event->button == 3 && press) {
 511             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 512                                     (jint) event->x, (jint) event->y,
 513                                     (jint) event->x_root, (jint) event->y_root,
 514                                     JNI_FALSE);
 515             CHECK_JNI_EXCEPTION(mainEnv)
 516         }
 517     }
 518 }

 520 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 521     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 522     jint isDrag = glass_modifier & (
 523             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 524             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 525             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 526             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 527             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 528     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 529 
 530     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 531         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 532     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 533         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 534     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 535         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 536     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 537         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 538     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 539         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 540     }
 541 
 542     if (jview) {
 543         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 544                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 545                                 button,
 546                                 (jint) event->x, (jint) event->y,
 547                                 (jint) event->x_root, (jint) event->y_root,
 548                                 glass_modifier,
 549                                 JNI_FALSE,
 550                                 JNI_FALSE);
 551         CHECK_JNI_EXCEPTION(mainEnv)
 552     }
 553 }

 555 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 556     jdouble dx = 0;
 557     jdouble dy = 0;
 558 
 559     // converting direction to change in pixels
 560     switch (event->direction) {
 561 #if GTK_CHECK_VERSION(3, 4, 0)
 562         case GDK_SCROLL_SMOOTH:
 563             //FIXME 3.4 ???
 564             break;
 565 #endif
 566         case GDK_SCROLL_UP:
 567             dy = 1;
 568             break;
 569         case GDK_SCROLL_DOWN:
 570             dy = -1;
 571             break;
 572         case GDK_SCROLL_LEFT:
 573             dx = 1;
 574             break;
 575         case GDK_SCROLL_RIGHT:
 576             dx = -1;
 577             break;
 578     }
 579     if (event->state & GDK_SHIFT_MASK) {
 580         jdouble t = dy;
 581         dy = dx;
 582         dx = t;
 583     }
 584     if (jview) {
 585         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 586                                 (jint) event->x, (jint) event->y,
 587                                 (jint) event->x_root, (jint) event->y_root,
 588                                 dx, dy,
 589                                 gdk_modifier_mask_to_glass(event->state),
 590                                 (jint) 0, (jint) 0,
 591                                 (jint) 0, (jint) 0,
 592                                 (jdouble) 40.0, (jdouble) 40.0);
 593         CHECK_JNI_EXCEPTION(mainEnv)
 594     }
 595 }

 597 void WindowContext::process_mouse_cross(GdkEventCrossing *event) {
 598     bool enter = event->type == GDK_ENTER_NOTIFY;
 599 
 600     if (jview) {
 601         guint state = event->state;
 602         if (enter) { // workaround for RT-21590
 603             state &= ~MOUSE_BUTTONS_MASK;
 604         }
 605 
 606         if (enter != is_mouse_entered) {

 622 void WindowContext::process_key(GdkEventKey *event) {
 623     bool press = event->type == GDK_KEY_PRESS;
 624     jint glassKey = get_glass_key(event);
 625     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 626     if (press) {
 627         glassModifier |= glass_key_to_modifier(glassKey);
 628     } else {
 629         glassModifier &= ~glass_key_to_modifier(glassKey);
 630     }
 631     jcharArray jChars = NULL;
 632     jchar key = gdk_keyval_to_unicode(event->keyval);
 633     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 634         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 635     } else {
 636 #ifdef GLASS_GTK2
 637         if (key == 0) {
 638             // Work around "bug" fixed in gtk-3.0:
 639             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 640             switch (event->keyval) {
 641             case 0xFF08 /* Backspace */: key =  '\b';
 642             case 0xFF09 /* Tab       */: key =  '\t';
 643             case 0xFF0A /* Linefeed  */: key =  '\n';
 644             case 0xFF0B /* Vert. Tab */: key =  '\v';
 645             case 0xFF0D /* Return    */: key =  '\r';
 646             case 0xFF1B /* Escape    */: key =  '\033';
 647             case 0xFFFF /* Delete    */: key =  '\177';
 648             }
 649         }
 650 #endif
 651     }
 652 
 653     if (key > 0) {
 654         jChars = mainEnv->NewCharArray(1);
 655         if (jChars) {
 656             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 657             CHECK_JNI_EXCEPTION(mainEnv)
 658         }
 659     } else {
 660         jChars = mainEnv->NewCharArray(0);
 661     }
 662     if (jview) {
 663         if (press) {
 664             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 665                                     com_sun_glass_events_KeyEvent_PRESS,
 666                                     glassKey,
 667                                     jChars,
 668                                     glassModifier);
 669             CHECK_JNI_EXCEPTION(mainEnv)
 670 
 671             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 672                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 673                                         com_sun_glass_events_KeyEvent_TYPED,
 674                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 675                                         jChars,
 676                                         glassModifier);
 677                 CHECK_JNI_EXCEPTION(mainEnv)
 678             }
 679         } else {
 680             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 681                                     com_sun_glass_events_KeyEvent_RELEASE,
 682                                     glassKey,
 683                                     jChars,
 684                                     glassModifier);
 685             CHECK_JNI_EXCEPTION(mainEnv)
 686         }
 687     }
 688 }

 690 void WindowContext::process_state(GdkEventWindowState *event) {
 691     if (event->changed_mask &
 692         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 693 
 694         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 695             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 696         }
 697         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 698             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 699         }
 700 
 701         jint stateChangeEvent;
 702 
 703         if (is_iconified) {
 704             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 705         } else if (is_maximized) {
 706             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 707         } else {
 708             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 709             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 710                 // in this case - the window manager will not support the programatic
 711                 // request to iconify - so we need to restore it now.
 712                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 713             }
 714         }
 715 
 716         notify_state(stateChangeEvent);
 717     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 718         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 719     }
 720 }

 722 void WindowContext::process_net_wm_property() {
 723     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 724 
 725     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 726     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 727     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 728     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 729 
 730     gint length;
 731     glong *atoms = NULL;
 732 
 733     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 734                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 735 
 736         bool is_hidden = false;
 737         bool is_above = false;
 738         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 739             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 740                 is_hidden = true;
 741             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 742                 is_above = true;
 743             }
 744         }
 745 
 746         g_free(atoms);
 747 
 748         if (is_iconified != is_hidden) {
 749             is_iconified = is_hidden;
 750 
 751             notify_state((is_hidden)
 752                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 753                          : com_sun_glass_events_WindowEvent_RESTORE);
 754         }
 755 
 756         notify_on_top(is_above);
 757     }
 758 }

 932 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 933     calculate_adjustments();
 934 
 935     // newW / newH always content sizes compatible with GTK+
 936     // if window has no decoration, adjustments will be ZERO
 937     int newW = w > 0 ? w - geometry.adjust_w : cw;
 938     int newH = h > 0 ? h - geometry.adjust_h : ch;
 939 
 940     gboolean size_changed = FALSE;
 941     gboolean pos_changed = FALSE;
 942 
 943     if (newW > 0 && newH > 0) {
 944         size_changed = TRUE;
 945 
 946         geometry.current_cw = newW;
 947         geometry.current_ch = newH;
 948         geometry.current_w = newW + geometry.adjust_w;
 949         geometry.current_h = newH + geometry.adjust_h;
 950 
 951         if (visible_received) {
 952             // update constraints if not resized by the user interface so it will
 953             // let gtk_window_resize succeed, because it's bound to geometry constraints
 954             apply_geometry();
 955             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
 956         } else {
 957             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
 958         }
 959     }
 960 
 961     if (xSet || ySet) {
 962         int newX = (xSet) ? x : geometry.current_x;
 963         int newY = (ySet) ? y : geometry.current_y;
 964 
 965         if (newX != geometry.current_x || newY != geometry.current_y) {
 966             pos_changed = TRUE;
 967             geometry.current_x = newX;
 968             geometry.current_y = newY;
 969             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
 970         }
 971     }
 972 
 973     size_position_notify(size_changed, pos_changed);
 974 }

1177 
1178 // This function calculate the deltas between window and window + decoration (titleblar, borders).
1179 // Only useful if the window manager does not support the frame extents extension - in this case
1180 // it uses GDK calculation, that is primary based on _NET_FRAME_EXTENTS but will try harder
1181 // if that's not available.
1182 void WindowContext::calculate_adjustments() {
1183     if (frame_type != TITLED || geometry.frame_extents_received) {
1184         return;
1185     }
1186 
1187     gint x, y;
1188     gdk_window_get_origin(gdk_window, &x, &y);
1189 
1190     gint rx, ry;
1191     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1192 
1193     if (rx != x || ry != y) {
1194         // the left extends are correct - the right one is guessed to be the same
1195         geometry.adjust_w = (rx - x) * 2;
1196         // guess that bottom size is the same as left and right
1197         geometry.adjust_h = (ry - y) + (rx - x);
1198 
1199         // those will be correct
1200         geometry.view_x = (rx - x);
1201         geometry.view_y = (ry - y);
1202 
1203         save_cached_extents();
1204     } else {
1205         CachedExtents c = (window_type == NORMAL) ? normal_extents : utility_extents;
1206         geometry.adjust_w = c.adjust_w;
1207         geometry.adjust_h = c.adjust_h;
1208         geometry.view_x = c.view_x;
1209         geometry.view_y = c.view_y;
1210     }
1211 
1212     apply_geometry();
1213 }
1214 
1215 void WindowContext::save_cached_extents() {
1216     if (frame_type != TITLED) {
1217         return;
1218     }
1219 
1220     if (window_type == NORMAL) {
1221         normal_extents.adjust_w = geometry.adjust_w;
1222         normal_extents.adjust_h = geometry.adjust_h;
1223         normal_extents.view_x = geometry.view_x;
1224         normal_extents.view_y = geometry.view_y;
1225     } else {
1226         utility_extents.adjust_w = geometry.adjust_w;
1227         utility_extents.adjust_h = geometry.adjust_h;
1228         utility_extents.view_x = geometry.view_x;

1233 void WindowContext::apply_geometry() {
1234     if (!map_received) {
1235         return;
1236     }
1237 
1238     GdkGeometry gdk_geometry;
1239     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1240 
1241     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1242         // not resizeable
1243         int w = (geometry.current_w - geometry.adjust_w) > 0
1244                 ? geometry.current_w - geometry.adjust_w
1245                 : geometry.current_cw;
1246 
1247         int h = (geometry.current_h - geometry.adjust_h) > 0
1248                 ? geometry.current_h - geometry.adjust_h
1249                 : geometry.current_ch;
1250 
1251         gdk_geometry.min_width = gdk_geometry.max_width = w;
1252         gdk_geometry.min_height = gdk_geometry.max_height = h;
1253     } else {
1254         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1255                                  ? geometry.minw - geometry.adjust_w : 1;
1256         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1257                                   ? geometry.minh - geometry.adjust_h : 1;
1258 
1259         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1260                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1261         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1262                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1263     }
1264 
1265     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1266                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1267 }

1295 void WindowContext::activate_window() {
1296     Display *display = GDK_DISPLAY_XDISPLAY(gdk_window_get_display(gdk_window));
1297     Atom navAtom = XInternAtom(display, "_NET_ACTIVE_WINDOW", True);
1298     if (navAtom != None) {
1299         XClientMessageEvent clientMessage;
1300         memset(&clientMessage, 0, sizeof(clientMessage));
1301 
1302         clientMessage.type = ClientMessage;
1303         clientMessage.window = GDK_WINDOW_XID(gdk_window);
1304         clientMessage.message_type = navAtom;
1305         clientMessage.format = 32;
1306         clientMessage.data.l[0] = 1;
1307         clientMessage.data.l[1] = gdk_x11_get_server_time(gdk_window);
1308         clientMessage.data.l[2] = 0;
1309 
1310         XSendEvent(display, XDefaultRootWindow(display), False,
1311                    SubstructureRedirectMask | SubstructureNotifyMask,
1312                    (XEvent * ) & clientMessage);
1313         XFlush(display);
1314     }

1317 void WindowContext::size_position_notify(bool size_changed, bool pos_changed) {
1318     if (jview) {
1319         if (size_changed) {
1320             mainEnv->CallVoidMethod(jview, jViewNotifyResize, geometry.current_cw, geometry.current_ch);
1321             CHECK_JNI_EXCEPTION(mainEnv);
1322         }
1323 
1324         if (pos_changed) {
1325             mainEnv->CallVoidMethod(jview, jViewNotifyView, com_sun_glass_events_ViewEvent_MOVE);
1326             CHECK_JNI_EXCEPTION(mainEnv)
1327         }
1328     }
1329 
1330     if (jwindow) {
1331         if (size_changed || is_maximized) {
1332             mainEnv->CallVoidMethod(jwindow, jWindowNotifyResize,
1333                                     (is_maximized)
1334                                     ? com_sun_glass_events_WindowEvent_MAXIMIZE
1335                                     : com_sun_glass_events_WindowEvent_RESIZE,
1336                                     geometry.current_w, geometry.current_h);
1337             CHECK_JNI_EXCEPTION(mainEnv)
1338         }
1339 
1340         if (pos_changed) {
1341             mainEnv->CallVoidMethod(jwindow, jWindowNotifyMove, geometry.current_x, geometry.current_y);
1342             CHECK_JNI_EXCEPTION(mainEnv)
1343         }
1344     }

1355 bool WindowContext::on_top_inherited() {
1356     WindowContext *o = owner;
1357     while (o) {
1358         WindowContext *topO = dynamic_cast<WindowContext *>(o);
1359         if (!topO) break;
1360         if (topO->on_top) {
1361             return true;
1362         }
1363         o = topO->owner;
1364     }
1365     return false;
1366 }

1368 bool WindowContext::effective_on_top() {
1369     if (owner) {
1370         WindowContext *topO = dynamic_cast<WindowContext *>(owner);
1371         return (topO && topO->effective_on_top()) || on_top;
1372     }
1373     return on_top;
1374 }
