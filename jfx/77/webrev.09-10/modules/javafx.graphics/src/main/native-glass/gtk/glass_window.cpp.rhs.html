 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 171 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 172                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 173         screen(_screen),
 174         frame_type(_frame_type),
 175         window_type(type),
 176         owner(_owner),
 177         jview(NULL),
 178         map_received(false),
 179         visible_received(false),
 180         on_top(false),
 181         is_fullscreen(false),
 182         is_iconified(false),
 183         is_maximized(false),
 184         is_mouse_entered(false),
 185         can_be_deleted(false),
 186         events_processing_cnt(0),
 187         win_group(NULL) {
 188 
 189     jwindow = mainEnv->NewGlobalRef(_jwindow);
 190 
 191     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 192 
 193     if (type != POPUP && owner == NULL) {
 194         win_group = gtk_window_group_new();
 195     }
 196 
 197     if (gchar * app_name = get_application_name()) {
 198         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 199         g_free(app_name);
 200     }
 201 
 202     if (owner) {
 203         owner->add_child(this);
 204         if (on_top_inherited()) {
 205             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 206         }
 207     }
 208 
 209     if (type == UTILITY) {
 210         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 211     }
 212 
 213     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 214 
 215     if (xvisualID != 0) {
 216         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 217         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 218     }
 219 
 220     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 221     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 222     if (frame_type != TITLED) {
 223         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 224     }
 225 
 226     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 227     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 228 
 229     gdk_window = gtk_widget_get_window(gtk_widget);
 230 
 231     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 232 
 233     glass_dnd_attach_context(this);
 234 
 235     gdk_windowManagerFunctions = wmf;
 236     if (wmf) {
 237         gdk_window_set_functions(gdk_window, wmf);
 238     }
 239 
 240     connect_signals(gtk_widget, this);
 241 }

 243 void WindowContext::paint(void *data, jint width, jint height) {
 244 #if GTK_CHECK_VERSION(3, 0, 0)
 245     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 246 #if GTK_CHECK_VERSION(3, 22, 0)
 247     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 248     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 249 #else
 250     gdk_window_begin_paint_region(gdk_window, region);
 251     cairo_t* context = gdk_cairo_create(gdk_window);
 252 #endif
 253 #else
 254     cairo_t *context = gdk_cairo_create(gdk_window);
 255 #endif
 256 
 257     cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue, (frame_type == TRANSPARENT) ? 0 : 1);
 258     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 259     cairo_paint(context);
 260 
 261     cairo_surface_t *cairo_surface;
 262     cairo_surface = cairo_image_surface_create_for_data(
 263             (unsigned char *) data,
 264             CAIRO_FORMAT_ARGB32,
 265             width, height, width * 4);
 266 
 267     cairo_set_source_surface(context, cairo_surface, 0, 0);
 268 
 269     applyShapeMask(data, width, height);
 270     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 271     cairo_paint(context);
 272 
 273 #if GTK_CHECK_VERSION(3, 0, 0)
 274 #if GTK_CHECK_VERSION(3, 22, 0)
 275     gdk_window_end_draw_frame(gdk_window, dcontext);
 276     cairo_region_destroy(region);
 277 #else
 278     gdk_window_end_paint(gdk_window);
 279     cairo_region_destroy(region);
 280     cairo_destroy(context);
 281 #endif
 282 #else
 283     cairo_destroy(context);
 284 #endif
 285 
 286     cairo_surface_destroy(cairo_surface);
 287 }

 328 void WindowContext::process_focus(GdkEventFocus *event) {
 329     if (!event->in) {
 330         ungrab_focus();
 331     }
 332 
 333     if (xim.enabled && xim.ic) {
 334         if (event->in) {
 335             XSetICFocus(xim.ic);
 336         } else {
 337             XUnsetICFocus(xim.ic);
 338         }
 339     }
 340 
 341     if (jwindow) {
 342         if (!event->in || isEnabled()) {
 343             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 344                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 345                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 346             CHECK_JNI_EXCEPTION(mainEnv)
 347         } else {
 348             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 349             CHECK_JNI_EXCEPTION(mainEnv)
 350         }
 351     }
 352 }

 354 void WindowContext::process_property_notify(GdkEventProperty *event) {
 355     if (event->window == gdk_window) {
 356         if (event->atom == atom_net_wm_state) {
 357             process_net_wm_property();
 358         } else if (event->atom == atom_net_wm_frame_extents) {
 359             if (frame_type != TITLED) {
 360                 return;
 361             }
 362 
 363             int top, left, bottom, right;


 364 
 365             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 366                 if (top + left + bottom + right > 0) {
 367                     geometry.frame_extents_received = true;
 368                     geometry.adjust_w = left + right;
 369                     geometry.adjust_h = top + bottom;
 370                     geometry.view_x = left;
 371                     geometry.view_y = top;
 372 
 373                     save_cached_extents();


 374 
 375                     // set bounds again to set to correct window size that must
 376                     // be the total width and height accounting extents
 377                     set_bounds(0, 0, false, false, -1, -1, -1, -1);
 378                 }
 379             }
 380         }


 381     }
 382 }

 384 void WindowContext::process_configure() {
 385     calculate_adjustments();
 386 
 387     gint x, y, w, h, gtk_w, gtk_h;
 388 
 389     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 390     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 391     w = gtk_w + geometry.adjust_w;
 392     h = gtk_h + geometry.adjust_h;
 393 
 394     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 395     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 396                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 397 
 398     geometry.current_x = x;
 399     geometry.current_y = y;
 400     geometry.current_w = w;
 401     geometry.current_h = h;
 402     geometry.current_cw = gtk_w;
 403     geometry.current_ch = gtk_h;
 404 
 405     size_position_notify(size_changed, pos_changed);
 406 }

 408 void WindowContext::process_destroy() {
 409     if (owner) {
 410         owner->remove_child(this);
 411     }
 412 
 413     ungrab_focus();
 414 
 415     std::set<WindowContext *>::iterator it;
 416     for (it = children.begin(); it != children.end(); ++it) {
 417         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 418         // WindowContext::process_destroy() to call remove_child() (because children
 419         // is being iterated here) but also prevents gtk_window_set_transient_for from
 420         // being called - this causes the crash on gnome.
 421         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 422         (*it)->set_owner(NULL);
 423         destroy_and_delete_ctx(*it);
 424     }
 425     children.clear();
 426 
 427     if (jwindow) {
 428         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 429         EXCEPTION_OCCURED(mainEnv);
 430     }
 431 
 432     if (jview) {
 433         mainEnv->DeleteGlobalRef(jview);
 434         jview = NULL;
 435     }
 436 
 437     if (jwindow) {
 438         mainEnv->DeleteGlobalRef(jwindow);
 439         jwindow = NULL;
 440     }
 441 
 442     can_be_deleted = true;
 443 }

 448         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 449         CHECK_JNI_EXCEPTION(mainEnv)
 450     }

 461 void WindowContext::process_mouse_button(GdkEventButton *event) {


















 462     bool press = event->type == GDK_BUTTON_PRESS;
 463     guint state = event->state;
 464     guint mask = 0;
 465 
 466     // We need to add/remove current mouse button from the modifier flags
 467     // as X lib state represents the state just prior to the event and
 468     // glass needs the state just after the event
 469     switch (event->button) {
 470         case 1:
 471             mask = GDK_BUTTON1_MASK;
 472             break;
 473         case 2:
 474             mask = GDK_BUTTON2_MASK;
 475             break;
 476         case 3:
 477             mask = GDK_BUTTON3_MASK;
 478             break;
 479         case MOUSE_BACK_BTN:
 480             mask = GDK_BUTTON4_MASK;
 481             break;
 482         case MOUSE_FORWARD_BTN:
 483             mask = GDK_BUTTON5_MASK;
 484             break;
 485     }
 486 
 487     if (press) {
 488         state |= mask;
 489     } else {
 490         state &= ~mask;
 491     }
 492 
 493     jint button = gtk_button_number_to_mouse_button(event->button);
 494 
 495     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 496         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 497                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 498                                 button,
 499                                 (jint) event->x, (jint) event->y,
 500                                 (jint) event->x_root, (jint) event->y_root,
 501                                 gdk_modifier_mask_to_glass(state),
 502                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 503                                 JNI_FALSE);
 504         CHECK_JNI_EXCEPTION(mainEnv)
 505 
 506         if (jview && event->button == 3 && press) {
 507             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 508                                     (jint) event->x, (jint) event->y,
 509                                     (jint) event->x_root, (jint) event->y_root,
 510                                     JNI_FALSE);
 511             CHECK_JNI_EXCEPTION(mainEnv)
 512         }
 513     }
 514 }

 516 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 517     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 518     jint isDrag = glass_modifier & (
 519             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 520             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 521             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 522             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 523             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 524     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 525 
 526     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 527         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 528     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 529         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 530     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 531         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 532     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 533         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 534     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 535         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 536     }
 537 
 538     if (jview) {
 539         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 540                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 541                                 button,
 542                                 (jint) event->x, (jint) event->y,
 543                                 (jint) event->x_root, (jint) event->y_root,
 544                                 glass_modifier,
 545                                 JNI_FALSE,
 546                                 JNI_FALSE);
 547         CHECK_JNI_EXCEPTION(mainEnv)
 548     }
 549 }

 551 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 552     jdouble dx = 0;
 553     jdouble dy = 0;
 554 
 555     // converting direction to change in pixels
 556     switch (event->direction) {
 557 #if GTK_CHECK_VERSION(3, 4, 0)
 558         case GDK_SCROLL_SMOOTH:
 559             //FIXME 3.4 ???
 560             break;
 561 #endif
 562         case GDK_SCROLL_UP:
 563             dy = 1;
 564             break;
 565         case GDK_SCROLL_DOWN:
 566             dy = -1;
 567             break;
 568         case GDK_SCROLL_LEFT:
 569             dx = 1;
 570             break;
 571         case GDK_SCROLL_RIGHT:
 572             dx = -1;
 573             break;
 574     }
 575     if (event->state & GDK_SHIFT_MASK) {
 576         jdouble t = dy;
 577         dy = dx;
 578         dx = t;
 579     }
 580     if (jview) {
 581         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 582                                 (jint) event->x, (jint) event->y,
 583                                 (jint) event->x_root, (jint) event->y_root,
 584                                 dx, dy,
 585                                 gdk_modifier_mask_to_glass(event->state),
 586                                 (jint) 0, (jint) 0,
 587                                 (jint) 0, (jint) 0,
 588                                 (jdouble) 40.0, (jdouble) 40.0);
 589         CHECK_JNI_EXCEPTION(mainEnv)
 590     }
 591 }

 593 void WindowContext::process_mouse_cross(GdkEventCrossing *event) {
 594     bool enter = event->type == GDK_ENTER_NOTIFY;
 595 
 596     if (jview) {
 597         guint state = event->state;
 598         if (enter) { // workaround for RT-21590
 599             state &= ~MOUSE_BUTTONS_MASK;
 600         }
 601 
 602         if (enter != is_mouse_entered) {
 603             is_mouse_entered = enter;

 618 void WindowContext::process_key(GdkEventKey *event) {
 619     bool press = event->type == GDK_KEY_PRESS;
 620     jint glassKey = get_glass_key(event);
 621     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 622     if (press) {
 623         glassModifier |= glass_key_to_modifier(glassKey);
 624     } else {
 625         glassModifier &= ~glass_key_to_modifier(glassKey);
 626     }
 627     jcharArray jChars = NULL;
 628     jchar key = gdk_keyval_to_unicode(event->keyval);
 629     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 630         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 631     } else {
 632 #ifdef GLASS_GTK2
 633         if (key == 0) {
 634             // Work around "bug" fixed in gtk-3.0:
 635             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 636             switch (event->keyval) {
 637             case 0xFF08 /* Backspace */: key =  '\b';
 638             case 0xFF09 /* Tab       */: key =  '\t';
 639             case 0xFF0A /* Linefeed  */: key =  '\n';
 640             case 0xFF0B /* Vert. Tab */: key =  '\v';
 641             case 0xFF0D /* Return    */: key =  '\r';
 642             case 0xFF1B /* Escape    */: key =  '\033';
 643             case 0xFFFF /* Delete    */: key =  '\177';
 644             }
 645         }
 646 #endif
 647     }
 648 
 649     if (key > 0) {
 650         jChars = mainEnv->NewCharArray(1);
 651         if (jChars) {
 652             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 653             CHECK_JNI_EXCEPTION(mainEnv)
 654         }
 655     } else {
 656         jChars = mainEnv->NewCharArray(0);
 657     }
 658     if (jview) {
 659         if (press) {
 660             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 661                                     com_sun_glass_events_KeyEvent_PRESS,
 662                                     glassKey,
 663                                     jChars,
 664                                     glassModifier);
 665             CHECK_JNI_EXCEPTION(mainEnv)
 666 
 667             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 668                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 669                                         com_sun_glass_events_KeyEvent_TYPED,
 670                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 671                                         jChars,
 672                                         glassModifier);
 673                 CHECK_JNI_EXCEPTION(mainEnv)
 674             }
 675         } else {
 676             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 677                                     com_sun_glass_events_KeyEvent_RELEASE,
 678                                     glassKey,
 679                                     jChars,
 680                                     glassModifier);
 681             CHECK_JNI_EXCEPTION(mainEnv)
 682         }
 683     }
 684 }

 686 void WindowContext::process_state(GdkEventWindowState *event) {
 687     if (event->changed_mask &
 688         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {























































































































































































































































































































































































































































 689 
 690         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 691             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;






 692         }
 693         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 694             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;







 695         }
 696 
 697         jint stateChangeEvent;























 698 
 699         if (is_iconified) {
 700             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 701         } else if (is_maximized) {
 702             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 703         } else {
 704             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 705             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 706                 // in this case - the window manager will not support the programatic
 707                 // request to iconify - so we need to restore it now.
 708                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 709             }
 710         }
 711 
 712         notify_state(stateChangeEvent);
 713     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 714         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 715     }
 716 }

 718 void WindowContext::process_net_wm_property() {
 719     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 720 
 721     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 722     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 723     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 724     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 725 
 726     gint length;
 727     glong *atoms = NULL;


 728 
 729     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 730                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {

 731 
 732         bool is_hidden = false;
 733         bool is_above = false;
 734         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 735             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 736                 is_hidden = true;
 737             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 738                 is_above = true;
 739             }
 740         }
 741 
 742         g_free(atoms);
 743 
 744         if (is_iconified != is_hidden) {
 745             is_iconified = is_hidden;
 746 
 747             notify_state((is_hidden)
 748                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 749                          : com_sun_glass_events_WindowEvent_RESTORE);
 750         }
 751 
 752         notify_on_top(is_above);
 753     }
 754 }

 923 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 924     calculate_adjustments();
 925 
 926     // newW / newH always content sizes compatible with GTK+
 927     // if window has no decoration, adjustments will be ZERO
 928     int newW = w > 0 ? w - geometry.adjust_w : cw;
 929     int newH = h > 0 ? h - geometry.adjust_h : ch;
 930 
 931     gboolean size_changed = FALSE;
 932     gboolean pos_changed = FALSE;
 933 
 934     if (newW > 0 && newH > 0) {
 935         size_changed = TRUE;
 936 
 937         geometry.current_cw = newW;
 938         geometry.current_ch = newH;
 939         geometry.current_w = newW + geometry.adjust_w;
 940         geometry.current_h = newH + geometry.adjust_h;
 941 
 942         if (visible_received) {
 943             // update constraints if not resized by the user interface so it will
 944             // let gtk_window_resize succeed, because it's bound to geometry constraints
 945             apply_geometry();
 946             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
 947         } else {
 948             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
 949         }
 950     }
 951 
 952     if (xSet || ySet) {
 953         int newX = (xSet) ? x : geometry.current_x;
 954         int newY = (ySet) ? y : geometry.current_y;
 955 
 956         if (newX != geometry.current_x || newY != geometry.current_y) {
 957             pos_changed = TRUE;
 958             geometry.current_x = newX;
 959             geometry.current_y = newY;
 960             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
 961         }
 962     }
 963 
 964     size_position_notify(size_changed, pos_changed);
 965 }

1169 // This function calculate the deltas between window and window + decoration (titleblar, borders).
1170 // Only useful if the window manager does not support the frame extents extension - in this case
1171 // it uses GDK calculation, that is primary based on _NET_FRAME_EXTENTS but will try harder
1172 // if that's not available.
1173 void WindowContext::calculate_adjustments() {
1174     if (frame_type != TITLED || geometry.frame_extents_received) {
1175         return;
1176     }
1177 
1178     gint x, y;
1179     gdk_window_get_origin(gdk_window, &x, &y);
1180 
1181     gint rx, ry;
1182     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1183 
1184     if (rx != x || ry != y) {
1185         // the left extends are correct - the right one is guessed to be the same
1186         geometry.adjust_w = (rx - x) * 2;
1187         // guess that bottom size is the same as left and right
1188         geometry.adjust_h = (ry - y) + (rx - x);
1189 
1190         // those will be correct
1191         geometry.view_x = (rx - x);
1192         geometry.view_y = (ry - y);
1193 
1194         save_cached_extents();
1195     } else {
1196         CachedExtents c = (window_type == NORMAL) ? normal_extents : utility_extents;
1197         geometry.adjust_w = c.adjust_w;
1198         geometry.adjust_h = c.adjust_h;
1199         geometry.view_x = c.view_x;
1200         geometry.view_y = c.view_y;
1201     }
1202 
1203     apply_geometry();
1204 }
1205 
1206 void WindowContext::save_cached_extents() {
1207     if (frame_type != TITLED) {
1208         return;
1209     }
1210 
1211     if (window_type == NORMAL) {
1212         normal_extents.adjust_w = geometry.adjust_w;
1213         normal_extents.adjust_h = geometry.adjust_h;
1214         normal_extents.view_x = geometry.view_x;
1215         normal_extents.view_y = geometry.view_y;
1216     } else {
1217         utility_extents.adjust_w = geometry.adjust_w;
1218         utility_extents.adjust_h = geometry.adjust_h;
1219         utility_extents.view_x = geometry.view_x;
1220         utility_extents.view_y = geometry.view_y;
1221     }

1224 void WindowContext::apply_geometry() {
1225     if (!map_received) {
1226         return;
1227     }

1246                                  ? geometry.minw - geometry.adjust_w : 1;

1253                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;


1254     }

1257                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));

1346 bool WindowContext::on_top_inherited() {
1347     WindowContext *o = owner;
1348     while (o) {
1349         WindowContext *topO = dynamic_cast<WindowContext *>(o);
1350         if (!topO) break;
1351         if (topO->on_top) {
1352             return true;
1353         }
1354         o = topO->owner;
1355     }
1356     return false;
