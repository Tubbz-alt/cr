 160 void WindowContextBase::notify_state(jint glass_state) {
 161     if (glass_state == com_sun_glass_events_WindowEvent_RESTORE) {
 162         if (is_maximized) {
 163             glass_state = com_sun_glass_events_WindowEvent_MAXIMIZE;

 170        mainEnv->CallVoidMethod(jwindow,
 171                jGtkWindowNotifyStateChanged,

 174     }

 188 void WindowContextBase::process_state(GdkEventWindowState* event) {
 189     if (event->changed_mask &
 190             (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 191 
 192         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 193             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 194         }
 195         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 196             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 197         }
 198 
 199         jint stateChangeEvent;


 200 
 201         if (is_iconified) {
 202             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 203         } else if (is_maximized) {
 204             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 205         } else {
 206             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 207             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 208                 // in this case - the window manager will not support the programatic
 209                 // request to iconify - so we need to restore it now.
 210                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 211             }
 212         }
 213 
 214         notify_state(stateChangeEvent);
 215     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 216         notify_on_top( event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 217     }
 218 }

 220 void WindowContextBase::process_focus(GdkEventFocus* event) {
 221     if (!event->in) {
 222         ungrab_focus();
 223     }

 225     if (xim.enabled && xim.ic) {

 235             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,

 237             CHECK_JNI_EXCEPTION(mainEnv)
 238         } else {
 239             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 240             CHECK_JNI_EXCEPTION(mainEnv)
 241         }
 242     }

 273 void WindowContextBase::process_destroy() {




























 274     ungrab_focus();
 275 
 276     std::set<WindowContextTop*>::iterator it;
 277     for (it = children.begin(); it != children.end(); ++it) {
 278         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 279         // WindowContextTop::process_destroy() to call remove_child() (because children
 280         // is being iterated here) but also prevents gtk_window_set_transient_for from
 281         // being called - this causes the crash on gnome.
 282         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 283         (*it)->set_owner(NULL);
 284         destroy_and_delete_ctx(*it);
 285     }
 286     children.clear();
 287 
 288     if (jwindow) {
 289         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 290         EXCEPTION_OCCURED(mainEnv);
 291     }
 292 
 293     if (jview) {
 294         mainEnv->DeleteGlobalRef(jview);
 295         jview = NULL;
 296     }
 297 
 298     if (jwindow) {
 299         mainEnv->DeleteGlobalRef(jwindow);
 300         jwindow = NULL;
 301     }
 302 
 303     can_be_deleted = true;
 304 }

 322 static inline jint gtk_button_number_to_mouse_button(guint button) {
 323     switch (button) {
 324         case 1:
 325             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 326         case 2:
 327             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 328         case 3:
 329             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 330         case MOUSE_BACK_BTN:
 331             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 332         case MOUSE_FORWARD_BTN:
 333             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 334         default:
 335             // Other buttons are not supported by quantum and are not reported by other platforms
 336             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 337     }
 338 }

 340 void WindowContextBase::process_mouse_button(GdkEventButton* event) {
 341     bool press = event->type == GDK_BUTTON_PRESS;
 342     guint state = event->state;
 343     guint mask = 0;
 344 
 345     // We need to add/remove current mouse button from the modifier flags
 346     // as X lib state represents the state just prior to the event and
 347     // glass needs the state just after the event
 348     switch (event->button) {
 349         case 1:
 350             mask = GDK_BUTTON1_MASK;
 351             break;
 352         case 2:
 353             mask = GDK_BUTTON2_MASK;
 354             break;
 355         case 3:
 356             mask = GDK_BUTTON3_MASK;
 357             break;
 358         case MOUSE_BACK_BTN:
 359             mask = GDK_BUTTON4_MASK;
 360             break;
 361         case MOUSE_FORWARD_BTN:
 362             mask = GDK_BUTTON5_MASK;
 363             break;
 364     }
 365 
 366     if (press) {
 367         state |= mask;
 368     } else {
 369         state &= ~mask;
 370     }
 371 
 372     jint button = gtk_button_number_to_mouse_button(event->button);
 373 
 374     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 375         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 376                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 377                 button,
 378                 (jint) event->x, (jint) event->y,
 379                 (jint) event->x_root, (jint) event->y_root,
 380                 gdk_modifier_mask_to_glass(state),
 381                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 382                 JNI_FALSE);
 383         CHECK_JNI_EXCEPTION(mainEnv)
 384 
 385         if (jview && event->button == 3 && press) {
 386             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 387                     (jint)event->x, (jint)event->y,
 388                     (jint)event->x_root, (jint)event->y_root,
 389                     JNI_FALSE);
 390             CHECK_JNI_EXCEPTION(mainEnv)
 391         }
 392     }
 393 }

 395 void WindowContextBase::process_mouse_motion(GdkEventMotion* event) {
 396     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 397     jint isDrag = glass_modifier & (
 398             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 399             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 400             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 401             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 402             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 403     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 404 
 405     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 406         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 407     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 408         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 409     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 410         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 411     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 412         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 413     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 414         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 415     }
 416 
 417     if (jview) {
 418         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 419                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 420                 button,
 421                 (jint) event->x, (jint) event->y,
 422                 (jint) event->x_root, (jint) event->y_root,
 423                 glass_modifier,
 424                 JNI_FALSE,
 425                 JNI_FALSE);
 426         CHECK_JNI_EXCEPTION(mainEnv)
 427     }
 428 }

 430 void WindowContextBase::process_mouse_scroll(GdkEventScroll* event) {
 431     jdouble dx = 0;
 432     jdouble dy = 0;
 433 
 434     // converting direction to change in pixels
 435     switch (event->direction) {
 436 #if GTK_CHECK_VERSION(3, 4, 0)
 437         case GDK_SCROLL_SMOOTH:
 438             //FIXME 3.4 ???
 439             break;
 440 #endif
 441         case GDK_SCROLL_UP:
 442             dy = 1;
 443             break;
 444         case GDK_SCROLL_DOWN:
 445             dy = -1;
 446             break;
 447         case GDK_SCROLL_LEFT:
 448             dx = 1;
 449             break;
 450         case GDK_SCROLL_RIGHT:
 451             dx = -1;
 452             break;
 453     }
 454     if (event->state & GDK_SHIFT_MASK) {
 455         jdouble t = dy;
 456         dy = dx;
 457         dx = t;
 458     }
 459     if (jview) {
 460         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 461                 (jint) event->x, (jint) event->y,
 462                 (jint) event->x_root, (jint) event->y_root,
 463                 dx, dy,
 464                 gdk_modifier_mask_to_glass(event->state),
 465                 (jint) 0, (jint) 0,
 466                 (jint) 0, (jint) 0,
 467                 (jdouble) 40.0, (jdouble) 40.0);
 468         CHECK_JNI_EXCEPTION(mainEnv)
 469     }
 470 }

 472 void WindowContextBase::process_mouse_cross(GdkEventCrossing* event) {
 473     bool enter = event->type == GDK_ENTER_NOTIFY;
 474 
 475     if (jview) {
 476         guint state = event->state;

 485                     com_sun_glass_events_MouseEvent_BUTTON_NONE,

 487                     (jint) event->x_root, (jint) event->y_root,
 488                     gdk_modifier_mask_to_glass(state),
 489                     JNI_FALSE,
 490                     JNI_FALSE);

 491             CHECK_JNI_EXCEPTION(mainEnv)
 492         }

 496 void WindowContextBase::process_key(GdkEventKey* event) {
 497     bool press = event->type == GDK_KEY_PRESS;
 498     jint glassKey = get_glass_key(event);
 499     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 500     if (press) {
 501         glassModifier |= glass_key_to_modifier(glassKey);
 502     } else {
 503         glassModifier &= ~glass_key_to_modifier(glassKey);
 504     }
 505     jcharArray jChars = NULL;
 506     jchar key = gdk_keyval_to_unicode(event->keyval);
 507     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 508         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 509     } else {
 510 #ifdef GLASS_GTK2
 511         if (key == 0) {
 512             // Work around "bug" fixed in gtk-3.0:
 513             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 514             switch (event->keyval) {
 515             case 0xFF08 /* Backspace */: key =  '\b';
 516             case 0xFF09 /* Tab       */: key =  '\t';

 519             case 0xFF0D /* Return    */: key =  '\r';
 520             case 0xFF1B /* Escape    */: key =  '\033';
 521             case 0xFFFF /* Delete    */: key =  '\177';
 522             }
 523         }
 524 #endif
 525     }
 526 
 527     if (key > 0) {
 528         jChars = mainEnv->NewCharArray(1);
 529         if (jChars) {
 530             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 531             CHECK_JNI_EXCEPTION(mainEnv)
 532         }
 533     } else {
 534         jChars = mainEnv->NewCharArray(0);
 535     }
 536     if (jview) {
 537         if (press) {
 538             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 539                     com_sun_glass_events_KeyEvent_PRESS,
 540                     glassKey,
 541                     jChars,
 542                     glassModifier);
 543             CHECK_JNI_EXCEPTION(mainEnv)
 544 
 545             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 546                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 547                         com_sun_glass_events_KeyEvent_TYPED,
 548                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 549                         jChars,
 550                         glassModifier);
 551                 CHECK_JNI_EXCEPTION(mainEnv)
 552             }
 553         } else {
 554             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 555                     com_sun_glass_events_KeyEvent_RELEASE,
 556                     glassKey,
 557                     jChars,
 558                     glassModifier);
 559             CHECK_JNI_EXCEPTION(mainEnv)
 560         }
 561     }
 562 }

 564 void WindowContextBase::paint(void* data, jint width, jint height) {
 565 #if GTK_CHECK_VERSION(3, 0, 0)
 566     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 567 #if GTK_CHECK_VERSION(3, 22, 0)
 568     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 569     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 570 #else
 571     gdk_window_begin_paint_region(gdk_window, region);
 572     cairo_t* context = gdk_cairo_create(gdk_window);
 573 #endif
 574 #else
 575     cairo_t* context = gdk_cairo_create(gdk_window);
 576 #endif
 577     if (bg_color.is_set) {
 578         cairo_set_source_rgb(context, bg_color.red, bg_color.green, bg_color.blue);
 579         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 580         cairo_paint(context);
 581     }
 582 
 583     cairo_surface_t* cairo_surface;
 584     cairo_surface = cairo_image_surface_create_for_data(
 585             (unsigned char*)data,
 586             CAIRO_FORMAT_ARGB32,
 587             width, height, width * 4);
 588 
 589     applyShapeMask(data, width, height);
 590     cairo_set_source_surface(context, cairo_surface, 0, 0);
 591     cairo_set_operator(context, CAIRO_OPERATOR_ATOP);
 592     cairo_paint(context);
 593 
 594 #if GTK_CHECK_VERSION(3, 0, 0)
 595 #if GTK_CHECK_VERSION(3, 22, 0)
 596     gdk_window_end_draw_frame(gdk_window, dcontext);
 597     cairo_region_destroy(region);
 598 #else
 599     gdk_window_end_paint(gdk_window);
 600     cairo_region_destroy(region);
 601     cairo_destroy(context);
 602 #endif
 603 #else
 604     cairo_destroy(context);
 605 #endif
 606 
 607     cairo_surface_destroy(cairo_surface);
 608 }

 619 void WindowContextBase::remove_child(WindowContextTop* child) {
 620     children.erase(child);
 621     gtk_window_set_transient_for(child->get_gtk_window(), NULL);

 824 // This function calculate the deltas between window and window + decoration (titleblar, borders).
 825 // Only useful if the window manager does not support the frame extents extension - in this case
 826 // it uses GDK calculation, that is primary based on _NET_FRAME_EXTENTS but will try harder
 827 // if that's not available.
 828 void WindowContextTop::calculate_adjustments() {
 829     if (frame_type != TITLED || geometry.frame_extents_received) {
 830         return;
 831     }
 832 
 833     gint x, y;
 834     gdk_window_get_origin(gdk_window, &x, &y);
 835 
 836     gint rx, ry;
 837     gdk_window_get_root_origin(gdk_window, &rx, &ry);
 838 
 839     if (rx != x || ry != y) {
 840         // the left extends are correct - the right one is guessed to be the same
 841         geometry.adjust_w = (rx - x) * 2;
 842         // guess that bottom size is the same as left and right
 843         geometry.adjust_h = (ry - y) + (rx - x);
 844 
 845         // those will be correct
 846         geometry.view_x = (rx - x);
 847         geometry.view_y = (ry - y);
 848 
 849         save_cached_extents();
 850     } else {
 851         CachedExtents c = (window_type == NORMAL) ? normal_extents : utility_extents;
 852         geometry.adjust_w = c.adjust_w;
 853         geometry.adjust_h = c.adjust_h;
 854         geometry.view_x = c.view_x;
 855         geometry.view_y = c.view_y;
 856     }
 857 
 858     apply_geometry();
 859 }

 868         normal_extents.adjust_h = geometry.adjust_h;
 869         normal_extents.view_x = geometry.view_x;
 870         normal_extents.view_y = geometry.view_y;
 871     } else {
 872         utility_extents.adjust_w = geometry.adjust_w;
 873         utility_extents.adjust_h = geometry.adjust_h;
 874         utility_extents.view_x = geometry.view_x;

 879 void WindowContextTop::apply_geometry() {
 880     if (!map_received) {
 881         return;
 882     }
 883 
 884     GdkGeometry gdk_geometry;
 885     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
 886 
 887     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
 888         // not resizeable
 889         int w = (geometry.current_w - geometry.adjust_w) > 0
 890                     ? geometry.current_w - geometry.adjust_w
 891                     : geometry.current_cw;
 892 
 893         int h = (geometry.current_h - geometry.adjust_h) > 0
 894                     ? geometry.current_h - geometry.adjust_h
 895                     : geometry.current_ch;
 896 
 897         gdk_geometry.min_width = gdk_geometry.max_width = w;
 898         gdk_geometry.min_height = gdk_geometry.max_height = h;
 899     } else {
 900         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
 901                                     ? geometry.minw - geometry.adjust_w : 1;
 902         gdk_geometry.min_height = (geometry.minh -  geometry.adjust_h) > 0
 903                                     ? geometry.minh - geometry.adjust_h : 1;
 904 
 905         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
 906                                     ? geometry.maxw - geometry.adjust_w : G_MAXINT;
 907         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h> 0)
 908                                     ? geometry.maxh - geometry.adjust_h : G_MAXINT;
 909     }
 910 
 911     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
 912         (GdkWindowHints) (GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
 913 }

 993 void WindowContextTop::process_net_wm_property() {
 994     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 995 
 996     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 997     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 998     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 999     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
1000 
1001     gint length;
1002     glong* atoms = NULL;
1003 
1004     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
1005             0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar**) &atoms)) {
1006 
1007         bool is_hidden = false;
1008         bool is_above = false;
1009         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
1010             if (atom_net_wm_state_hidden == (GdkAtom)atoms[i]) {
1011                 is_hidden = true;
1012             } else if (atom_net_wm_state_above == (GdkAtom)atoms[i]) {
1013                 is_above = true;
1014             }
1015         }
1016 
1017         g_free(atoms);
1018 
1019         if (is_iconified != is_hidden) {
1020             is_iconified = is_hidden;
1021 
1022             notify_state((is_hidden)
1023                     ? com_sun_glass_events_WindowEvent_MINIMIZE
1024                     : com_sun_glass_events_WindowEvent_RESTORE);
1025         }
1026 
1027         notify_on_top(is_above);
1028     }
1029 }

 408 void WindowContext::process_destroy() {
 409     if (owner) {
 410         owner->remove_child(this);
 411     }

 413     ungrab_focus();
 414 
 415     std::set<WindowContext *>::iterator it;
 416     for (it = children.begin(); it != children.end(); ++it) {

 461 void WindowContext::process_mouse_button(GdkEventButton *event) {


















 462     bool press = event->type == GDK_BUTTON_PRESS;
 463     guint state = event->state;
 464     guint mask = 0;
 465 
 466     // We need to add/remove current mouse button from the modifier flags
 467     // as X lib state represents the state just prior to the event and
 468     // glass needs the state just after the event
 469     switch (event->button) {
 470         case 1:
 471             mask = GDK_BUTTON1_MASK;
 472             break;
 473         case 2:
 474             mask = GDK_BUTTON2_MASK;
 475             break;
 476         case 3:
 477             mask = GDK_BUTTON3_MASK;
 478             break;
 479         case MOUSE_BACK_BTN:
 480             mask = GDK_BUTTON4_MASK;
 481             break;
 482         case MOUSE_FORWARD_BTN:
 483             mask = GDK_BUTTON5_MASK;
 484             break;
 485     }
 486 
 487     if (press) {
 488         state |= mask;
 489     } else {
 490         state &= ~mask;
 491     }
 492 
 493     jint button = gtk_button_number_to_mouse_button(event->button);
 494 
 495     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 496         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 497                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 498                                 button,
 499                                 (jint) event->x, (jint) event->y,
 500                                 (jint) event->x_root, (jint) event->y_root,
 501                                 gdk_modifier_mask_to_glass(state),
 502                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 503                                 JNI_FALSE);
 504         CHECK_JNI_EXCEPTION(mainEnv)
 505 
 506         if (jview && event->button == 3 && press) {

 618 void WindowContext::process_key(GdkEventKey *event) {
 619     bool press = event->type == GDK_KEY_PRESS;
 620     jint glassKey = get_glass_key(event);
 621     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 622     if (press) {
 623         glassModifier |= glass_key_to_modifier(glassKey);
 624     } else {
 625         glassModifier &= ~glass_key_to_modifier(glassKey);
 626     }
 627     jcharArray jChars = NULL;
 628     jchar key = gdk_keyval_to_unicode(event->keyval);
 629     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 630         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 631     } else {
 632 #ifdef GLASS_GTK2
 633         if (key == 0) {
 634             // Work around "bug" fixed in gtk-3.0:
 635             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 636             switch (event->keyval) {
 637             case 0xFF08 /* Backspace */: key =  '\b';
 638             case 0xFF09 /* Tab       */: key =  '\t';

 641             case 0xFF0D /* Return    */: key =  '\r';
 642             case 0xFF1B /* Escape    */: key =  '\033';
 643             case 0xFFFF /* Delete    */: key =  '\177';
 644             }
 645         }
 646 #endif
 647     }
 648 
 649     if (key > 0) {
 650         jChars = mainEnv->NewCharArray(1);
 651         if (jChars) {
 652             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 653             CHECK_JNI_EXCEPTION(mainEnv)
 654         }
 655     } else {
 656         jChars = mainEnv->NewCharArray(0);
 657     }
 658     if (jview) {
 659         if (press) {
 660             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 661                                     com_sun_glass_events_KeyEvent_PRESS,
 662                                     glassKey,
 663                                     jChars,
 664                                     glassModifier);
 665             CHECK_JNI_EXCEPTION(mainEnv)
 666 
 667             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 668                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 669                                         com_sun_glass_events_KeyEvent_TYPED,
 670                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 671                                         jChars,
 672                                         glassModifier);
 673                 CHECK_JNI_EXCEPTION(mainEnv)
 674             }
 675         } else {
 676             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 677                                     com_sun_glass_events_KeyEvent_RELEASE,
 678                                     glassKey,
 679                                     jChars,
 680                                     glassModifier);
 681             CHECK_JNI_EXCEPTION(mainEnv)
 682         }
 683     }
 684 }

 687     if (event->changed_mask &
 688         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {

 718 void WindowContext::process_net_wm_property() {
 719     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 720 
 721     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 722     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 723     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");

 725 
 726     gint length;
 727     glong *atoms = NULL;


 728 
 729     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 730                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {

 731 
 732         bool is_hidden = false;
 733         bool is_above = false;
 734         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 735             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 736                 is_hidden = true;
 737             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 738                 is_above = true;
 739             }
 740         }
 741 
 742         g_free(atoms);
 743 
 744         if (is_iconified != is_hidden) {
 745             is_iconified = is_hidden;
 746 
 747             notify_state((is_hidden)
 748                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 749                          : com_sun_glass_events_WindowEvent_RESTORE);
 750         }
 751 
 752         notify_on_top(is_above);
 753     }

 923 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 924     calculate_adjustments();
 925 
 926     // newW / newH always content sizes compatible with GTK+
 927     // if window has no decoration, adjustments will be ZERO
 928     int newW = w > 0 ? w - geometry.adjust_w : cw;
 929     int newH = h > 0 ? h - geometry.adjust_h : ch;
 930 
 931     gboolean size_changed = FALSE;
 932     gboolean pos_changed = FALSE;
 933 
 934     if (newW > 0 && newH > 0) {
 935         size_changed = TRUE;
 936 
 937         geometry.current_cw = newW;
 938         geometry.current_ch = newH;
 939         geometry.current_w = newW + geometry.adjust_w;
 940         geometry.current_h = newH + geometry.adjust_h;
 941 
 942         if (visible_received) {
 943             // update constraints if not resized by the user interface so it will
 944             // let gtk_window_resize succeed, because it's bound to geometry constraints
 945             apply_geometry();
 946             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
 947         } else {
 948             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
 949         }
 950     }
 951 
 952     if (xSet || ySet) {
 953         int newX = (xSet) ? x : geometry.current_x;
 954         int newY = (ySet) ? y : geometry.current_y;
 955 
 956         if (newX != geometry.current_x || newY != geometry.current_y) {
 957             pos_changed = TRUE;
 958             geometry.current_x = newX;
 959             geometry.current_y = newY;
 960             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
 961         }
 962     }
 963 
 964     size_position_notify(size_changed, pos_changed);
 965 }

1169 // This function calculate the deltas between window and window + decoration (titleblar, borders).
1170 // Only useful if the window manager does not support the frame extents extension - in this case
1171 // it uses GDK calculation, that is primary based on _NET_FRAME_EXTENTS but will try harder
1172 // if that's not available.
1173 void WindowContext::calculate_adjustments() {
1174     if (frame_type != TITLED || geometry.frame_extents_received) {
1175         return;
1176     }
1177 
1178     gint x, y;
1179     gdk_window_get_origin(gdk_window, &x, &y);
1180 
1181     gint rx, ry;
1182     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1183 
1184     if (rx != x || ry != y) {
1185         // the left extends are correct - the right one is guessed to be the same
1186         geometry.adjust_w = (rx - x) * 2;
1187         // guess that bottom size is the same as left and right
1188         geometry.adjust_h = (ry - y) + (rx - x);
1189 
1190         // those will be correct
1191         geometry.view_x = (rx - x);
1192         geometry.view_y = (ry - y);
1193 
1194         save_cached_extents();
1195     } else {
1196         CachedExtents c = (window_type == NORMAL) ? normal_extents : utility_extents;
1197         geometry.adjust_w = c.adjust_w;
1198         geometry.adjust_h = c.adjust_h;
1199         geometry.view_x = c.view_x;
1200         geometry.view_y = c.view_y;
1201     }
1202 
1203     apply_geometry();

1224 void WindowContext::apply_geometry() {
1225     if (!map_received) {
1226         return;
1227     }
1228 
1229     GdkGeometry gdk_geometry;
1230     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1231 
1232     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1233         // not resizeable

1235                 ? geometry.current_w - geometry.adjust_w
