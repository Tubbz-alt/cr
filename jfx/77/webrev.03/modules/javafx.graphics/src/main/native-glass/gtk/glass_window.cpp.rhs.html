 151     if (jwindow) {

 153         LOG_EXCEPTION(mainEnv)
 154         return result;
 155     } else {
 156         return false;

 160 void WindowContextBase::notify_state(jint glass_state) {
 161     if (glass_state == com_sun_glass_events_WindowEvent_RESTORE) {
 162         if (is_maximized) {
 163             glass_state = com_sun_glass_events_WindowEvent_MAXIMIZE;
 164         }
 165 
 166         notify_repaint();







 167     }
 168 
 169     if (jwindow) {
 170        mainEnv->CallVoidMethod(jwindow,
 171                jGtkWindowNotifyStateChanged,
 172                glass_state);
 173        CHECK_JNI_EXCEPTION(mainEnv);
 174     }
 175 }

 177 void WindowContextBase::notify_repaint() {
 178     int w, h;
 179     glass_gdk_window_get_size(gdk_window, &w, &h);
 180     if (jview) {
 181         mainEnv->CallVoidMethod(jview,
 182                 jViewNotifyRepaint,
 183                 0, 0, w, h);
 184         CHECK_JNI_EXCEPTION(mainEnv);
 185     }
 186 }

 188 void WindowContextBase::process_state(GdkEventWindowState* event) {
 189     if (event->changed_mask &
 190             (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 191 
 192         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 193             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 194         }
 195         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 196             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 197         }
 198 
 199         jint stateChangeEvent;
 200 
 201         if (is_iconified) {
 202             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 203         } else if (is_maximized) {
 204             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 205         } else {
 206             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 207             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 208                 // in this case - the window manager will not support the programatic
 209                 // request to iconify - so we need to restore it now.
 210                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 211             }
 212         }
 213 
 214         notify_state(stateChangeEvent);
 215     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 216         notify_on_top( event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 217     }
 218 }

 220 void WindowContextBase::process_focus(GdkEventFocus* event) {
 221     if (!event->in) {



 222         ungrab_focus();
 223     }
 224 
 225     if (xim.enabled && xim.ic) {
 226         if (event->in) {
 227             XSetICFocus(xim.ic);
 228         } else {
 229             XUnsetICFocus(xim.ic);
 230         }
 231     }
 232 
 233     if (jwindow) {
 234         if (!event->in || isEnabled()) {
 235             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 236                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 237             CHECK_JNI_EXCEPTION(mainEnv)
 238         } else {
 239             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 240             CHECK_JNI_EXCEPTION(mainEnv)
 241         }
 242     }
 243 }

 261 void destroy_and_delete_ctx(WindowContext* ctx) {
 262     if (ctx) {
 263         ctx->process_destroy();
 264 
 265         if (!ctx->get_events_count()) {
 266             delete ctx;
 267         }
 268         // else: ctx will be deleted in EventsCounterHelper after completing
 269         // an event processing
 270     }
 271 }

 273 void WindowContextBase::process_destroy() {
 274     ungrab_focus();






 275 
 276     std::set<WindowContextTop*>::iterator it;
 277     for (it = children.begin(); it != children.end(); ++it) {
 278         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 279         // WindowContextTop::process_destroy() to call remove_child() (because children
 280         // is being iterated here) but also prevents gtk_window_set_transient_for from
 281         // being called - this causes the crash on gnome.
 282         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 283         (*it)->set_owner(NULL);
 284         destroy_and_delete_ctx(*it);
 285     }
 286     children.clear();
 287 
 288     if (jwindow) {
 289         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 290         EXCEPTION_OCCURED(mainEnv);
 291     }
 292 
 293     if (jview) {
 294         mainEnv->DeleteGlobalRef(jview);
 295         jview = NULL;
 296     }
 297 
 298     if (jwindow) {
 299         mainEnv->DeleteGlobalRef(jwindow);
 300         jwindow = NULL;
 301     }
 302 
 303     can_be_deleted = true;
 304 }

 309         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 310         CHECK_JNI_EXCEPTION(mainEnv)
 311     }

 322 static inline jint gtk_button_number_to_mouse_button(guint button) {
 323     switch (button) {
 324         case 1:
 325             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 326         case 2:
 327             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 328         case 3:
 329             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 330         case MOUSE_BACK_BTN:
 331             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 332         case MOUSE_FORWARD_BTN:
 333             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 334         default:
 335             // Other buttons are not supported by quantum and are not reported by other platforms
 336             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 337     }
 338 }

 340 void WindowContextBase::process_mouse_button(GdkEventButton* event) {
 341     bool press = event->type == GDK_BUTTON_PRESS;
 342     guint state = event->state;
 343     guint mask = 0;
 344 
 345     // We need to add/remove current mouse button from the modifier flags
 346     // as X lib state represents the state just prior to the event and
 347     // glass needs the state just after the event
 348     switch (event->button) {
 349         case 1:
 350             mask = GDK_BUTTON1_MASK;
 351             break;
 352         case 2:
 353             mask = GDK_BUTTON2_MASK;
 354             break;
 355         case 3:
 356             mask = GDK_BUTTON3_MASK;
 357             break;
 358         case MOUSE_BACK_BTN:
 359             mask = GDK_BUTTON4_MASK;
 360             break;
 361         case MOUSE_FORWARD_BTN:
 362             mask = GDK_BUTTON5_MASK;
 363             break;
 364     }
 365 
 366     if (press) {
 367         state |= mask;
 368     } else {
 369         state &= ~mask;
 370     }
 371 






























 372     jint button = gtk_button_number_to_mouse_button(event->button);
 373 
 374     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 375         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 376                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 377                 button,
 378                 (jint) event->x, (jint) event->y,
 379                 (jint) event->x_root, (jint) event->y_root,
 380                 gdk_modifier_mask_to_glass(state),
 381                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 382                 JNI_FALSE);
 383         CHECK_JNI_EXCEPTION(mainEnv)
 384 
 385         if (jview && event->button == 3 && press) {
 386             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 387                     (jint)event->x, (jint)event->y,
 388                     (jint)event->x_root, (jint)event->y_root,
 389                     JNI_FALSE);
 390             CHECK_JNI_EXCEPTION(mainEnv)
 391         }
 392     }
 393 }

 395 void WindowContextBase::process_mouse_motion(GdkEventMotion* event) {
 396     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 397     jint isDrag = glass_modifier & (
 398             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 399             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 400             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 401             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 402             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 403     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 404 
 405     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 406         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 407     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 408         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 409     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 410         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 411     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 412         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 413     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 414         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 415     }
 416 
 417     if (jview) {
 418         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 419                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 420                 button,
 421                 (jint) event->x, (jint) event->y,
 422                 (jint) event->x_root, (jint) event->y_root,
 423                 glass_modifier,
 424                 JNI_FALSE,
 425                 JNI_FALSE);
 426         CHECK_JNI_EXCEPTION(mainEnv)
 427     }
 428 }

 430 void WindowContextBase::process_mouse_scroll(GdkEventScroll* event) {
 431     jdouble dx = 0;
 432     jdouble dy = 0;
 433 
 434     // converting direction to change in pixels
 435     switch (event->direction) {
 436 #if GTK_CHECK_VERSION(3, 4, 0)
 437         case GDK_SCROLL_SMOOTH:
 438             //FIXME 3.4 ???
 439             break;
 440 #endif
 441         case GDK_SCROLL_UP:
 442             dy = 1;
 443             break;
 444         case GDK_SCROLL_DOWN:
 445             dy = -1;
 446             break;
 447         case GDK_SCROLL_LEFT:
 448             dx = 1;
 449             break;
 450         case GDK_SCROLL_RIGHT:
 451             dx = -1;
 452             break;
 453     }
 454     if (event->state & GDK_SHIFT_MASK) {
 455         jdouble t = dy;
 456         dy = dx;
 457         dx = t;
 458     }
 459     if (jview) {
 460         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 461                 (jint) event->x, (jint) event->y,
 462                 (jint) event->x_root, (jint) event->y_root,
 463                 dx, dy,
 464                 gdk_modifier_mask_to_glass(event->state),
 465                 (jint) 0, (jint) 0,
 466                 (jint) 0, (jint) 0,
 467                 (jdouble) 40.0, (jdouble) 40.0);
 468         CHECK_JNI_EXCEPTION(mainEnv)
 469     }

 470 }

 472 void WindowContextBase::process_mouse_cross(GdkEventCrossing* event) {
 473     bool enter = event->type == GDK_ENTER_NOTIFY;
 474 
 475     if (jview) {
 476         guint state = event->state;
 477         if (enter) { // workaround for RT-21590
 478             state &= ~MOUSE_BUTTONS_MASK;
 479         }
 480 
 481         if (enter != is_mouse_entered) {
 482             is_mouse_entered = enter;
 483             mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 484                     enter ? com_sun_glass_events_MouseEvent_ENTER : com_sun_glass_events_MouseEvent_EXIT,
 485                     com_sun_glass_events_MouseEvent_BUTTON_NONE,
 486                     (jint) event->x, (jint) event->y,
 487                     (jint) event->x_root, (jint) event->y_root,
 488                     gdk_modifier_mask_to_glass(state),
 489                     JNI_FALSE,
 490                     JNI_FALSE);
 491             CHECK_JNI_EXCEPTION(mainEnv)
 492         }
 493     }
 494 }

 496 void WindowContextBase::process_key(GdkEventKey* event) {
 497     bool press = event->type == GDK_KEY_PRESS;
 498     jint glassKey = get_glass_key(event);
 499     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 500     if (press) {
 501         glassModifier |= glass_key_to_modifier(glassKey);
 502     } else {
 503         glassModifier &= ~glass_key_to_modifier(glassKey);
 504     }
 505     jcharArray jChars = NULL;
 506     jchar key = gdk_keyval_to_unicode(event->keyval);
 507     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 508         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 509     } else {
 510 #ifdef GLASS_GTK2
 511         if (key == 0) {
 512             // Work around "bug" fixed in gtk-3.0:
 513             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 514             switch (event->keyval) {
 515             case 0xFF08 /* Backspace */: key =  '\b';
 516             case 0xFF09 /* Tab       */: key =  '\t';
 517             case 0xFF0A /* Linefeed  */: key =  '\n';
 518             case 0xFF0B /* Vert. Tab */: key =  '\v';
 519             case 0xFF0D /* Return    */: key =  '\r';
 520             case 0xFF1B /* Escape    */: key =  '\033';
 521             case 0xFFFF /* Delete    */: key =  '\177';
 522             }
 523         }
 524 #endif
 525     }
 526 
 527     if (key > 0) {
 528         jChars = mainEnv->NewCharArray(1);
 529         if (jChars) {
 530             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 531             CHECK_JNI_EXCEPTION(mainEnv)
 532         }
 533     } else {
 534         jChars = mainEnv->NewCharArray(0);
 535     }
 536     if (jview) {
 537         if (press) {
 538             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 539                     com_sun_glass_events_KeyEvent_PRESS,
 540                     glassKey,
 541                     jChars,
 542                     glassModifier);
 543             CHECK_JNI_EXCEPTION(mainEnv)
 544 
 545             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 546                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 547                         com_sun_glass_events_KeyEvent_TYPED,
 548                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 549                         jChars,
 550                         glassModifier);
 551                 CHECK_JNI_EXCEPTION(mainEnv)
 552             }
 553         } else {
 554             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 555                     com_sun_glass_events_KeyEvent_RELEASE,
 556                     glassKey,
 557                     jChars,
 558                     glassModifier);
 559             CHECK_JNI_EXCEPTION(mainEnv)
 560         }
 561     }
 562 }

 564 void WindowContextBase::paint(void* data, jint width, jint height) {
 565 #if GTK_CHECK_VERSION(3, 0, 0)




 566     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 567 #if GTK_CHECK_VERSION(3, 22, 0)
 568     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 569     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 570 #else
 571     gdk_window_begin_paint_region(gdk_window, region);
 572     cairo_t* context = gdk_cairo_create(gdk_window);
 573 #endif
 574 #else
 575     cairo_t* context = gdk_cairo_create(gdk_window);
 576 #endif



 577     cairo_surface_t* cairo_surface;
 578     cairo_surface = cairo_image_surface_create_for_data(
 579             (unsigned char*)data,
 580             CAIRO_FORMAT_ARGB32,
 581             width, height, width * 4);
 582 
 583     applyShapeMask(data, width, height);
 584 #ifdef GLASS_GTK3
 585     if (bg_color.is_set) {
 586         cairo_set_source_rgb(context, bg_color.red, bg_color.green, bg_color.blue);
 587     }
 588 #endif
 589     cairo_set_source_surface(context, cairo_surface, 0, 0);
 590     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 591     cairo_paint(context);
 592

 822 // This function calculate the deltas between window and window + decoration (titleblar, borders)
 823 // Only useful if the window manager does not support the frame extents extension
 824 void WindowContextTop::calculate_adjustments() {
 825     if (frame_type == TITLED && !geometry.frame_extents_received) {
 826         GdkRectangle er;
 827         gdk_window_get_frame_extents(gdk_window, &er);
 828 
 829         int w, h;
 830         gtk_window_get_size(GTK_WINDOW(gtk_widget), &w, &h);
 831 
 832         int px, py;
 833         gdk_window_get_origin(gdk_window, &px, &py);
 834 
 835         geometry.adjust_w = (er.width - w > 0) ? er.width - w : 0;
 836         geometry.adjust_h = (er.height - h > 0) ? er.height - h : 0;
 837         geometry.view_x = (px - er.x > 0) ? px - er.x : 0;
 838         geometry.view_y = (py - er.y > 0) ? py - er.y : 0;
 839 
 840         apply_geometry();
 841     }
 842 }

 844 void WindowContextTop::apply_geometry() {
 845     GdkGeometry gdk_geometry;
 846     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;




 847 
 848     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
 849         gdk_geometry.min_width = geometry.current_w;
 850         gdk_geometry.min_height = geometry.current_h;
 851         gdk_geometry.max_width = geometry.current_w;
 852         gdk_geometry.max_height = geometry.current_h;
 853     } else {
 854         gdk_geometry.min_width = (geometry.minw > 0) ? geometry.minw - geometry.adjust_w : 1;
 855         gdk_geometry.min_height = (geometry.minh > 0) ? geometry.minh - geometry.adjust_h : 1;
 856         gdk_geometry.max_width = (geometry.maxw > 0) ? geometry.maxw  - geometry.adjust_w : G_MAXINT;
 857         gdk_geometry.max_height = (geometry.maxh > 0) ? geometry.maxh - geometry.adjust_h : G_MAXINT;
 858     }
 859 
 860     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), gtk_widget, &gdk_geometry,
 861         (GdkWindowHints) (GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
 862 }

 908 bool WindowContextTop::get_frame_extents_property(int *top, int *left,



































 909         int *bottom, int *right) {
 910     unsigned long *extents;
 911 
 912     if (gdk_property_get(gdk_window,
 913             atom_net_wm_frame_extents,
 914             gdk_atom_intern("CARDINAL", FALSE),
 915             0,
 916             sizeof (unsigned long) * 4,
 917             FALSE,
 918             NULL,
 919             NULL,
 920             NULL,
 921             (guchar**) & extents)) {
 922         *left = extents [0];
 923         *right = extents [1];
 924         *top = extents [2];
 925         *bottom = extents [3];

 934 void WindowContextTop::process_net_wm_property() {
 935     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 936 
 937     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 938     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 939     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 940     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 941 
 942     gint length;

 943     glong* atoms = NULL;
 944 
 945     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 946             0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar**) &atoms)) {
 947 
 948         bool is_hidden = false;
 949         bool is_above = false;
 950         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 951             if (atom_net_wm_state_hidden == (GdkAtom)atoms[i]) {
 952                 is_hidden = true;
 953             } else if (atom_net_wm_state_above == (GdkAtom)atoms[i]) {
 954                 is_above = true;
 955             }
 956         }
 957 
 958         g_free(atoms);
 959 
 960         if (is_iconified != is_hidden) {
 961             is_iconified = is_hidden;
 962 
 963             notify_state((is_hidden)
 964                     ? com_sun_glass_events_WindowEvent_MINIMIZE
 965                     : com_sun_glass_events_WindowEvent_RESTORE);
 966         }
 967 
 968         notify_on_top(is_above);
 969     }
 970 }

 972 void WindowContextTop::process_property_notify(GdkEventProperty* event) {
 973     if (event->window == gdk_window) {
 974         if (event->atom == atom_net_wm_state) {
 975             process_net_wm_property();
 976         } else if (event->atom == atom_net_wm_frame_extents) {
 977             if (frame_type != TITLED) {
 978                 return;
 979             }
 980 
 981             int top, left, bottom, right;
 982 
 983             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 984                 if (top + left + bottom + right > 0) {
 985                     geometry.frame_extents_received = true;
 986                     geometry.adjust_w = left + right;
 987                     geometry.adjust_h = top + bottom;
 988                     geometry.view_x = left;
 989                     geometry.view_y = top;
 990 
 991                     // set bounds again to set to correct window size that must
 992                     // be the total width and height accounting extents
 993                     set_bounds(0, 0,
 994                                false, false,
 995                                geometry.current_w, geometry.current_h,
 996                                -1, -1);
 997                 }
 998             }
 999         }
1000     }
1001 }

1070 void WindowContextTop::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
1071     calculate_adjustments();














1072 
1073     // newW / newH always content sizes compatible with GTK+
1074     int newW = w > 0 ? w - geometry.adjust_w : cw;
1075     int newH = h > 0 ? h - geometry.adjust_h : ch;


























1076 
1077     gboolean changed = FALSE;
1078     if (newW > 0 && newH > 0) {
1079         changed = TRUE;

1080 
1081         geometry.current_cw = newW;
1082         geometry.current_ch = newH;
1083         geometry.current_w = newW + geometry.adjust_w;
1084         geometry.current_h = newH + geometry.adjust_h;
1085 
1086         if (map_received) {
1087             apply_geometry(); // update constraints if not resized by the user interface
1088         }
1089 
1090         gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);



1091     }
1092 
1093     if (xSet || ySet) {
1094         int newX = (xSet) ? x : geometry.current_x;
1095         int newY = (ySet) ? y : geometry.current_y;
1096 
1097         if (newX != geometry.current_x || newY != geometry.current_y) {
1098             changed = TRUE;
1099             geometry.current_x = newX;
1100             geometry.current_y = newY;
1101             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1102         }
1103     }
1104 
1105     if (changed) {
1106         size_position_notify();
1107     }
1108 }

1268 bool WindowContextTop::on_top_inherited() {
1269     WindowContext* o = owner;
1270     while (o) {
1271         WindowContextTop* topO = dynamic_cast<WindowContextTop*>(o);
1272         if (!topO) break;
1273         if (topO->on_top) {
1274             return true;
1275         }
1276         o = topO->owner;
1277     }
1278     return false;
1279 }

1281 bool WindowContextTop::effective_on_top() {
1282     if (owner) {
1283         WindowContextTop* topO = dynamic_cast<WindowContextTop*>(owner);
1284         return (topO && topO->effective_on_top()) || on_top;
1285     }
1286     return on_top;
1287 }

1289 void WindowContextTop::notify_on_top(bool top) {
1290     // Do not report effective (i.e. native) values to the FX, only if the user sets it manually
1291     if (top != effective_on_top() && jwindow) {
1292         if (on_top_inherited() && !top) {
1293             // Disallow user's "on top" handling on windows that inherited the property
1294             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
1295         } else {
1296             on_top = top;
1297             update_ontop_tree(top);
1298             mainEnv->CallVoidMethod(jwindow,
1299                     jWindowNotifyLevelChanged,
1300                     top ? com_sun_glass_ui_Window_Level_FLOATING :  com_sun_glass_ui_Window_Level_NORMAL);
1301             CHECK_JNI_EXCEPTION(mainEnv);
1302         }
1303     }
1304 }

1306 void WindowContextTop::set_level(int level) {
1307     if (level == com_sun_glass_ui_Window_Level_NORMAL) {
1308         on_top = false;
1309     } else if (level == com_sun_glass_ui_Window_Level_FLOATING
1310             || level == com_sun_glass_ui_Window_Level_TOPMOST) {
1311         on_top = true;
1312     }
1313     // We need to emulate always on top behaviour on child windows
1314 
1315     if (!on_top_inherited()) {
1316         update_ontop_tree(on_top);
1317     }
1318 }
