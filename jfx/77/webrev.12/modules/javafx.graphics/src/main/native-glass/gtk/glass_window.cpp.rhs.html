 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 171 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 172                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 173         screen(_screen),
 174         frame_type(_frame_type),
 175         window_type(type),
 176         owner(_owner),
 177         jview(NULL),
 178         map_received(false),
 179         visible_received(false),
 180         on_top(false),
 181         is_fullscreen(false),
 182         is_iconified(false),
 183         is_maximized(false),
 184         is_mouse_entered(false),
 185         can_be_deleted(false),
 186         events_processing_cnt(0),
 187         win_group(NULL),
 188         pointer_device(NULL) {
 189 
 190     jwindow = mainEnv->NewGlobalRef(_jwindow);
 191 
 192     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 193 
 194     if (type != POPUP && owner == NULL) {
 195         win_group = gtk_window_group_new();
 196     }
 197 
 198     if (gchar * app_name = get_application_name()) {
 199         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 200         g_free(app_name);
 201     }
 202 
 203     if (owner) {
 204         owner->add_child(this);
 205         if (on_top_inherited()) {
 206             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 207         }
 208     }
 209 
 210     if (type == UTILITY) {
 211         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 212     }
 213 
 214     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 215 
 216     if (xvisualID != 0) {
 217         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 218         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 219     }
 220 
 221     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 222     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 223 
 224     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 225     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 226 
 227     gdk_window = gtk_widget_get_window(gtk_widget);
 228     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 229 
 230     glass_dnd_attach_context(this);
 231 
 232     gdk_windowManagerFunctions = wmf;
 233     if (wmf) {
 234         gdk_window_set_functions(gdk_window, wmf);
 235     }
 236 
 237     if (frame_type != TITLED) {
 238         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 239     }
 240 
 241     connect_signals(gtk_widget, this);
 242 }

 244 void WindowContext::paint(void *data, jint width, jint height) {
 245 #if GTK_CHECK_VERSION(3, 0, 0)
 246     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 247 #if GTK_CHECK_VERSION(3, 22, 0)
 248     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 249     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 250 #else
 251     gdk_window_begin_paint_region(gdk_window, region);
 252     cairo_t* context = gdk_cairo_create(gdk_window);
 253 #endif
 254 #else
 255     cairo_t *context = gdk_cairo_create(gdk_window);
 256 #endif
 257 
 258     if (bg_color.is_set) {
 259         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 260                                 (frame_type == TRANSPARENT) ? 0 : 1);
 261         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 262         cairo_paint(context);
 263     }
 264 
 265     cairo_surface_t *cairo_surface;
 266     cairo_surface = cairo_image_surface_create_for_data(
 267             (unsigned char *) data,
 268             CAIRO_FORMAT_ARGB32,
 269             width, height, width * 4);
 270 
 271     cairo_set_source_surface(context, cairo_surface, 0, 0);
 272 
 273     applyShapeMask(data, width, height);
 274     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 275     cairo_paint(context);
 276 
 277 #if GTK_CHECK_VERSION(3, 0, 0)
 278 #if GTK_CHECK_VERSION(3, 22, 0)
 279     gdk_window_end_draw_frame(gdk_window, dcontext);
 280     cairo_region_destroy(region);
 281 #else
 282     gdk_window_end_paint(gdk_window);
 283     cairo_region_destroy(region);
 284     cairo_destroy(context);
 285 #endif
 286 #else
 287     cairo_destroy(context);
 288 #endif
 289 
 290     cairo_surface_destroy(cairo_surface);
 291 }

 332 void WindowContext::process_focus(GdkEventFocus *event) {
 333     if (!event->in) {
 334         ungrab_focus();
 335         pointer_device_ungrab();
 336     }
 337 
 338     if (xim.enabled && xim.ic) {
 339         if (event->in) {
 340             XSetICFocus(xim.ic);
 341         } else {
 342             XUnsetICFocus(xim.ic);
 343         }
 344     }
 345 
 346     if (jwindow) {
 347         if (!event->in || isEnabled()) {
 348             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 349                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 350                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 351             CHECK_JNI_EXCEPTION(mainEnv)
 352         } else {
 353             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 354             CHECK_JNI_EXCEPTION(mainEnv)
 355         }
 356     }
 357 }

 359 void WindowContext::process_property_notify(GdkEventProperty *event) {
 360     if (event->window == gdk_window) {
 361         if (event->atom == atom_net_wm_state) {
 362             process_net_wm_property();
 363         } else if (event->atom == atom_net_wm_frame_extents) {
 364             if (frame_type != TITLED) {
 365                 return;
 366             }
 367 
 368             int top, left, bottom, right;


 369 
 370             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 371                 if (top + left + bottom + right > 0) {
 372                     geometry.frame_extents_received = true;
 373                     geometry.adjust_w = left + right;
 374                     geometry.adjust_h = top + bottom;
 375                     geometry.view_x = left;
 376                     geometry.view_y = top;
 377 
 378                     save_cached_extents();


 379 
 380                     // set bounds again to set to correct window size that must
 381                     // be the total width and height accounting extents
 382                     set_bounds(0, 0, false, false, -1, -1, -1, -1);
 383                 }
 384             }
 385         }


 386     }
 387 }

 389 void WindowContext::process_configure() {
 390     calculate_adjustments();


 391 
 392     gint x, y, w, h, gtk_w, gtk_h;
 393 
 394     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 395     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 396     w = gtk_w + geometry.adjust_w;
 397     h = gtk_h + geometry.adjust_h;
 398 
 399     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 400     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 401                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 402 
 403     geometry.current_x = x;
 404     geometry.current_y = y;
 405     geometry.current_w = w;
 406     geometry.current_h = h;
 407     geometry.current_cw = gtk_w;
 408     geometry.current_ch = gtk_h;
 409 
 410     size_position_notify(size_changed, pos_changed);
 411 }

 413 void WindowContext::process_destroy() {
 414     if (owner) {
 415         owner->remove_child(this);
 416     }
 417 
 418     ungrab_focus();
 419     pointer_device_ungrab();
 420 
 421     std::set<WindowContext *>::iterator it;
 422     for (it = children.begin(); it != children.end(); ++it) {
 423         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 424         // WindowContext::process_destroy() to call remove_child() (because children
 425         // is being iterated here) but also prevents gtk_window_set_transient_for from
 426         // being called - this causes the crash on gnome.
 427         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 428         (*it)->set_owner(NULL);
 429         destroy_and_delete_ctx(*it);
 430     }
 431     children.clear();
 432 
 433     if (jwindow) {
 434         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 435         EXCEPTION_OCCURED(mainEnv);
 436     }
 437 
 438     if (jview) {
 439         mainEnv->DeleteGlobalRef(jview);
 440         jview = NULL;
 441     }
 442 
 443     if (jwindow) {
 444         mainEnv->DeleteGlobalRef(jwindow);
 445         jwindow = NULL;
 446     }
 447 
 448     can_be_deleted = true;
 449 }

 467 void WindowContext::process_mouse_button(GdkEventButton *event) {


















 468     bool press = event->type == GDK_BUTTON_PRESS;
 469     bool release = event->type == GDK_BUTTON_RELEASE;
 470 
 471     guint state = event->state;
 472     guint mask = 0;
 473 
 474     // We need to add/remove current mouse button from the modifier flags
 475     // as X lib state represents the state just prior to the event and
 476     // glass needs the state just after the event
 477     switch (event->button) {
 478         case 1:
 479             mask = GDK_BUTTON1_MASK;
 480             break;
 481         case 2:
 482             mask = GDK_BUTTON2_MASK;
 483             break;
 484         case 3:
 485             mask = GDK_BUTTON3_MASK;
 486             break;
 487         case MOUSE_BACK_BTN:
 488             mask = GDK_BUTTON4_MASK;
 489             break;
 490         case MOUSE_FORWARD_BTN:
 491             mask = GDK_BUTTON5_MASK;
 492             break;
 493     }
 494 
 495     if (press) {
 496         state |= mask;
 497     } else {
 498         state &= ~mask;
 499     }
 500 











 501     // Upper layers expects from us Windows behavior:
 502     // all mouse events should be delivered to window where drag begins
 503     // and no exit/enter event should be reported during this drag.
 504     // We can grab mouse pointer for these needs.
 505     if (press) {
 506         pointer_device_grab();
 507     } else {
 508         if ((event->state & MOUSE_BUTTONS_MASK)
 509             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 510             pointer_device_ungrab();
 511         } else if (event->button == 8 || event->button == 9) {
 512             // GDK X backend interprets button press events for buttons 4-7 as
 513             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 514             // be set on the event->state from GDK. Thus we cannot check if all
 515             // buttons have been released in the usual way (as above).
 516             pointer_device_ungrab();
 517         }
 518     }
 519 
 520     jint button = gtk_button_number_to_mouse_button(event->button);
 521 
 522     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 523         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 524                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 525                                 button,
 526                                 (jint) event->x, (jint) event->y,
 527                                 (jint) event->x_root, (jint) event->y_root,
 528                                 gdk_modifier_mask_to_glass(state),
 529                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 530                                 JNI_FALSE);
 531         CHECK_JNI_EXCEPTION(mainEnv)
 532 
 533         if (jview && event->button == 3 && press) {
 534             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 535                                     (jint) event->x, (jint) event->y,
 536                                     (jint) event->x_root, (jint) event->y_root,
 537                                     JNI_FALSE);
 538             CHECK_JNI_EXCEPTION(mainEnv)
 539         }
 540     }
 541 }

 543 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 544     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 545     jint isDrag = glass_modifier & (
 546             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 547             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 548             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 549             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 550             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 551     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 552 
 553     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 554         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 555     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 556         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 557     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 558         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 559     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 560         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 561     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 562         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 563     }
 564 
 565     if (jview) {
 566         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 567                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 568                                 button,
 569                                 (jint) event->x, (jint) event->y,
 570                                 (jint) event->x_root, (jint) event->y_root,
 571                                 glass_modifier,
 572                                 JNI_FALSE,
 573                                 JNI_FALSE);
 574         CHECK_JNI_EXCEPTION(mainEnv)
 575     }
 576 }

 578 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 579     jdouble dx = 0;
 580     jdouble dy = 0;
 581 
 582     // converting direction to change in pixels
 583     switch (event->direction) {
 584 #if GTK_CHECK_VERSION(3, 4, 0)
 585         case GDK_SCROLL_SMOOTH:
 586             //FIXME 3.4 ???
 587             break;
 588 #endif
 589         case GDK_SCROLL_UP:
 590             dy = 1;
 591             break;
 592         case GDK_SCROLL_DOWN:
 593             dy = -1;
 594             break;
 595         case GDK_SCROLL_LEFT:
 596             dx = 1;
 597             break;
 598         case GDK_SCROLL_RIGHT:
 599             dx = -1;
 600             break;
 601     }
 602     if (event->state & GDK_SHIFT_MASK) {
 603         jdouble t = dy;

 645 void WindowContext::process_key(GdkEventKey *event) {
 646     bool press = event->type == GDK_KEY_PRESS;
 647     jint glassKey = get_glass_key(event);
 648     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 649     if (press) {
 650         glassModifier |= glass_key_to_modifier(glassKey);
 651     } else {
 652         glassModifier &= ~glass_key_to_modifier(glassKey);
 653     }
 654     jcharArray jChars = NULL;
 655     jchar key = gdk_keyval_to_unicode(event->keyval);
 656     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 657         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 658     } else {
 659 #ifdef GLASS_GTK2
 660         if (key == 0) {
 661             // Work around "bug" fixed in gtk-3.0:
 662             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 663             switch (event->keyval) {
 664             case 0xFF08 /* Backspace */: key =  '\b';
 665             case 0xFF09 /* Tab       */: key =  '\t';
 666             case 0xFF0A /* Linefeed  */: key =  '\n';
 667             case 0xFF0B /* Vert. Tab */: key =  '\v';
 668             case 0xFF0D /* Return    */: key =  '\r';
 669             case 0xFF1B /* Escape    */: key =  '\033';
 670             case 0xFFFF /* Delete    */: key =  '\177';
 671             }
 672         }
 673 #endif
 674     }
 675 
 676     if (key > 0) {
 677         jChars = mainEnv->NewCharArray(1);
 678         if (jChars) {
 679             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 680             CHECK_JNI_EXCEPTION(mainEnv)
 681         }
 682     } else {
 683         jChars = mainEnv->NewCharArray(0);
 684     }
 685     if (jview) {
 686         if (press) {
 687             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 688                                     com_sun_glass_events_KeyEvent_PRESS,
 689                                     glassKey,
 690                                     jChars,
 691                                     glassModifier);
 692             CHECK_JNI_EXCEPTION(mainEnv)
 693 
 694             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 695                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 696                                         com_sun_glass_events_KeyEvent_TYPED,
 697                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 698                                         jChars,
 699                                         glassModifier);
 700                 CHECK_JNI_EXCEPTION(mainEnv)
 701             }
 702         } else {
 703             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 704                                     com_sun_glass_events_KeyEvent_RELEASE,
 705                                     glassKey,
 706                                     jChars,
 707                                     glassModifier);
 708             CHECK_JNI_EXCEPTION(mainEnv)
 709         }
 710     }
 711 }

 745 void WindowContext::process_net_wm_property() {
 746     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 747 
 748     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 749     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 750     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 751     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 752 
 753     gint length;
 754     glong *atoms = NULL;

 755 
 756     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 757                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 758 
 759         bool is_hidden = false;
 760         bool is_above = false;
 761         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 762             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 763                 is_hidden = true;
 764             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 765                 is_above = true;
 766             }
 767         }
 768 
 769         g_free(atoms);
 770 
 771         if (is_iconified != is_hidden) {
 772             is_iconified = is_hidden;
 773 
 774             notify_state((is_hidden)
 775                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 776                          : com_sun_glass_events_WindowEvent_RESTORE);
 777         }
 778 
 779         notify_on_top(is_above);
 780     }

 955 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 956     calculate_adjustments();
 957 
 958     // newW / newH always content sizes compatible with GTK+
 959     // if window has no decoration, adjustments will be ZERO
 960     int newW = w > 0 ? w - geometry.adjust_w : cw;
 961     int newH = h > 0 ? h - geometry.adjust_h : ch;
 962 
 963     gboolean size_changed = FALSE;
 964     gboolean pos_changed = FALSE;
 965 
 966     if (newW > 0 && newH > 0) {
 967         size_changed = TRUE;
 968 
 969         geometry.current_cw = newW;
 970         geometry.current_ch = newH;
 971         geometry.current_w = newW + geometry.adjust_w;
 972         geometry.current_h = newH + geometry.adjust_h;
 973 
 974         if (visible_received) {
 975             // update constraints if not resized by the user interface so it will
 976             // let gtk_window_resize succeed, because it's bound to geometry constraints
 977             apply_geometry();
 978             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
 979         } else {
 980             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
 981         }
 982     }

 983 
 984     if (xSet || ySet) {
 985         int newX = (xSet) ? x : geometry.current_x;
 986         int newY = (ySet) ? y : geometry.current_y;
 987 
 988         if (newX != geometry.current_x || newY != geometry.current_y) {
 989             pos_changed = TRUE;
 990             geometry.current_x = newX;
 991             geometry.current_y = newY;
 992             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
 993         }
 994     }

 995 
 996     size_position_notify(size_changed, pos_changed);

 997 }

1201 // This function calculate the deltas between window and window + decoration (titleblar, borders).
1202 // Only useful if the window manager does not support the frame extents extension - in this case
1203 // it uses GDK calculation, that is primary based on _NET_FRAME_EXTENTS but will try harder
1204 // if that's not available.
1205 void WindowContext::calculate_adjustments() {
1206     if (frame_type != TITLED || geometry.frame_extents_received) {
1207         return;

1208     }
1209 
1210     gint x, y;
1211     gdk_window_get_origin(gdk_window, &x, &y);
1212 
1213     gint rx, ry;
1214     gdk_window_get_root_origin(gdk_window, &rx, &ry);

1215 
1216     if (rx != x || ry != y) {
1217         // the left extends are correct - the right one is guessed to be the same
1218         geometry.adjust_w = (rx - x) * 2;
1219         // guess that bottom size is the same as left and right
1220         geometry.adjust_h = (ry - y) + (rx - x);


1221 
1222         // those will be correct
1223         geometry.view_x = (rx - x);
1224         geometry.view_y = (ry - y);
1225 
1226         save_cached_extents();
1227     } else {
1228         CachedExtents c = (window_type == NORMAL) ? normal_extents : utility_extents;
1229         geometry.adjust_w = c.adjust_w;
1230         geometry.adjust_h = c.adjust_h;
1231         geometry.view_x = c.view_x;
1232         geometry.view_y = c.view_y;
1233     }
1234 
1235     apply_geometry();
1236 }

1256 void WindowContext::apply_geometry() {
1257     if (!map_received) {
1258         return;








1259     }
1260 
1261     GdkGeometry gdk_geometry;
1262     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;


1263 
1264     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1265         // not resizeable
1266         int w = (geometry.current_w - geometry.adjust_w) > 0
1267                 ? geometry.current_w - geometry.adjust_w
1268                 : geometry.current_cw;

1269 
1270         int h = (geometry.current_h - geometry.adjust_h) > 0
1271                 ? geometry.current_h - geometry.adjust_h
1272                 : geometry.current_ch;













1273 
1274         gdk_geometry.min_width = gdk_geometry.max_width = w;
1275         gdk_geometry.min_height = gdk_geometry.max_height = h;
1276     } else {
1277         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1278                                  ? geometry.minw - geometry.adjust_w : 1;
1279         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1280                                   ? geometry.minh - geometry.adjust_h : 1;
1281 
1282         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1283                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1284         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1285                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1286     }
1287 
1288     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1289                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1290 }

1378 bool WindowContext::on_top_inherited() {
1379     WindowContext *o = owner;
1380     while (o) {
1381         WindowContext *topO = dynamic_cast<WindowContext *>(o);
1382         if (!topO) break;
1383         if (topO->on_top) {
1384             return true;
1385         }
1386         o = topO->owner;
1387     }
1388     return false;
