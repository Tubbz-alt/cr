 148     /**
 149      * Creates a new TableColumnHeader instance to visually represent the given
 150      * {@link TableColumnBase} instance.
 151      *
 152      * @param tc The table column to be visually represented by this instance.
 153      */
 154     public TableColumnHeader(final TableColumnBase tc) {
 155         setTableColumn(tc);
 156         setFocusTraversable(false);
 157 
 158         initStyleClasses();
 159         initUI();
 160 
 161         // change listener for multiple properties
 162         changeListenerHandler = new LambdaMultiplePropertyChangeListenerHandler();
 163         changeListenerHandler.registerChangeListener(sceneProperty(), e -> updateScene());
 164 
 165         if (getTableColumn() != null) {
 166             changeListenerHandler.registerChangeListener(tc.idProperty(), e -> setId(tc.getId()));
 167             changeListenerHandler.registerChangeListener(tc.styleProperty(), e -> setStyle(tc.getStyle()));
 168             changeListenerHandler.registerChangeListener(tc.widthProperty(), e -> {
 169                 // It is this that ensures that when a column is resized that the header
 170                 // visually adjusts its width as necessary.
 171                 isSizeDirty = true;
 172                 requestLayout();
 173             });
 174             changeListenerHandler.registerChangeListener(tc.visibleProperty(), e -> setVisible(getTableColumn().isVisible()));
 175             changeListenerHandler.registerChangeListener(tc.sortNodeProperty(), e -> updateSortGrid());
 176             changeListenerHandler.registerChangeListener(tc.sortableProperty(), e -> {
 177                 // we need to notify all headers that a sortable state has changed,
 178                 // in case the sort grid in other columns needs to be updated.
 179                 if (TableSkinUtils.getSortOrder(getTableSkin()).contains(getTableColumn())) {
 180                     NestedTableColumnHeader root = getTableHeaderRow().getRootHeader();
 181                     updateAllHeaders(root);
 182                 }
 183             });
 184             changeListenerHandler.registerChangeListener(tc.textProperty(), e -> label.setText(tc.getText()));
 185             changeListenerHandler.registerChangeListener(tc.graphicProperty(), e -> label.setGraphic(tc.getGraphic()));
 186 
 187             setId(tc.getId());
 188             setStyle(tc.getStyle());
 189             /* Having TableColumn role parented by TableColumn causes VoiceOver to be unhappy */
 190             setAccessibleRole(AccessibleRole.TABLE_COLUMN);
 191         }
 192     }

 231     private static final EventHandler<MouseEvent> mousePressedHandler = me -> {
 232         TableColumnHeader header = (TableColumnHeader) me.getSource();
 233         TableColumnBase tableColumn = header.getTableColumn();
 234 
 235         ContextMenu menu = tableColumn.getContextMenu();
 236         if (menu != null && menu.isShowing()) {
 237             menu.hide();
 238         }
 239 
 240         if (me.isConsumed()) return;
 241         me.consume();
 242 
 243         header.getTableHeaderRow().columnDragLock = true;
 244 
 245         // pass focus to the table, so that the user immediately sees
 246         // the focus rectangle around the table control.
 247         header.getTableSkin().getSkinnable().requestFocus();
 248 
 249         if (me.isPrimaryButtonDown() && header.isColumnReorderingEnabled()) {
 250             header.columnReorderingStarted(me.getX());
 251         }
 252     };
 253 
 254     private static final EventHandler<MouseEvent> mouseDraggedHandler = me -> {
 255         if (me.isConsumed()) return;
 256         me.consume();
 257 
 258         TableColumnHeader header = (TableColumnHeader) me.getSource();
 259 
 260         if (me.isPrimaryButtonDown() && header.isColumnReorderingEnabled()) {
 261             header.columnReordering(me.getSceneX(), me.getSceneY());
 262         }
 263     };
 264 
 265     private static final EventHandler<MouseEvent> mouseReleasedHandler = me -> {
 266         if (me.isPopupTrigger()) return;
 267         if (me.isConsumed()) return;
 268         me.consume();
 269 
 270         TableColumnHeader header = (TableColumnHeader) me.getSource();
 271         header.getTableHeaderRow().columnDragLock = false;
 272 
 273         if (header.getTableHeaderRow().isReordering() && header.isColumnReorderingEnabled()) {
 274             header.columnReorderingComplete();
 275         } else if (me.isStillSincePress()) {
 276             header.sortColumn(me.isShiftDown());
 277         }
 278     };
 279 
 280     private static final EventHandler<ContextMenuEvent> contextMenuRequestedHandler = me -> {
 281         TableColumnHeader header = (TableColumnHeader) me.getSource();
 282         TableColumnBase tableColumn = header.getTableColumn();
 283 
 284         ContextMenu menu = tableColumn.getContextMenu();
 285         if (menu != null) {
 286             menu.show(header, me.getScreenX(), me.getScreenY());
 287             me.consume();
 288         }
 289     };

 362     /** {@inheritDoc} */
 363     @Override protected void layoutChildren() {
 364         if (isSizeDirty) {
 365             resize(getTableColumn().getWidth(), getHeight());
 366             isSizeDirty = false;
 367         }
 368 
 369         double sortWidth = 0;
 370         double w = snapSizeX(getWidth()) - (snappedLeftInset() + snappedRightInset());
 371         double h = getHeight() - (snappedTopInset() + snappedBottomInset());
 372         double x = w;
 373 
 374         // a bit hacky, but we REALLY don't want the arrow shape to fluctuate
 375         // in size
 376         if (arrow != null) {
 377             arrow.setMaxSize(arrow.prefWidth(-1), arrow.prefHeight(-1));
 378         }
 379 
 380         if (sortArrow != null && sortArrow.isVisible()) {
 381             sortWidth = sortArrow.prefWidth(-1);
 382             x -= sortWidth;
 383             sortArrow.resize(sortWidth, sortArrow.prefHeight(-1));
 384             positionInArea(sortArrow, x, snappedTopInset(),
 385                     sortWidth, h, 0, HPos.CENTER, VPos.CENTER);
 386         }
 387 
 388         if (label != null) {
 389             double labelWidth = w - sortWidth;
 390             label.resizeRelocate(snappedLeftInset(), 0, labelWidth, getHeight());
 391         }
 392     }

 394     /** {@inheritDoc} */
 395     @Override protected double computePrefWidth(double height) {
 396         if (getNestedColumnHeader() != null) {
 397             double width = getNestedColumnHeader().prefWidth(height);
 398 
 399             if (getTableColumn() != null) {
 400                 TableColumnBaseHelper.setWidth(getTableColumn(), width);
 401             }
 402 
 403             return width;
 404         } else if (getTableColumn() != null && getTableColumn().isVisible()) {
 405             return snapSizeX(getTableColumn().getWidth());
 406         }
 407 
 408         return 0;
 409     }

 477     private void updateTableSkin() {
 478         // when we get the table header row, we are also given the skin,
 479         // so this is the time to hook up listeners, etc.
 480         TableViewSkinBase<?,?,?,?,?> tableSkin = getTableSkin();
 481         if (tableSkin == null) return;
 482 
 483         updateColumnIndex();
 484         this.columnReorderLine = tableSkin.getColumnReorderLine();
 485 
 486         if (getTableColumn() != null) {
 487             updateSortPosition();
 488             TableSkinUtils.getSortOrder(tableSkin).addListener(weakSortOrderListener);
 489             TableSkinUtils.getVisibleLeafColumns(tableSkin).addListener(weakVisibleLeafColumnsListener);
 490         }
 491     }

 507     // RT-29682: When the sortable property of a TableColumnBase changes this
 508     // may impact other TableColumnHeaders, as they may need to change their
 509     // sort order representation. Rather than install listeners across all
 510     // TableColumn in the sortOrder list for their sortable property, we simply
 511     // update the sortPosition of all headers whenever the sortOrder property
 512     // changes, assuming the column is within the sortOrder list.
 513     private void updateAllHeaders(TableColumnHeader header) {
 514         if (header instanceof NestedTableColumnHeader) {
 515             List<TableColumnHeader> children = ((NestedTableColumnHeader)header).getColumnHeaders();
 516             for (int i = 0; i < children.size(); i++) {
 517                 updateAllHeaders(children.get(i));
 518             }
 519         } else {
 520             header.updateSortPosition();
 521         }
 522     }

 524     private void updateScene() {
 525         // RT-17684: If the TableColumn widths are all currently the default,
 526         // we attempt to 'auto-size' based on the preferred width of the first
 527         // n rows (we can't do all rows, as that could conceivably be an unlimited
 528         // number of rows retrieved from a very slow (e.g. remote) data source.
 529         // Obviously, the bigger the value of n, the more likely the default
 530         // width will be suitable for most values in the column
 531         final int n = 30;
 532         if (! autoSizeComplete) {
 533             if (getTableColumn() == null || getTableColumn().getWidth() != DEFAULT_COLUMN_WIDTH || getScene() == null) {
 534                 return;
 535             }
 536             doColumnAutoSize(n);
 537             autoSizeComplete = true;
 538         }
 539     }

 542         TableViewSkinBase tableSkin = getTableSkin();
 543         if (tableSkin != null) {
 544             TableSkinUtils.getVisibleLeafColumns(tableSkin).removeListener(weakVisibleLeafColumnsListener);
 545             TableSkinUtils.getSortOrder(tableSkin).removeListener(weakSortOrderListener);
 546         }

 548         changeListenerHandler.dispose();
 549     }

 563     private void initUI() {
 564         // TableColumn will be null if we are dealing with the root NestedTableColumnHeader
 565         if (getTableColumn() == null) return;
 566 
 567         // set up mouse events
 568         setOnMousePressed(mousePressedHandler);
 569         setOnMouseDragged(mouseDraggedHandler);
 570         setOnDragDetected(event -> event.consume());
 571         setOnContextMenuRequested(contextMenuRequestedHandler);
 572         setOnMouseReleased(mouseReleasedHandler);
 573 
 574         // --- label
 575         label = new Label();
 576         label.setText(getTableColumn().getText());
 577         label.setGraphic(getTableColumn().getGraphic());
 578         label.setVisible(getTableColumn().isVisible());
 579 
 580         // ---- container for the sort arrow (which is not supported on embedded
 581         // platforms)
 582         if (isSortingEnabled()) {
 583             // put together the grid
 584             updateSortGrid();
 585         }
 586     }

 588     private void doColumnAutoSize(int cellsToMeasure) {
 589         double prefWidth = getTableColumn().getPrefWidth();
 590 
 591         // if the prefWidth has been set, we do _not_ autosize columns
 592         if (prefWidth == DEFAULT_COLUMN_WIDTH) {
 593             resizeColumnToFitContent(cellsToMeasure);
 594         }
 595     }

 597     /**
 598      * Resizes this {@code TableColumnHeader}'s column to fit the width of its content.
 599      *
 600      * @implSpec The resulting column width for this implementation is the maximum of the preferred width of the header
 601      * cell and the preferred width of the first {@code maxRow} cells.
 602      * <p>
 603      * Subclasses can either use this method or override it (without the need to call {@code super()}) to provide their
 604      * custom implementation (such as ones that exclude the header, exclude {@code null} content, compute the minimum
 605      * width etc.).
 606      *
 607      * @param maxRows the number of rows considered when resizing. If -1 is given, all rows are considered.
 608      * @since 14
 609      */
 610     protected void resizeColumnToFitContent(int maxRows) {
 611         TableColumnBase<?, ?> tc = getTableColumn();
 612         if (!tc.isResizable()) return;
 613 
 614         Object control = this.getTableSkin().getSkinnable();
 615         if (control instanceof TableView) {
 616             resizeColumnToFitContent((TableView) control, (TableColumn) tc, this.getTableSkin(), maxRows);
 617         } else if (control instanceof TreeTableView) {
 618             resizeColumnToFitContent((TreeTableView) control, (TreeTableColumn) tc, this.getTableSkin(), maxRows);
 619         }
 620     }

 622     private <T,S> void resizeColumnToFitContent(TableView<T> tv, TableColumn<T, S> tc, TableViewSkinBase tableSkin, int maxRows) {
 623         List<?> items = tv.getItems();
 624         if (items == null || items.isEmpty()) return;
 625 
 626         Callback/*<TableColumn<T, ?>, TableCell<T,?>>*/ cellFactory = tc.getCellFactory();
 627         if (cellFactory == null) return;
 628 
 629         TableCell<T,?> cell = (TableCell<T, ?>) cellFactory.call(tc);
 630         if (cell == null) return;
 631 
 632         // set this property to tell the TableCell we want to know its actual
 633         // preferred width, not the width of the associated TableColumnBase
 634         cell.getProperties().put(Properties.DEFER_TO_PARENT_PREF_WIDTH, Boolean.TRUE);
 635 
 636         // determine cell padding
 637         double padding = 10;
 638         Node n = cell.getSkin() == null ? null : cell.getSkin().getNode();
 639         if (n instanceof Region) {
 640             Region r = (Region) n;
 641             padding = r.snappedLeftInset() + r.snappedRightInset();
 642         }
 643 
 644         int rows = maxRows == -1 ? items.size() : Math.min(items.size(), maxRows);
 645         double maxWidth = 0;
 646         for (int row = 0; row < rows; row++) {
 647             cell.updateTableColumn(tc);
 648             cell.updateTableView(tv);
 649             cell.updateIndex(row);
 650 
 651             if ((cell.getText() != null && !cell.getText().isEmpty()) || cell.getGraphic() != null) {
 652                 tableSkin.getChildren().add(cell);
 653                 cell.applyCss();
 654                 maxWidth = Math.max(maxWidth, cell.prefWidth(-1));
 655                 tableSkin.getChildren().remove(cell);
 656             }
 657         }
 658 
 659         // dispose of the cell to prevent it retaining listeners (see RT-31015)
 660         cell.updateIndex(-1);
 661 
 662         // RT-36855 - take into account the column header text / graphic widths.
 663         // Magic 10 is to allow for sort arrow to appear without text truncation.
 664         TableColumnHeader header = tableSkin.getTableHeaderRow().getColumnHeaderFor(tc);
 665         double headerTextWidth = Utils.computeTextWidth(header.label.getFont(), tc.getText(), -1);
 666         Node graphic = header.label.getGraphic();
 667         double headerGraphicWidth = graphic == null ? 0 : graphic.prefWidth(-1) + header.label.getGraphicTextGap();
 668         double headerWidth = headerTextWidth + headerGraphicWidth + 10 + header.snappedLeftInset() + header.snappedRightInset();
 669         maxWidth = Math.max(maxWidth, headerWidth);

 693     private <T,S> void resizeColumnToFitContent(TreeTableView<T> ttv, TreeTableColumn<T, S> tc, TableViewSkinBase tableSkin, int maxRows) {
 694         List<?> items = new TreeTableViewBackingList(ttv);
 695         if (items == null || items.isEmpty()) return;
 696 
 697         Callback cellFactory = tc.getCellFactory();
 698         if (cellFactory == null) return;
 699 
 700         TreeTableCell<T,S> cell = (TreeTableCell) cellFactory.call(tc);
 701         if (cell == null) return;

 703         // set this property to tell the TableCell we want to know its actual
 704         // preferred width, not the width of the associated TableColumnBase
 705         cell.getProperties().put(Properties.DEFER_TO_PARENT_PREF_WIDTH, Boolean.TRUE);

 719         double maxWidth = 0;
 720         for (int row = 0; row < rows; row++) {
 721             treeTableRow.updateIndex(row);
 722             treeTableRow.updateTreeItem(ttv.getTreeItem(row));

 726             cell.updateTreeTableRow(treeTableRow);

 911     void moveColumn(TableColumnBase column, final int newColumnPos) {
 912         if (column == null || newColumnPos < 0) return;
 913 
 914         ObservableList<TableColumnBase<?,?>> columns = getColumns(column);
 915 
 916         final int columnsCount = columns.size();
 917         final int currentPos = columns.indexOf(column);
 918 
 919         int actualNewColumnPos = newColumnPos;
 920 
 921         // Fix for RT-35141: We need to account for hidden columns.
 922         // We keep iterating until we see 'requiredVisibleColumns' number of visible columns
 923         final int requiredVisibleColumns = actualNewColumnPos;
 924         int visibleColumnsSeen = 0;
 925         for (int i = 0; i < columnsCount; i++) {
 926             if (visibleColumnsSeen == (requiredVisibleColumns + 1)) {
 927                 break;
 928             }
 929 
 930             if (columns.get(i).isVisible()) {
 931                 visibleColumnsSeen++;
 932             } else {
 933                 actualNewColumnPos++;
 934             }
 935         }
 936         // --- end of RT-35141 fix
 937 
 938         if (actualNewColumnPos >= columnsCount) {
 939             actualNewColumnPos = columnsCount - 1;
 940         } else if (actualNewColumnPos < 0) {
 941             actualNewColumnPos = 0;
 942         }
 943 
 944         if (actualNewColumnPos == currentPos) return;
 945 
 946         List<TableColumnBase<?,?>> tempList = new ArrayList<>(columns);
 947         tempList.remove(column);
 948         tempList.add(actualNewColumnPos, column);
 949 
 950         columns.setAll(tempList);
 951     }

 990     private void sortColumn(final boolean addColumn) {
 991         if (! isSortingEnabled()) return;
 992 
 993         // we only allow sorting on the leaf columns and columns
 994         // that actually have comparators defined, and are sortable
 995         if (getTableColumn() == null || getTableColumn().getColumns().size() != 0 || getTableColumn().getComparator() == null || !getTableColumn().isSortable()) return;
 996 //        final int sortPos = getTable().getSortOrder().indexOf(column);
 997 //        final boolean isSortColumn = sortPos != -1;
 998 
 999         final ObservableList<TableColumnBase<?,?>> sortOrder = TableSkinUtils.getSortOrder(getTableSkin());
1000 
1001         // addColumn is true e.g. when the user is holding down Shift
1002         if (addColumn) {
1003             if (!isSortColumn) {
1004                 setSortType(getTableColumn(), TableColumn.SortType.ASCENDING);
1005                 sortOrder.add(getTableColumn());
1006             } else if (isAscending(getTableColumn())) {
1007                 setSortType(getTableColumn(), TableColumn.SortType.DESCENDING);
1008             } else {
1009                 int i = sortOrder.indexOf(getTableColumn());
1010                 if (i != -1) {
1011                     sortOrder.remove(i);
1012                 }
1013             }
1014         } else {
1015             // the user has clicked on a column header - we should add this to
1016             // the TableView sortOrder list if it isn't already there.
1017             if (isSortColumn && sortOrder.size() == 1) {
1018                 // the column is already being sorted, and it's the only column.
1019                 // We therefore move through the 2nd or 3rd states:
1020                 //   1st click: sort ascending
1021                 //   2nd click: sort descending
1022                 //   3rd click: natural sorting (sorting is switched off)
1023                 if (isAscending(getTableColumn())) {
1024                     setSortType(getTableColumn(), TableColumn.SortType.DESCENDING);
1025                 } else {
1026                     // remove from sort
1027                     sortOrder.remove(getTableColumn());
1028                 }
1029             } else if (isSortColumn) {
1030                 // the column is already being used to sort, so we toggle its
1031                 // sortAscending property, and also make the column become the
1032                 // primary sort column
1033                 if (isAscending(getTableColumn())) {
1034                     setSortType(getTableColumn(), TableColumn.SortType.DESCENDING);
1035                 } else if (isDescending(getTableColumn())) {
1036                     setSortType(getTableColumn(), TableColumn.SortType.ASCENDING);
1037                 }
1038 
1039                 // to prevent multiple sorts, we make a copy of the sort order
1040                 // list, moving the column value from the current position to
1041                 // its new position at the front of the list
1042                 List<TableColumnBase<?,?>> sortOrderCopy = new ArrayList<TableColumnBase<?,?>>(sortOrder);
1043                 sortOrderCopy.remove(getTableColumn());
1044                 sortOrderCopy.add(0, getTableColumn());
1045                 sortOrder.setAll(getTableColumn());
1046             } else {
1047                 // add to the sort order, in ascending form
1048                 setSortType(getTableColumn(), TableColumn.SortType.ASCENDING);
1049                 sortOrder.setAll(getTableColumn());
1050             }
1051         }
1052     }

1054     // Because it is possible that some columns are in the sortOrder list but are
1055     // not themselves sortable, we cannot just do sortOrderList.indexOf(column).
1056     // Therefore, this method does the proper work required of iterating through
1057     // and ignoring non-sortable (and null) columns in the sortOrder list.
1058     private int getSortPosition() {
1059         if (getTableColumn() == null) {
1060             return -1;
1061         }
1062 
1063         final List<TableColumnBase> sortOrder = getVisibleSortOrderColumns();
1064         int pos = 0;
1065         for (int i = 0; i < sortOrder.size(); i++) {
1066             TableColumnBase _tc = sortOrder.get(i);
1067 
1068             if (getTableColumn().equals(_tc)) {
1069                 return pos;
1070             }
1071 
1072             pos++;
1073         }
1074 
1075         return -1;
1076     }

1078     private List<TableColumnBase> getVisibleSortOrderColumns() {
1079         final ObservableList<TableColumnBase<?,?>> sortOrder = TableSkinUtils.getSortOrder(getTableSkin());
1080 
1081         List<TableColumnBase> visibleSortOrderColumns = new ArrayList<>();
1082         for (int i = 0; i < sortOrder.size(); i++) {
1083             TableColumnBase _tc = sortOrder.get(i);
1084             if (_tc == null || ! _tc.isSortable() || ! _tc.isVisible()) {
1085                 continue;
1086             }
1087 
1088             visibleSortOrderColumns.add(_tc);
1089         }
1090 
1091         return visibleSortOrderColumns;
1092     }

1122     // package for testing
1123     void columnReordering(double sceneX, double sceneY) {
1124         if (! getTableColumn().isReorderable()) return;
1125 
1126         // this is for handling the column drag to reorder columns.
1127         // It shows a line to indicate where the 'drop' will be.
1128 
1129         // indicate that we've started dragging so that the dragging
1130         // line overlay is shown
1131         getTableHeaderRow().setReordering(true);
1132 
1133         // Firstly we need to determine where to draw the line.
1134         // Find which column we're over
1135         TableColumnHeader hoverHeader = null;
1136 
1137         // x represents where the mouse is relative to the parent
1138         // NestedTableColumnHeader
1139         final double x = getParentHeader().sceneToLocal(sceneX, sceneY).getX();
1140 
1141         // calculate where the ghost column header should be
1142         double dragX = getTableSkin().getSkinnable().sceneToLocal(sceneX, sceneY).getX() - dragOffset;
1143         getTableHeaderRow().setDragHeaderX(dragX);
1144 
1145         double startX = 0;
1146         double endX = 0;
1147         double headersWidth = 0;
1148         newColumnPos = 0;
1149         for (TableColumnHeader header : getParentHeader().getColumnHeaders()) {
1150             if (! header.isVisible()) continue;
1151 
1152             double headerWidth = header.prefWidth(-1);
1153             headersWidth += headerWidth;
1154 
1155             startX = header.getBoundsInParent().getMinX();
1156             endX = startX + headerWidth;
1157 
1158             if (x >= startX && x < endX) {
1159                 hoverHeader = header;
1160                 break;
1161             }
1162             newColumnPos++;
1163         }
1164 
1165         // hoverHeader will be null if the drag occurs outside of the
1166         // tableview. In this case we handle the newColumnPos specially
1167         // and then short-circuit. This results in the drop action
1168         // resulting in the correct result (the column will drop at
1169         // the start or end of the table).
1170         if (hoverHeader == null) {
1171             newColumnPos = x > headersWidth ? (getParentHeader().getColumns().size() - 1) : 0;
1172             return;
1173         }
1174 
1175         // This is the x-axis value midway through hoverHeader. It's
1176         // used to determine whether the drop should be to the left
1177         // or the right of hoverHeader.
1178         double midPoint = startX + (endX - startX) / 2;
1179         boolean beforeMidPoint = x <= midPoint;
1180 
1181         // Based on where the mouse actually is, we have to shuffle
1182         // where we want the column to end up. This code handles that.
1183         int currentPos = getIndex(getTableColumn());
1184         newColumnPos += newColumnPos > currentPos && beforeMidPoint ?
1185             -1 : (newColumnPos < currentPos && !beforeMidPoint ? 1 : 0);
1186 
1187         double lineX = getTableHeaderRow().sceneToLocal(hoverHeader.localToScene(hoverHeader.getBoundsInLocal())).getMinX();
1188         lineX = lineX + ((beforeMidPoint) ? (0) : (hoverHeader.getWidth()));
1189 
1190         if (lineX >= -0.5 && lineX <= getTableSkin().getSkinnable().getWidth()) {
1191             columnReorderLine.setTranslateX(lineX);
1192 
1193             // then if this is the first event, we set the property to true
1194             // so that the line becomes visible until the drop is completed.
1195             // We also set reordering to true so that the various reordering
1196             // effects become visible (ghost, transparent overlay, etc).
1197             columnReorderLine.setVisible(true);
1198         }
1199 
1200         getTableHeaderRow().setReordering(true);
1201     }
