 148     /**
 149      * Creates a new TableColumnHeader instance to visually represent the given
 150      * {@link TableColumnBase} instance.
 151      *
 152      * @param tc The table column to be visually represented by this instance.
 153      */
 154     public TableColumnHeader(final TableColumnBase tc) {
 155         setTableColumn(tc);
 156         setFocusTraversable(false);
 157 
 158         initStyleClasses();
 159         initUI();
 160 
 161         // change listener for multiple properties
 162         changeListenerHandler = new LambdaMultiplePropertyChangeListenerHandler();
 163         changeListenerHandler.registerChangeListener(sceneProperty(), e -> updateScene());
 164 
 165         if (getTableColumn() != null) {
 166             changeListenerHandler.registerChangeListener(tc.idProperty(), e -> setId(tc.getId()));
 167             changeListenerHandler.registerChangeListener(tc.styleProperty(), e -> setStyle(tc.getStyle()));
 168             changeListenerHandler.registerChangeListener(tc.widthProperty(), e -> {
 169                 // It is this that ensures that when a column is resized that the header
 170                 // visually adjusts its width as necessary.
 171                 isSizeDirty = true;
 172                 requestLayout();
 173             });
 174             changeListenerHandler.registerChangeListener(tc.visibleProperty(), e -> setVisible(getTableColumn().isVisible()));
 175             changeListenerHandler.registerChangeListener(tc.sortNodeProperty(), e -> updateSortGrid());
 176             changeListenerHandler.registerChangeListener(tc.sortableProperty(), e -> {
 177                 // we need to notify all headers that a sortable state has changed,
 178                 // in case the sort grid in other columns needs to be updated.
 179                 if (TableSkinUtils.getSortOrder(getTableSkin()).contains(getTableColumn())) {
 180                     NestedTableColumnHeader root = getTableHeaderRow().getRootHeader();
 181                     updateAllHeaders(root);
 182                 }
 183             });
 184             changeListenerHandler.registerChangeListener(tc.textProperty(), e -> label.setText(tc.getText()));
 185             changeListenerHandler.registerChangeListener(tc.graphicProperty(), e -> label.setGraphic(tc.getGraphic()));
 186 
 187             setId(tc.getId());
 188             setStyle(tc.getStyle());
 189             /* Having TableColumn role parented by TableColumn causes VoiceOver to be unhappy */
 190             setAccessibleRole(AccessibleRole.TABLE_COLUMN);
 191         }
 192     }

 231     private static final EventHandler<MouseEvent> mousePressedHandler = me -> {
 232         TableColumnHeader header = (TableColumnHeader) me.getSource();
 233         TableColumnBase tableColumn = header.getTableColumn();
 234 
 235         ContextMenu menu = tableColumn.getContextMenu();
 236         if (menu != null && menu.isShowing()) {
 237             menu.hide();
 238         }
 239 
 240         if (me.isConsumed()) return;
 241         me.consume();
 242 
 243         header.getTableHeaderRow().columnDragLock = true;
 244 
 245         // pass focus to the table, so that the user immediately sees
 246         // the focus rectangle around the table control.
 247         header.getTableSkin().getSkinnable().requestFocus();
 248 
 249         if (me.isPrimaryButtonDown() && header.isColumnReorderingEnabled()) {
 250             header.columnReorderingStarted(me.getX());
 251         }
 252     };
 253 
 254     private static final EventHandler<MouseEvent> mouseDraggedHandler = me -> {
 255         if (me.isConsumed()) return;
 256         me.consume();
 257 
 258         TableColumnHeader header = (TableColumnHeader) me.getSource();
 259 
 260         if (me.isPrimaryButtonDown() && header.isColumnReorderingEnabled()) {
 261             header.columnReordering(me.getSceneX(), me.getSceneY());
 262         }
 263     };
 264 
 265     private static final EventHandler<MouseEvent> mouseReleasedHandler = me -> {
 266         if (me.isPopupTrigger()) return;
 267         if (me.isConsumed()) return;
 268         me.consume();
 269 
 270         TableColumnHeader header = (TableColumnHeader) me.getSource();
 271         header.getTableHeaderRow().columnDragLock = false;
 272 
 273         if (header.getTableHeaderRow().isReordering() && header.isColumnReorderingEnabled()) {
 274             header.columnReorderingComplete();
 275         } else if (me.isStillSincePress()) {
 276             header.sortColumn(me.isShiftDown());
 277         }
 278     };
 279 
 280     private static final EventHandler<ContextMenuEvent> contextMenuRequestedHandler = me -> {
 281         TableColumnHeader header = (TableColumnHeader) me.getSource();
 282         TableColumnBase tableColumn = header.getTableColumn();
 283 
 284         ContextMenu menu = tableColumn.getContextMenu();
 285         if (menu != null) {
 286             menu.show(header, me.getScreenX(), me.getScreenY());
 287             me.consume();
 288         }
 289     };

 362     /** {@inheritDoc} */
 363     @Override protected void layoutChildren() {
 364         if (isSizeDirty) {
 365             resize(getTableColumn().getWidth(), getHeight());
 366             isSizeDirty = false;
 367         }
 368 
 369         double sortWidth = 0;
 370         double w = snapSizeX(getWidth()) - (snappedLeftInset() + snappedRightInset());
 371         double h = getHeight() - (snappedTopInset() + snappedBottomInset());
 372         double x = w;
 373 
 374         // a bit hacky, but we REALLY don't want the arrow shape to fluctuate
 375         // in size
 376         if (arrow != null) {
 377             arrow.setMaxSize(arrow.prefWidth(-1), arrow.prefHeight(-1));
 378         }
 379 
 380         if (sortArrow != null && sortArrow.isVisible()) {
 381             sortWidth = sortArrow.prefWidth(-1);
 382             x -= sortWidth;
 383             sortArrow.resize(sortWidth, sortArrow.prefHeight(-1));
 384             positionInArea(sortArrow, x, snappedTopInset(),
 385                     sortWidth, h, 0, HPos.CENTER, VPos.CENTER);
 386         }
 387 
 388         if (label != null) {
 389             double labelWidth = w - sortWidth;
 390             label.resizeRelocate(snappedLeftInset(), 0, labelWidth, getHeight());
 391         }
 392     }

 394     /** {@inheritDoc} */
 395     @Override protected double computePrefWidth(double height) {
 396         if (getNestedColumnHeader() != null) {
 397             double width = getNestedColumnHeader().prefWidth(height);
 398 
 399             if (getTableColumn() != null) {
 400                 TableColumnBaseHelper.setWidth(getTableColumn(), width);
 401             }
 402 
 403             return width;
 404         } else if (getTableColumn() != null && getTableColumn().isVisible()) {
 405             return snapSizeX(getTableColumn().getWidth());
 406         }
 407 
 408         return 0;
 409     }

 477     private void updateTableSkin() {
 478         // when we get the table header row, we are also given the skin,
 479         // so this is the time to hook up listeners, etc.
 480         TableViewSkinBase<?,?,?,?,?> tableSkin = getTableSkin();
 481         if (tableSkin == null) return;
 482 
 483         updateColumnIndex();
 484         this.columnReorderLine = tableSkin.getColumnReorderLine();
 485 
 486         if (getTableColumn() != null) {
 487             updateSortPosition();
 488             TableSkinUtils.getSortOrder(tableSkin).addListener(weakSortOrderListener);
 489             TableSkinUtils.getVisibleLeafColumns(tableSkin).addListener(weakVisibleLeafColumnsListener);
 490         }
 491     }

 507     // RT-29682: When the sortable property of a TableColumnBase changes this
 508     // may impact other TableColumnHeaders, as they may need to change their
 509     // sort order representation. Rather than install listeners across all
 510     // TableColumn in the sortOrder list for their sortable property, we simply
 511     // update the sortPosition of all headers whenever the sortOrder property
 512     // changes, assuming the column is within the sortOrder list.
 513     private void updateAllHeaders(TableColumnHeader header) {
 514         if (header instanceof NestedTableColumnHeader) {
 515             List<TableColumnHeader> children = ((NestedTableColumnHeader)header).getColumnHeaders();
 516             for (int i = 0; i < children.size(); i++) {
 517                 updateAllHeaders(children.get(i));
 518             }
 519         } else {
 520             header.updateSortPosition();
 521         }
 522     }

 524     private void updateScene() {
 525         // RT-17684: If the TableColumn widths are all currently the default,
 526         // we attempt to 'auto-size' based on the preferred width of the first
 527         // n rows (we can't do all rows, as that could conceivably be an unlimited
 528         // number of rows retrieved from a very slow (e.g. remote) data source.
 529         // Obviously, the bigger the value of n, the more likely the default
 530         // width will be suitable for most values in the column
 531         final int n = 30;
 532         if (! autoSizeComplete) {
 533             if (getTableColumn() == null || getTableColumn().getWidth() != DEFAULT_COLUMN_WIDTH || getScene() == null) {
 534                 return;
 535             }
 536             doColumnAutoSize(n);
 537             autoSizeComplete = true;
 538         }
 539     }

 542         TableViewSkinBase tableSkin = getTableSkin();
 543         if (tableSkin != null) {
 544             TableSkinUtils.getVisibleLeafColumns(tableSkin).removeListener(weakVisibleLeafColumnsListener);
 545             TableSkinUtils.getSortOrder(tableSkin).removeListener(weakSortOrderListener);
 546         }

 548         changeListenerHandler.dispose();
 549     }

 563     private void initUI() {
 564         // TableColumn will be null if we are dealing with the root NestedTableColumnHeader
 565         if (getTableColumn() == null) return;
 566 
 567         // set up mouse events
 568         setOnMousePressed(mousePressedHandler);
 569         setOnMouseDragged(mouseDraggedHandler);
 570         setOnDragDetected(event -> event.consume());
 571         setOnContextMenuRequested(contextMenuRequestedHandler);
 572         setOnMouseReleased(mouseReleasedHandler);
 573 
 574         // --- label
 575         label = new Label();
 576         label.setText(getTableColumn().getText());
 577         label.setGraphic(getTableColumn().getGraphic());
 578         label.setVisible(getTableColumn().isVisible());
 579 
 580         // ---- container for the sort arrow (which is not supported on embedded
 581         // platforms)
 582         if (isSortingEnabled()) {
 583             // put together the grid
 584             updateSortGrid();
 585         }
 586     }

 588     private void doColumnAutoSize(int cellsToMeasure) {
 589         double prefWidth = getTableColumn().getPrefWidth();
 590 
 591         // if the prefWidth has been set, we do _not_ autosize columns
 592         if (prefWidth == DEFAULT_COLUMN_WIDTH) {
 593             resizeColumnToFitContent(cellsToMeasure);
 594         }
 595     }

 597     /**
 598      * Resizes this {@code TableColumnHeader}'s column based on content of header and content of cells. This
 599      * implementation measures the preferred width of the header, the preferred width of the first {@code maxRow} cells
 600      * and sizes the column to the maximum width of all measured values. Subclass can either call this method or
 601      * override it (no need to call {@code super()}) to provide their custom implementation (exclude headers, exclude
 602      * null content, use min).
 603      *
 604      * @param maxRows the number of rows considered when resizing. If -1 is given, all rows are considered.
 605      * @since 14
 606      */
 607     protected void resizeColumnToFitContent(int maxRows) {
 608         TableColumnBase<?, ?> tc = getTableColumn();
 609         if (!tc.isResizable()) return;
 610 
 611         Object control = this.getTableSkin().getSkinnable();
 612         if (control instanceof TableView) {
 613             resizeColumnToFitContent((TableView)control, (TableColumn)tc, this.getTableSkin(), maxRows);
 614         } else if (control instanceof TreeTableView) {
 615             resizeColumnToFitContent((TreeTableView)control, (TreeTableColumn)tc, this.getTableSkin(), maxRows);
 616         }
 617     }

 619     private <T,S> void resizeColumnToFitContent(TableView<T> tv, TableColumn<T, S> tc, TableViewSkinBase tableSkin, int maxRows) {
 620         List<?> items = tv.getItems();
 621         if (items == null || items.isEmpty()) return;
 622 
 623         Callback/*<TableColumn<T, ?>, TableCell<T,?>>*/ cellFactory = tc.getCellFactory();
 624         if (cellFactory == null) return;
 625 
 626         TableCell<T,?> cell = (TableCell<T, ?>) cellFactory.call(tc);
 627         if (cell == null) return;
 628 
 629         // set this property to tell the TableCell we want to know its actual
 630         // preferred width, not the width of the associated TableColumnBase
 631         cell.getProperties().put(Properties.DEFER_TO_PARENT_PREF_WIDTH, Boolean.TRUE);
 632 
 633         // determine cell padding
 634         double padding = 10;
 635         Node n = cell.getSkin() == null ? null : cell.getSkin().getNode();
 636         if (n instanceof Region) {
 637             Region r = (Region) n;
 638             padding = r.snappedLeftInset() + r.snappedRightInset();
 639         }
 640 
 641         int rows = maxRows == -1 ? items.size() : Math.min(items.size(), maxRows);
 642         double maxWidth = 0;
 643         for (int row = 0; row < rows; row++) {
 644             cell.updateTableColumn(tc);
 645             cell.updateTableView(tv);
 646             cell.updateIndex(row);
 647 
 648             if ((cell.getText() != null && !cell.getText().isEmpty()) || cell.getGraphic() != null) {
 649                 tableSkin.getChildren().add(cell);
 650                 cell.applyCss();
 651                 maxWidth = Math.max(maxWidth, cell.prefWidth(-1));
 652                 tableSkin.getChildren().remove(cell);
 653             }
 654         }
 655 
 656         // dispose of the cell to prevent it retaining listeners (see RT-31015)
 657         cell.updateIndex(-1);
 658 
 659         // RT-36855 - take into account the column header text / graphic widths.
 660         // Magic 10 is to allow for sort arrow to appear without text truncation.
 661         TableColumnHeader header = tableSkin.getTableHeaderRow().getColumnHeaderFor(tc);
 662         double headerTextWidth = Utils.computeTextWidth(header.label.getFont(), tc.getText(), -1);
 663         Node graphic = header.label.getGraphic();
 664         double headerGraphicWidth = graphic == null ? 0 : graphic.prefWidth(-1) + header.label.getGraphicTextGap();
 665         double headerWidth = headerTextWidth + headerGraphicWidth + 10 + header.snappedLeftInset() + header.snappedRightInset();
 666         maxWidth = Math.max(maxWidth, headerWidth);
 667 
 668         // RT-23486
 669         maxWidth += padding;

 690     private <T,S> void resizeColumnToFitContent(TreeTableView<T> ttv, TreeTableColumn<T, S> tc, TableViewSkinBase tableSkin, int maxRows) {
 691         List<?> items = new TreeTableViewBackingList(ttv);
 692         if (items == null || items.isEmpty()) return;
 693 
 694         Callback cellFactory = tc.getCellFactory();
 695         if (cellFactory == null) return;
 696 
 697         TreeTableCell<T,S> cell = (TreeTableCell) cellFactory.call(tc);
 698         if (cell == null) return;
 699 
 700         // set this property to tell the TableCell we want to know its actual
 701         // preferred width, not the width of the associated TableColumnBase
 702         cell.getProperties().put(Properties.DEFER_TO_PARENT_PREF_WIDTH, Boolean.TRUE);
 703 
 704         // determine cell padding
 705         double padding = 10;
 706         Node n = cell.getSkin() == null ? null : cell.getSkin().getNode();
 707         if (n instanceof Region) {
 708             Region r = (Region) n;
 709             padding = r.snappedLeftInset() + r.snappedRightInset();
 710         }
 711 
 712         TreeTableRow<T> treeTableRow = new TreeTableRow<>();
 713         treeTableRow.updateTreeTableView(ttv);
 714 
 715         int rows = maxRows == -1 ? items.size() : Math.min(items.size(), maxRows);
 716         double maxWidth = 0;
 717         for (int row = 0; row < rows; row++) {
 718             treeTableRow.updateIndex(row);
 719             treeTableRow.updateTreeItem(ttv.getTreeItem(row));
 720 
 721             cell.updateTreeTableColumn(tc);
 722             cell.updateTreeTableView(ttv);
 723             cell.updateTreeTableRow(treeTableRow);
 724             cell.updateIndex(row);
 725 
 726             if ((cell.getText() != null && !cell.getText().isEmpty()) || cell.getGraphic() != null) {
 727                 tableSkin.getChildren().add(cell);
 728                 cell.applyCss();
 729 
 730                 double w = cell.prefWidth(-1);
 731 
 732                 maxWidth = Math.max(maxWidth, w);
 733                 tableSkin.getChildren().remove(cell);
 734             }
 735         }
 736 
 737         // dispose of the cell to prevent it retaining listeners (see RT-31015)
 738         cell.updateIndex(-1);
 739 
 740         // RT-36855 - take into account the column header text / graphic widths.
 741         // Magic 10 is to allow for sort arrow to appear without text truncation.
 742         TableColumnHeader header = tableSkin.getTableHeaderRow().getColumnHeaderFor(tc);
 743         double headerTextWidth = Utils.computeTextWidth(header.label.getFont(), tc.getText(), -1);
 744         Node graphic = header.label.getGraphic();
 745         double headerGraphicWidth = graphic == null ? 0 : graphic.prefWidth(-1) + header.label.getGraphicTextGap();
 746         double headerWidth = headerTextWidth + headerGraphicWidth + 10 + header.snappedLeftInset() + header.snappedRightInset();
 747         maxWidth = Math.max(maxWidth, headerWidth);
 748 
 749         // RT-23486
 750         maxWidth += padding;
 751         if (ttv.getColumnResizePolicy() == TreeTableView.CONSTRAINED_RESIZE_POLICY && ttv.getWidth() > 0) {
 752 
 753             if (maxWidth > tc.getMaxWidth()) {
 754                 maxWidth = tc.getMaxWidth();
 755             }
 756 
 757             int size = tc.getColumns().size();
 758             if (size > 0) {
 759                 TableColumnHeader columnHeader = getTableHeaderRow().getColumnHeaderFor(tc.getColumns().get(size - 1));
 760                 if (columnHeader != null) {
 761                     columnHeader.resizeColumnToFitContent(maxRows);
 762                 }
 763                 return;
 764             }
 765 
 766             TableSkinUtils.resizeColumn(tableSkin, tc, Math.round(maxWidth - tc.getWidth()));
 767         } else {
 768             TableColumnBaseHelper.setWidth(tc, maxWidth);
 769         }
 770     }

 908     void moveColumn(TableColumnBase column, final int newColumnPos) {
 909         if (column == null || newColumnPos < 0) return;
 910 
 911         ObservableList<TableColumnBase<?,?>> columns = getColumns(column);
 912 
 913         final int columnsCount = columns.size();
 914         final int currentPos = columns.indexOf(column);
 915 
 916         int actualNewColumnPos = newColumnPos;
 917 
 918         // Fix for RT-35141: We need to account for hidden columns.
 919         // We keep iterating until we see 'requiredVisibleColumns' number of visible columns
 920         final int requiredVisibleColumns = actualNewColumnPos;
 921         int visibleColumnsSeen = 0;
 922         for (int i = 0; i < columnsCount; i++) {
 923             if (visibleColumnsSeen == (requiredVisibleColumns + 1)) {
 924                 break;
 925             }
 926 
 927             if (columns.get(i).isVisible()) {
 928                 visibleColumnsSeen++;
 929             } else {
 930                 actualNewColumnPos++;
 931             }
 932         }
 933         // --- end of RT-35141 fix
 934 
 935         if (actualNewColumnPos >= columnsCount) {
 936             actualNewColumnPos = columnsCount - 1;
 937         } else if (actualNewColumnPos < 0) {
 938             actualNewColumnPos = 0;
 939         }
 940 
 941         if (actualNewColumnPos == currentPos) return;
 942 
 943         List<TableColumnBase<?,?>> tempList = new ArrayList<>(columns);
 944         tempList.remove(column);
 945         tempList.add(actualNewColumnPos, column);
 946 
 947         columns.setAll(tempList);
 948     }

 987     private void sortColumn(final boolean addColumn) {
 988         if (! isSortingEnabled()) return;
 989 
 990         // we only allow sorting on the leaf columns and columns
 991         // that actually have comparators defined, and are sortable
 992         if (getTableColumn() == null || getTableColumn().getColumns().size() != 0 || getTableColumn().getComparator() == null || !getTableColumn().isSortable()) return;
 993 //        final int sortPos = getTable().getSortOrder().indexOf(column);
 994 //        final boolean isSortColumn = sortPos != -1;
 995 
 996         final ObservableList<TableColumnBase<?,?>> sortOrder = TableSkinUtils.getSortOrder(getTableSkin());
 997 
 998         // addColumn is true e.g. when the user is holding down Shift
 999         if (addColumn) {
1000             if (!isSortColumn) {
1001                 setSortType(getTableColumn(), TableColumn.SortType.ASCENDING);
1002                 sortOrder.add(getTableColumn());
1003             } else if (isAscending(getTableColumn())) {
1004                 setSortType(getTableColumn(), TableColumn.SortType.DESCENDING);
1005             } else {
1006                 int i = sortOrder.indexOf(getTableColumn());
1007                 if (i != -1) {
1008                     sortOrder.remove(i);
1009                 }
1010             }
1011         } else {
1012             // the user has clicked on a column header - we should add this to
1013             // the TableView sortOrder list if it isn't already there.
1014             if (isSortColumn && sortOrder.size() == 1) {
1015                 // the column is already being sorted, and it's the only column.
1016                 // We therefore move through the 2nd or 3rd states:
1017                 //   1st click: sort ascending
1018                 //   2nd click: sort descending
1019                 //   3rd click: natural sorting (sorting is switched off)
1020                 if (isAscending(getTableColumn())) {
1021                     setSortType(getTableColumn(), TableColumn.SortType.DESCENDING);
1022                 } else {
1023                     // remove from sort
1024                     sortOrder.remove(getTableColumn());
1025                 }
1026             } else if (isSortColumn) {
1027                 // the column is already being used to sort, so we toggle its
1028                 // sortAscending property, and also make the column become the
1029                 // primary sort column
1030                 if (isAscending(getTableColumn())) {
1031                     setSortType(getTableColumn(), TableColumn.SortType.DESCENDING);
1032                 } else if (isDescending(getTableColumn())) {
1033                     setSortType(getTableColumn(), TableColumn.SortType.ASCENDING);
1034                 }
1035 
1036                 // to prevent multiple sorts, we make a copy of the sort order
1037                 // list, moving the column value from the current position to
1038                 // its new position at the front of the list
1039                 List<TableColumnBase<?,?>> sortOrderCopy = new ArrayList<TableColumnBase<?,?>>(sortOrder);
1040                 sortOrderCopy.remove(getTableColumn());
1041                 sortOrderCopy.add(0, getTableColumn());
1042                 sortOrder.setAll(getTableColumn());
1043             } else {
1044                 // add to the sort order, in ascending form
1045                 setSortType(getTableColumn(), TableColumn.SortType.ASCENDING);
1046                 sortOrder.setAll(getTableColumn());
1047             }
1048         }
1049     }

1051     // Because it is possible that some columns are in the sortOrder list but are
1052     // not themselves sortable, we cannot just do sortOrderList.indexOf(column).
1053     // Therefore, this method does the proper work required of iterating through
1054     // and ignoring non-sortable (and null) columns in the sortOrder list.
1055     private int getSortPosition() {
1056         if (getTableColumn() == null) {
1057             return -1;
1058         }
1059 
1060         final List<TableColumnBase> sortOrder = getVisibleSortOrderColumns();
1061         int pos = 0;
1062         for (int i = 0; i < sortOrder.size(); i++) {
1063             TableColumnBase _tc = sortOrder.get(i);
1064 
1065             if (getTableColumn().equals(_tc)) {
1066                 return pos;
1067             }
1068 
1069             pos++;
1070         }
1071 
1072         return -1;
1073     }

1075     private List<TableColumnBase> getVisibleSortOrderColumns() {
1076         final ObservableList<TableColumnBase<?,?>> sortOrder = TableSkinUtils.getSortOrder(getTableSkin());
1077 
1078         List<TableColumnBase> visibleSortOrderColumns = new ArrayList<>();
1079         for (int i = 0; i < sortOrder.size(); i++) {
1080             TableColumnBase _tc = sortOrder.get(i);
1081             if (_tc == null || ! _tc.isSortable() || ! _tc.isVisible()) {
1082                 continue;
1083             }
1084 
1085             visibleSortOrderColumns.add(_tc);
1086         }
1087 
1088         return visibleSortOrderColumns;
1089     }

1119     // package for testing
1120     void columnReordering(double sceneX, double sceneY) {
1121         if (! getTableColumn().isReorderable()) return;
1122 
1123         // this is for handling the column drag to reorder columns.
1124         // It shows a line to indicate where the 'drop' will be.
1125 
1126         // indicate that we've started dragging so that the dragging
1127         // line overlay is shown
1128         getTableHeaderRow().setReordering(true);
1129 
1130         // Firstly we need to determine where to draw the line.
1131         // Find which column we're over
1132         TableColumnHeader hoverHeader = null;
1133 
1134         // x represents where the mouse is relative to the parent
1135         // NestedTableColumnHeader
1136         final double x = getParentHeader().sceneToLocal(sceneX, sceneY).getX();
1137 
1138         // calculate where the ghost column header should be
1139         double dragX = getTableSkin().getSkinnable().sceneToLocal(sceneX, sceneY).getX() - dragOffset;
1140         getTableHeaderRow().setDragHeaderX(dragX);
1141 
1142         double startX = 0;
1143         double endX = 0;
1144         double headersWidth = 0;
1145         newColumnPos = 0;
1146         for (TableColumnHeader header : getParentHeader().getColumnHeaders()) {
1147             if (! header.isVisible()) continue;
1148 
1149             double headerWidth = header.prefWidth(-1);
1150             headersWidth += headerWidth;
1151 
1152             startX = header.getBoundsInParent().getMinX();
1153             endX = startX + headerWidth;
1154 
1155             if (x >= startX && x < endX) {
1156                 hoverHeader = header;
1157                 break;
1158             }
1159             newColumnPos++;
1160         }
1161 
1162         // hoverHeader will be null if the drag occurs outside of the
1163         // tableview. In this case we handle the newColumnPos specially
1164         // and then short-circuit. This results in the drop action
1165         // resulting in the correct result (the column will drop at
1166         // the start or end of the table).
1167         if (hoverHeader == null) {
1168             newColumnPos = x > headersWidth ? (getParentHeader().getColumns().size() - 1) : 0;
1169             return;
1170         }
1171 
1172         // This is the x-axis value midway through hoverHeader. It's
1173         // used to determine whether the drop should be to the left
1174         // or the right of hoverHeader.
1175         double midPoint = startX + (endX - startX) / 2;
1176         boolean beforeMidPoint = x <= midPoint;
1177 
1178         // Based on where the mouse actually is, we have to shuffle
1179         // where we want the column to end up. This code handles that.
1180         int currentPos = getIndex(getTableColumn());
1181         newColumnPos += newColumnPos > currentPos && beforeMidPoint ?
1182             -1 : (newColumnPos < currentPos && !beforeMidPoint ? 1 : 0);
1183 
1184         double lineX = getTableHeaderRow().sceneToLocal(hoverHeader.localToScene(hoverHeader.getBoundsInLocal())).getMinX();
1185         lineX = lineX + ((beforeMidPoint) ? (0) : (hoverHeader.getWidth()));
1186 
1187         if (lineX >= -0.5 && lineX <= getTableSkin().getSkinnable().getWidth()) {
1188             columnReorderLine.setTranslateX(lineX);
1189 
1190             // then if this is the first event, we set the property to true
1191             // so that the line becomes visible until the drop is completed.
1192             // We also set reordering to true so that the various reordering
1193             // effects become visible (ghost, transparent overlay, etc).
1194             columnReorderLine.setVisible(true);
1195         }
1196 
1197         getTableHeaderRow().setReordering(true);
1198     }
