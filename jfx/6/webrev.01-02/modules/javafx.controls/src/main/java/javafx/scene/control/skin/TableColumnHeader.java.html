 148     /**
 149      * Creates a new TableColumnHeader instance to visually represent the given
 150      * {@link TableColumnBase} instance.
 151      *
 152      * @param tc The table column to be visually represented by this instance.
 153      */
 154     public TableColumnHeader(final TableColumnBase tc) {
 155         setTableColumn(tc);
 156         setFocusTraversable(false);
 157 
 158         initStyleClasses();
 159         initUI();
 160 
 161         // change listener for multiple properties
 162         changeListenerHandler = new LambdaMultiplePropertyChangeListenerHandler();
 163         changeListenerHandler.registerChangeListener(sceneProperty(), e -> updateScene());
 164 
 165         if (getTableColumn() != null) {
 166             changeListenerHandler.registerChangeListener(tc.idProperty(), e -> setId(tc.getId()));
 167             changeListenerHandler.registerChangeListener(tc.styleProperty(), e -> setStyle(tc.getStyle()));
 168             changeListenerHandler.registerChangeListener(tc.widthProperty(), e -> {
 169                 // It is this that ensures that when a column is resized that the header
 170                 // visually adjusts its width as necessary.
 171                 isSizeDirty = true;
 172                 requestLayout();
 173             });
 174             changeListenerHandler.registerChangeListener(tc.visibleProperty(), e -> setVisible(getTableColumn().isVisible()));
 175             changeListenerHandler.registerChangeListener(tc.sortNodeProperty(), e -> updateSortGrid());
 176             changeListenerHandler.registerChangeListener(tc.sortableProperty(), e -> {
 177                 // we need to notify all headers that a sortable state has changed,
 178                 // in case the sort grid in other columns needs to be updated.
 179                 if (TableSkinUtils.getSortOrder(getTableSkin()).contains(getTableColumn())) {
 180                     NestedTableColumnHeader root = getTableHeaderRow().getRootHeader();
 181                     updateAllHeaders(root);
 182                 }
 183             });
 184             changeListenerHandler.registerChangeListener(tc.textProperty(), e -> label.setText(tc.getText()));
 185             changeListenerHandler.registerChangeListener(tc.graphicProperty(), e -> label.setGraphic(tc.getGraphic()));
 186 
 187             setId(tc.getId());
 188             setStyle(tc.getStyle());
 189             /* Having TableColumn role parented by TableColumn causes VoiceOver to be unhappy */
 190             setAccessibleRole(AccessibleRole.TABLE_COLUMN);
 191         }
 192     }

 231     private static final EventHandler<MouseEvent> mousePressedHandler = me -> {
 232         TableColumnHeader header = (TableColumnHeader) me.getSource();
 233         TableColumnBase tableColumn = header.getTableColumn();
 234 
 235         ContextMenu menu = tableColumn.getContextMenu();
 236         if (menu != null && menu.isShowing()) {
 237             menu.hide();
 238         }
 239 
 240         if (me.isConsumed()) return;
 241         me.consume();
 242 
 243         header.getTableHeaderRow().columnDragLock = true;
 244 
 245         // pass focus to the table, so that the user immediately sees
 246         // the focus rectangle around the table control.
 247         header.getTableSkin().getSkinnable().requestFocus();
 248 
 249         if (me.isPrimaryButtonDown() && header.isColumnReorderingEnabled()) {
 250             header.columnReorderingStarted(me.getX());
 251         }
 252     };
 253 
 254     private static final EventHandler<MouseEvent> mouseDraggedHandler = me -> {
 255         if (me.isConsumed()) return;
 256         me.consume();
 257 
 258         TableColumnHeader header = (TableColumnHeader) me.getSource();
 259 
 260         if (me.isPrimaryButtonDown() && header.isColumnReorderingEnabled()) {
 261             header.columnReordering(me.getSceneX(), me.getSceneY());
 262         }
 263     };
 264 
 265     private static final EventHandler<MouseEvent> mouseReleasedHandler = me -> {
 266         if (me.isPopupTrigger()) return;
 267         if (me.isConsumed()) return;
 268         me.consume();
 269 
 270         TableColumnHeader header = (TableColumnHeader) me.getSource();
 271         header.getTableHeaderRow().columnDragLock = false;
 272 
 273         if (header.getTableHeaderRow().isReordering() && header.isColumnReorderingEnabled()) {
 274             header.columnReorderingComplete();
 275         } else if (me.isStillSincePress()) {
 276             header.sortColumn(me.isShiftDown());
 277         }
 278     };
 279 
 280     private static final EventHandler<ContextMenuEvent> contextMenuRequestedHandler = me -> {
 281         TableColumnHeader header = (TableColumnHeader) me.getSource();
 282         TableColumnBase tableColumn = header.getTableColumn();
 283 
 284         ContextMenu menu = tableColumn.getContextMenu();
 285         if (menu != null) {
 286             menu.show(header, me.getScreenX(), me.getScreenY());
 287             me.consume();
 288         }
 289     };

 362     /** {@inheritDoc} */
 363     @Override protected void layoutChildren() {
 364         if (isSizeDirty) {
 365             resize(getTableColumn().getWidth(), getHeight());
 366             isSizeDirty = false;
 367         }
 368 
 369         double sortWidth = 0;
 370         double w = snapSizeX(getWidth()) - (snappedLeftInset() + snappedRightInset());
 371         double h = getHeight() - (snappedTopInset() + snappedBottomInset());
 372         double x = w;
 373 
 374         // a bit hacky, but we REALLY don't want the arrow shape to fluctuate
 375         // in size
 376         if (arrow != null) {
 377             arrow.setMaxSize(arrow.prefWidth(-1), arrow.prefHeight(-1));
 378         }
 379 
 380         if (sortArrow != null && sortArrow.isVisible()) {
 381             sortWidth = sortArrow.prefWidth(-1);
 382             x -= sortWidth;
 383             sortArrow.resize(sortWidth, sortArrow.prefHeight(-1));
 384             positionInArea(sortArrow, x, snappedTopInset(),
 385                     sortWidth, h, 0, HPos.CENTER, VPos.CENTER);
 386         }
 387 
 388         if (label != null) {
 389             double labelWidth = w - sortWidth;
 390             label.resizeRelocate(snappedLeftInset(), 0, labelWidth, getHeight());
 391         }
 392     }

 394     /** {@inheritDoc} */
 395     @Override protected double computePrefWidth(double height) {
 396         if (getNestedColumnHeader() != null) {
 397             double width = getNestedColumnHeader().prefWidth(height);
 398 
 399             if (getTableColumn() != null) {
 400                 TableColumnBaseHelper.setWidth(getTableColumn(), width);
 401             }
 402 
 403             return width;
 404         } else if (getTableColumn() != null && getTableColumn().isVisible()) {
 405             return snapSizeX(getTableColumn().getWidth());
 406         }
 407 
 408         return 0;
 409     }

 477     private void updateTableSkin() {
 478         // when we get the table header row, we are also given the skin,
 479         // so this is the time to hook up listeners, etc.
 480         TableViewSkinBase<?,?,?,?,?> tableSkin = getTableSkin();
 481         if (tableSkin == null) return;
 482 
 483         updateColumnIndex();
 484         this.columnReorderLine = tableSkin.getColumnReorderLine();
 485 
 486         if (getTableColumn() != null) {
 487             updateSortPosition();
 488             TableSkinUtils.getSortOrder(tableSkin).addListener(weakSortOrderListener);
 489             TableSkinUtils.getVisibleLeafColumns(tableSkin).addListener(weakVisibleLeafColumnsListener);
 490         }
 491     }

 507     // RT-29682: When the sortable property of a TableColumnBase changes this
 508     // may impact other TableColumnHeaders, as they may need to change their
 509     // sort order representation. Rather than install listeners across all
 510     // TableColumn in the sortOrder list for their sortable property, we simply
 511     // update the sortPosition of all headers whenever the sortOrder property
 512     // changes, assuming the column is within the sortOrder list.
 513     private void updateAllHeaders(TableColumnHeader header) {
 514         if (header instanceof NestedTableColumnHeader) {
 515             List<TableColumnHeader> children = ((NestedTableColumnHeader)header).getColumnHeaders();
 516             for (int i = 0; i < children.size(); i++) {
 517                 updateAllHeaders(children.get(i));
 518             }
 519         } else {
 520             header.updateSortPosition();
 521         }
 522     }

 524     private void updateScene() {
 525         // RT-17684: If the TableColumn widths are all currently the default,
 526         // we attempt to 'auto-size' based on the preferred width of the first
 527         // n rows (we can't do all rows, as that could conceivably be an unlimited
 528         // number of rows retrieved from a very slow (e.g. remote) data source.
 529         // Obviously, the bigger the value of n, the more likely the default
 530         // width will be suitable for most values in the column
 531         final int n = 30;
 532         if (! autoSizeComplete) {
 533             if (getTableColumn() == null || getTableColumn().getWidth() != DEFAULT_COLUMN_WIDTH || getScene() == null) {
 534                 return;
 535             }
 536             doColumnAutoSize(n);
 537             autoSizeComplete = true;
 538         }
 539     }

 542         TableViewSkinBase tableSkin = getTableSkin();
 543         if (tableSkin != null) {
 544             TableSkinUtils.getVisibleLeafColumns(tableSkin).removeListener(weakVisibleLeafColumnsListener);

 548         changeListenerHandler.dispose();

 563     private void initUI() {
 564         // TableColumn will be null if we are dealing with the root NestedTableColumnHeader
 565         if (getTableColumn() == null) return;
 566 
 567         // set up mouse events
 568         setOnMousePressed(mousePressedHandler);
 569         setOnMouseDragged(mouseDraggedHandler);
 570         setOnDragDetected(event -> event.consume());
 571         setOnContextMenuRequested(contextMenuRequestedHandler);
 572         setOnMouseReleased(mouseReleasedHandler);
 573 
 574         // --- label
 575         label = new Label();
 576         label.setText(getTableColumn().getText());
 577         label.setGraphic(getTableColumn().getGraphic());
 578         label.setVisible(getTableColumn().isVisible());

 580         // ---- container for the sort arrow (which is not supported on embedded
 581         // platforms)
 582         if (isSortingEnabled()) {
 583             // put together the grid
 584             updateSortGrid();
 585         }
 586     }

 588     private void doColumnAutoSize(int cellsToMeasure) {
 589         double prefWidth = getTableColumn().getPrefWidth();
 590 
 591         // if the prefWidth has been set, we do _not_ autosize columns
 592         if (prefWidth == DEFAULT_COLUMN_WIDTH) {
 593             resizeColumnToFitContent(cellsToMeasure);
 594         }
 595     }

 597     /**
 598      * Resizes this {@code TableColumnHeader}'s column to fit the width of its content. The resulting column width is
 599      * the maximum of the preferred width of the header cell and the preferred width of the first {@code maxRow} cells.
 600      * <p>
 601      * Subclasses can either use this method or override it (without the need to call {@code super()}) to provide their
 602      * custom implementation (such as ones that exclude the header, exclude {@code null} content, compute the minimum
 603      * width etc.).
 604      *
 605      * @param maxRows the number of rows considered when resizing. If -1 is given, all rows are considered.
 606      * @since 14
 607      */
 608     protected void resizeColumnToFitContent(int maxRows) {
 609         TableColumnBase<?, ?> tc = getTableColumn();
 610         if (!tc.isResizable()) return;
 611 
 612         Object control = this.getTableSkin().getSkinnable();
 613         if (control instanceof TableView) {
 614             resizeColumnToFitContent((TableView) control, (TableColumn) tc, this.getTableSkin(), maxRows);
 615         } else if (control instanceof TreeTableView) {
 616             resizeColumnToFitContent((TreeTableView) control, (TreeTableColumn) tc, this.getTableSkin(), maxRows);
 617         }
 618     }

 620     private <T,S> void resizeColumnToFitContent(TableView<T> tv, TableColumn<T, S> tc, TableViewSkinBase tableSkin, int maxRows) {
 621         List<?> items = tv.getItems();
 622         if (items == null || items.isEmpty()) return;
 623 
 624         Callback/*<TableColumn<T, ?>, TableCell<T,?>>*/ cellFactory = tc.getCellFactory();
 625         if (cellFactory == null) return;
 626 
 627         TableCell<T,?> cell = (TableCell<T, ?>) cellFactory.call(tc);
 628         if (cell == null) return;
 629 
 630         // set this property to tell the TableCell we want to know its actual
 631         // preferred width, not the width of the associated TableColumnBase
 632         cell.getProperties().put(Properties.DEFER_TO_PARENT_PREF_WIDTH, Boolean.TRUE);
 633 
 634         // determine cell padding
 635         double padding = 10;
 636         Node n = cell.getSkin() == null ? null : cell.getSkin().getNode();
 637         if (n instanceof Region) {
 638             Region r = (Region) n;
 639             padding = r.snappedLeftInset() + r.snappedRightInset();
 640         }
 641 
 642         int rows = maxRows == -1 ? items.size() : Math.min(items.size(), maxRows);
 643         double maxWidth = 0;
 644         for (int row = 0; row < rows; row++) {
 645             cell.updateTableColumn(tc);
 646             cell.updateTableView(tv);
 647             cell.updateIndex(row);
 648 
 649             if ((cell.getText() != null && !cell.getText().isEmpty()) || cell.getGraphic() != null) {
 650                 tableSkin.getChildren().add(cell);
 651                 cell.applyCss();
 652                 maxWidth = Math.max(maxWidth, cell.prefWidth(-1));
 653                 tableSkin.getChildren().remove(cell);
 654             }
 655         }
 656 
 657         // dispose of the cell to prevent it retaining listeners (see RT-31015)
 658         cell.updateIndex(-1);
 659 
 660         // RT-36855 - take into account the column header text / graphic widths.
 661         // Magic 10 is to allow for sort arrow to appear without text truncation.
 662         TableColumnHeader header = tableSkin.getTableHeaderRow().getColumnHeaderFor(tc);
 663         double headerTextWidth = Utils.computeTextWidth(header.label.getFont(), tc.getText(), -1);
 664         Node graphic = header.label.getGraphic();
 665         double headerGraphicWidth = graphic == null ? 0 : graphic.prefWidth(-1) + header.label.getGraphicTextGap();
 666         double headerWidth = headerTextWidth + headerGraphicWidth + 10 + header.snappedLeftInset() + header.snappedRightInset();
 667         maxWidth = Math.max(maxWidth, headerWidth);
 668 
 669         // RT-23486

 691     private <T,S> void resizeColumnToFitContent(TreeTableView<T> ttv, TreeTableColumn<T, S> tc, TableViewSkinBase tableSkin, int maxRows) {
 692         List<?> items = new TreeTableViewBackingList(ttv);
 693         if (items == null || items.isEmpty()) return;
 694 
 695         Callback cellFactory = tc.getCellFactory();
 696         if (cellFactory == null) return;
 697 
 698         TreeTableCell<T,S> cell = (TreeTableCell) cellFactory.call(tc);
 699         if (cell == null) return;
 700 
 701         // set this property to tell the TableCell we want to know its actual
 702         // preferred width, not the width of the associated TableColumnBase
 703         cell.getProperties().put(Properties.DEFER_TO_PARENT_PREF_WIDTH, Boolean.TRUE);
 704 
 705         // determine cell padding
 706         double padding = 10;
 707         Node n = cell.getSkin() == null ? null : cell.getSkin().getNode();
 708         if (n instanceof Region) {
 709             Region r = (Region) n;
 710             padding = r.snappedLeftInset() + r.snappedRightInset();
 711         }
 712 
 713         TreeTableRow<T> treeTableRow = new TreeTableRow<>();
 714         treeTableRow.updateTreeTableView(ttv);
 715 
 716         int rows = maxRows == -1 ? items.size() : Math.min(items.size(), maxRows);
 717         double maxWidth = 0;
 718         for (int row = 0; row < rows; row++) {
 719             treeTableRow.updateIndex(row);
 720             treeTableRow.updateTreeItem(ttv.getTreeItem(row));
 721 
 722             cell.updateTreeTableColumn(tc);
 723             cell.updateTreeTableView(ttv);
 724             cell.updateTreeTableRow(treeTableRow);
 725             cell.updateIndex(row);
 726 
 727             if ((cell.getText() != null && !cell.getText().isEmpty()) || cell.getGraphic() != null) {
 728                 tableSkin.getChildren().add(cell);
 729                 cell.applyCss();
 730 
 731                 double w = cell.prefWidth(-1);
 732 
 733                 maxWidth = Math.max(maxWidth, w);
 734                 tableSkin.getChildren().remove(cell);
 735             }
 736         }
 737 
 738         // dispose of the cell to prevent it retaining listeners (see RT-31015)
 739         cell.updateIndex(-1);
 740 
 741         // RT-36855 - take into account the column header text / graphic widths.
 742         // Magic 10 is to allow for sort arrow to appear without text truncation.
 743         TableColumnHeader header = tableSkin.getTableHeaderRow().getColumnHeaderFor(tc);
 744         double headerTextWidth = Utils.computeTextWidth(header.label.getFont(), tc.getText(), -1);
 745         Node graphic = header.label.getGraphic();
 746         double headerGraphicWidth = graphic == null ? 0 : graphic.prefWidth(-1) + header.label.getGraphicTextGap();
 747         double headerWidth = headerTextWidth + headerGraphicWidth + 10 + header.snappedLeftInset() + header.snappedRightInset();
 748         maxWidth = Math.max(maxWidth, headerWidth);
 749 
 750         // RT-23486
 751         maxWidth += padding;
 752         if (ttv.getColumnResizePolicy() == TreeTableView.CONSTRAINED_RESIZE_POLICY && ttv.getWidth() > 0) {
 753 
 754             if (maxWidth > tc.getMaxWidth()) {
 755                 maxWidth = tc.getMaxWidth();
 756             }
 757 
 758             int size = tc.getColumns().size();
 759             if (size > 0) {
 760                 TableColumnHeader columnHeader = getTableHeaderRow().getColumnHeaderFor(tc.getColumns().get(size - 1));
 761                 if (columnHeader != null) {
 762                     columnHeader.resizeColumnToFitContent(maxRows);
 763                 }
 764                 return;
 765             }
 766 
 767             TableSkinUtils.resizeColumn(tableSkin, tc, Math.round(maxWidth - tc.getWidth()));
 768         } else {
 769             TableColumnBaseHelper.setWidth(tc, maxWidth);
 770         }
 771     }

 909     void moveColumn(TableColumnBase column, final int newColumnPos) {
 910         if (column == null || newColumnPos < 0) return;
 911 
 912         ObservableList<TableColumnBase<?,?>> columns = getColumns(column);
 913 
 914         final int columnsCount = columns.size();
 915         final int currentPos = columns.indexOf(column);
 916 
 917         int actualNewColumnPos = newColumnPos;
 918 
 919         // Fix for RT-35141: We need to account for hidden columns.
 920         // We keep iterating until we see 'requiredVisibleColumns' number of visible columns
 921         final int requiredVisibleColumns = actualNewColumnPos;
 922         int visibleColumnsSeen = 0;
 923         for (int i = 0; i < columnsCount; i++) {
 924             if (visibleColumnsSeen == (requiredVisibleColumns + 1)) {
 925                 break;
 926             }
 927 
 928             if (columns.get(i).isVisible()) {
 929                 visibleColumnsSeen++;
 930             } else {
 931                 actualNewColumnPos++;
 932             }
 933         }
 934         // --- end of RT-35141 fix
 935 
 936         if (actualNewColumnPos >= columnsCount) {
 937             actualNewColumnPos = columnsCount - 1;
 938         } else if (actualNewColumnPos < 0) {
 939             actualNewColumnPos = 0;
 940         }
 941 
 942         if (actualNewColumnPos == currentPos) return;
 943 
 944         List<TableColumnBase<?,?>> tempList = new ArrayList<>(columns);
 945         tempList.remove(column);
 946         tempList.add(actualNewColumnPos, column);
 947 
 948         columns.setAll(tempList);
 949     }

 988     private void sortColumn(final boolean addColumn) {
 989         if (! isSortingEnabled()) return;
 990 
 991         // we only allow sorting on the leaf columns and columns
 992         // that actually have comparators defined, and are sortable
 993         if (getTableColumn() == null || getTableColumn().getColumns().size() != 0 || getTableColumn().getComparator() == null || !getTableColumn().isSortable()) return;
 994 //        final int sortPos = getTable().getSortOrder().indexOf(column);
 995 //        final boolean isSortColumn = sortPos != -1;
 996 
 997         final ObservableList<TableColumnBase<?,?>> sortOrder = TableSkinUtils.getSortOrder(getTableSkin());
 998 
 999         // addColumn is true e.g. when the user is holding down Shift
1000         if (addColumn) {
1001             if (!isSortColumn) {
1002                 setSortType(getTableColumn(), TableColumn.SortType.ASCENDING);
1003                 sortOrder.add(getTableColumn());
1004             } else if (isAscending(getTableColumn())) {
1005                 setSortType(getTableColumn(), TableColumn.SortType.DESCENDING);
1006             } else {
1007                 int i = sortOrder.indexOf(getTableColumn());
1008                 if (i != -1) {
1009                     sortOrder.remove(i);
1010                 }
1011             }
1012         } else {
1013             // the user has clicked on a column header - we should add this to
1014             // the TableView sortOrder list if it isn't already there.
1015             if (isSortColumn && sortOrder.size() == 1) {
1016                 // the column is already being sorted, and it's the only column.
1017                 // We therefore move through the 2nd or 3rd states:
1018                 //   1st click: sort ascending
1019                 //   2nd click: sort descending
1020                 //   3rd click: natural sorting (sorting is switched off)
1021                 if (isAscending(getTableColumn())) {
1022                     setSortType(getTableColumn(), TableColumn.SortType.DESCENDING);
1023                 } else {
1024                     // remove from sort
1025                     sortOrder.remove(getTableColumn());
1026                 }
1027             } else if (isSortColumn) {
1028                 // the column is already being used to sort, so we toggle its
1029                 // sortAscending property, and also make the column become the
1030                 // primary sort column
1031                 if (isAscending(getTableColumn())) {
1032                     setSortType(getTableColumn(), TableColumn.SortType.DESCENDING);
1033                 } else if (isDescending(getTableColumn())) {
1034                     setSortType(getTableColumn(), TableColumn.SortType.ASCENDING);
1035                 }
1036 
1037                 // to prevent multiple sorts, we make a copy of the sort order
1038                 // list, moving the column value from the current position to
1039                 // its new position at the front of the list
1040                 List<TableColumnBase<?,?>> sortOrderCopy = new ArrayList<TableColumnBase<?,?>>(sortOrder);
1041                 sortOrderCopy.remove(getTableColumn());
1042                 sortOrderCopy.add(0, getTableColumn());
1043                 sortOrder.setAll(getTableColumn());
1044             } else {
1045                 // add to the sort order, in ascending form
1046                 setSortType(getTableColumn(), TableColumn.SortType.ASCENDING);
1047                 sortOrder.setAll(getTableColumn());
1048             }
1049         }
1050     }

1052     // Because it is possible that some columns are in the sortOrder list but are
1053     // not themselves sortable, we cannot just do sortOrderList.indexOf(column).
1054     // Therefore, this method does the proper work required of iterating through
1055     // and ignoring non-sortable (and null) columns in the sortOrder list.
1056     private int getSortPosition() {
1057         if (getTableColumn() == null) {
1058             return -1;
1059         }
1060 
1061         final List<TableColumnBase> sortOrder = getVisibleSortOrderColumns();
1062         int pos = 0;
1063         for (int i = 0; i < sortOrder.size(); i++) {
1064             TableColumnBase _tc = sortOrder.get(i);
1065 
1066             if (getTableColumn().equals(_tc)) {
1067                 return pos;
1068             }
1069 
1070             pos++;
1071         }
1072 
1073         return -1;
1074     }

1076     private List<TableColumnBase> getVisibleSortOrderColumns() {
1077         final ObservableList<TableColumnBase<?,?>> sortOrder = TableSkinUtils.getSortOrder(getTableSkin());
1078 
1079         List<TableColumnBase> visibleSortOrderColumns = new ArrayList<>();
1080         for (int i = 0; i < sortOrder.size(); i++) {
1081             TableColumnBase _tc = sortOrder.get(i);
1082             if (_tc == null || ! _tc.isSortable() || ! _tc.isVisible()) {
1083                 continue;
1084             }
1085 
1086             visibleSortOrderColumns.add(_tc);
1087         }
1088 
1089         return visibleSortOrderColumns;
1090     }

1120     // package for testing
1121     void columnReordering(double sceneX, double sceneY) {
1122         if (! getTableColumn().isReorderable()) return;
1123 
1124         // this is for handling the column drag to reorder columns.
1125         // It shows a line to indicate where the 'drop' will be.
1126 
1127         // indicate that we've started dragging so that the dragging
1128         // line overlay is shown
1129         getTableHeaderRow().setReordering(true);
1130 
1131         // Firstly we need to determine where to draw the line.
1132         // Find which column we're over
1133         TableColumnHeader hoverHeader = null;
1134 
1135         // x represents where the mouse is relative to the parent
1136         // NestedTableColumnHeader
1137         final double x = getParentHeader().sceneToLocal(sceneX, sceneY).getX();
1138 
1139         // calculate where the ghost column header should be
1140         double dragX = getTableSkin().getSkinnable().sceneToLocal(sceneX, sceneY).getX() - dragOffset;
1141         getTableHeaderRow().setDragHeaderX(dragX);
1142 
1143         double startX = 0;
1144         double endX = 0;
1145         double headersWidth = 0;
1146         newColumnPos = 0;
1147         for (TableColumnHeader header : getParentHeader().getColumnHeaders()) {
1148             if (! header.isVisible()) continue;
1149 
1150             double headerWidth = header.prefWidth(-1);
1151             headersWidth += headerWidth;
1152 
1153             startX = header.getBoundsInParent().getMinX();
1154             endX = startX + headerWidth;
1155 
1156             if (x >= startX && x < endX) {
1157                 hoverHeader = header;
1158                 break;
1159             }
1160             newColumnPos++;
1161         }
1162 
1163         // hoverHeader will be null if the drag occurs outside of the
1164         // tableview. In this case we handle the newColumnPos specially
1165         // and then short-circuit. This results in the drop action
1166         // resulting in the correct result (the column will drop at
1167         // the start or end of the table).
1168         if (hoverHeader == null) {
1169             newColumnPos = x > headersWidth ? (getParentHeader().getColumns().size() - 1) : 0;
1170             return;
1171         }
1172 
1173         // This is the x-axis value midway through hoverHeader. It's
1174         // used to determine whether the drop should be to the left
1175         // or the right of hoverHeader.
1176         double midPoint = startX + (endX - startX) / 2;
1177         boolean beforeMidPoint = x <= midPoint;
1178 
1179         // Based on where the mouse actually is, we have to shuffle
1180         // where we want the column to end up. This code handles that.
1181         int currentPos = getIndex(getTableColumn());
1182         newColumnPos += newColumnPos > currentPos && beforeMidPoint ?
1183             -1 : (newColumnPos < currentPos && !beforeMidPoint ? 1 : 0);
1184 
1185         double lineX = getTableHeaderRow().sceneToLocal(hoverHeader.localToScene(hoverHeader.getBoundsInLocal())).getMinX();
1186         lineX = lineX + ((beforeMidPoint) ? (0) : (hoverHeader.getWidth()));
1187 
1188         if (lineX >= -0.5 && lineX <= getTableSkin().getSkinnable().getWidth()) {
1189             columnReorderLine.setTranslateX(lineX);
1190 
1191             // then if this is the first event, we set the property to true
1192             // so that the line becomes visible until the drop is completed.
1193             // We also set reordering to true so that the various reordering
1194             // effects become visible (ghost, transparent overlay, etc).
1195             columnReorderLine.setVisible(true);
1196         }
1197 
1198         getTableHeaderRow().setReordering(true);
1199     }
