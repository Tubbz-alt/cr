25 package javafx.scene.control.skin;
26 
27 import javafx.scene.control.TableColumn;
28 import javafx.scene.control.TableColumnBase;
29 
30 public class TableColumnHeaderShim {
31 
32     public static int getColumnIndex(TableColumnHeader colHeader) {
33         return colHeader == null ? -1 : colHeader.columnIndex;
34     }
35 
36     public static void moveColumn(TableColumn col, TableColumnHeader colHeader, int newPos) {
37         colHeader.moveColumn(col, newPos);
38     }
39 
40     public static int getSortPos(TableColumnHeader header) {
41         return header.sortPos;
42     }
43 
44     public static TableColumnHeader getColumnHeaderFor(TableHeaderRow header, final TableColumnBase<?,?> col) {
45         return header.getColumnHeaderFor(col);
46     }
47 
48     public static void columnReorderingStarted(TableColumnHeader header, double dragOffset) {
49         header.columnReorderingStarted(dragOffset);
50     }
51 
52     public static void columnReordering(TableColumnHeader header, double sceneX, double sceneY) {
53         header.columnReordering(sceneX, sceneY);
54     }
55 
56     public static void columnReorderingComplete(TableColumnHeader header) {
57         header.columnReorderingComplete();
58     }




59 }
