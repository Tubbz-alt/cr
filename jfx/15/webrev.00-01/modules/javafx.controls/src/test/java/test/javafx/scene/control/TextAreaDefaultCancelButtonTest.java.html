26 package test.javafx.scene.control;
27 
28 import javafx.scene.control.TextArea;
29 
30 /**
31  * Test for interplay of ENTER/ESCAPE handlers on TextArea with
32  * default/cancel button actions.
33  */
34 public class TextAreaDefaultCancelButtonTest extends DefaultCancelButtonTestBase<TextArea> {
35 
36     public TextAreaDefaultCancelButtonTest(ButtonType buttonType,
37             boolean consume, boolean registerAfterShowing) {
38         super(buttonType, consume, registerAfterShowing);
39     }
40 
41     /**
42      * Overridden to back out for ENTER (which is handled internally always)
43      */
44     @Override
45     public void testFallbackFilter() {
46         if (isEnter()) return;
47         super.testFallbackFilter();
48     }
49 
50     /**
51      * Overridden to back out for ENTER (which is handled internally always)
52      */
53     @Override
54     public void testFallbackHandler() {
55         if (isEnter()) return;
56         super.testFallbackHandler();
57     }
58 
59     /**
60      * Overridden to back out for ENTER (which is handled internally always)
61      */
62     @Override
63     public void testFallbackSingletonHandler() {
64         if (isEnter()) return;
65         super.testFallbackSingletonHandler();
66     }
67 
68     /**
69      * Overridden to back out for ENTER (which is handled internally always)
70      */
71     @Override
72     public void testFallbackNoHandler() {
73         if (isEnter()) return;
74         super.testFallbackNoHandler();
75     }
76 
77     @Override
78     protected TextArea createControl() {
79         return new TextArea();
80     }
81 
82 }
