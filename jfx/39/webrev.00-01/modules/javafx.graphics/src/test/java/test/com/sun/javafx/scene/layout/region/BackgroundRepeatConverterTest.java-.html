 26 package test.com.sun.javafx.scene.layout.region;
 27 
 28 import javafx.scene.layout.BackgroundRepeat;
 29 import org.junit.Test;
 30 import javafx.css.ParsedValue;
 31 import com.sun.javafx.css.ParsedValueImpl;
 32 import com.sun.javafx.scene.layout.region.RepeatStruct;
 33 
 34 import static org.junit.Assert.assertEquals;
 35 import com.sun.javafx.scene.layout.region.RepeatStructConverter;
 36 import com.sun.javafx.scene.layout.region.RepeatStructConverter;
 37 
 38 /**
 39  */
 40 public class BackgroundRepeatConverterTest {
 41     /*
 42         -fx-background-repeat:
 43      */
 44     @Test public void scenario1() {
 45         ParsedValueImpl<ParsedValue<String,BackgroundRepeat>[][], RepeatStruct[]> value =
 46                 new ParsedValueImpl<ParsedValue<String,BackgroundRepeat>[][], RepeatStruct[]>(
 47                         new ParsedValueImpl[0][0], null
 48                 );
 49         RepeatStruct[] results = RepeatStructConverter.getInstance().convert(value, null);
 50         assertEquals(0, results.length, 0);
 51     }
 52 
 53     /*
 54         -fx-background-repeat: null
 55      */
 56     @Test(expected=NullPointerException.class)
 57     public void scenario2() {
 58         ParsedValue<String,BackgroundRepeat>[][] values = new ParsedValueImpl[][] {
 59                 {null}
 60         };
 61 
 62         ParsedValueImpl<ParsedValue<String,BackgroundRepeat>[][], RepeatStruct[]> value =
 63                 new ParsedValueImpl<ParsedValue<String,BackgroundRepeat>[][], RepeatStruct[]>(
 64                         values, null
 65                 );
 66         RepeatStruct[] results = RepeatStructConverter.getInstance().convert(value, null);
 67     }
 68 
 69     /*
 70         -fx-background-repeat: repeat round
 71      */
 72     @Test public void scenario3() {
 73         ParsedValue<String,BackgroundRepeat>[][] values = new ParsedValueImpl[][] {
 74                 { new ParsedValueImpl("repeat", null), new ParsedValueImpl("round", null) }
 75         };
 76 
 77         ParsedValue<ParsedValue<String,BackgroundRepeat>[][], RepeatStruct[]> value =
 78                 new ParsedValueImpl<ParsedValue<String,BackgroundRepeat>[][], RepeatStruct[]>(
 79                         values, null
 80                 );
 81         RepeatStruct[] results = RepeatStructConverter.getInstance().convert(value, null);
 82         assertEquals(1, results.length, 0);
 83         assertEquals(BackgroundRepeat.REPEAT, results[0].repeatX);
 84         assertEquals(BackgroundRepeat.ROUND, results[0].repeatY);
 85     }
 86 
 87     /*
 88         -fx-background-repeat: space no-repeat
 89      */
 90     @Test public void scenario4() {
 91         ParsedValue<String,BackgroundRepeat>[][] values = new ParsedValueImpl[][] {
 92                 { new ParsedValueImpl("space", null), new ParsedValueImpl("no-repeat", null) }
 93         };
 94 
 95         ParsedValue<ParsedValue<String,BackgroundRepeat>[][], RepeatStruct[]> value =
 96                 new ParsedValueImpl<ParsedValue<String,BackgroundRepeat>[][], RepeatStruct[]>(
 97                         values, null
 98                 );
 99         RepeatStruct[] results = RepeatStructConverter.getInstance().convert(value, null);
100         assertEquals(1, results.length, 0);
101         assertEquals(BackgroundRepeat.SPACE, results[0].repeatX);
102         assertEquals(BackgroundRepeat.NO_REPEAT, results[0].repeatY);
103     }
104 
105     /*
106         -fx-background-repeat: no-repeat repeat, space round
107      */
108     @Test public void scenario5() {
109         ParsedValue<String,BackgroundRepeat>[][] values = new ParsedValueImpl[][] {
110                 { new ParsedValueImpl("no-repeat", null), new ParsedValueImpl("repeat", null) },
111                 { new ParsedValueImpl("space", null), new ParsedValueImpl("round", null) }
112         };
113 
114         ParsedValue<ParsedValue<String,BackgroundRepeat>[][], RepeatStruct[]> value =
115                 new ParsedValueImpl<ParsedValue<String,BackgroundRepeat>[][], RepeatStruct[]>(
116                         values, null
117                 );
118         RepeatStruct[] results = RepeatStructConverter.getInstance().convert(value, null);
119         assertEquals(2, results.length, 0);
120         assertEquals(BackgroundRepeat.NO_REPEAT, results[0].repeatX);
121         assertEquals(BackgroundRepeat.REPEAT, results[0].repeatY);
122         assertEquals(BackgroundRepeat.SPACE, results[1].repeatX);
123         assertEquals(BackgroundRepeat.ROUND, results[1].repeatY);
124     }
125 
126 }
