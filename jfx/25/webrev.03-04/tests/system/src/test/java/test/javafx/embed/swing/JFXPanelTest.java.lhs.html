 54 public class JFXPanelTest {
 55     // Used to launch the application before running any test
 56     private static final CountDownLatch launchLatch = new CountDownLatch(1);
 57 
 58     // Application class. An instance is created and initialized before running
 59     // the first test, and it lives through the execution of all tests.
 60     public static class MyApp extends Application {
 61         @Override
 62         public void start(Stage primaryStage) throws Exception {
 63             Platform.setImplicitExit(false);
 64             Assert.assertTrue(Platform.isFxApplicationThread());
 65             Assert.assertNotNull(primaryStage);
 66 
 67             launchLatch.countDown();
 68         }
 69     }
 70 
 71     @BeforeClass
 72     public static void doSetupOnce() {
 73         // Start the Application
 74         new Thread(() -> Application.launch(MyApp.class, (String[]) null)).start();
 75 
 76         try {
 77             if (!launchLatch.await(5000, TimeUnit.MILLISECONDS)) {
 78                 throw new AssertionFailedError("Timeout waiting for Application to launch");
 79             }
 80         } catch (InterruptedException ex) {
 81             AssertionFailedError err = new AssertionFailedError("Unexpected exception");
 82             err.initCause(ex);
 83             throw err;
 84         }
 85 
 86         Assert.assertEquals(0, launchLatch.getCount());
 87     }
 88 
 89     @AfterClass
 90     public static void doTeardownOnce() {
 91         Platform.exit();
 92     }
 93 
 94     class TestFXPanel extends JFXPanel {
 95         protected void processMouseEventPublic(MouseEvent e) {
 96             processMouseEvent(e);
 97         }
 98     };
 99 
100     @Test
101     public void testNoDoubleClickOnFirstClick() throws Exception {
102 
103         CountDownLatch firstPressedEventLatch = new CountDownLatch(1);
104 
105         // It's an array, so we can mutate it inside of lambda statement
106         int[] pressedEventCounter = {0};
107 
108         SwingUtilities.invokeLater(() -> {


109             TestFXPanel fxPnl = new TestFXPanel();
110             fxPnl.setPreferredSize(new Dimension(100, 100));
111             JFrame jframe = new JFrame();
112             JPanel jpanel = new JPanel();

113             jpanel.add(fxPnl);
114             jframe.setContentPane(jpanel);
115             jframe.pack();
116             jframe.setVisible(true);
117 
118             Platform.runLater(() -> {
119                 Group grp = new Group();

120                 Scene scene = new Scene(new Group());

121 
122                 scene.addEventHandler(javafx.scene.input.MouseEvent.MOUSE_PRESSED, (event -> {
123                     pressedEventCounter[0] += 1;
124                     firstPressedEventLatch.countDown();
125                 }));
126 
127                 fxPnl.setScene(scene);
128 
129                 SwingUtilities.invokeLater(() -> {
130                     MouseEvent e = new MouseEvent(fxPnl, MouseEvent.MOUSE_PRESSED, 0, MouseEvent.BUTTON1_DOWN_MASK,
131                             5, 5, 1, false, MouseEvent.BUTTON1);
132 
133                     fxPnl.processMouseEventPublic(e);
134                 });
135             });
136         });
137 
138         if(!firstPressedEventLatch.await(5000, TimeUnit.MILLISECONDS)) {
139             throw new Exception();
140         };
141 
142         Thread.sleep(500); // there should be no pressed event after the initial one. Let's wait for 0.5s and check again.
143 
144         Assert.assertEquals(1, pressedEventCounter[0]);
145     }
146 }
