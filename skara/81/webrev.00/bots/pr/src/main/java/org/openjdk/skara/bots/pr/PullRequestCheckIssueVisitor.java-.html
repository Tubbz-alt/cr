 23 package org.openjdk.skara.bots.pr;
 24 
 25 import org.openjdk.skara.host.*;
 26 import org.openjdk.skara.jcheck.*;
 27 import org.openjdk.skara.jcheck.Check;
 28 import org.openjdk.skara.vcs.Hash;
 29 
 30 import java.util.*;
 31 import java.util.logging.Logger;
 32 import java.util.stream.Collectors;
 33 
 34 class PullRequestCheckIssueVisitor implements IssueVisitor {
 35     private final Set<String> messages = new HashSet<>();
 36     private final List<CheckAnnotation> annotations = new LinkedList<>();
 37     private final Set<Check> enabledChecks;
 38     private final Set<Class<? extends Check>> failedChecks = new HashSet<>();
 39 
 40     private boolean readyForReview;
 41 
 42     private final Logger log = Logger.getLogger("org.openjdk.skara.bots.pr");
 43 
 44     private final Set<Class<? extends Check>> displayedChecks = Set.of(
 45             DuplicateIssuesCheck.class,
 46             ReviewersCheck.class,
 47             WhitespaceCheck.class,
 48             IssuesCheck.class
 49     );
 50 
 51     PullRequestCheckIssueVisitor(Set<Check> enabledChecks) {
 52         this.enabledChecks = enabledChecks;
 53         readyForReview = true;
 54     }
 55 
 56     List<String> getIssues() {
 57         return new ArrayList<>(messages);
 58     }
 59 
 60     Map<String, Boolean> getChecks() {
 61         return enabledChecks.stream()
 62                             .filter(check -> displayedChecks.contains(check.getClass()))
 63                             .collect(Collectors.toMap(Check::description,
 64                                                       check -> !failedChecks.contains(check.getClass())));
 65     }
 66 
 67     List<CheckAnnotation> getAnnotations() { return annotations; }
 68 
 69     boolean isReadyForReview() {
 70         return readyForReview;
 71     }
 72 
 73     public void visit(DuplicateIssuesIssue e) {
 74         var id = e.issue().id();
 75         var other = e.hashes()
 76                      .stream()
 77                      .map(Hash::abbreviate)
 78                      .map(s -> "         - " + s)
 79                      .collect(Collectors.toList());
 80 
 81         var output = new StringBuilder();
 82         output.append("Issue id ").append(id).append(" is already used in these commits:\n");
 83         other.forEach(h -> output.append(" * ").append(h).append("\n"));
 84         messages.add(output.toString());
 85         failedChecks.add(e.check().getClass());
 86         readyForReview = false;
 87     }
 88 
 89     @Override
 90     public void visit(TagIssue e) {
 91         log.fine("ignored: illegal tag name: " + e.tag().name());
 92     }
 93 
 94     @Override
 95     public void visit(BranchIssue e) {
 96         log.fine("ignored: illegal branch name: " + e.branch().name());
 97     }
 98 
 99     @Override
100     public void visit(SelfReviewIssue e) {
101         log.fine("ignored: self-reviews are not allowed");
102     }
103 
104     @Override
105     public void visit(TooFewReviewersIssue e) {
106         messages.add(String.format("Too few reviewers found (have %d, need at least %d)", e.numActual(), e.numRequired()));
107         failedChecks.add(e.check().getClass());
108     }
109 
110     @Override
111     public void visit(InvalidReviewersIssue e) {
112         log.fine("ignored: invalid reviewers: " + e.invalid());
113     }
114 
115     @Override
116     public void visit(MergeMessageIssue e) {
117         var hex = e.commit().hash().abbreviate();
118         log.fine("ignored: " + hex + ": merge commits should only have commit message 'Merge'");
119     }
120 
121     @Override
122     public void visit(HgTagCommitIssue e) {
123         log.fine("ignored: invalid tag commit");
124     }
125 
126     @Override
127     public void visit(CommitterIssue e) {
128         log.fine("ignored: invalid author: " + e.commit().author().name());
129     }
130 
131     @Override
132     public void visit(CommitterNameIssue issue) {
133         log.fine("ignored: invalid committer name");
134     }
135 
136     @Override
137     public void visit(CommitterEmailIssue issue) {
138         log.fine("ignored: invalid committer email");
139     }
140 
141     @Override
142     public void visit(AuthorNameIssue issue) {
143         log.fine("ignored: invalid author name");
144     }
145 
146     @Override
147     public void visit(AuthorEmailIssue issue) {
148         log.fine("ignored: invalid author email");
149     }
150 
151     @Override
152     public void visit(WhitespaceIssue e) {
153         var startColumn = Integer.MAX_VALUE;
154         var endColumn = Integer.MIN_VALUE;
155         var details = new LinkedList<String>();
156         for (var error : e.errors()) {
157             startColumn = Math.min(error.index(), startColumn);
158             endColumn = Math.max(error.index(), endColumn);
159             details.add("Column " + error.index() + ": " + error.kind().toString());
160         }
161 
162         var annotationBuilder = CheckAnnotationBuilder.create(
163                 e.path().toString(),
164                 e.row(),
165                 e.row(),
166                 CheckAnnotationLevel.FAILURE,
167                 String.join("  \n", details));
168 
169         if (startColumn < Integer.MAX_VALUE) {
170             annotationBuilder.startColumn(startColumn);
171         }
172         if (endColumn > Integer.MIN_VALUE) {
173             annotationBuilder.endColumn(endColumn);
174         }
175 
176         var annotation = annotationBuilder.title("Whitespace error").build();
177         annotations.add(annotation);
178 
179         messages.add("Whitespace errors");
180         failedChecks.add(e.check().getClass());
181         readyForReview = false;
182     }
183 
184     @Override
185     public void visit(MessageIssue issue) {
186         log.fine("ignored: incorrectly formatted commit message");
187     }
188 
189     @Override
190     public void visit(IssuesIssue issue) {
191         messages.add("The commit message does not reference any issue");
192         failedChecks.add(issue.check().getClass());
193         readyForReview = false;
194     }
195 
196     @Override
197     public void visit(ExecutableIssue issue) {
198         messages.add(String.format("Executable files are not allowed (file: %s)", issue.path()));
199         failedChecks.add(issue.check().getClass());
200         readyForReview = false;
201     }
202 
203     @Override
204     public void visit(BlacklistIssue issue) {
205         log.fine("ignored: blacklisted commit");
206     }
207 }
