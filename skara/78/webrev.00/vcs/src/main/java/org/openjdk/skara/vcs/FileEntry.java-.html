23 package org.openjdk.skara.vcs;
24 
25 import java.util.Objects;
26 import java.nio.file.Path;
27 
28 public class FileEntry {
29     private final FileType type;
30     private final Hash hash;
31     private final Path path;
32 
33     public FileEntry(FileType type, Hash hash, Path path) {
34         this.type = type;
35         this.hash = hash;
36         this.path = path;
37     }
38 
39     public FileType type() {
40         return type;
41     }
42 
43     public Hash hash() {
44         return hash;
45     }
46 
47     public Path path() {
48         return path;
49     }
50 
51     @Override
52     public int hashCode() {
53         return Objects.hash(type, hash, path);
54     }
55 
56     @Override
57     public boolean equals(Object o) {
58         if (!(o instanceof FileEntry)) {
59             return false;
60         }
61 
62         var e = (FileEntry) o;
63         return Objects.equals(type, e.type) &&
64                Objects.equals(hash, e.hash) &&
65                Objects.equals(path, e.path);
66     }
67 
68     @Override
69     public String toString() {
70         return type.toString() + " " + hash.toString() + "\t" + path.toString();
71     }
72 }
