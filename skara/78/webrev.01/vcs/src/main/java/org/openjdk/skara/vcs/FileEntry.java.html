23 package org.openjdk.skara.vcs;
24 
25 import java.util.Objects;
26 import java.nio.file.Path;
27 
28 public class FileEntry {
29     private final Hash commit;
30     private final FileType type;
31     private final Hash hash;
32     private final Path path;
33 
34     public FileEntry(Hash commit, FileType type, Hash hash, Path path) {
35         this.commit = commit;
36         this.type = type;
37         this.hash = hash;
38         this.path = path;
39     }
40 
41     public Hash commit() {
42         return commit;
43     }
44 
45     public FileType type() {
46         return type;
47     }
48 
49     public Hash hash() {
50         return hash;
51     }
52 
53     public Path path() {
54         return path;
55     }
56 
57     @Override
58     public int hashCode() {
59         return Objects.hash(commit, type, hash, path);
60     }
61 
62     @Override
63     public boolean equals(Object o) {
64         if (!(o instanceof FileEntry)) {
65             return false;
66         }
67 
68         var e = (FileEntry) o;
69         return Objects.equals(commit, e.commit) &&
70                Objects.equals(type, e.type) &&
71                Objects.equals(hash, e.hash) &&
72                Objects.equals(path, e.path);
73     }
74 
75     @Override
76     public String toString() {
77         return type.toString() + " blob " + hash.toString() + "\t" + path.toString();
78     }
79 }
