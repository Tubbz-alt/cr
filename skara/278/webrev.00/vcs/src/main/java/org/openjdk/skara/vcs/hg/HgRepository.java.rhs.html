  48     private java.lang.Process start(String... cmd) throws IOException {
  49         return start(Arrays.asList(cmd));
  50     }

  52     private java.lang.Process start(List<String> cmd) throws IOException {
  53         log.fine("Executing " + String.join(" ", cmd));
  54         var pb = new ProcessBuilder(cmd);
  55         pb.directory(dir.toFile());
  56         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  57         pb.environment().put("HGRCPATH", "");
  58         pb.environment().put("HGPLAIN", "");
  59         return pb.start();
  60     }

  62     private static void stop(java.lang.Process p) throws IOException {
  63         if (p != null && p.isAlive()) {
  64             var stream = p.getInputStream();
  65             var read = 0;
  66             var buf = new byte[128];
  67             while (read != -1) {
  68                 read = stream.read(buf);
  69             }
  70             try {
  71                 p.waitFor();
  72             } catch (InterruptedException e) {
  73                 throw new IOException(e);
  74             }
  75         }
  76     }

  86     private static Execution capture(Path cwd, List<String> cmd) {
  87         return capture(cwd, cmd.toArray(new String[0]));

  89     private static Execution capture(Path cwd, String... cmd) {
  90         return Process.capture(cmd)
  91                       .environ("HGRCPATH", "")
  92                       .environ("HGPLAIN", "")
  93                       .workdir(cwd)
  94                       .execute();
  95     }

  97     private static Execution.Result await(Execution e) throws IOException {
  98         var result = e.await();
  99         if (result.status() != 0) {
 100             if (result.exception().isPresent()) {
 101                 throw new IOException("Unexpected exit code\n" + result, result.exception().get());
 102             } else {
 103                 throw new IOException("Unexpected exit code\n" + result);
 104             }
 105         }
 106         return result;
 107     }
 108 
 109     private static void await(java.lang.Process p) throws IOException {
 110         try {
 111             var res = p.waitFor();
 112             if (res != 0) {
 113                 throw new IOException("Unexpected exit code: " + res);
 114             }
 115         } catch (InterruptedException e) {
 116             throw new IOException(e);
 117         }
 118     }

 157     private void checkout(String ref, boolean force) throws IOException {
 158         var cmd = new ArrayList<String>();
 159         cmd.addAll(List.of("hg", "update"));
 160         if (!force) {
 161             cmd.add("--check");
 162         }
 163         cmd.add(ref);
 164         try (var p = capture(cmd)) {
 165             await(p);
 166         }
 167     }

 180     public Optional<Hash> resolve(String ref) throws IOException {
 181         try (var p = capture("hg", "log", "--rev=" + ref, "--template={node}\n")) {
 182             var res = p.await();
 183             if (res.status() == 0 && res.stdout().size() == 1) {
 184                 return Optional.of(new Hash(res.stdout().get(0)));
 185             }
 186             return Optional.empty();

 253     @Override
 254     public List<CommitMetadata> commitMetadata() throws IOException {
 255         var ext = Files.createTempFile("ext", ".py");
 256         copyResource(EXT_PY, ext);
 257 
 258         var p = start("hg", "--config", "extensions.dump=" + ext.toAbsolutePath().toString(), "metadata");
 259         var reader = new UnixStreamReader(p.getInputStream());
 260         var result = new ArrayList<CommitMetadata>();
 261 
 262         var line = reader.readLine();
 263         while (line != null) {
 264             result.add(HgCommitMetadata.read(reader));
 265             line = reader.readLine();
 266         }
 267 
 268         await(p);
 269         return result;
 270     }

 272     @Override
 273     public boolean isEmpty() throws IOException {
 274         var numBranches = branches().size();
 275         var numTags = tags().size();
 276 
 277         if (numBranches > 0 || numTags > 1) {
 278             return false;
 279         }
 280 
 281         var tip = resolve("tip");
 282         return tip.isEmpty() || tip.get().hex().equals("0".repeat(40));
 283     }

 292     @Override
 293     public void clean() throws IOException {
 294         try (var p = capture("hg", "merge", "--abort")) {
 295             p.await();
 296         }
 297 
 298         try (var p = capture("hg", "recover")) {
 299             p.await();
 300         }
 301 
 302         try (var p = capture("hg", "status", "--ignored", "--no-status")) {
 303             var root = root().toString();
 304             for (var filename : await(p).stdout()) {
 305                 Files.delete(Path.of(root, filename));
 306             }
 307         }
 308 
 309         try (var p = capture("hg", "status", "--unknown", "--no-status")) {
 310             var root = root().toString();
 311             for (var filename : await(p).stdout()) {
 312                 Files.delete(Path.of(root, filename));
 313             }
 314         }
 315 
 316         try (var p = capture("hg", "revert", "--no-backup", "--all")) {
 317             await(p);
 318         }
 319     }

 336     @Override
 337     public Hash fetch(URI uri, String refspec) throws IOException {
 338         var oldHeads = new HashSet<Hash>(heads());
 339 
 340         var cmd = new ArrayList<String>();
 341         cmd.add("hg");
 342         cmd.add("pull");
 343         if (refspec != null) {
 344             cmd.add("--rev");
 345             cmd.add(refspec);
 346         }
 347         if (uri != null) {
 348             cmd.add(uri.toString());
 349         }
 350         try (var p = capture(cmd)) {
 351             await(p);
 352         }
 353 
 354         var newHeads = new HashSet<Hash>(heads());
 355         newHeads.removeAll(oldHeads);
 356 
 357         if (newHeads.size() > 1) {
 358             throw new IllegalStateException("fetching multiple heads is not supported");
 359         } else if (newHeads.size() == 0) {
 360             // no new head was fetched, return current head
 361             return head();
 362         }
 363         return newHeads.iterator().next();
 364     }

 366     @Override
 367     public void fetchAll() throws IOException {
 368         var pullPaths = new ArrayList<URI>();
 369         try (var p = capture("hg", "paths")) {
 370             var res = await(p);
 371             for (var line : res.stdout()) {
 372                 var parts = line.split("=");
 373                 var name = parts[0].trim();
 374                 var uri = parts[1].trim();
 375                 if (!name.endsWith("-push")) {
 376                     pullPaths.add(URI.create(uri));
 377                 }
 378             }
 379         }
 380 
 381         for (var uri : pullPaths) {
 382             fetch(uri, null);
 383         }
 384     }

 452     private void export(String revset, Path to) throws IOException {
 453         var cmd = List.of("hg", "export", "--git", "--rev", revset);
 454         log.fine("Executing " + String.join(" ", cmd));
 455         var pb = new ProcessBuilder(cmd);
 456         pb.directory(dir.toFile());
 457         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
 458         pb.redirectOutput(to.toFile());
 459         pb.environment().put("HGRCPATH", "");
 460         pb.environment().put("HGPLAIN", "");
 461         var p = pb.start();
 462         try {
 463             await(p);
 464         } catch (Throwable t) {
 465             if (p.isAlive()) {
 466                 try {
 467                     p.waitFor();
 468                 } catch (InterruptedException e) {
 469                     throw new IOException(e);
 470                 }
 471             }
 472 
 473             throw new IOException(t);
 474         }
 475     }

 477     @Override
 478     public void squash(Hash h) throws IOException {
 479         var revset = ".:" + h.hex() + " and not .";
 480         var patch = Files.createTempFile("squash", ".patch");
 481         export(revset, patch);
 482 
 483         try (var p = capture("hg", "--config", "extensions.mq=", "strip", "--rev", revset)) {
 484             await(p);
 485         }
 486 
 487         try (var p = capture("hg", "import", "--no-commit", patch.toString())) {
 488             await(p);
 489         }
 490     }

 498     @Override
 499     public Hash commit(String message, String authorName, String authorEmail, ZonedDateTime authorDate)  throws IOException {
 500         var user = authorEmail == null ? authorName : authorName + " <" + authorEmail + ">";
 501         var cmd = new ArrayList<String>();
 502         cmd.addAll(List.of("hg", "commit", "--message=" + message, "--user=" + user));
 503         if (authorDate != null) {
 504             var formatter = DateTimeFormatter.ISO_OFFSET_DATE_TIME;
 505             cmd.add("--date=" + authorDate.format(formatter));
 506         }
 507         try (var p = capture(cmd)) {
 508             await(p);
 509         }
 510         return resolve("tip").orElseThrow(() -> new IOException("Could not resolve 'tip'"));
 511     }

 643     @Override
 644     public Optional<byte[]> show(Path path, Hash hash) throws IOException {
 645         var output = Files.createTempFile("hg-cat-rev-" + hash.abbreviate(), ".bin");
 646         try (var p = capture("hg", "cat", "--output=" + output, "--rev=" + hash.hex(), path.toString())) {
 647             var res = p.await();
 648             if (res.status() == 0 && Files.exists(output)) {
 649                 var bytes = Files.readAllBytes(output);
 650                 Files.delete(output);
 651                 return Optional.of(bytes);
 652             }
 653 
 654             if (Files.exists(output)) {
 655                 Files.delete(output);
 656             }
 657             return Optional.empty();
 658         }
 659     }

 661     private List<FileEntry> allFiles(Hash hash, List<Path> paths) throws IOException {
 662         var ext = Files.createTempFile("ext", ".py");
 663         copyResource(EXT_PY, ext);
 664 
 665         var include = new HashSet<>(paths);
 666 
 667         try (var p = capture("hg", "--config", "extensions.ls-tree=" + ext, "ls-tree", hash.hex())) {
 668             var res = await(p);
 669             var entries = new ArrayList<FileEntry>();
 670             for (var line : res.stdout()) {
 671                 var parts = line.split("\t");
 672                 var metadata = parts[0].split(" ");
 673                 var path = Path.of(parts[1]);
 674                 if (include.isEmpty() || include.contains(path)) {
 675                     var entry = new FileEntry(hash,
 676                                               FileType.fromOctal(metadata[0]),
 677                                               new Hash(metadata[2]),
 678                                               path);
 679                     entries.add(entry);
 680                 }
 681             }
 682             return entries;
 683         }
 684     }

 686     @Override
 687     public List<FileEntry> files(Hash hash, List<Path> paths) throws IOException {
 688         if (paths.isEmpty()) {
 689             return allFiles(hash, paths);
 690         }
 691 
 692         var entries = new ArrayList<FileEntry>();
 693         var batchSize = 64;
 694         var start = 0;
 695         while (start < paths.size()) {
 696             var end = start + batchSize;
 697             if (end > paths.size()) {
 698                 end = paths.size();
 699             }
 700             entries.addAll(allFiles(hash, paths.subList(start, end)));
 701             start = end;
 702         }
 703         return entries;
 704     }

 877     @Override
 878     public void setPaths(String remote, String pullPath, String pushPath) throws IOException {
 879         var hgrc = Path.of(root().toString(), ".hg", "hgrc");
 880         if (!Files.exists(hgrc)) {
 881             Files.createFile(hgrc);
 882         }
 883 
 884         var lines = Files.readAllLines(hgrc);
 885         var newLines = new ArrayList<String>();
 886 
 887         var isInPathsSection = false;
 888         var hasPathsSection = false;
 889         for (var line : lines) {
 890             var isSectionHeader = line.startsWith("[") && line.endsWith("]");
 891             if (isSectionHeader && !isInPathsSection) {
 892                 isInPathsSection = line.equals("[paths]");
 893                 if (isInPathsSection) {
 894                     newLines.add(line);
 895                     newLines.add(remote + " = " + (pullPath == null ? "" : pullPath));
 896                     newLines.add(remote + "-push = " + (pushPath == null ? "" : pushPath));
 897                     hasPathsSection = true;
 898                     continue;
 899                 }
 900             }
 901 
 902             if (isInPathsSection && line.startsWith(remote)) {
 903                 if (line.startsWith(remote + "-push")) {
 904                     // skip
 905                 } else if (line.startsWith(remote + ":pushurl")) {
 906                     // skip
 907                 } else if (line.startsWith(remote + " ") || line.startsWith(remote + "=")) {
 908                     // skip
 909                 } else {
 910                     newLines.add(line);
 911                 }
 912             } else {
 913                 newLines.add(line);
 914             }
 915         }
 916 
 917         Files.write(hgrc, newLines, StandardOpenOption.WRITE, StandardOpenOption.TRUNCATE_EXISTING);
 918         if (!hasPathsSection) {
 919             var section = List.of("[paths]",
 920                                   remote + " = " + (pullPath == null ? "" : pullPath),
 921                                   remote + "-push = " + (pushPath == null ? "" : pushPath));
 922             Files.write(hgrc, section, StandardOpenOption.WRITE, StandardOpenOption.APPEND);
 923         }
 924     }

1003     private void batch(Operation op, List<Path> args) throws IOException {
1004         var batchSize = 64;
1005         var start = 0;
1006         while (start < args.size()) {
1007             var end = start + batchSize;
1008             if (end > args.size()) {
1009                 end = args.size();
1010             }
1011             op.execute(args.subList(start, end));
1012             start = end;
1013         }
1014     }

1159         var hgSub = root().resolve(".hgsub");
1160         var hgSubState = root().resolve(".hgsubstate");
1161         if (!(Files.exists(hgSub) && Files.exists(hgSubState))) {

1166         for (var line : Files.readAllLines(hgSub)) {
1167             var parts = line.split("=");
1168             var path = parts[0].trim();
1169             var url = parts[1].trim();
1170             urls.put(path, url);
1171         }
1172 
1173         var hashes = new HashMap<String, String>();
1174         for (var line : Files.readAllLines(hgSubState)) {
1175             var parts = line.split(" ");
1176             var hash = parts[0];
1177             var path = parts[1];
1178             hashes.put(path, hash);
1179         }
1180 
1181         var modules = new ArrayList<Submodule>();
1182         for (var path : urls.keySet()) {
1183             var url = urls.get(path);
1184             var hash = hashes.get(path);
1185             modules.add(new Submodule(new Hash(hash), Path.of(path), url));
1186         }

1193         var hgtags = root().resolve(".hgtags");
1194         if (!Files.exists(hgtags)) {
1195             return Optional.empty();
1196         }
1197         try (var p = capture("hg", "annotate", hgtags.toString())) {
1198             var reversed = new ArrayList<>(await(p).stdout());
1199             Collections.reverse(reversed);
1200             for (var line : reversed) {
1201                 var parts = line.split(" ");
1202                 var tagName = parts[2];
1203                 if (tagName.equals(tag.name())) {
1204                     var target = new Hash(parts[1]);
1205                     var rev = parts[0].substring(0, parts[0].length() - 1).trim(); // skip last ':' and ev. whitespace
1206                     var hash = resolve(rev).orElseThrow(IOException::new);
1207                     var commit = lookup(hash).orElseThrow(IOException::new);
1208                     var message = String.join("\n", commit.message()) + "\n";
1209                     return Optional.of(new Tag.Annotated(tagName, target, commit.author(), commit.date(), message));
1210                 }
1211             }
1212         }
1213         return Optional.empty();
