23 package org.openjdk.skara.vcs;
24 
25 import java.io.*;
26 import java.util.*;
27 import java.util.stream.*;
28 
29 public interface Commits extends Closeable, Iterable<Commit> {
30     default List<Commit> asList() throws IOException {
31         var result = new ArrayList<Commit>();
32         for (var commit : this) {
33             result.add(commit);
34         }
35 
36         close();
37 
38         return result;
39     }
40 
41     default Stream<Commit> stream() {
42         return StreamSupport.stream(spliterator(), false);
43     }
44 }
