23 package org.openjdk.skara.bots.notify;
24 
25 import org.openjdk.skara.forge.HostedRepository;
26 import org.openjdk.skara.vcs.*;
27 import org.openjdk.skara.vcs.openjdk.OpenJDKTag;
28 
29 import java.util.List;
30 
31 public interface UpdateConsumer {
32     void handleCommits(HostedRepository repository, List<Commit> commits, Branch branch);
33     void handleOpenJDKTagCommits(HostedRepository repository, List<Commit> commits, OpenJDKTag tag, Tag.Annotated annotated);
34     void handleTagCommit(HostedRepository repository, Commit commit, Tag tag, Tag.Annotated annotation);
35     void handleNewBranch(HostedRepository repository, List<Commit> commits, Branch parent, Branch branch);
36 
37     default void handleOpenJDKTagCommits(HostedRepository repository, List<Commit> commits, OpenJDKTag tag) {
38         handleOpenJDKTagCommits(repository, commits, tag, null);
39     }
40     default void handleTagCommit(HostedRepository repository, Commit commit, Tag tag) {
41         handleTagCommit(repository, commit, tag, null);
42     }
43 }
