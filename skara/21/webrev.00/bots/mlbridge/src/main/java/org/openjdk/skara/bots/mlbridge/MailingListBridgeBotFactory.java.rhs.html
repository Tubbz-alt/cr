23 package org.openjdk.skara.bots.mlbridge;
24 
25 import org.openjdk.skara.bot.*;
26 import org.openjdk.skara.email.EmailAddress;
27 import org.openjdk.skara.host.HostedRepository;
28 import org.openjdk.skara.host.network.URIBuilder;
29 import org.openjdk.skara.json.*;
30 import org.openjdk.skara.mailinglist.MailingListServerFactory;
31 
32 import java.nio.file.Path;
33 import java.util.*;
34 import java.util.regex.Pattern;
35 import java.util.stream.Collectors;
36 
37 public class MailingListBridgeBotFactory implements BotFactory {
38     @Override
39     public String name() {
40         return "mlbridge";
41     }
42 
43     @Override
44     public List<Bot> create(BotConfiguration configuration) {
45         var ret = new ArrayList<Bot>();
46         var specific = configuration.specific();
47 
48         var from = EmailAddress.from(specific.get("name").asString(), specific.get("mail").asString());
49         var ignoredUsers = specific.get("ignored").stream()
50                                    .map(JSONValue::asString)
51                                    .collect(Collectors.toSet());
52         var listArchive = URIBuilder.base(specific.get("server").get("archive").asString()).build();
53         var listSmtp = specific.get("server").get("smtp").asString();
54 
55         var webrevRepo = specific.get("webrevs").get("repository").asString();
56         var webrevRef = specific.get("webrevs").get("ref").asString();
57         var webrevWeb = specific.get("webrevs").get("web").asString();
58 
59         var allListNames = new HashSet<EmailAddress>();
60         var allRepositories = new HashSet<HostedRepository>();
61 
62         var readyLabels = specific.get("ready").get("labels").stream()
63                 .map(JSONValue::asString)
64                 .collect(Collectors.toSet());
65         var readyComments = specific.get("ready").get("comments").stream()
66                 .map(JSONValue::asObject)
67                 .collect(Collectors.toMap(obj -> obj.get("user").asString(),
68                                           obj -> Pattern.compile(obj.get("pattern").asString())));
69 
70         for (var repoConfig : specific.get("repositories").asArray()) {
71             var repo = repoConfig.get("repository").asString();
72             var archive = repoConfig.get("archive").asString();
73             var list = EmailAddress.parse(repoConfig.get("list").asString());
74             var bot = new MailingListBridgeBot(from, configuration.repository(repo), configuration.repository(archive),
75                                                list, ignoredUsers, listArchive, listSmtp,
76                                                configuration.repository(webrevRepo), webrevRef, Path.of(repo),
77                                                URIBuilder.base(webrevWeb).build(), readyLabels, readyComments);
78             ret.add(bot);
79 
80             allListNames.add(list);
81             allRepositories.add(configuration.repository(repo));
82         }
83 
84         var mailmanServer = MailingListServerFactory.createMailmanServer(listArchive, listSmtp);
85         var allLists = allListNames.stream()
86                                    .map(name -> mailmanServer.getList(name.toString()))
87                                    .collect(Collectors.toSet());
88 
89         var bot = new MailingListArchiveReaderBot(from, allLists, allRepositories);
90         ret.add(bot);
91 
92         return ret;
93     }
94 }
