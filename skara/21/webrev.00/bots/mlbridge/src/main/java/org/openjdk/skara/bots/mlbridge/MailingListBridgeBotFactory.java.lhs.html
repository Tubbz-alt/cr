23 package org.openjdk.skara.bots.mlbridge;
24 
25 import org.openjdk.skara.bot.*;
26 import org.openjdk.skara.email.EmailAddress;
27 import org.openjdk.skara.host.HostedRepository;
28 import org.openjdk.skara.host.network.URIBuilder;
29 import org.openjdk.skara.json.JSONValue;
30 import org.openjdk.skara.mailinglist.MailingListServerFactory;
31 
32 import java.nio.file.Path;

33 import java.util.*;
34 import java.util.stream.Collectors;
35 
36 public class MailingListBridgeBotFactory implements BotFactory {
37     @Override
38     public String name() {
39         return "mlbridge";
40     }
41 
42     @Override
43     public List<Bot> create(BotConfiguration configuration) {
44         var ret = new ArrayList<Bot>();
45         var specific = configuration.specific();
46 
47         var from = EmailAddress.from(specific.get("name").asString(), specific.get("mail").asString());
48         var ignoredUsers = specific.get("ignored").stream()
49                                    .map(JSONValue::asString)
50                                    .collect(Collectors.toSet());
51         var listArchive = URIBuilder.base(specific.get("server").get("archive").asString()).build();
52         var listSmtp = specific.get("server").get("smtp").asString();
53 
54         var webrevRepo = specific.get("webrevs").get("repository").asString();
55         var webrevRef = specific.get("webrevs").get("ref").asString();
56         var webrevWeb = specific.get("webrevs").get("web").asString();
57 
58         var allListNames = new HashSet<EmailAddress>();
59         var allRepositories = new HashSet<HostedRepository>();








60 
61         for (var repoConfig : specific.get("repositories").asArray()) {
62             var repo = repoConfig.get("repository").asString();
63             var archive = repoConfig.get("archive").asString();
64             var list = EmailAddress.parse(repoConfig.get("list").asString());
65             var bot = new MailingListBridgeBot(from, configuration.repository(repo), configuration.repository(archive),
66                                                list, ignoredUsers, listArchive, listSmtp,
67                                                configuration.repository(webrevRepo), webrevRef, Path.of(repo),
68                                                URIBuilder.base(webrevWeb).build());
69             ret.add(bot);
70 
71             allListNames.add(list);
72             allRepositories.add(configuration.repository(repo));
73         }
74 
75         var mailmanServer = MailingListServerFactory.createMailmanServer(listArchive, listSmtp);
76         var allLists = allListNames.stream()
77                                    .map(name -> mailmanServer.getList(name.toString()))
78                                    .collect(Collectors.toSet());
79 
80         var bot = new MailingListArchiveReaderBot(from, allLists, allRepositories);
81         ret.add(bot);
82 
83         return ret;
84     }
85 }
