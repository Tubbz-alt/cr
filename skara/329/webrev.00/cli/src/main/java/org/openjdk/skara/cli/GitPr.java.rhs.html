  86     private static Optional<Issue> getIssue(Commit commit, String project) throws IOException {
  87         var message = CommitMessageParsers.v1.parse(commit.message());
  88         var issues = message.issues();
  89         if (issues.isEmpty()) {
  90             return getIssue(message.title(), project);
  91         } else if (issues.size() == 1) {
  92             var issue = issues.get(0);
  93             return getIssue(issue.id(), project);
  94         }
  95         return Optional.empty();
  96     }
  97 
  98     private static Optional<Issue> getIssue(Branch b, String project) throws IOException {
  99         return getIssue(b.name(), project);
 100     }

 102     private static Optional<Issue> getIssue(String s, String project) throws IOException {
 103         var issueIdPattern = Pattern.compile("([A-Za-z][A-Za-z0-9]+)?-([0-9]+)");
 104         var m = issueIdPattern.matcher(s);
 105         if (m.matches()) {
 106             var id = m.group(2);
 107             if (project == null) {
 108                 project = m.group(1);
 109             }
 110             var issueTracker = IssueTracker.from("jira", URI.create("https://bugs.openjdk.java.net"));
 111             return issueTracker.project(project).issue(id);
 112         }
 113 
 114         return Optional.empty();
 115     }

 117     private static void await(Process p) throws IOException {
 118         try {
 119             var res = p.waitFor();
 120             if (res != 0) {
 121                 throw new IOException("Unexpected exit code " + res);
 122             }
 123         } catch (InterruptedException e) {
 124             throw new IOException(e);
 125         }
 126     }

 128     private static boolean spawnEditor(ReadOnlyRepository repo, Path file) throws IOException {
 129         String editor = null;
 130         var lines = repo.config("core.editor");
 131         if (lines.size() == 1) {
 132             editor = lines.get(0);
 133         }
 134         if (editor == null) {
 135             editor = System.getenv("GIT_EDITOR");
 136         }
 137         if (editor == null) {
 138             editor = System.getenv("EDITOR");
 139         }
 140         if (editor == null) {
 141             editor = System.getenv("VISUAL");
 142         }
 143         if (editor == null) {
 144             editor = "vi";
 145         }
 146 
 147         var pb = new ProcessBuilder(editor, file.toString());
 148         pb.inheritIO();
 149         var p = pb.start();
 150         try {
 151             return p.waitFor() == 0;
 152         } catch (InterruptedException e) {
 153             throw new IOException(e);
 154         }
 155     }

 157     private static String projectName(URI uri) {
 158         var name = uri.getPath().toString().substring(1);
 159         if (name.endsWith(".git")) {
 160             name = name.substring(0, name.length() - ".git".length());
 161         }
 162         return name;
 163     }

 165     private static HostedRepository getHostedRepositoryFor(URI uri, ReadOnlyRepository repo, Forge host) throws IOException {
 166         HostedRepository targetRepo = null;
 167 
 168         try {
 169             var upstream = Remote.toWebURI(repo.pullPath("upstream"));
 170             targetRepo = host.repository(projectName(upstream)).orElse(null);
 171         } catch (IOException e) {
 172             // do nothing
 173         }
 174 
 175         if (targetRepo == null) {
 176             var remoteRepo = host.repository(projectName(uri)).orElseThrow(() ->
 177                     new IOException("Could not find repository at: " + uri.toString())
 178             );
 179             var parentRepo = remoteRepo.parent();
 180             targetRepo = parentRepo.isPresent() ? parentRepo.get() : remoteRepo;
 181         }
 182 
 183         return targetRepo;
 184     }

 186     private static PullRequest getPullRequest(URI uri, ReadOnlyRepository repo, Forge host, String prId) throws IOException {
 187         var pr = getHostedRepositoryFor(uri, repo, host).pullRequest(prId);
 188         if (pr == null) {
 189             exit("error: could not fetch PR information");
 190         }
 191 
 192         return pr;
 193     }

 195     private static void show(String ref, Hash hash) throws IOException {
 196         show(ref, hash, null);

 198     private static void show(String ref, Hash hash, Path dir) throws IOException {
 199         var pb = new ProcessBuilder("git", "diff", "--binary",
 200                                                    "--patch",
 201                                                    "--find-renames=50%",
 202                                                    "--find-copies=50%",
 203                                                    "--find-copies-harder",
 204                                                    "--abbrev",
 205                                                    ref + "..." + hash.hex());
 206         if (dir != null) {
 207             pb.directory(dir.toFile());
 208         }
 209         pb.inheritIO();
 210         await(pb.start());
 211     }

 225     private static List<String> hgTags() throws IOException, InterruptedException {
 226         var pb = new ProcessBuilder("hg", "tags", "--quiet");
 227         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
 228         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 229         var p = pb.start();
 230         var bytes = p.getInputStream().readAllBytes();
 231         var exited = p.waitFor(1, TimeUnit.MINUTES);
 232         var exitValue = p.exitValue();
 233         if (!exited || exitValue != 0) {
 234             throw new IOException("'hg tags' exited with value: " + exitValue);
 235         }
 236 
 237         return Arrays.asList(new String(bytes, StandardCharsets.UTF_8).split("\n"));
 238     }

 240     private static String hgResolve(String ref) throws IOException, InterruptedException {
 241         var pb = new ProcessBuilder("hg", "log", "-r", ref, "--template", "{node}");
 242         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
 243         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 244         var p = pb.start();
 245         var bytes = p.getInputStream().readAllBytes();
 246         var exited = p.waitFor(1, TimeUnit.MINUTES);
 247         var exitValue = p.exitValue();
 248         if (!exited || exitValue != 0) {
 249             throw new IOException("'hg log' exited with value: " + exitValue);
 250         }
 251 
 252         return new String(bytes, StandardCharsets.UTF_8);
 253     }

 259     private static Path diff(String ref, Hash hash, Path dir) throws IOException {
 260         var patch = Files.createTempFile(hash.hex(), ".patch");
 261         var pb = new ProcessBuilder("git", "diff", "--binary",
 262                                                    "--patch",
 263                                                    "--find-renames=50%",
 264                                                    "--find-copies=50%",
 265                                                    "--find-copies-harder",
 266                                                    "--abbrev",
 267                                                    ref + "..." + hash.hex());
 268         if (dir != null) {
 269             pb.directory(dir.toFile());
 270         }
 271         pb.redirectOutput(patch.toFile());
 272         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 273         await(pb.start());
 274         return patch;
 275     }

 287     public static void main(String[] args) throws IOException, InterruptedException {
 288         var flags = List.of(
 289             Option.shortcut("u")
 290                   .fullname("username")
 291                   .describe("NAME")
 292                   .helptext("Username on host")
 293                   .optional(),
 294             Option.shortcut("r")
 295                   .fullname("remote")
 296                   .describe("NAME")
 297                   .helptext("Name of remote, defaults to 'origin'")
 298                   .optional(),
 299             Option.shortcut("b")
 300                   .fullname("branch")
 301                   .describe("NAME")
 302                   .helptext("Name of target branch, defaults to 'master'")
 303                   .optional(),
 304             Option.shortcut("")
 305                   .fullname("authors")
 306                   .describe("LIST")
 307                   .helptext("Comma separated list of authors")
 308                   .optional(),
 309             Option.shortcut("")
 310                   .fullname("assignees")
 311                   .describe("LIST")
 312                   .helptext("Comma separated list of assignees")
 313                   .optional(),
 314             Option.shortcut("")
 315                   .fullname("labels")
 316                   .describe("LIST")
 317                   .helptext("Comma separated list of labels")
 318                   .optional(),
 319             Option.shortcut("")
 320                   .fullname("columns")
 321                   .describe("id,title,author,assignees,labels")
 322                   .helptext("Comma separated list of columns to show")
 323                   .optional(),
 324             Switch.shortcut("")
 325                   .fullname("no-decoration")
 326                   .helptext("Hide any decorations when listing PRs")
 327                   .optional(),
 328             Switch.shortcut("")
 329                   .fullname("ignore-workspace")
 330                   .helptext("Ignore local changes in worktree and staging area when creating pull request")
 331                   .optional(),
 332             Switch.shortcut("")
 333                   .fullname("ignore-local-commits")
 334                   .helptext("Ignore local commits not pushed when creating pull request")
 335                   .optional(),
 336             Switch.shortcut("")
 337                   .fullname("publish")
 338                   .helptext("Publish the local branch before creating the pull request")
 339                   .optional(),
 340             Switch.shortcut("")
 341                   .fullname("no-token")
 342                   .helptext("Do not use a personal access token (PAT). Only works for read-only operations.")
 343                   .optional(),
 344             Switch.shortcut("")
 345                   .fullname("mercurial")
 346                   .helptext("Force use of Mercurial (hg)")
 347                   .optional(),
 348             Switch.shortcut("")
 349                   .fullname("verbose")
 350                   .helptext("Turn on verbose output")
 351                   .optional(),
 352             Switch.shortcut("")
 353                   .fullname("debug")
 354                   .helptext("Turn on debugging output")
 355                   .optional(),
 356             Switch.shortcut("")
 357                   .fullname("version")
 358                   .helptext("Print the version of this tool")
 359                   .optional());
 360 
 361         var inputs = List.of(
 362             Input.position(0)
 363                  .describe("list|fetch|show|checkout|apply|integrate|approve|create|close|update|test")
 364                  .singular()
 365                  .required(),
 366             Input.position(1)
 367                  .describe("ID")
 368                  .singular()
 369                  .optional()
 370         );
 371 
 372         var parser = new ArgumentParser("git-pr", flags, inputs);
 373         var arguments = parser.parse(args);
 374 
 375         if (arguments.contains("version")) {
 376             System.out.println("git-pr version: " + Version.fromManifest().orElse("unknown"));
 377             System.exit(0);
 378         }
 379 
 380         if (arguments.contains("verbose") || arguments.contains("debug")) {
 381             var level = arguments.contains("debug") ? Level.FINER : Level.FINE;
 382             Logging.setup(level);
 383         }
 384 
 385         HttpProxy.setup();
 386 
 387         var isMercurial = arguments.contains("mercurial");
 388         var cwd = Path.of("").toAbsolutePath();
 389         var repo = Repository.get(cwd).orElseThrow(() -> new IOException("no git repository found at " + cwd.toString()));
 390         var remote = arguments.get("remote").orString(isMercurial ? "default" : "origin");
 391         var remotePullPath = repo.pullPath(remote);
 392         var username = arguments.contains("username") ? arguments.get("username").asString() : null;
 393         var token = isMercurial ? System.getenv("HG_TOKEN") :  System.getenv("GIT_TOKEN");
 394         var uri = Remote.toWebURI(remotePullPath);
 395         var shouldUseToken = !arguments.contains("no-token");
 396         var credentials = !shouldUseToken ?
 397             null :
 398             GitCredentials.fill(uri.getHost(), uri.getPath(), username, token, uri.getScheme());
 399         var forgeURI = URI.create(uri.getScheme() + "://" + uri.getHost());
 400         var forge = credentials == null ?
 401             Forge.from(forgeURI) :
 402             Forge.from(forgeURI, new Credential(credentials.username(), credentials.password()));
 403         if (forge.isEmpty()) {
 404             if (!shouldUseToken) {
 405                 if (arguments.contains("verbose")) {
 406                     System.err.println("");
 407                 }
 408                 System.err.println("warning: using git-pr with --no-token may result in rate limiting from " + forgeURI);
 409                 if (!arguments.contains("verbose")) {
 410                     System.err.println("         Re-run git-pr with --verbose to see if you are being rate limited");
 411                     System.err.println("");
 412                 }
 413             }
 414             exit("error: failed to connect to host: " + forgeURI);
 415         }
 416         var host = forge.get();
 417 
 418         var action = arguments.at(0).asString();
 419         if (!shouldUseToken &&
 420             !List.of("list", "fetch", "show", "checkout", "apply").contains(action)) {
 421             System.err.println("error: --no-token can only be used with read-only operations");
 422             System.exit(1);
 423         }
 424 
 425         if (action.equals("create")) {
 426             if (isMercurial) {
 427                 var currentBookmark = repo.currentBookmark();
 428                 if (!currentBookmark.isPresent()) {
 429                     System.err.println("error: no bookmark is active, you must be on an active bookmark");
 430                     System.err.println("");
 431                     System.err.println("To create a bookmark and activate it, run:");
 432                     System.err.println("");
 433                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
 434                     System.err.println("");
 435                     System.exit(1);
 436                 }
 437 
 438                 var bookmark = currentBookmark.get();
 439                 if (bookmark.equals(new Bookmark("master"))) {
 440                     System.err.println("error: you should not create pull requests from the 'master' bookmark");
 441                     System.err.println("To create a bookmark and activate it, run:");
 442                     System.err.println("");
 443                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
 444                     System.err.println("");
 445                     System.exit(1);
 446                 }
 447 
 448                 var tags = hgTags();
 449                 var upstreams = tags.stream()
 450                                     .filter(t -> t.endsWith(bookmark.name()))
 451                                     .collect(Collectors.toList());
 452                 if (upstreams.isEmpty()) {
 453                     System.err.println("error: there is no remote branch for the local bookmark '" + bookmark.name() + "'");
 454                     System.err.println("");
 455                     System.err.println("To create a remote branch and push the commits for your local branch, run:");
 456                     System.err.println("");
 457                     System.err.println("    hg push --bookmark " + bookmark.name());
 458                     System.err.println("");
 459                     System.exit(1);
 460                 }
 461 
 462                 var tagsAndHashes = new HashMap<String, String>();
 463                 for (var tag : tags) {
 464                     tagsAndHashes.put(tag, hgResolve(tag));
 465                 }
 466                 var bookmarkHash = hgResolve(bookmark.name());
 467                 if (!tagsAndHashes.containsValue(bookmarkHash)) {
 468                     System.err.println("error: there are local commits on bookmark '" + bookmark.name() + "' not present in a remote repository");
 469                     System.err.println("");
 470 
 471                     if (upstreams.size() == 1) {
 472                         System.err.println("To push the local commits to the remote repository, run:");
 473                         System.err.println("");
 474                         System.err.println("    hg push --bookmark " + bookmark.name() + " " + upstreams.get(0));
 475                         System.err.println("");
 476                     } else {
 477                         System.err.println("The following paths contains the " + bookmark.name() + " bookmark:");
 478                         System.err.println("");
 479                         for (var upstream : upstreams) {
 480                             System.err.println("- " + upstream.replace("/" + bookmark.name(), ""));
 481                         }
 482                         System.err.println("");
 483                         System.err.println("To push the local commits to a remote repository, run:");
 484                         System.err.println("");
 485                         System.err.println("    hg push --bookmark " + bookmark.name() + " <PATH>");
 486                         System.err.println("");
 487                     }
 488                     System.exit(1);
 489                 }
 490 
 491                 var targetBranch = arguments.get("branch").orString("master");
 492                 var targetHash = hgResolve(targetBranch);
 493                 var commits = repo.commits(targetHash + ".." + bookmarkHash + "-" + targetHash).asList();
 494                 if (commits.isEmpty()) {
 495                     System.err.println("error: no difference between bookmarks " + targetBranch + " and " + bookmark.name());
 496                     System.err.println("       Cannot create an empty pull request, have you committed?");
 497                     System.exit(1);
 498                 }
 499 
 500                 var diff = repo.diff(repo.head());
 501                 if (!diff.patches().isEmpty()) {
 502                     System.err.println("error: there are uncommitted changes in your working tree:");
 503                     System.err.println("");
 504                     for (var patch : diff.patches()) {
 505                         var path = patch.target().path().isPresent() ?
 506                             patch.target().path().get() : patch.source().path().get();
 507                         System.err.println("    " + patch.status().toString() + " " + path.toString());
 508                     }
 509                     System.err.println("");
 510                     System.err.println("If these changes are meant to be part of the pull request, run:");
 511                     System.err.println("");
 512                     System.err.println("    hg commit --amend");
 513                     System.err.println("    hg git-cleanup");
 514                     System.err.println("    hg push --bookmark " + bookmark.name() + " <PATH>");
 515                     System.err.println("    hg gimport");
 516                     System.err.println("");
 517                     System.err.println("If these changes are *not* meant to be part of the pull request, run:");
 518                     System.err.println("");
 519                     System.err.println("    hg shelve");
 520                     System.err.println("");
 521                     System.err.println("(You can later restore the changes by running: hg unshelve)");
 522                     System.exit(1);
 523                 }
 524 
 525                 var remoteRepo = host.repository(projectName(uri)).orElseThrow(() ->
 526                         new IOException("Could not find repository at " + uri.toString())
 527                 );
 528                 if (token == null) {
 529                     GitCredentials.approve(credentials);
 530                 }
 531                 var parentRepo = remoteRepo.parent().orElseThrow(() ->
 532                         new IOException("error: remote repository " + remotePullPath + " is not a fork of any repository"));
 533 
 534                 var file = Files.createTempFile("PULL_REQUEST_", ".txt");
 535                 if (commits.size() == 1) {
 536                     var commit = commits.get(0);
 537                     var message = CommitMessageParsers.v1.parse(commit.message());
 538                     Files.writeString(file, message.title() + "\n");
 539                     if (!message.summaries().isEmpty()) {
 540                         Files.write(file, message.summaries(), StandardOpenOption.APPEND);
 541                     }
 542                     if (!message.additional().isEmpty()) {
 543                         Files.write(file, message.additional(), StandardOpenOption.APPEND);
 544                     }
 545                 } else {
 546                     Files.write(file, List.of(""));
 547                 }
 548                 Files.write(file, List.of(
 549                     "# Please enter the pull request message for your changes. Lines starting",
 550                     "# with '#' will be ignored, and an empty message aborts the pull request.",
 551                     "# The first line will be considered the subject, use a blank line to separate",
 552                     "# the subject from the body.",
 553                     "#",
 554                     "# Commits to be included from branch '" + bookmark.name() + "'"
 555                     ),
 556                     StandardOpenOption.APPEND
 557                 );
 558                 for (var commit : commits) {
 559                     var desc = commit.hash().abbreviate() + ": " + commit.message().get(0);
 560                     Files.writeString(file, "# - " + desc + "\n", StandardOpenOption.APPEND);
 561                 }
 562                 Files.writeString(file, "#\n", StandardOpenOption.APPEND);
 563                 Files.writeString(file, "# Target repository: " + remotePullPath + "\n", StandardOpenOption.APPEND);
 564                 Files.writeString(file, "# Target branch: " + targetBranch + "\n", StandardOpenOption.APPEND);
 565                 var success = spawnEditor(repo, file);
 566                 if (!success) {
 567                     System.err.println("error: editor exited with non-zero status code, aborting");
 568                     System.exit(1);
 569                 }
 570                 var lines = Files.readAllLines(file)
 571                                  .stream()
 572                                  .filter(l -> !l.startsWith("#"))
 573                                  .collect(Collectors.toList());
 574                 var isEmpty = lines.stream().allMatch(String::isEmpty);
 575                 if (isEmpty) {
 576                     System.err.println("error: no message present, aborting");
 577                     System.exit(1);
 578                 }
 579 
 580                 var title = lines.get(0);
 581                 List<String> body = null;
 582                 if (lines.size() > 1) {
 583                     body = lines.subList(1, lines.size())
 584                                 .stream()
 585                                 .dropWhile(String::isEmpty)
 586                                 .collect(Collectors.toList());
 587                 } else {
 588                     body = Collections.emptyList();
 589                 }
 590 
 591                 var pr = remoteRepo.createPullRequest(parentRepo, targetBranch, bookmark.name(), title, body);
 592                 if (arguments.contains("assignees")) {
 593                     var usernames = Arrays.asList(arguments.get("assignees").asString().split(","));
 594                     var assignees = usernames.stream()
 595                                              .map(u -> host.user(u))
 596                                              .collect(Collectors.toList());
 597                     pr.setAssignees(assignees);
 598                 }
 599                 System.out.println(pr.webUrl().toString());
 600                 Files.deleteIfExists(file);
 601 
 602                 System.exit(0);
 603             }
 604             var currentBranch = repo.currentBranch().orElseGet(() -> {
 605                     System.err.println("error: the repository is in a detached HEAD state");
 606                     System.exit(1);
 607                     return null;
 608             });
 609             if (currentBranch.equals(repo.defaultBranch())) {
 610                 System.err.println("error: you should not create pull requests from the 'master' branch");
 611                 System.err.println("");
 612                 System.err.println("To create a local branch for your changes and restore the 'master' branch, run:");
 613                 System.err.println("");
 614                 System.err.println("    git checkout -b NAME-FOR-YOUR-LOCAL-BRANCH");
 615                 System.err.println("    git branch --force master origin/master");
 616                 System.err.println("");
 617                 System.exit(1);
 618             }
