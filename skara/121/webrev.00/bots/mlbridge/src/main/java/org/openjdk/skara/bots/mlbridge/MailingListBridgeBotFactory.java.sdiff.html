 32 import java.nio.file.Path;
 33 import java.util.*;
 34 import java.util.regex.Pattern;
 35 import java.util.stream.Collectors;

 37 public class MailingListBridgeBotFactory implements BotFactory {
 38     @Override
 39     public String name() {
 40         return "mlbridge";
 41     }
 42 
 43     @Override
 44     public List<Bot> create(BotConfiguration configuration) {
 45         var ret = new ArrayList<Bot>();
 46         var specific = configuration.specific();
 47 
 48         var from = EmailAddress.from(specific.get("name").asString(), specific.get("mail").asString());
 49         var ignoredUsers = specific.get("ignored").stream()
 50                                    .map(JSONValue::asString)
 51                                    .collect(Collectors.toSet());




 52         var listArchive = URIBuilder.base(specific.get("server").get("archive").asString()).build();
 53         var listSmtp = specific.get("server").get("smtp").asString();
 54 
 55         var webrevRepo = configuration.repository(specific.get("webrevs").get("repository").asString());
 56         var webrevRef = configuration.repositoryRef(specific.get("webrevs").get("repository").asString());
 57         var webrevWeb = specific.get("webrevs").get("web").asString();
 58 
 59         var archiveRepo = configuration.repository(specific.get("archive").asString());
 60 
 61         var allListNames = new HashSet<EmailAddress>();
 62         var allRepositories = new HashSet<HostedRepository>();
 63 
 64         var readyLabels = specific.get("ready").get("labels").stream()
 65                 .map(JSONValue::asString)
 66                 .collect(Collectors.toSet());
 67         var readyComments = specific.get("ready").get("comments").stream()
 68                 .map(JSONValue::asObject)
 69                 .collect(Collectors.toMap(obj -> obj.get("user").asString(),
 70                                           obj -> Pattern.compile(obj.get("pattern").asString())));
 71 
 72         for (var repoConfig : specific.get("repositories").asArray()) {
 73             var repo = repoConfig.get("repository").asString();



 74             var list = EmailAddress.parse(repoConfig.get("list").asString());
 75             var folder = repoConfig.contains("folder") ? repoConfig.get("folder").asString() : configuration.repositoryName(repo);
 76             var bot = new MailingListBridgeBot(from, configuration.repository(repo), archiveRepo,
 77                                                list, ignoredUsers, listArchive, listSmtp,

 78                                                webrevRepo, webrevRef, Path.of(folder),
 79                                                URIBuilder.base(webrevWeb).build(), readyLabels, readyComments);
 80             ret.add(bot);
 81 
 82             allListNames.add(list);
 83             allRepositories.add(configuration.repository(repo));
 84         }
 85 
 86         var mailmanServer = MailingListServerFactory.createMailmanServer(listArchive, listSmtp);
 87         var allLists = allListNames.stream()
 88                                    .map(name -> mailmanServer.getList(name.toString()))
 89                                    .collect(Collectors.toSet());
 90 
 91         var bot = new MailingListArchiveReaderBot(from, allLists, allRepositories);
 92         ret.add(bot);
 93 
 94         return ret;
 95     }
