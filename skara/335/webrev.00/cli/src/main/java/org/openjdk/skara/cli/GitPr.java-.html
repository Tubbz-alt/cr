  86     private static Optional<Issue> getIssue(Commit commit, String project) throws IOException {
  87         var message = CommitMessageParsers.v1.parse(commit.message());
  88         var issues = message.issues();
  89         if (issues.isEmpty()) {
  90             return getIssue(message.title(), project);
  91         } else if (issues.size() == 1) {
  92             var issue = issues.get(0);
  93             return getIssue(issue.id(), project);
  94         }
  95         return Optional.empty();
  96     }

  98     private static Optional<Issue> getIssue(Branch b, String project) throws IOException {
  99         return getIssue(b.name(), project);
 100     }

 102     private static Optional<Issue> getIssue(String s, String project) throws IOException {
 103         var issueIdPattern = Pattern.compile("([A-Za-z][A-Za-z0-9]+)?-([0-9]+)");
 104         var m = issueIdPattern.matcher(s);
 105         if (m.matches()) {
 106             var id = m.group(2);
 107             if (project == null) {
 108                 project = m.group(1);
 109             }
 110             var issueTracker = IssueTracker.from("jira", URI.create("https://bugs.openjdk.java.net"));
 111             return issueTracker.project(project).issue(id);
 112         }
 113 
 114         return Optional.empty();
 115     }

 117     private static void await(Process p) throws IOException {
 118         try {
 119             var res = p.waitFor();
 120             if (res != 0) {
 121                 throw new IOException("Unexpected exit code " + res);
 122             }
 123         } catch (InterruptedException e) {
 124             throw new IOException(e);
 125         }
 126     }

 128     private static boolean spawnEditor(ReadOnlyRepository repo, Path file) throws IOException {
 129         String editor = null;
 130         var lines = repo.config("core.editor");
 131         if (lines.size() == 1) {
 132             editor = lines.get(0);
 133         }
 134         if (editor == null) {
 135             editor = System.getenv("GIT_EDITOR");
 136         }
 137         if (editor == null) {
 138             editor = System.getenv("EDITOR");
 139         }
 140         if (editor == null) {
 141             editor = System.getenv("VISUAL");
 142         }
 143         if (editor == null) {
 144             editor = "vi";
 145         }
 146 
 147         var pb = new ProcessBuilder(editor, file.toString());
 148         pb.inheritIO();
 149         var p = pb.start();
 150         try {
 151             return p.waitFor() == 0;
 152         } catch (InterruptedException e) {
 153             throw new IOException(e);
 154         }
 155     }

 157     private static String projectName(URI uri) {
 158         var name = uri.getPath().toString().substring(1);
 159         if (name.endsWith(".git")) {
 160             name = name.substring(0, name.length() - ".git".length());
 161         }
 162         return name;
 163     }

 165     private static HostedRepository getHostedRepositoryFor(URI uri, ReadOnlyRepository repo, Forge host) throws IOException {
 166         HostedRepository targetRepo = null;
 167 
 168         try {
 169             var upstream = Remote.toWebURI(repo.pullPath("upstream"));
 170             targetRepo = host.repository(projectName(upstream)).orElse(null);
 171         } catch (IOException e) {
 172             // do nothing
 173         }
 174 
 175         if (targetRepo == null) {
 176             var remoteRepo = host.repository(projectName(uri)).orElseThrow(() ->
 177                     new IOException("Could not find repository at: " + uri.toString())
 178             );
 179             var parentRepo = remoteRepo.parent();
 180             targetRepo = parentRepo.isPresent() ? parentRepo.get() : remoteRepo;
 181         }
 182 
 183         return targetRepo;
 184     }

 186     private static PullRequest getPullRequest(URI uri, ReadOnlyRepository repo, Forge host, String prId) throws IOException {

 188         if (pr == null) {
 189             exit("error: could not fetch PR information");
 190         }
 191 
 192         return pr;

 198     private static void show(String ref, Hash hash, Path dir) throws IOException {
 199         var pb = new ProcessBuilder("git", "diff", "--binary",
 200                                                    "--patch",
 201                                                    "--find-renames=50%",
 202                                                    "--find-copies=50%",
 203                                                    "--find-copies-harder",
 204                                                    "--abbrev",
 205                                                    ref + "..." + hash.hex());
 206         if (dir != null) {
 207             pb.directory(dir.toFile());
 208         }
 209         pb.inheritIO();
 210         await(pb.start());
 211     }

 225     private static List<String> hgTags() throws IOException, InterruptedException {
 226         var pb = new ProcessBuilder("hg", "tags", "--quiet");
 227         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
 228         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 229         var p = pb.start();
 230         var bytes = p.getInputStream().readAllBytes();
 231         var exited = p.waitFor(1, TimeUnit.MINUTES);
 232         var exitValue = p.exitValue();
 233         if (!exited || exitValue != 0) {
 234             throw new IOException("'hg tags' exited with value: " + exitValue);
 235         }
 236 
 237         return Arrays.asList(new String(bytes, StandardCharsets.UTF_8).split("\n"));
 238     }

 240     private static String hgResolve(String ref) throws IOException, InterruptedException {
 241         var pb = new ProcessBuilder("hg", "log", "-r", ref, "--template", "{node}");
 242         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
 243         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 244         var p = pb.start();
 245         var bytes = p.getInputStream().readAllBytes();
 246         var exited = p.waitFor(1, TimeUnit.MINUTES);
 247         var exitValue = p.exitValue();
 248         if (!exited || exitValue != 0) {
 249             throw new IOException("'hg log' exited with value: " + exitValue);
 250         }
 251 
 252         return new String(bytes, StandardCharsets.UTF_8);
 253     }

 259     private static Path diff(String ref, Hash hash, Path dir) throws IOException {
 260         var patch = Files.createTempFile(hash.hex(), ".patch");
 261         var pb = new ProcessBuilder("git", "diff", "--binary",
 262                                                    "--patch",
 263                                                    "--find-renames=50%",
 264                                                    "--find-copies=50%",
 265                                                    "--find-copies-harder",
 266                                                    "--abbrev",
 267                                                    ref + "..." + hash.hex());
 268         if (dir != null) {
 269             pb.directory(dir.toFile());
 270         }
 271         pb.redirectOutput(patch.toFile());
 272         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 273         await(pb.start());
 274         return patch;
 275     }

 287     public static void main(String[] args) throws IOException, InterruptedException {
 288         var flags = List.of(
 289             Option.shortcut("u")
 290                   .fullname("username")
 291                   .describe("NAME")
 292                   .helptext("Username on host")
 293                   .optional(),
 294             Option.shortcut("r")
 295                   .fullname("remote")
 296                   .describe("NAME")
 297                   .helptext("Name of remote, defaults to 'origin'")
 298                   .optional(),
 299             Option.shortcut("b")
 300                   .fullname("branch")
 301                   .describe("NAME")
 302                   .helptext("Name of target branch, defaults to 'master'")
 303                   .optional(),
 304             Option.shortcut("")
 305                   .fullname("authors")
 306                   .describe("LIST")
 307                   .helptext("Comma separated list of authors")
 308                   .optional(),
 309             Option.shortcut("")
 310                   .fullname("assignees")
 311                   .describe("LIST")
 312                   .helptext("Comma separated list of assignees")
 313                   .optional(),
 314             Option.shortcut("")
 315                   .fullname("labels")
 316                   .describe("LIST")
 317                   .helptext("Comma separated list of labels")
 318                   .optional(),
 319             Option.shortcut("")
 320                   .fullname("columns")
 321                   .describe("id,title,author,assignees,labels")
 322                   .helptext("Comma separated list of columns to show")
 323                   .optional(),
 324             Switch.shortcut("")
 325                   .fullname("no-decoration")
 326                   .helptext("Hide any decorations when listing PRs")
 327                   .optional(),
 328             Switch.shortcut("")
 329                   .fullname("ignore-workspace")
 330                   .helptext("Ignore local changes in worktree and staging area when creating pull request")
 331                   .optional(),
 332             Switch.shortcut("")
 333                   .fullname("ignore-local-commits")
 334                   .helptext("Ignore local commits not pushed when creating pull request")
 335                   .optional(),
 336             Switch.shortcut("")
 337                   .fullname("publish")
 338                   .helptext("Publish the local branch before creating the pull request")
 339                   .optional(),
 340             Switch.shortcut("")
 341                   .fullname("jcheck")
 342                   .helptext("Run jcheck before creating the pull request")
 343                   .optional(),
 344             Switch.shortcut("")
 345                   .fullname("no-token")
 346                   .helptext("Do not use a personal access token (PAT). Only works for read-only operations.")
 347                   .optional(),
 348             Switch.shortcut("")
 349                   .fullname("mercurial")
 350                   .helptext("Force use of Mercurial (hg)")
 351                   .optional(),
 352             Switch.shortcut("")
 353                   .fullname("verbose")
 354                   .helptext("Turn on verbose output")
 355                   .optional(),
 356             Switch.shortcut("")
 357                   .fullname("debug")
 358                   .helptext("Turn on debugging output")
 359                   .optional(),
 360             Switch.shortcut("")
 361                   .fullname("version")
 362                   .helptext("Print the version of this tool")
 363                   .optional());
 364 
 365         var inputs = List.of(
 366             Input.position(0)
 367                  .describe("list|fetch|show|checkout|apply|integrate|approve|create|close|update|test")
 368                  .singular()
 369                  .optional(),
 370             Input.position(1)
 371                  .describe("ID")
 372                  .singular()
 373                  .optional()
 374         );
 375 
 376         var parser = new ArgumentParser("git-pr", flags, inputs);
 377         var arguments = parser.parse(args);
 378 
 379         if (arguments.contains("version")) {
 380             System.out.println("git-pr version: " + Version.fromManifest().orElse("unknown"));
 381             System.exit(0);
 382         }
 383 
 384         if (arguments.contains("verbose") || arguments.contains("debug")) {
 385             var level = arguments.contains("debug") ? Level.FINER : Level.FINE;
 386             Logging.setup(level);
 387         }
 388 
 389         HttpProxy.setup();
 390 
 391         var isMercurial = arguments.contains("mercurial");
 392         var cwd = Path.of("").toAbsolutePath();
 393         var repo = Repository.get(cwd).orElseThrow(() -> new IOException("no git repository found at " + cwd.toString()));
 394         var remote = arguments.get("remote").orString(isMercurial ? "default" : "origin");
 395         var remotePullPath = repo.pullPath(remote);
 396         var username = arguments.contains("username") ? arguments.get("username").asString() : null;
 397         var token = isMercurial ? System.getenv("HG_TOKEN") :  System.getenv("GIT_TOKEN");
 398         var uri = Remote.toWebURI(remotePullPath);
 399         var shouldUseToken = !arguments.contains("no-token");
 400         var credentials = !shouldUseToken ?
 401             null :
 402             GitCredentials.fill(uri.getHost(), uri.getPath(), username, token, uri.getScheme());
 403         var forgeURI = URI.create(uri.getScheme() + "://" + uri.getHost());
 404         var forge = credentials == null ?
 405             Forge.from(forgeURI) :
 406             Forge.from(forgeURI, new Credential(credentials.username(), credentials.password()));
 407         if (forge.isEmpty()) {
 408             if (!shouldUseToken) {
 409                 if (arguments.contains("verbose")) {
 410                     System.err.println("");
 411                 }
 412                 System.err.println("warning: using git-pr with --no-token may result in rate limiting from " + forgeURI);
 413                 if (!arguments.contains("verbose")) {
 414                     System.err.println("         Re-run git-pr with --verbose to see if you are being rate limited");
 415                     System.err.println("");
 416                 }
 417             }
 418             exit("error: failed to connect to host: " + forgeURI);
 419         }
 420         var host = forge.get();
 421 
 422         var action = arguments.at(0).isPresent() ? arguments.at(0).asString() : null;
 423         if (action == null) {
 424             var lines = repo.config("pr.default");
 425             if (lines.size() == 1) {
 426                 action = lines.get(0);
 427             }
 428         }
 429 
 430         if (!shouldUseToken &&
 431             !List.of("list", "fetch", "show", "checkout", "apply").contains(action)) {
 432             System.err.println("error: --no-token can only be used with read-only operations");
 433             System.exit(1);
 434         }
 435 
 436         if (action.equals("create")) {
 437             if (isMercurial) {
 438                 var currentBookmark = repo.currentBookmark();
 439                 if (!currentBookmark.isPresent()) {
 440                     System.err.println("error: no bookmark is active, you must be on an active bookmark");
 441                     System.err.println("");
 442                     System.err.println("To create a bookmark and activate it, run:");
 443                     System.err.println("");
 444                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
 445                     System.err.println("");
 446                     System.exit(1);
 447                 }
 448 
 449                 var bookmark = currentBookmark.get();
 450                 if (bookmark.equals(new Bookmark("master"))) {
 451                     System.err.println("error: you should not create pull requests from the 'master' bookmark");
 452                     System.err.println("To create a bookmark and activate it, run:");
 453                     System.err.println("");
 454                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
 455                     System.err.println("");
 456                     System.exit(1);
 457                 }
 458 
 459                 var tags = hgTags();
 460                 var upstreams = tags.stream()
 461                                     .filter(t -> t.endsWith(bookmark.name()))
 462                                     .collect(Collectors.toList());
 463                 if (upstreams.isEmpty()) {
 464                     System.err.println("error: there is no remote branch for the local bookmark '" + bookmark.name() + "'");
 465                     System.err.println("");
 466                     System.err.println("To create a remote branch and push the commits for your local branch, run:");
 467                     System.err.println("");
 468                     System.err.println("    hg push --bookmark " + bookmark.name());
 469                     System.err.println("");
 470                     System.exit(1);
 471                 }
 472 
 473                 var tagsAndHashes = new HashMap<String, String>();
 474                 for (var tag : tags) {
 475                     tagsAndHashes.put(tag, hgResolve(tag));
 476                 }
 477                 var bookmarkHash = hgResolve(bookmark.name());
 478                 if (!tagsAndHashes.containsValue(bookmarkHash)) {
 479                     System.err.println("error: there are local commits on bookmark '" + bookmark.name() + "' not present in a remote repository");
 480                     System.err.println("");
 481 
 482                     if (upstreams.size() == 1) {
 483                         System.err.println("To push the local commits to the remote repository, run:");
 484                         System.err.println("");
 485                         System.err.println("    hg push --bookmark " + bookmark.name() + " " + upstreams.get(0));
 486                         System.err.println("");
 487                     } else {
 488                         System.err.println("The following paths contains the " + bookmark.name() + " bookmark:");
 489                         System.err.println("");
 490                         for (var upstream : upstreams) {
 491                             System.err.println("- " + upstream.replace("/" + bookmark.name(), ""));
 492                         }
 493                         System.err.println("");
 494                         System.err.println("To push the local commits to a remote repository, run:");
 495                         System.err.println("");
 496                         System.err.println("    hg push --bookmark " + bookmark.name() + " <PATH>");
 497                         System.err.println("");
 498                     }
 499                     System.exit(1);
 500                 }
 501 
 502                 var targetBranch = arguments.get("branch").orString("master");
 503                 var targetHash = hgResolve(targetBranch);
 504                 var commits = repo.commits(targetHash + ".." + bookmarkHash + "-" + targetHash).asList();
 505                 if (commits.isEmpty()) {
 506                     System.err.println("error: no difference between bookmarks " + targetBranch + " and " + bookmark.name());
 507                     System.err.println("       Cannot create an empty pull request, have you committed?");
 508                     System.exit(1);
 509                 }
 510 
 511                 var diff = repo.diff(repo.head());
 512                 if (!diff.patches().isEmpty()) {
 513                     System.err.println("error: there are uncommitted changes in your working tree:");
 514                     System.err.println("");
 515                     for (var patch : diff.patches()) {
 516                         var path = patch.target().path().isPresent() ?
 517                             patch.target().path().get() : patch.source().path().get();
 518                         System.err.println("    " + patch.status().toString() + " " + path.toString());
 519                     }
 520                     System.err.println("");
 521                     System.err.println("If these changes are meant to be part of the pull request, run:");
 522                     System.err.println("");
 523                     System.err.println("    hg commit --amend");
 524                     System.err.println("    hg git-cleanup");
 525                     System.err.println("    hg push --bookmark " + bookmark.name() + " <PATH>");
 526                     System.err.println("    hg gimport");
 527                     System.err.println("");
 528                     System.err.println("If these changes are *not* meant to be part of the pull request, run:");
 529                     System.err.println("");
 530                     System.err.println("    hg shelve");
 531                     System.err.println("");
 532                     System.err.println("(You can later restore the changes by running: hg unshelve)");
 533                     System.exit(1);
 534                 }
 535 
 536                 var remoteRepo = host.repository(projectName(uri)).orElseThrow(() ->
 537                         new IOException("Could not find repository at " + uri.toString())
 538                 );
 539                 if (token == null) {
 540                     GitCredentials.approve(credentials);
 541                 }
 542                 var parentRepo = remoteRepo.parent().orElseThrow(() ->
 543                         new IOException("error: remote repository " + remotePullPath + " is not a fork of any repository"));
 544 
 545                 var file = Files.createTempFile("PULL_REQUEST_", ".txt");
 546                 if (commits.size() == 1) {
 547                     var commit = commits.get(0);
 548                     var message = CommitMessageParsers.v1.parse(commit.message());
 549                     Files.writeString(file, message.title() + "\n");
 550                     if (!message.summaries().isEmpty()) {
 551                         Files.write(file, message.summaries(), StandardOpenOption.APPEND);
 552                     }
 553                     if (!message.additional().isEmpty()) {
 554                         Files.write(file, message.additional(), StandardOpenOption.APPEND);
 555                     }
 556                 } else {
 557                     Files.write(file, List.of(""));
 558                 }
 559                 Files.write(file, List.of(
 560                     "# Please enter the pull request message for your changes. Lines starting",
 561                     "# with '#' will be ignored, and an empty message aborts the pull request.",
 562                     "# The first line will be considered the subject, use a blank line to separate",
 563                     "# the subject from the body.",
 564                     "#",
 565                     "# Commits to be included from branch '" + bookmark.name() + "'"
 566                     ),
 567                     StandardOpenOption.APPEND
 568                 );
 569                 for (var commit : commits) {
 570                     var desc = commit.hash().abbreviate() + ": " + commit.message().get(0);
 571                     Files.writeString(file, "# - " + desc + "\n", StandardOpenOption.APPEND);
 572                 }
 573                 Files.writeString(file, "#\n", StandardOpenOption.APPEND);
 574                 Files.writeString(file, "# Target repository: " + remotePullPath + "\n", StandardOpenOption.APPEND);
 575                 Files.writeString(file, "# Target branch: " + targetBranch + "\n", StandardOpenOption.APPEND);
 576                 var success = spawnEditor(repo, file);
 577                 if (!success) {
 578                     System.err.println("error: editor exited with non-zero status code, aborting");
 579                     System.exit(1);
 580                 }
 581                 var lines = Files.readAllLines(file)
 582                                  .stream()
 583                                  .filter(l -> !l.startsWith("#"))
 584                                  .collect(Collectors.toList());
 585                 var isEmpty = lines.stream().allMatch(String::isEmpty);
 586                 if (isEmpty) {
 587                     System.err.println("error: no message present, aborting");
 588                     System.exit(1);
 589                 }
 590 
 591                 var title = lines.get(0);
 592                 List<String> body = null;
 593                 if (lines.size() > 1) {
 594                     body = lines.subList(1, lines.size())
 595                                 .stream()
 596                                 .dropWhile(String::isEmpty)
 597                                 .collect(Collectors.toList());
 598                 } else {
 599                     body = Collections.emptyList();
 600                 }
 601 
 602                 var pr = remoteRepo.createPullRequest(parentRepo, targetBranch, bookmark.name(), title, body);
 603                 if (arguments.contains("assignees")) {
 604                     var usernames = Arrays.asList(arguments.get("assignees").asString().split(","));
 605                     var assignees = usernames.stream()
 606                                              .map(u -> host.user(u))
 607                                              .collect(Collectors.toList());
 608                     pr.setAssignees(assignees);
 609                 }
 610                 System.out.println(pr.webUrl().toString());
 611                 Files.deleteIfExists(file);
 612 
 613                 System.exit(0);
 614             }
 615             var currentBranch = repo.currentBranch().orElseGet(() -> {
 616                     System.err.println("error: the repository is in a detached HEAD state");
 617                     System.exit(1);
 618                     return null;
 619             });
 620             if (currentBranch.equals(repo.defaultBranch())) {
 621                 System.err.println("error: you should not create pull requests from the 'master' branch");
 622                 System.err.println("");

 844         } else if (action.equals("integrate") || action.equals("approve") || action.equals("test")) {
