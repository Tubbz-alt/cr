  49     private java.lang.Process start(List<String> cmd) throws IOException {
  50         log.fine("Executing " + String.join(" ", cmd));
  51         var pb = new ProcessBuilder(cmd);
  52         pb.directory(dir.toFile());
  53         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  54         return pb.start();
  55     }

  57     private static void stop(java.lang.Process p) throws IOException {
  58         if (p != null && p.isAlive()) {
  59             var stream = p.getInputStream();
  60             var read = 0;
  61             var buf = new byte[128];
  62             while (read != -1) {
  63                 read = stream.read(buf);
  64             }
  65             try {
  66                 p.waitFor();
  67             } catch (InterruptedException e) {
  68                 throw new IOException(e);
  69             }
  70         }
  71     }

  91     private static Execution.Result await(Execution e) throws IOException {
  92         var result = e.await();
  93         if (result.status() != 0) {
  94             throw new IOException("Unexpected exit code\n" + result);
  95         }
  96         return result;
  97     }

  99     private static void await(java.lang.Process p) throws IOException {
 100         try {
 101             var res = p.waitFor();
 102             if (res != 0) {
 103                 throw new IOException("Unexpected exit code: " + res);
 104             }
 105         } catch (InterruptedException e) {
 106             throw new IOException(e);
 107         }
 108     }

 193     public List<CommitMetadata> commitMetadata() throws IOException {
 194         var revisions = "--all";
 195         var p = start("git", "rev-list", "--format=" + GitCommitMetadata.FORMAT, "--no-abbrev", "--reverse", "--no-color", revisions);
 196         var reader = new UnixStreamReader(p.getInputStream());
 197         var result = new ArrayList<CommitMetadata>();
 198 
 199         var line = reader.readLine();
 200         while (line != null) {
 201             if (!line.startsWith("commit")) {
 202                 throw new IOException("Unexpected line: " + line);
 203             }
 204 
 205             result.add(GitCommitMetadata.read(reader));
 206             line = reader.readLine();
 207         }
 208 
 209         await(p);
 210         return result;
 211     }

 213     private List<Hash> refs() throws IOException {
 214         try (var p = capture("git", "show-ref", "--hash", "--abbrev")) {
 215             var res = p.await();
 216             if (res.status() == -1) {
 217                 if (res.stdout().size() != 0) {
 218                     throw new IOException("Unexpected output\n" + res);
 219                 }
 220                 return new ArrayList<>();
 221             } else {
 222                 return res.stdout().stream()
 223                           .map(Hash::new)
 224                           .collect(Collectors.toList());
 225             }
 226         }
 227     }

 229     @Override
 230     public boolean isEmpty() throws IOException {
 231         int numLooseObjects = -1;
 232         int numPackedObjects = -1;
 233 
 234         try (var p = capture("git", "count-objects", "-v")) {
 235             var res = await(p);
 236             var stdout = res.stdout();
 237 
 238             for (var line : stdout) {
 239                 if (line.startsWith("count: ")) {
 240                     try {
 241                         numLooseObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 242                     } catch (NumberFormatException e) {
 243                         throw new IOException("Unexpected 'count' value\n" + res, e);
 244                     }
 245 
 246                 } else if (line.startsWith("in-pack: ")) {
 247                     try {
 248                         numPackedObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 249                     } catch (NumberFormatException e) {
 250                         throw new IOException("Unexpected 'in-pack' value\n" + res, e);
 251                     }
 252                 }
 253             }
 254         }
 255 
 256         return numLooseObjects == 0 && numPackedObjects == 0 && refs().size() == 0;
 257     }

 259     @Override
 260     public boolean isHealthy() throws IOException {
 261         var refs = refs();
 262         if (refs.size() == 0) {
 263             return true;
 264         }
 265 
 266         var name = "health-check";
 267         try (var p = capture("git", "branch", name, refs.get(0).hex())) {
 268             if (p.await().status() != 0) {
 269                 return false;
 270             }
 271         }
 272         try (var p = capture("git", "branch", "-D", name)) {
 273             if (p.await().status() != 0) {
 274                 return false;
 275             }
 276         }
 277 
 278         return true;
 279     }

 435     @Override
 436     public Path root() throws IOException {
 437         if (cachedRoot != null) {
 438             return cachedRoot;
 439         }
 440 
 441         try (var p = capture("git", "rev-parse", "--show-toplevel")) {
 442             var res = await(p);
 443             if (res.stdout().size() != 1) {
 444                 // Perhaps this is a bare repository
 445                 try (var p2 = capture("git", "rev-parse", "--git-dir")) {
 446                     var res2 = await(p2);
 447                     if (res2.stdout().size() != 1) {
 448                         throw new IOException("Unexpected output\n" + res2);
 449                     }
 450                     cachedRoot = dir.resolve(Path.of(res2.stdout().get(0)));
 451                     return cachedRoot;
 452                 }
 453             }
 454 
 455             cachedRoot = Path.of(res.stdout().get(0));
 456             return cachedRoot;
 457         }
 458     }

 467     @FunctionalInterface
 468     private static interface Operation {
 469         void execute(List<Path> args) throws IOException;
 470     }
 471 
 472     private void batch(Operation op, List<Path> args) throws IOException {
 473         var batchSize = 64;
 474         var start = 0;
 475         while (start < args.size()) {
 476             var end = start + batchSize;
 477             if (end > args.size()) {
 478                 end = args.size();
 479             }
 480             op.execute(args.subList(start, end));
 481             start = end;
 482         }
 483     }

 548     @Override
 549     public Hash commit(String message,
 550                        String authorName,
 551                        String authorEmail,
 552                        ZonedDateTime authorDate,
 553                        String committerName,
 554                        String committerEmail,
 555                        ZonedDateTime committerDate) throws IOException {
 556         var cmd = Process.capture("git", "commit", "--message=" + message)
 557                          .workdir(dir)
 558                          .environ("GIT_AUTHOR_NAME", authorName)
 559                          .environ("GIT_AUTHOR_EMAIL", authorEmail)
 560                          .environ("GIT_COMMITTER_NAME", committerName)
 561                          .environ("GIT_COMMITTER_EMAIL", committerEmail);
 562         if (authorDate != null) {
 563             cmd = cmd.environ("GIT_AUTHOR_DATE",
 564                               authorDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 565         }
 566         if (committerDate != null) {
 567             cmd = cmd.environ("GIT_COMMITTER_DATE",
 568                               committerDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 569         }
 570         try (var p = cmd.execute()) {
 571             await(p);
 572             return head();
 573         }
 574     }

 699     private String treeEntry(Path path, Hash hash) throws IOException {
 700         try (var p = Process.capture("git", "ls-tree", hash.hex(), path.toString())
 701                             .workdir(root())
 702                             .execute()) {
 703             var res = await(p);
 704             if (res.stdout().size() == 0) {
 705                 return null;
 706             }
 707             if (res.stdout().size() > 1) {
 708                 throw new IOException("Unexpected output\n" + res);
 709             }
 710             return res.stdout().get(0);
 711         }
 712     }

 714     private List<FileEntry> allFiles(Hash hash, List<Path> paths) throws IOException {
 715         var cmd = new ArrayList<String>();
 716         cmd.addAll(List.of("git", "ls-tree", "-r"));
 717         cmd.add(hash.hex());
 718         cmd.addAll(paths.stream().map(Path::toString).collect(Collectors.toList()));
 719         try (var p = Process.capture(cmd.toArray(new String[0]))
 720                             .workdir(root())
 721                             .execute()) {
 722             var res = await(p);
 723             var entries = new ArrayList<FileEntry>();
 724             for (var line : res.stdout()) {
 725                 var parts = line.split("\t");
 726                 var metadata = parts[0].split(" ");
 727                 var filename = parts[1];
 728 
 729                 var entry = new FileEntry(hash,
 730                                           FileType.fromOctal(metadata[0]),
 731                                           new Hash(metadata[2]),
 732                                           Path.of(filename));
 733                 entries.add(entry);
 734             }
 735             return entries;

 739     @Override
 740     public List<FileEntry> files(Hash hash, List<Path> paths) throws IOException {
 741         if (paths.isEmpty()) {
 742             return allFiles(hash, paths);
 743         }
 744 
 745         var entries = new ArrayList<FileEntry>();
 746         var batchSize = 64;
 747         var start = 0;
 748         while (start < paths.size()) {
 749             var end = start + batchSize;
 750             if (end > paths.size()) {
 751                 end = paths.size();
 752             }
 753             entries.addAll(allFiles(hash, paths.subList(start, end)));
 754             start = end;
 755         }
 756         return entries;
 757     }

 759     private Path unpackFile(String blob) throws IOException {
 760         try (var p = capture("git", "unpack-file", blob)) {
 761             var res = await(p);
 762             if (res.stdout().size() != 1) {
 763                 throw new IOException("Unexpected output\n" + res);
 764             }
 765 
 766             return Path.of(root().toString(), res.stdout().get(0));

 770     @Override
 771     public Optional<byte[]> show(Path path, Hash hash) throws IOException {
 772         var entries = files(hash, path);
 773         if (entries.size() == 0) {
 774             return Optional.empty();
 775         } else if (entries.size() > 1) {
 776             throw new IOException("Multiple files match path " + path.toString() + " in commit " + hash.hex());
 777         }
 778 
 779         var entry = entries.get(0);
 780         var type = entry.type();
 781         if (type.isVCSLink()) {
 782             var content = "Subproject commit " + entry.hash().hex() + " " + entry.path().toString();
 783             return Optional.of(content.getBytes(StandardCharsets.UTF_8));
 784         } else if (type.isRegular()) {
 785             var tmp = unpackFile(entry.hash().hex());
 786             var content = Files.readAllBytes(tmp);
 787             Files.delete(tmp);
 788             return Optional.of(content);
 789         }
 790 
 791         return Optional.empty();
 792     }

 822     public Diff diff(Hash from, Hash to) throws IOException {
 823         var cmd = new ArrayList<>(List.of("git", "diff", "--patch",
 824                                                          "--find-renames=99%",
 825                                                          "--find-copies=99%",
 826                                                          "--find-copies-harder",
 827                                                          "--binary",
 828                                                          "--raw",
 829                                                          "--no-abbrev",
 830                                                          "--unified=0",
 831                                                          "--no-color",
 832                                                          from.hex()));
 833         if (to != null) {
 834             cmd.add(to.hex());
 835         }
 836 
 837         var p = start(cmd);
 838         try {
 839             var patches = UnifiedDiffParser.parseGitRaw(p.getInputStream());
 840             await(p);
 841             return new Diff(from, to, patches);
 842         } catch (Throwable t) {
 843             stop(p);
 844             throw t;
 845         }
 846     }
