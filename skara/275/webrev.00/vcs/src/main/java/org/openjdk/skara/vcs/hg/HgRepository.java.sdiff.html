 752     @Override
 753     public Diff diff(Hash from, Hash to, List<Path> files) throws IOException {
 754         var ext = Files.createTempFile("ext", ".py");
 755         copyResource(EXT_PY, ext);
 756 
 757         var cmd = new ArrayList<>(List.of("hg", "--config", "extensions.diff-git-raw=" + ext.toAbsolutePath(),
 758                                                 "diff-git-raw", "--patch", from.hex()));
 759         if (to != null) {
 760             cmd.add(to.hex());
 761         }
 762 
 763         if (files != null) {
 764             var filenames = files.stream().map(Path::toString).collect(Collectors.toList());
 765             cmd.add("--files=" + String.join(",", filenames));
 766         }
 767 
 768         var p = start(cmd);
 769         try {
 770             var patches = UnifiedDiffParser.parseGitRaw(p.getInputStream());
 771             await(p);
 772             return new Diff(from, to, patches);
 773         } catch (Throwable t) {
 774             throw new IOException(t);
 775         }
 776     }
