48 
49         var hostname = uri.getHost();
50         var knownHostFactories = factories.stream()
51                                           .filter(f -> f.knownHosts().contains(hostname))
52                                           .collect(Collectors.toList());
53         if (knownHostFactories.size() == 1) {
54             var factory = knownHostFactories.get(0);
55             return Optional.of(factory.create(uri, credential, configuration));
56         }
