23 package org.openjdk.skara.forge;
24 
25 import org.junit.jupiter.api.Test;
26 import org.openjdk.skara.host.Credential;
27 import org.openjdk.skara.json.JSONObject;
28 
29 import java.net.URI;
30 import java.util.*;
31 import java.util.stream.Collectors;
32 
33 import static org.junit.jupiter.api.Assertions.assertEquals;
34 
35 class ForgeTests {
36     @Test
37     void sortTest() {
38         var allFactories = List.of(new ForgeFactory() {
39                                        @Override
40                                        public String name() {
41                                            return "something";
42                                        }
43 
44                                        @Override
45                                        public Set<String> knownHosts() {
46                                            return Set.of();
47                                        }
48 
49                                        @Override
50                                        public Forge create(URI uri, Credential credential, JSONObject configuration) {
51                                            return null;
52                                        }
53                                    },
54                                    new ForgeFactory() {
55                                        @Override
56                                        public String name() {
57                                            return "other";
58                                        }
59 
60                                        @Override
61                                        public Set<String> knownHosts() {
62                                            return Set.of();
63                                        }
64 
65                                        @Override
66                                        public Forge create(URI uri, Credential credential, JSONObject configuration) {
67                                            return null;
68                                        }
69                                    });
70 
71         var sorted = allFactories.stream()
72                                  .sorted(Comparator.comparing(f -> !f.name().contains("other")))
73                                  .collect(Collectors.toList());
74 
75         assertEquals("something", allFactories.get(0).name());
76         assertEquals("other", sorted.get(0).name());
77     }
78 }
