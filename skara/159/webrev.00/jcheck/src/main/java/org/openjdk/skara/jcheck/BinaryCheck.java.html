23 package org.openjdk.skara.jcheck;
24 
25 import org.openjdk.skara.vcs.Commit;
26 import org.openjdk.skara.vcs.openjdk.CommitMessage;
27 
28 import java.util.Iterator;
29 import java.util.ArrayList;
30 import java.util.logging.Logger;
31 
32 public class BinaryCheck extends CommitCheck {
33     private final Logger log = Logger.getLogger("org.openjdk.skara.jcheck.binary");
34 
35     @Override
36     Iterator<Issue> check(Commit commit, CommitMessage message, JCheckConfiguration conf) {
37         var metadata = CommitIssue.metadata(commit, message, conf, this);
38 
39         var issues = new ArrayList<Issue>();
40         for (var diff : commit.parentDiffs()) {
41             for (var patch : diff.patches()) {
42                 if (patch.isBinary() &&
43                     (patch.status().isAdded() || patch.status().isCopied())) {
44                     issues.add(new BinaryIssue(patch.target().path().get(), metadata));
45                 }
46             }
47         }
48 
49         return issues.iterator();
50     }
51 
52     @Override
53     public String name() {
54         return "binary";
55     }
56 
57     @Override
58     public String description() {
59         return "Files should not be binary";
60     }
61 }
