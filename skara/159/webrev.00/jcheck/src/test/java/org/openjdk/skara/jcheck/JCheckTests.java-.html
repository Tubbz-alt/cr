 41     static class CheckableRepository {
 42         public static Repository create(Path path, VCS vcs) throws IOException {
 43             var repo = Repository.init(path, vcs);
 44 
 45             Files.createDirectories(path.resolve(".jcheck"));
 46             var checkConf = path.resolve(".jcheck/conf");
 47             try (var output = new FileWriter(checkConf.toFile())) {
 48                 output.append("[general]\n");
 49                 output.append("project=test\n");
 50                 output.append("\n");
 51                 output.append("[checks]\n");
 52                 output.append("error=reviewers,whitespace\n");
 53                 output.append("\n");
 54                 output.append("[census]\n");
 55                 output.append("version=0\n");
 56                 output.append("domain=openjdk.java.net\n");
 57                 output.append("\n");
 58                 output.append("[checks \"whitespace\"]\n");
 59                 output.append("suffixes=.txt\n");
 60                 output.append("\n");
 61                 output.append("[checks \"reviewers\"]\n");
 62                 output.append("minimum=1\n");
 63             }
 64             repo.add(checkConf);
 65 
 66             repo.commit("Initial commit", "duke", "duke@openjdk.java.net");
 67 
 68             return repo;
 69         }
 70     }

236     @ParameterizedTest
237     @EnumSource(VCS.class)
238     void checksForCommit(VCS vcs) throws Exception {
239         try (var dir = new TemporaryDirectory()) {
240             var repoPath = dir.path().resolve("repo");
241             var repo = CheckableRepository.create(repoPath, vcs);
242 
243             var readme = repoPath.resolve("README");
244             Files.write(readme, List.of("Hello, readme!"));
245             repo.add(readme);
246             var first = repo.commit("Add README", "duke", "duke@openjdk.java.net");
247 
248             var censusPath = dir.path().resolve("census");
249             Files.createDirectories(censusPath);
250             CensusCreator.populateCensusDirectory(censusPath);
251             var census = Census.parse(censusPath);
252 
253             var checks = JCheck.checks(repo, census, first);
254             var checkNames = checks.stream()
255                                    .map(Check::name)
256                                    .collect(Collectors.toSet());
257             assertEquals(Set.of("whitespace", "reviewers"), checkNames);
258         }
259     }

261     @ParameterizedTest
262     @EnumSource(VCS.class)
263     void checkRemoval(VCS vcs) throws Exception {
264         try (var dir = new TemporaryDirectory()) {
265             var repoPath = dir.path().resolve("repo");
266             var repo = CheckableRepository.create(repoPath, vcs);
267 
268             var file = repoPath.resolve("file.txt");
269             Files.write(file, List.of("Hello, file!"));
270             repo.add(file);
271             var first = repo.commit("Add file", "duke", "duke@openjdk.java.net");
272 
273             Files.delete(file);
274             repo.remove(file);
275             var second = repo.commit("Remove file", "duke", "duke@openjdk.java.net");
276 
277             var censusPath = dir.path().resolve("census");
278             Files.createDirectories(censusPath);
279             CensusCreator.populateCensusDirectory(censusPath);
280             var census = Census.parse(censusPath);
281 
282             var visitor = new TestVisitor();
283             try (var issues = JCheck.check(repo, census, CommitMessageParsers.v1, first.hex() + ".." + second.hex())) {
284                 for (var issue : issues) {
285                     issue.accept(visitor);
286                 }
287             }
288             assertEquals(Set.of("org.openjdk.skara.jcheck.TooFewReviewersIssue"), visitor.issueNames());
289         }
