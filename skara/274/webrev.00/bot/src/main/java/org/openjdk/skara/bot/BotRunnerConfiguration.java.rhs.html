 41 public class BotRunnerConfiguration {
 42     private final Logger log;
 43     private final JSONObject config;
 44     private final Map<String, Forge> repositoryHosts;
 45     private final Map<String, IssueTracker> issueHosts;
 46     private final Map<String, ContinuousIntegration> continuousIntegrations;
 47     private final Map<String, HostedRepository> repositories;
 48 
 49     private BotRunnerConfiguration(JSONObject config, Path cwd) throws ConfigurationError {
 50         this.config = config;
 51         log = Logger.getLogger("org.openjdk.skara.bot");
 52 
 53         repositoryHosts = parseRepositoryHosts(config, cwd);
 54         issueHosts = parseIssueHosts(config, cwd);
 55         continuousIntegrations = parseContinuousIntegrations(config, cwd);
 56         repositories = parseRepositories(config);
 57     }

 59     private Map<String, Forge> parseRepositoryHosts(JSONObject config, Path cwd) throws ConfigurationError {
 60         Map<String, Forge> ret = new HashMap<>();
 61 
 62         if (!config.contains("forges")) {
 63             return ret;
 64         }
 65 
 66         for (var entry : config.get("forges").fields()) {
 67             if (entry.value().contains("gitlab")) {
 68                 var gitlab = entry.value().get("gitlab");
 69                 var uri = URIBuilder.base(gitlab.get("url").asString()).build();
 70                 var pat = new Credential(gitlab.get("username").asString(), gitlab.get("pat").asString());
 71                 ret.put(entry.name(), Forge.from("gitlab", uri, pat, gitlab.asObject()));
 72             } else if (entry.value().contains("github")) {
 73                 var github = entry.value().get("github");
 74                 URI uri;
 75                 if (github.contains("url")) {
 76                     uri = URIBuilder.base(github.get("url").asString()).build();
 77                 } else {
 78                     uri = URIBuilder.base("https://github.com/").build();
 79                 }
 80 
 81                 if (github.contains("app")) {
 82                     var keyFile = cwd.resolve(github.get("app").get("key").asString());
 83                     try {
 84                         var keyContents = Files.readString(keyFile, StandardCharsets.UTF_8);
 85                         var pat = new Credential(github.get("app").get("id").asString() + ";" +
 86                                                          github.get("app").get("installation").asString(),
 87                                                  keyContents);
 88                         ret.put(entry.name(), Forge.from("github", uri, pat, github.asObject()));
 89                     } catch (IOException e) {
 90                         throw new ConfigurationError("Cannot find key file: " + keyFile);
 91                     }
 92                 } else {
 93                     var pat = new Credential(github.get("username").asString(), github.get("pat").asString());
 94                     ret.put(entry.name(), Forge.from("github", uri, pat, github.asObject()));
 95                 }
 96             } else {
 97                 throw new ConfigurationError("Host " + entry.name());
 98             }
 99         }
100 
101         return ret;
102     }

104     private Map<String, IssueTracker> parseIssueHosts(JSONObject config, Path cwd) throws ConfigurationError {
105         Map<String, IssueTracker> ret = new HashMap<>();
106 
107         if (!config.contains("issuetrackers")) {
108             return ret;
109         }
110 
111         for (var entry : config.get("issuetrackers").fields()) {
112             if (entry.value().contains("jira")) {
113                 var jira = entry.value().get("jira");
114                 var uri = URIBuilder.base(jira.get("url").asString()).build();
115                 Credential credential = null;
116                 if (jira.contains("username")) {
117                     credential = new Credential(jira.get("username").asString(), jira.get("password").asString());
118                 }
119                 ret.put(entry.name(), IssueTracker.from("jira", uri, credential, jira.asObject()));
120             } else {
121                 throw new ConfigurationError("Host " + entry.name());
122             }
123         }
124 
125         return ret;
126     }

128     private Map<String, ContinuousIntegration> parseContinuousIntegrations(JSONObject config, Path cwd) throws ConfigurationError {
129         Map<String, ContinuousIntegration> ret = new HashMap<>();
130 
131         if (!config.contains("ci")) {
132             return ret;
133         }
134 
135         for (var entry : config.get("ci").fields()) {
136             var url = entry.value().get("url").asString();
137             var ci = ContinuousIntegration.from(URI.create(url), entry.value().asObject());
138             if (ci.isPresent()) {
139                 ret.put(entry.name(), ci.get());
140             } else {
141                 throw new ConfigurationError("No continuous integration named with url: " + url);
142             }
143         }
144 
145         return ret;
146     }
147 
148     private Map<String, HostedRepository> parseRepositories(JSONObject config) throws ConfigurationError {
149         Map<String, HostedRepository> ret = new HashMap<>();
150 
151         if (!config.contains("repositories")) {
152             return ret;
153         }
154 
155         for (var entry : config.get("repositories").fields()) {
156             var hostName = entry.value().get("host").asString();
157             if (!repositoryHosts.containsKey(hostName)) {
158                 throw new ConfigurationError("Repository " + entry.name() + " uses undefined host '" + hostName + "'");
159             }
160             var host = repositoryHosts.get(hostName);
161             var repo = host.repository(entry.value().get("repository").asString()).orElseThrow(() ->
162                     new ConfigurationError("Repository " + entry.value().get("repository").asString() + " is not available at " + hostName)
163             );
164             ret.put(entry.name(), repo);
165         }
166 
167         return ret;
168     }
169 
170     private static class RepositoryEntry {
171         HostedRepository repository;
172         String ref;
173     }
174 
175     private RepositoryEntry parseRepositoryEntry(String entry) throws ConfigurationError {
176         var ret = new RepositoryEntry();
177         var refSeparatorIndex = entry.indexOf(':');
178         if (refSeparatorIndex >= 0) {
179             ret.ref = entry.substring(refSeparatorIndex + 1);
180             entry = entry.substring(0, refSeparatorIndex);
181         }
182         var hostSeparatorIndex = entry.indexOf('/');
183         if (hostSeparatorIndex >= 0) {
184             var hostName = entry.substring(0, hostSeparatorIndex);
185             var host = repositoryHosts.get(hostName);
186             if (!repositoryHosts.containsKey(hostName)) {
187                 throw new ConfigurationError("Repository entry " + entry + " uses undefined host '" + hostName + "'");
188             }
189             var repositoryName = entry.substring(hostSeparatorIndex + 1);
190             ret.repository = host.repository(repositoryName).orElseThrow(() ->
191                     new ConfigurationError("Repository " + repositoryName + " is not available at " + hostName)
192             );
193         } else {
194             if (!repositories.containsKey(entry)) {
195                 throw new ConfigurationError("Repository " + entry + " is not defined!");
196             }
197             ret.repository = repositories.get(entry);
198         }
199 
200         if (ret.ref == null) {
201             ret.ref = ret.repository.repositoryType() == VCS.GIT ? "master" : "default";
202         }
203 
204         return ret;
205     }

207     private IssueProject parseIssueProjectEntry(String entry) throws ConfigurationError {
208         var hostSeparatorIndex = entry.indexOf('/');
209         if (hostSeparatorIndex >= 0) {
210             var hostName = entry.substring(0, hostSeparatorIndex);
211             var host = issueHosts.get(hostName);
212             if (!issueHosts.containsKey(hostName)) {
213                 throw new ConfigurationError("Issue project entry " + entry + " uses undefined host '" + hostName + "'");
214             }
215             var issueProjectName = entry.substring(hostSeparatorIndex + 1);
216             return host.project(issueProjectName);
217         } else {
218             throw new ConfigurationError("Malformed issue project entry");
219         }
220     }

222     public static BotRunnerConfiguration parse(JSONObject config, Path cwd) throws ConfigurationError {
223         return new BotRunnerConfiguration(config, cwd);
224     }

226     public static BotRunnerConfiguration parse(JSONObject config) throws ConfigurationError {
227         return parse(config, Paths.get("."));
228     }

230     public BotConfiguration perBotConfiguration(String botName) throws ConfigurationError {
231         if (!config.contains(botName)) {
232             throw new ConfigurationError("No configuration for bot name: " + botName);
233         }
234 
235         return new BotConfiguration() {
236             @Override
237             public Path storageFolder() {
238                 if (!config.contains("storage") || !config.get("storage").contains("path")) {
239                     try {
240                         return Files.createTempDirectory("storage-" + botName);
241                     } catch (IOException e) {
242                         throw new UncheckedIOException(e);
243                     }
244                 }
245                 return Paths.get(config.get("storage").get("path").asString()).resolve(botName);
246             }
247 
248             @Override
249             public HostedRepository repository(String name) {
250                 try {
251                     var entry = parseRepositoryEntry(name);
252                     return entry.repository;
253                 } catch (ConfigurationError configurationError) {
254                     throw new RuntimeException("Couldn't find repository with name: " + name, configurationError);
255                 }
256             }
257 
258             @Override
259             public IssueProject issueProject(String name) {
260                 try {
261                     return parseIssueProjectEntry(name);
262                 } catch (ConfigurationError configurationError) {
263                     throw new RuntimeException("Couldn't find issue project with name: " + name, configurationError);
264                 }
265             }
266 
267             @Override
268             public ContinuousIntegration continuousIntegration(String name) {
269                 if (continuousIntegrations.containsKey(name)) {
270                     return continuousIntegrations.get(name);
271                 }
272                 throw new RuntimeException("Couldn't find continuous integration with name: " + name);
273             }
274 
275             @Override
276             public String repositoryRef(String name) {
277                 try {
278                     var entry = parseRepositoryEntry(name);
279                     return entry.ref;
280                 } catch (ConfigurationError configurationError) {
281                     throw new RuntimeException("Couldn't find repository with name: " + name, configurationError);
282                 }
283             }
284 
285             @Override
286             public String repositoryName(String name) {
287                 var refIndex = name.indexOf(':');
288                 if (refIndex >= 0) {
289                     name = name.substring(0, refIndex);
290                 }
291                 var orgIndex = name.lastIndexOf('/');
292                 if (orgIndex >= 0) {
293                     name = name.substring(orgIndex + 1);
294                 }
295                 return name;
296             }
297 
298             @Override
299             public JSONObject specific() {
300                 return config.get(botName).asObject();
301             }
302         };
303     }

305     /**
306      * The amount of time to wait between each invocation of Bot.getPeriodicItems.
307      * @return
308      */
309     Duration scheduledExecutionPeriod() {
310         if (!config.contains("runner") || !config.get("runner").contains("interval")) {
311             log.info("No WorkItem invocation period defined, using default value");
312             return Duration.ofSeconds(10);
313         } else {
314             return Duration.parse(config.get("runner").get("interval").asString());
315         }
316     }
317 
318     /**
319      * Number of WorkItems to execute in parallel.
320      * @return
321      */
322     Integer concurrency() {
323         if (!config.contains("runner") || !config.get("runner").contains("concurrency")) {
324             log.info("WorkItem concurrency not defined, using default value");
325             return 2;
326         } else {
327             return config.get("runner").get("concurrency").asInt();
328         }
329     }
330 
331     /**
332      * Folder that WorkItems may use to store temporary data.
333      * @return
334      */
335     Path scratchFolder() {
336         if (!config.contains("scratch") || !config.get("scratch").contains("path")) {
337             try {
338                 log.warning("No scratch folder defined, creating a temporary folder");
339                 return Files.createTempDirectory("botrunner");
340             } catch (IOException e) {
341                 throw new UncheckedIOException(e);
342             }
343         }
344         return Paths.get(config.get("scratch").get("path").asString());
345     }
346 
347     Optional<Integer> restReceiverPort() {
348         if (!config.contains("webhooks")) {
349             return Optional.empty();
350         }
351         return Optional.of(config.get("webhooks").get("port").asInt());
352     }
353 
354     Duration watchdogTimeout() {
355         if (!config.contains("runner") || !config.get("runner").contains("watchdog")) {
356             log.info("No WorkItem watchdog timeout defined, using default value");
357             return Duration.ofHours(1);
358         } else {
359             return Duration.parse(config.get("runner").get("watchdog").asString());
360         }
361     }
