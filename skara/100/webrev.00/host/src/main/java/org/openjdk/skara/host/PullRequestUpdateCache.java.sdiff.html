45     public synchronized boolean needsUpdate(PullRequest pr) {
46         // GitLab CE does not update this field on events such as adding an award
47         if (pr instanceof GitLabMergeRequest) {
48             return true;

50 
51         var uniqueId = getUniqueId(pr);

53
