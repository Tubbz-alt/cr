467     @Test
468     void retryOnFailure(TestInfo testInfo) throws IOException {
469         try (var credentials = new HostCredentials(testInfo);
470              var tempFolder = new TemporaryDirectory()) {
471 
472             var author = credentials.getHostedRepository();
473             var integrator = credentials.getHostedRepository();
474             var censusBuilder = credentials.getCensusBuilder()
475                                            .addCommitter(author.host().getCurrentUserDetails().id())
476                                            .addReviewer(integrator.host().getCurrentUserDetails().id());
477             var mergeBot = new PullRequestBot(integrator, censusBuilder.build(), "master");
478 
479             // Populate the projects repository
480             var localRepo = CheckableRepository.init(tempFolder.path(), author.getRepositoryType());
481             var masterHash = localRepo.resolve("master").orElseThrow();
482             assertFalse(CheckableRepository.hasBeenEdited(localRepo));
483             localRepo.push(masterHash, author.getUrl(), "master", true);
484 
485             // Make a change with a corresponding PR
486             var editHash = CheckableRepository.appendAndCommit(localRepo);
487             localRepo.push(editHash, author.getUrl(), "refs/heads/edit", true);
488             var pr = credentials.createPullRequest(author, "master", "edit", "This is a pull request");
489 
490             // Approve it as another user
491             var approvalPr = integrator.getPullRequest(pr.getId());
492             approvalPr.addReview(Review.Verdict.APPROVED, "Approved");
493 
494             // Let the bot check it
495             TestBotRunner.runPeriodicItems(mergeBot);
496 
497             // Pre-push to cause a failure
498             localRepo.push(editHash, author.getUrl(), "master");
499 
500             // Attempt a merge (without triggering another check)
501             pr.addComment("/integrate");
502             assertThrows(RuntimeException.class, () -> TestBotRunner.runPeriodicItems(mergeBot, wi -> wi instanceof CheckWorkItem));
503 
504             // Restore the master branch
505             localRepo.push(masterHash, author.getUrl(), "master", true);
506 
507             // The bot should now retry
508             TestBotRunner.runPeriodicItems(mergeBot, wi -> wi instanceof CheckWorkItem);
509 
510             // The bot should reply with an ok message
511             var pushed = pr.getComments().stream()
512                            .filter(comment -> comment.body().contains("Pushed as commit"))
513                            .count();
514             assertEquals(1, pushed);
515         }
516     }
