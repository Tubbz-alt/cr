483     @Override
484     public Hash commit(String message,
485                        String authorName,
486                        String authorEmail,
487                        Instant authorDate,
488                        String committerName,
489                        String committerEmail,
490                        Instant committerDate) throws IOException {
491         var cmd = Process.capture("git", "commit", "--message=" + message)
492                          .workdir(dir)
493                          .environ("GIT_AUTHOR_NAME", authorName)
494                          .environ("GIT_AUTHOR_EMAIL", authorEmail)
495                          .environ("GIT_COMMITTER_NAME", committerName)
496                          .environ("GIT_COMMITTER_EMAIL", committerEmail);
497         if (authorDate != null) {
498             var epochSecond = ZonedDateTime.ofInstant(authorDate, ZoneOffset.UTC);
499             cmd = cmd.environ("GIT_AUTHOR_DATE", epochSecond + " +0000");
500         }
501         if (committerDate != null) {
502             var epochSecond = ZonedDateTime.ofInstant(committerDate, ZoneOffset.UTC);
503             cmd = cmd.environ("GIT_COMMITTER_DATE", epochSecond + " +0000");
504         }
505         try (var p = cmd.execute()) {
506             await(p);
507             return head();
508         }
509     }
