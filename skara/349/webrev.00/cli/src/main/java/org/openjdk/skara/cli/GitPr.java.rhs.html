  64     private static String gitConfig(String key) {
  65         try {
  66             var pb = new ProcessBuilder("git", "config", key);
  67             pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
  68             pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  69             var p = pb.start();
  70 
  71             var output = new String(p.getInputStream().readAllBytes(), StandardCharsets.UTF_8);
  72             var res = p.waitFor();
  73             if (res != 0) {
  74                 return null;
  75             }
  76 
  77             return output == null ? null : output.replace("\n", "");
  78         } catch (InterruptedException e) {
  79             return null;
  80         } catch (IOException e) {
  81             return null;
  82         }
  83     }

  89     private static String getOption(String name, String subsection, Arguments arguments) {
  90         if (arguments.contains(name)) {
  91             return arguments.get(name).asString();
  92         }
  93 
  94         if (subsection != null && !subsection.isEmpty()) {
  95             var subsectionSpecific = gitConfig("pr." + subsection + "." + name);
  96             if (subsectionSpecific != null) {
  97                 return subsectionSpecific;
  98             }
  99         }
 100 
 101         return gitConfig("fork." + name);
 102     }

 108     private static boolean getSwitch(String name, String subsection, Arguments arguments) {
 109         if (arguments.contains(name)) {
 110             return true;
 111         }
 112 
 113         if (subsection != null && !subsection.isEmpty()) {
 114             var subsectionSpecific = gitConfig("pr." + subsection + "." + name);
 115             if (subsectionSpecific != null) {
 116                 return subsectionSpecific.toLowerCase().equals("true");
 117             }
 118         }
 119 
 120         var sectionSpecific = gitConfig("fork." + name);
 121         return sectionSpecific != null && sectionSpecific.toLowerCase().equals("true");
 122     }

 159     private static String statusForPullRequest(PullRequest pr) {
 160         var labels = pr.labels();
 161         if (pr.isDraft()) {
 162             return "DRAFT";
 163         } else if (labels.contains("rfr")) {
 164             return "RFR";
 165         } else if (labels.contains("ready")) {
 166             return "READY";
 167         } else if (labels.contains("outdated")) {
 168             return "OUTDATED";
 169         } else if (labels.contains("oca")) {
 170             return "OCA";
 171         } else if (labels.contains("integrated")) {
 172             return "INTEGRATED";
 173         } else {
 174             var checks = pr.checks(pr.headHash());
 175             var jcheck = Optional.ofNullable(checks.get("jcheck"));
 176             if (jcheck.isPresent()) {
 177                 var checkStatus = jcheck.get().status();
 178                 if (checkStatus == CheckStatus.IN_PROGRESS) {
 179                     return "CHECKING";
 180                 } else if (checkStatus == CheckStatus.SUCCESS) {
 181                     return "RFR";
 182                 } else if (checkStatus == CheckStatus.FAILURE) {
 183                     return "FAILURE";
 184                 }
 185             } else {
 186                 return "CHECKING";
 187             }
 188         }
 189 
 190         return "UNKNOWN";
 191     }

 193     private static String statusForCheck(Check check) {
 194         var checkStatus = check.status();
 195         if (checkStatus == CheckStatus.IN_PROGRESS) {
 196             return "RUNNING";
 197         } else if (checkStatus == CheckStatus.SUCCESS) {
 198             return "OK";
 199         } else if (checkStatus == CheckStatus.FAILURE) {
 200             return "FAILED";
 201         } else if (checkStatus == CheckStatus.CANCELLED) {
 202             return "CANCELLED";
 203         }
 204 
 205         return "UNKNOWN";
 206     }

 208     private static List<String> issuesFromPullRequest(PullRequest pr) {
 209         var issueTitleIndex = -1;
 210         var lines = pr.body().split("\n");
 211         for (var i = 0; i < lines.length; i++) {
 212             if (lines[i].startsWith("## Issue")) {
 213                 issueTitleIndex = i;
 214                 break;
 215             }
 216         }
 217 
 218         if (issueTitleIndex == -1) {
 219             return List.of();
 220         }
 221 
 222         var issues = new ArrayList<String>();
 223         for (var i = issueTitleIndex + 1; i < lines.length; i++) {
 224             var m = ISSUE_MARKDOWN_PATTERN.matcher(lines[i]);
 225             if (m.matches()) {
 226                 issues.add(m.group(1));
 227             } else {
 228                 break;
 229             }
 230         }
 231 
 232         return issues;
 233     }

 243     private static Optional<Issue> getIssue(Commit commit, String project) throws IOException {
 244         var message = CommitMessageParsers.v1.parse(commit.message());
 245         var issues = message.issues();
 246         if (issues.isEmpty()) {
 247             return getIssue(message.title(), project);
 248         } else if (issues.size() == 1) {
 249             var issue = issues.get(0);
 250             return getIssue(issue.id(), project);
 251         }
 252         return Optional.empty();
 253     }

 255     private static Optional<Issue> getIssue(Branch b, String project) throws IOException {
 256         return getIssue(b.name(), project);
 257     }

 259     private static Optional<Issue> getIssue(String s, String project) throws IOException {
 260         var m = ISSUE_ID_PATTERN.matcher(s);
 261         if (m.matches()) {
 262             var id = m.group(2);
 263             if (project == null) {
 264                 project = m.group(1);
 265             }
 266             var issueTracker = IssueTracker.from("jira", URI.create("https://bugs.openjdk.java.net"));
 267             return issueTracker.project(project).issue(id);
 268         }
 269 
 270         return Optional.empty();
 271     }

 273     private static void await(Process p) throws IOException {
 274         try {
 275             var res = p.waitFor();
 276             if (res != 0) {
 277                 throw new IOException("Unexpected exit code " + res);
 278             }
 279         } catch (InterruptedException e) {
 280             throw new IOException(e);
 281         }
 282     }

 284     private static boolean spawnEditor(ReadOnlyRepository repo, Path file) throws IOException {
 285         String editor = null;
 286         var lines = repo.config("core.editor");
 287         if (lines.size() == 1) {
 288             editor = lines.get(0);
 289         }
 290         if (editor == null) {
 291             editor = System.getenv("GIT_EDITOR");
 292         }
 293         if (editor == null) {
 294             editor = System.getenv("EDITOR");
 295         }
 296         if (editor == null) {
 297             editor = System.getenv("VISUAL");
 298         }
 299         if (editor == null) {
 300             editor = "vi";
 301         }
 302 
 303         var pb = new ProcessBuilder(editor, file.toString());
 304         pb.inheritIO();
 305         var p = pb.start();
 306         try {
 307             return p.waitFor() == 0;
 308         } catch (InterruptedException e) {
 309             throw new IOException(e);
 310         }
 311     }

 321     private static HostedRepository getHostedRepositoryFor(URI uri, ReadOnlyRepository repo, Forge host) throws IOException {
 322         HostedRepository targetRepo = null;
 323 
 324         try {
 325             var upstream = Remote.toWebURI(repo.pullPath("upstream"));
 326             targetRepo = host.repository(projectName(upstream)).orElse(null);
 327         } catch (IOException e) {
 328             // do nothing
 329         }
 330 
 331         if (targetRepo == null) {
 332             var remoteRepo = host.repository(projectName(uri)).orElseThrow(() ->
 333                     new IOException("Could not find repository at: " + uri.toString())
 334             );
 335             var parentRepo = remoteRepo.parent();
 336             targetRepo = parentRepo.isPresent() ? parentRepo.get() : remoteRepo;
 337         }
 338 
 339         return targetRepo;
 340     }

 354     private static void show(String ref, Hash hash, Path dir) throws IOException {
 355         var pb = new ProcessBuilder("git", "diff", "--binary",
 356                                                    "--patch",
 357                                                    "--find-renames=50%",
 358                                                    "--find-copies=50%",
 359                                                    "--find-copies-harder",
 360                                                    "--abbrev",
 361                                                    ref + "..." + hash.hex());
 362         if (dir != null) {
 363             pb.directory(dir.toFile());
 364         }
 365         pb.inheritIO();
 366         await(pb.start());
 367     }

 381     private static List<String> hgTags() throws IOException, InterruptedException {
 382         var pb = new ProcessBuilder("hg", "tags", "--quiet");
 383         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
 384         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 385         var p = pb.start();
 386         var bytes = p.getInputStream().readAllBytes();
 387         var exited = p.waitFor(1, TimeUnit.MINUTES);
 388         var exitValue = p.exitValue();
 389         if (!exited || exitValue != 0) {
 390             throw new IOException("'hg tags' exited with value: " + exitValue);
 391         }
 392 
 393         return Arrays.asList(new String(bytes, StandardCharsets.UTF_8).split("\n"));
 394     }

 396     private static String hgResolve(String ref) throws IOException, InterruptedException {
 397         var pb = new ProcessBuilder("hg", "log", "-r", ref, "--template", "{node}");
 398         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
 399         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 400         var p = pb.start();
 401         var bytes = p.getInputStream().readAllBytes();
 402         var exited = p.waitFor(1, TimeUnit.MINUTES);
 403         var exitValue = p.exitValue();
 404         if (!exited || exitValue != 0) {
 405             throw new IOException("'hg log' exited with value: " + exitValue);
 406         }
 407 
 408         return new String(bytes, StandardCharsets.UTF_8);
 409     }

 415     private static Path diff(String ref, Hash hash, Path dir) throws IOException {
 416         var patch = Files.createTempFile(hash.hex(), ".patch");
 417         var pb = new ProcessBuilder("git", "diff", "--binary",
 418                                                    "--patch",
 419                                                    "--find-renames=50%",
 420                                                    "--find-copies=50%",
 421                                                    "--find-copies-harder",
 422                                                    "--abbrev",
 423                                                    ref + "..." + hash.hex());
 424         if (dir != null) {
 425             pb.directory(dir.toFile());
 426         }
 427         pb.redirectOutput(patch.toFile());
 428         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 429         await(pb.start());
 430         return patch;
 431     }

 449     public static void main(String[] args) throws IOException, InterruptedException {
 450         var flags = List.of(
 451             Option.shortcut("u")
 452                   .fullname("username")
 453                   .describe("NAME")
 454                   .helptext("Username on host")
 455                   .optional(),
 456             Option.shortcut("r")

 460                   .optional(),

 463                   .describe("NAME")
 464                   .helptext("Name of target branch, defaults to 'master'")
 465                   .optional(),
 466             Option.shortcut("")
 467                   .fullname("authors")
 468                   .describe("LIST")
 469                   .helptext("Comma separated list of authors")
 470                   .optional(),
 471             Option.shortcut("")
 472                   .fullname("assignees")
 473                   .describe("LIST")
 474                   .helptext("Comma separated list of assignees")
 475                   .optional(),
 476             Option.shortcut("")
 477                   .fullname("labels")
 478                   .describe("LIST")
 479                   .helptext("Comma separated list of labels")
 480                   .optional(),
 481             Option.shortcut("")
 482                   .fullname("issues")
 483                   .describe("LIST")
 484                   .helptext("Comma separated list of issues")
 485                   .optional(),
 486             Option.shortcut("")
 487                   .fullname("columns")
 488                   .describe("id,title,author,assignees,labels")
 489                   .helptext("Comma separated list of columns to show")
 490                   .optional(),
 491             Switch.shortcut("")
 492                   .fullname("no-decoration")
 493                   .helptext("Hide any decorations when listing PRs")
 494                   .optional(),
 495             Switch.shortcut("")
 496                   .fullname("no-draft")
 497                   .helptext("Hide all pull requests in draft state")
 498                   .optional(),
 499             Switch.shortcut("")
 500                   .fullname("ignore-workspace")
 501                   .helptext("Ignore local changes in worktree and staging area when creating pull request")
 502                   .optional(),
 503             Switch.shortcut("")
 504                   .fullname("ignore-local-commits")
 505                   .helptext("Ignore local commits not pushed when creating pull request")
 506                   .optional(),
 507             Switch.shortcut("")
 508                   .fullname("publish")
 509                   .helptext("Publish the local branch before creating the pull request")
 510                   .optional(),
 511             Switch.shortcut("")
 512                   .fullname("jcheck")
 513                   .helptext("Run jcheck before creating the pull request")
 514                   .optional(),
 515             Switch.shortcut("")
 516                   .fullname("no-token")
 517                   .helptext("Do not use a personal access token (PAT). Only works for read-only operations.")
 518                   .optional(),
 519             Switch.shortcut("")
 520                   .fullname("no-checks")
 521                   .helptext("Do not show check status as part of the 'git pr status' output")
 522                   .optional(),
 523             Switch.shortcut("")
 524                   .fullname("mercurial")
 525                   .helptext("Force use of Mercurial (hg)")
 526                   .optional(),
 527             Switch.shortcut("")
 528                   .fullname("verbose")
 529                   .helptext("Turn on verbose output")
 530                   .optional(),
 531             Switch.shortcut("")
 532                   .fullname("debug")
 533                   .helptext("Turn on debugging output")
 534                   .optional(),
 535             Switch.shortcut("")
 536                   .fullname("version")
 537                   .helptext("Print the version of this tool")
 538                   .optional());
 539 
 540         var inputs = List.of(
 541             Input.position(0)
 542                  .describe("list|fetch|show|checkout|apply|integrate|approve|create|close|update|test|status")
 543                  .singular()
 544                  .optional(),
 545             Input.position(1)
 546                  .describe("ID")
 547                  .singular()
 548                  .optional()
 549         );
 550 
 551         var parser = new ArgumentParser("git-pr", flags, inputs);
 552         var arguments = parser.parse(args);
 553 
 554         if (arguments.contains("version")) {
 555             System.out.println("git-pr version: " + Version.fromManifest().orElse("unknown"));
 556             System.exit(0);
 557         }
 558 
 559         if (arguments.contains("verbose") || arguments.contains("debug")) {
 560             var level = arguments.contains("debug") ? Level.FINER : Level.FINE;
 561             Logging.setup(level);
 562         }
 563 
 564         HttpProxy.setup();
 565 
 566         var isMercurial = getSwitch("mercurial", arguments);
 567         var cwd = Path.of("").toAbsolutePath();
 568         var repo = Repository.get(cwd).orElseThrow(() -> new IOException("no git repository found at " + cwd.toString()));
 569         var remote = getOption("remote", arguments);
 570         if (remote == null) {
 571             remote = isMercurial ? "default" : "origin";
 572         }
 573         var remotePullPath = repo.pullPath(remote);
 574         var username = getOption("username", arguments);
 575         var token = isMercurial ? System.getenv("HG_TOKEN") : System.getenv("GIT_TOKEN");
 576         var uri = Remote.toWebURI(remotePullPath);
 577         var shouldUseToken = !getSwitch("no-token", arguments);
 578         var credentials = !shouldUseToken ?
 579             null :
 580             GitCredentials.fill(uri.getHost(), uri.getPath(), username, token, uri.getScheme());
 581         var forgeURI = URI.create(uri.getScheme() + "://" + uri.getHost());
 582         var forge = credentials == null ?
 583             Forge.from(forgeURI) :
 584             Forge.from(forgeURI, new Credential(credentials.username(), credentials.password()));
 585         if (forge.isEmpty()) {
 586             if (!shouldUseToken) {
 587                 if (arguments.contains("verbose")) {
 588                     System.err.println("");
 589                 }
 590                 System.err.println("warning: using git-pr with --no-token may result in rate limiting from " + forgeURI);
 591                 if (!arguments.contains("verbose")) {
 592                     System.err.println("         Re-run git-pr with --verbose to see if you are being rate limited");
 593                     System.err.println("");
 594                 }
 595             }
 596             exit("error: failed to connect to host: " + forgeURI);
 597         }
 598         var host = forge.get();
 599 
 600         var action = arguments.at(0).isPresent() ? arguments.at(0).asString() : null;
 601         if (action == null) {
 602             var lines = repo.config("pr.default");
 603             if (lines.size() == 1) {
 604                 action = lines.get(0);
 605             }
 606         }
 607 
 608         if (!shouldUseToken &&
 609             !List.of("list", "fetch", "show", "checkout", "apply").contains(action)) {
 610             System.err.println("error: --no-token can only be used with read-only operations");
 611             System.exit(1);
 612         }
 613 
 614         if (action.equals("create")) {
 615             if (isMercurial) {
 616                 var currentBookmark = repo.currentBookmark();
 617                 if (!currentBookmark.isPresent()) {
 618                     System.err.println("error: no bookmark is active, you must be on an active bookmark");
 619                     System.err.println("");
 620                     System.err.println("To create a bookmark and activate it, run:");
 621                     System.err.println("");
 622                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
 623                     System.err.println("");
 624                     System.exit(1);
 625                 }
 626 
 627                 var bookmark = currentBookmark.get();
 628                 if (bookmark.equals(new Bookmark("master"))) {
 629                     System.err.println("error: you should not create pull requests from the 'master' bookmark");
 630                     System.err.println("To create a bookmark and activate it, run:");
 631                     System.err.println("");
 632                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
 633                     System.err.println("");
 634                     System.exit(1);
 635                 }
 636 
 637                 var tags = hgTags();
 638                 var upstreams = tags.stream()
 639                                     .filter(t -> t.endsWith(bookmark.name()))
 640                                     .collect(Collectors.toList());
 641                 if (upstreams.isEmpty()) {
 642                     System.err.println("error: there is no remote branch for the local bookmark '" + bookmark.name() + "'");
 643                     System.err.println("");
 644                     System.err.println("To create a remote branch and push the commits for your local branch, run:");
 645                     System.err.println("");
 646                     System.err.println("    hg push --bookmark " + bookmark.name());
 647                     System.err.println("");
 648                     System.exit(1);
 649                 }
 650 
 651                 var tagsAndHashes = new HashMap<String, String>();
 652                 for (var tag : tags) {
 653                     tagsAndHashes.put(tag, hgResolve(tag));
 654                 }
 655                 var bookmarkHash = hgResolve(bookmark.name());
 656                 if (!tagsAndHashes.containsValue(bookmarkHash)) {
 657                     System.err.println("error: there are local commits on bookmark '" + bookmark.name() + "' not present in a remote repository");
 658                     System.err.println("");
 659 
 660                     if (upstreams.size() == 1) {
 661                         System.err.println("To push the local commits to the remote repository, run:");
