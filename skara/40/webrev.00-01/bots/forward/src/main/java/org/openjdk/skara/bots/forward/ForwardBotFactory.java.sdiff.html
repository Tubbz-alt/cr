28 import java.nio.file.Path;
29 import java.util.*;
30 import java.util.logging.Logger;
31 
32 public class ForwardBotFactory implements BotFactory {
33     private final Logger log = Logger.getLogger("org.openjdk.skara.bots");;
34 
35     @Override
36     public String name() {
37         return "forward";
38     }
39 
40     @Override
41     public List<Bot> create(BotConfiguration configuration) {
42         var ret = new ArrayList<Bot>();






43         var specific = configuration.specific();
44 
45         for (var repo : specific.get("repositories").fields()) {
46             var repoName = repo.name();
47             var from = repo.value().get("from").asString().split(":");
48             var fromRepo = configuration.repository(from[0]);
49             var fromBranch = new Branch(from[1]);
50 
51             var to = repo.value().get("to").asString().split(":");
52             var toRepo = configuration.repository(to[0]);
53             var toBranch = new Branch(to[1]);
54 
55             var bot = new ForwardBot(fromRepo, fromBranch, toRepo, toBranch);
56             log.info("Setting up forwarding from " +
57                      fromRepo.getName() + ":" + fromBranch.name() +
58                      "to " + toRepo.getName() + ":" + toBranch.name());
59             ret.add(bot);
60         }
61 
62         return ret;
63     }
64 }

23 package org.openjdk.skara.bots.forward;
24 
25 import org.openjdk.skara.bot.*;
26 import org.openjdk.skara.vcs.Branch;
27 
28 import java.io.IOException;
29 import java.io.UncheckedIOException;
30 import java.nio.file.Files;
31 import java.nio.file.Path;
32 import java.util.*;
33 import java.util.logging.Logger;
34 
35 public class ForwardBotFactory implements BotFactory {
36     private final Logger log = Logger.getLogger("org.openjdk.skara.bots");;
37 
38     @Override
39     public String name() {
40         return "forward";
41     }
42 
43     @Override
44     public List<Bot> create(BotConfiguration configuration) {
45         var ret = new ArrayList<Bot>();
46         var storage = configuration.storageFolder();
47         try {
48             Files.createDirectories(storage);
49         } catch (IOException e) {
50             throw new UncheckedIOException(e);
51         }
52         var specific = configuration.specific();
53 
54         for (var repo : specific.get("repositories").fields()) {
55             var repoName = repo.name();
56             var from = repo.value().get("from").asString().split(":");
57             var fromRepo = configuration.repository(from[0]);
58             var fromBranch = new Branch(from[1]);
59 
60             var to = repo.value().get("to").asString().split(":");
61             var toRepo = configuration.repository(to[0]);
62             var toBranch = new Branch(to[1]);
63 
64             var bot = new ForwardBot(storage, fromRepo, fromBranch, toRepo, toBranch);
65             log.info("Setting up forwarding from " +
66                      fromRepo.getName() + ":" + fromBranch.name() +
67                      "to " + toRepo.getName() + ":" + toBranch.name());
68             ret.add(bot);
69         }
70 
71         return ret;
72     }
73 }
