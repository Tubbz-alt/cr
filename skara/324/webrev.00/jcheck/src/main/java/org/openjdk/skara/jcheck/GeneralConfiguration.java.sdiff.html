23 package org.openjdk.skara.jcheck;

25 import org.openjdk.skara.ini.Section;

27 import java.util.Optional;
28 
29 public class GeneralConfiguration {
30     private static final GeneralConfiguration DEFAULT = new GeneralConfiguration(null, null, null, null);
31 
32     private final String project;
33     private final String repository;
34     private final String jbs;
35     private final String version;
36 
37     private GeneralConfiguration(String project, String repository, String jbs, String version) {
38         this.project = project;
39         this.repository = repository;
40         this.jbs = jbs;
41         this.version = version;
42     }
43 
44     public String project() {
45         return project;
46     }
47 
48     public String repository() {
49         return repository;
50     }
51 
52     public String jbs() {
53         return jbs;
54     }
55 
56     public Optional<String> version() {
57         return Optional.ofNullable(version);
58     }
59 
60     static String name() {
61         return "general";
62     }
63 
64     static GeneralConfiguration parse(Section s) {
65         if (s == null) {
66             return DEFAULT;
67         }
68 
69         var project = s.get("project", DEFAULT.project());
70         var repository = s.get("repository", DEFAULT.repository());
71         var jbs = s.get("jbs", DEFAULT.jbs());
72         var version = s.get("version", DEFAULT.version().orElse(null));
73         return new GeneralConfiguration(project, repository, jbs, version);
74     }
75 }
