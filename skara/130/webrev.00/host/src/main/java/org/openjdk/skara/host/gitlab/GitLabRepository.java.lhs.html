 36 
 37 public class GitLabRepository implements HostedRepository {
 38     private final GitLabHost gitLabHost;
 39     private final String projectName;
 40     private final RestRequest request;
 41     private final JSONValue json;
 42     private final Pattern mergeRequestPattern;
 43 
 44     public GitLabRepository(GitLabHost gitLabHost, String projectName) {
 45         this.gitLabHost = gitLabHost;
 46         json = gitLabHost.getProjectInfo(projectName);
 47         this.projectName = json.get("path_with_namespace").asString();
 48 
 49         var id = json.get("id").asInt();
 50         var baseApi = URIBuilder.base(gitLabHost.getUri())
 51                 .setPath("/api/v4/projects/" + id + "/")
 52                 .build();
 53 
 54         request = gitLabHost.getPat()
 55                             .map(pat -> new RestRequest(baseApi, () -> Arrays.asList("Private-Token", pat.token())))
 56                             .orElseGet(() -> new RestRequest(baseApi));
 57 
 58         var urlPattern = URIBuilder.base(gitLabHost.getUri())
 59                                    .setPath("/" + projectName + "/merge_requests/").build();
 60         mergeRequestPattern = Pattern.compile(urlPattern.toString() + "(\\d+)");
 61     }
 62 
 63     @Override
 64     public Host host() {
 65         return gitLabHost;
 66     }
 67 
 68     @Override
 69     public Optional<HostedRepository> getParent() {
 70         if (json.contains("forked_from_project")) {
 71             var parent = json.get("forked_from_project").get("path_with_namespace").asString();
 72             return Optional.of(new GitLabRepository(gitLabHost, parent));
 73         }
 74         return Optional.empty();
 75     }
 76 
 77     @Override
 78     public PullRequest createPullRequest(HostedRepository target,
 79                                          String targetRef,
 80                                          String sourceRef,
 81                                          String title,
 82                                          List<String> body) {
 83         if (!(target instanceof GitLabRepository)) {
 84             throw new IllegalArgumentException("target must be a GitLab repository");
 85         }
 86 
 87         var pr = request.post("merge_requests")
 88                         .body("source_branch", sourceRef)
 89                         .body("target_branch", targetRef)
 90                         .body("title", title)
 91                         .body("description", String.join("\n", body))
 92                         .body("target_project_id", Long.toString(target.getId()))
 93                         .execute();
 94 
 95         var targetRepo = (GitLabRepository) target;
 96         return new GitLabMergeRequest(targetRepo, pr, targetRepo.request);
 97     }
 98 
 99     @Override
100     public PullRequest getPullRequest(String id) {
101         var pr = request.get("merge_requests/" + id).execute();
102         return new GitLabMergeRequest(this, pr, request);
103     }
104 
105     @Override
106     public List<PullRequest> getPullRequests() {
107         return request.get("merge_requests")
108                       .param("state", "opened")
109                       .execute().stream()
110                       .map(value -> new GitLabMergeRequest(this, value, request))
111                       .collect(Collectors.toList());
112     }
113 





114     @Override
115     public Optional<PullRequest> parsePullRequestUrl(String url) {
116         var matcher = mergeRequestPattern.matcher(url);
117         if (matcher.find()) {
118             return Optional.of(getPullRequest(matcher.group(1)));
119         } else {
120             return Optional.empty();
121         }
122     }

129     @Override
130     public URI getUrl() {
131         var builder = URIBuilder
132                 .base(gitLabHost.getUri())
133                 .setPath("/" + projectName + ".git");
134         gitLabHost.getPat().ifPresent(pat -> builder.setAuthentication(pat.userName() + ":" + pat.token()));
135         return builder.build();
136     }

145     @Override
146     public URI getWebUrl(Hash hash) {

148                          .setPath("/" + projectName + "/commit/" + hash.abbreviate())

152     @Override
153     public VCS getRepositoryType() {
154         return VCS.GIT;
155     }
156 
157     @Override
158     public String getFileContents(String filename, String ref) {
159         var confName = URLEncoder.encode(filename, StandardCharsets.UTF_8);
160         var conf = request.get("repository/files/" + confName)
161                           .param("ref", ref)
162                           .onError(response -> {
163                               // Retry once with additional escaping of the path fragment
164                               var escapedConfName = URLEncoder.encode(confName, StandardCharsets.UTF_8);
165                               return request.get("repository/files/" + escapedConfName)
166                                             .param("ref", ref).execute();
167                           })
168                           .execute();
169         var content = Base64.getDecoder().decode(conf.get("content").asString());
170         return new String(content, StandardCharsets.UTF_8);
171     }
172 
173     @Override
174     public String getNamespace() {
175         return URIBuilder.base(gitLabHost.getUri()).build().getHost();
176     }
177 
178     @Override
179     public Optional<WebHook> parseWebHook(JSONValue body) {
180         if (!body.contains("object_kind")) {
181             return Optional.empty();
182         }
183         if (!body.contains("project") || !body.get("project").contains("path_with_namespace")) {
184             return Optional.empty();
185         }
186         if (!body.get("project").get("path_with_namespace").asString().equals(projectName)) {
187             return Optional.empty();
188         }
189 
190         int id = -1;
191 
192         if (body.get("object_kind").asString().equals("merge_request")) {
193             if (!body.contains("object_attributes") || !body.get("object_attributes").contains("iid")) {
194                 return Optional.empty();
195             }
196             id = body.get("object_attributes").get("iid").asInt();
197         }
198 
199         if (body.contains("merge_request")) {
200             if (!body.get("merge_request").contains("iid")) {
201                 return Optional.empty();
202             }
203             id = body.get("merge_request").get("iid").asInt();
204         }
205 
206         if (id != -1) {
207             var pr = getPullRequest(Integer.toString(id));
208             var webHook = new WebHook(List.of(pr));
209             return Optional.of(webHook);
210         } else {
211             return Optional.empty();
212         }
213     }
214 
215     @Override
216     public HostedRepository fork() {
217         var namespace = gitLabHost.getCurrentUserDetails().userName();
218         request.post("fork")
219                .body("namespace", namespace)
220                .onError(r -> r.statusCode() == 409 ? JSON.object().put("exists", true) : null)
221                .execute();
222         var nameOnlyStart = projectName.lastIndexOf('/');
223         var nameOnly = nameOnlyStart >= 0 ? projectName.substring(nameOnlyStart + 1) : projectName;
224         var forkedRepoName = namespace + "/" + nameOnly;
225         while (!gitLabHost.isProjectForkComplete(forkedRepoName)) {
226             try {
227                 Thread.sleep(Duration.ofSeconds(1).toMillis());
228             } catch (InterruptedException e) {
229                 e.printStackTrace();
230             }
231         }
232         return gitLabHost.getRepository(forkedRepoName);
233     }

240     @Override
241     public Hash getBranchHash(String ref) {
242         var branch = request.get("repository/branches/" + ref).execute();

244     }
