 23 package org.openjdk.skara.host.github;
 24 
 25 import org.openjdk.skara.host.*;
 26 import org.openjdk.skara.host.network.*;
 27 import org.openjdk.skara.json.JSONValue;
 28 import org.openjdk.skara.vcs.*;
 29 
 30 import java.net.URI;
 31 import java.nio.charset.StandardCharsets;
 32 import java.util.*;
 33 import java.util.regex.Pattern;
 34 import java.util.stream.Collectors;
 35 
 36 public class GitHubRepository implements HostedRepository {
 37     private final GitHubHost gitHubHost;
 38     private final String repository;
 39     private final RestRequest request;
 40     private final JSONValue json;
 41     private final Pattern pullRequestPattern;
 42 
 43     GitHubRepository(GitHubHost gitHubHost, String repository) {
 44         this.gitHubHost = gitHubHost;
 45         this.repository = repository;
 46 
 47         var apiBase = URIBuilder
 48                 .base(gitHubHost.getURI())
 49                 .appendSubDomain("api")
 50                 .setPath("/repos/" + repository + "/")
 51                 .build();
 52         request = new RestRequest(apiBase, () -> Arrays.asList(
 53                 "Authorization", "token " + gitHubHost.getInstallationToken(),
 54                 "Accept", "application/vnd.github.machine-man-preview+json",
 55                 "Accept", "application/vnd.github.antiope-preview+json"));
 56         json = gitHubHost.getProjectInfo(repository);
 57         var urlPattern = gitHubHost.getWebURI("/" + repository + "/pull/").toString();
 58         pullRequestPattern = Pattern.compile(urlPattern + "(\\d+)");
 59     }
 60 
 61     @Override
 62     public Optional<HostedRepository> getParent() {
 63         if (json.get("fork").asBoolean()) {
 64             var parent = json.get("parent").get("full_name").asString();
 65             return Optional.of(new GitHubRepository(gitHubHost, parent));
 66         }
 67         return Optional.empty();
 68     }
 69 
 70     @Override
 71     public Host host() {
 72         return gitHubHost;
 73     }
 74 
 75     @Override
 76     public PullRequest createPullRequest(HostedRepository target,
 77                                          String targetRef,
 78                                          String sourceRef,
 79                                          String title,
 80                                          List<String> body) {
 81         if (!(target instanceof GitHubRepository)) {
 82             throw new IllegalArgumentException("target repository must be a GitHub repository");
 83         }
 84 
 85         var upstream = (GitHubRepository) target;
 86         var user = host().getCurrentUserDetails().userName();
 87         var namespace = user.endsWith("[bot]") ? "" : user + ":";
 88         var pr = upstream.request.post("pulls")
 89                                  .body("title", title)
 90                                  .body("head", namespace + sourceRef)
 91                                  .body("base", targetRef)
 92                                  .body("body", String.join("\n", body))
 93                                  .execute();
 94 
 95         return new GitHubPullRequest(upstream, pr, request);
 96     }
 97 
 98     @Override
 99     public PullRequest getPullRequest(String id) {
100         var pr = request.get("pulls/" + id).execute();
101         return new GitHubPullRequest(this, pr, request);
102     }
103 
104     @Override
105     public List<PullRequest> getPullRequests() {
106         return request.get("pulls").execute().asArray().stream()
107                       .map(jsonValue -> new GitHubPullRequest(this, jsonValue, request))
108                       .collect(Collectors.toList());
109     }
110 
111     @Override
112     public List<PullRequest> findPullRequestsWithComment(String author, String body) {
113         var query = "\"" + body + "\" in:comments type:pr repo:" + repository;
114         if (author != null) {
115             query += " commenter:" + author;
116         }
117         var result = gitHubHost.runSearch(query);
118         return result.get("items").stream()
119                 .map(jsonValue -> new GitHubPullRequest(this, jsonValue, request))
120                 .collect(Collectors.toList());
121     }
122 
123     @Override
124     public Optional<PullRequest> parsePullRequestUrl(String url) {
125         var matcher = pullRequestPattern.matcher(url);
126         if (matcher.find()) {
127             return Optional.of(getPullRequest(matcher.group(1)));
128         } else {
129             return Optional.empty();
130         }
131     }
132 
133     @Override
134     public String getName() {
135         return repository;
136     }
137 
138     @Override
139     public URI getUrl() {
140         return URIBuilder
141                 .base(gitHubHost.getURI())
142                 .setPath("/" + repository + ".git")
143                 .setAuthentication("x-access-token:" + gitHubHost.getInstallationToken())
144                 .build();
145     }
146 
147     @Override
148     public URI getWebUrl() {
149         var endpoint = "/" + repository;
150         return gitHubHost.getWebURI(endpoint);
151     }
152 
153     @Override
154     public URI getWebUrl(Hash hash) {
155         var endpoint = "/" + repository + "/commit/" + hash.abbreviate();
156         return gitHubHost.getWebURI(endpoint);
157     }
158 
159     @Override
160     public VCS getRepositoryType() {
161         return VCS.GIT;
162     }
163 
164     @Override
165     public String getFileContents(String filename, String ref) {
166         var conf = request.get("contents/" + filename)
167                           .param("ref", ref)
168                           .execute().asObject();
169         // Content may contain newline characters
170         return conf.get("content").asString().lines()
171                    .map(line -> new String(Base64.getDecoder().decode(line), StandardCharsets.UTF_8))
172                    .collect(Collectors.joining());
173     }
174 
175     @Override
176     public String getNamespace() {
177         return URIBuilder.base(gitHubHost.getURI()).build().getHost();
178     }
179 
180     @Override
181     public Optional<WebHook> parseWebHook(JSONValue body) {
182         throw new RuntimeException("not implemented yet");
183     }
184 
185     @Override
186     public HostedRepository fork() {
187         var response = request.post("forks").execute();
188         return gitHubHost.getRepository(response.get("full_name").asString());
189     }
190 
191     @Override
192     public long getId() {
193         return json.get("id").asLong();
194     }
195 
196     @Override
197     public Hash getBranchHash(String ref) {
198         var branch = request.get("branches/" + ref).execute();
199         return new Hash(branch.get("commit").get("sha").asString());
200     }
201 }
