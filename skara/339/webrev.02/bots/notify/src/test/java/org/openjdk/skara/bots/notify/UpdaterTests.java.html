  47     private List<Path> findJsonFiles(Path folder, String partialName) throws IOException {
  48         return Files.walk(folder)
  49                     .filter(path -> path.toString().endsWith(".json"))
  50                     .filter(path -> path.toString().contains(partialName))
  51                     .collect(Collectors.toList());
  52     }

  54     private StorageBuilder<Tag> createTagStorage(HostedRepository repository) {
  55         return new StorageBuilder<Tag>("tags.txt")
  56                 .remoteRepository(repository, "history", "Duke", "duke@openjdk.java.net", "Updated tags");
  57     }
  58 
  59     private StorageBuilder<ResolvedBranch> createBranchStorage(HostedRepository repository) {
  60         return new StorageBuilder<ResolvedBranch>("branches.txt")
  61                 .remoteRepository(repository, "history", "Duke", "duke@openjdk.java.net", "Updated branches");
  62     }
  63 
  64     private StorageBuilder<PullRequestIssues> createPullRequestIssuesStorage(HostedRepository repository) {
  65         return new StorageBuilder<PullRequestIssues>("prissues.txt")
  66                 .remoteRepository(repository, "history", "Duke", "duke@openjdk.java.net", "Updated prissues");
  67     }

 108     @Test
 109     void testJsonUpdaterTag(TestInfo testInfo) throws IOException {
 110         try (var credentials = new HostCredentials(testInfo);
 111              var tempFolder = new TemporaryDirectory()) {
 112             var repo = credentials.getHostedRepository();
 113             var localRepoFolder = tempFolder.path().resolve("repo");
 114             var localRepo = CheckableRepository.init(localRepoFolder, repo.repositoryType());
 115             credentials.commitLock(localRepo);
 116             var masterHash = localRepo.resolve("master").orElseThrow();
 117             localRepo.tag(masterHash, "jdk-12+1", "Added tag 1", "Duke", "duke@openjdk.java.net");
 118             localRepo.pushAll(repo.url());
 119 
 120             var tagStorage = createTagStorage(repo);
 121             var branchStorage = createBranchStorage(repo);
 122             var prIssuesStorage = createPullRequestIssuesStorage(repo);
 123             var jsonFolder = tempFolder.path().resolve("json");
 124             Files.createDirectory(jsonFolder);
 125             var storageFolder =tempFolder.path().resolve("storage");
 126 
 127             var updater = new JsonUpdater(jsonFolder, "12", "team");
 128             var notifyBot = new NotifyBot(repo, storageFolder, Pattern.compile("master"), tagStorage, branchStorage,
 129                                           prIssuesStorage, List.of(updater), List.of(), Set.of(), Map.of());
 130 
 131             TestBotRunner.runPeriodicItems(notifyBot);
 132             assertEquals(List.of(), findJsonFiles(jsonFolder, ""));
 133 
 134             var editHash = CheckableRepository.appendAndCommit(localRepo, "Another line", "23456789: More fixes");
 135             localRepo.fetch(repo.url(), "history:history");
 136             localRepo.tag(editHash, "jdk-12+2", "Added tag 2", "Duke", "duke@openjdk.java.net");
 137             var editHash2 = CheckableRepository.appendAndCommit(localRepo, "Another line", "34567890: Even more fixes");
 138             localRepo.tag(editHash2, "jdk-12+4", "Added tag 3", "Duke", "duke@openjdk.java.net");
 139             localRepo.pushAll(repo.url());
 140 
 141             TestBotRunner.runPeriodicItems(notifyBot);
 142             var jsonFiles = findJsonFiles(jsonFolder, "");
 143             assertEquals(3, jsonFiles.size());
 144 
 145             for (var file : jsonFiles) {
 146                 var jsonData = Files.readString(file, StandardCharsets.UTF_8);
 147                 var json = JSON.parse(jsonData);
 148 
 149                 if (json.asArray().get(0).contains("date")) {
 150                     assertEquals(2, json.asArray().size());
 151                     assertEquals(List.of("23456789"), json.asArray().get(0).get("issue").asArray().stream()
 152                                                           .map(JSONValue::asString)
 153                                                           .collect(Collectors.toList()));
 154                     assertEquals(repo.webUrl(editHash).toString(), json.asArray().get(0).get("url").asString());
 155                     assertEquals("team", json.asArray().get(0).get("build").asString());
 156                     assertEquals(List.of("34567890"), json.asArray().get(1).get("issue").asArray().stream()
 157                                                           .map(JSONValue::asString)
 158                                                           .collect(Collectors.toList()));
 159                     assertEquals(repo.webUrl(editHash2).toString(), json.asArray().get(1).get("url").asString());
 160                     assertEquals("team", json.asArray().get(1).get("build").asString());
 161                 } else {
 162                     assertEquals(1, json.asArray().size());
 163                     if (json.asArray().get(0).get("build").asString().equals("b02")) {
 164                         assertEquals(List.of("23456789"), json.asArray().get(0).get("issue").asArray().stream()
 165                                                               .map(JSONValue::asString)
 166                                                               .collect(Collectors.toList()));
 167                     } else {
 168                         assertEquals("b04", json.asArray().get(0).get("build").asString());
 169                         assertEquals(List.of("34567890"), json.asArray().get(0).get("issue").asArray().stream()
 170                                                               .map(JSONValue::asString)
 171                                                               .collect(Collectors.toList()));
 172                     }
 173                 }
 174             }
 175         }
 176     }

 178     @Test
 179     void testMailingList(TestInfo testInfo) throws IOException {
 180         try (var listServer = new TestMailmanServer();
 181              var credentials = new HostCredentials(testInfo);
 182              var tempFolder = new TemporaryDirectory()) {
 183             var repo = credentials.getHostedRepository();
 184             var repoFolder = tempFolder.path().resolve("repo");
 185             var localRepo = CheckableRepository.init(repoFolder, repo.repositoryType());
 186             var masterHash = localRepo.resolve("master").orElseThrow();
 187             credentials.commitLock(localRepo);
 188             localRepo.pushAll(repo.url());
 189 
 190             var listAddress = EmailAddress.parse(listServer.createList("test"));
 191             var mailmanServer = MailingListServerFactory.createMailmanServer(listServer.getArchive(), listServer.getSMTP(), Duration.ZERO);
 192             var mailmanList = mailmanServer.getList(listAddress.address());
 193             var tagStorage = createTagStorage(repo);
 194             var branchStorage = createBranchStorage(repo);
 195             var prIssuesStorage = createPullRequestIssuesStorage(repo);
 196             var storageFolder = tempFolder.path().resolve("storage");
 197 
 198             var sender = EmailAddress.from("duke", "duke@duke.duke");
 199             var updater = new MailingListUpdater(mailmanList, listAddress, sender, null, false, MailingListUpdater.Mode.ALL,
 200                                                  Map.of("extra1", "value1", "extra2", "value2"), Pattern.compile("none"));
 201             var notifyBot = new NotifyBot(repo, storageFolder, Pattern.compile("master"), tagStorage, branchStorage,
 202                                           prIssuesStorage, List.of(updater), List.of(), Set.of(), Map.of());
 203 
 204             // No mail should be sent on the first run as there is no history
 205             TestBotRunner.runPeriodicItems(notifyBot);
 206             assertThrows(RuntimeException.class, () -> listServer.processIncoming(Duration.ofMillis(1)));
 207 
 208             var editHash = CheckableRepository.appendAndCommit(localRepo, "Another line", "23456789: More fixes");
 209             localRepo.push(editHash, repo.url(), "master");
 210             TestBotRunner.runPeriodicItems(notifyBot);
 211             listServer.processIncoming();
 212 
 213             var conversations = mailmanList.conversations(Duration.ofDays(1));
 214             var email = conversations.get(0).first();
 215             assertEquals(listAddress, email.sender());
 216             assertEquals(sender, email.author());
 217             assertEquals(email.recipients(), List.of(listAddress));
 218             assertTrue(email.subject().contains(": 23456789: More fixes"));
 219             assertFalse(email.subject().contains("master"));
 220             assertTrue(email.body().contains("Changeset: " + editHash.abbreviate()));
 221             assertTrue(email.body().contains("23456789: More fixes"));
 222             assertFalse(email.body().contains("Committer"));
 223             assertFalse(email.body().contains(masterHash.abbreviate()));
 224             assertTrue(email.hasHeader("extra1"));
 225             assertEquals("value1", email.headerValue("extra1"));
 226             assertTrue(email.hasHeader("extra2"));
 227             assertEquals("value2", email.headerValue("extra2"));
 228         }

 412     @Test
 413     void testMailingListPROnly(TestInfo testInfo) throws IOException {
 414         try (var listServer = new TestMailmanServer();
 415              var credentials = new HostCredentials(testInfo);
 416              var tempFolder = new TemporaryDirectory()) {
 417             var repo = credentials.getHostedRepository();
 418             var repoFolder = tempFolder.path().resolve("repo");
 419             var localRepo = CheckableRepository.init(repoFolder, repo.repositoryType());
 420             var masterHash = localRepo.resolve("master").orElseThrow();
 421             credentials.commitLock(localRepo);
 422             localRepo.pushAll(repo.url());
 423 
 424             var listAddress = EmailAddress.parse(listServer.createList("test"));
 425             var mailmanServer = MailingListServerFactory.createMailmanServer(listServer.getArchive(), listServer.getSMTP(), Duration.ZERO);
 426             var mailmanList = mailmanServer.getList(listAddress.address());
 427             var tagStorage = createTagStorage(repo);
 428             var branchStorage = createBranchStorage(repo);
 429             var prIssuesStorage = createPullRequestIssuesStorage(repo);
 430             var storageFolder = tempFolder.path().resolve("storage");
 431 
 432             var sender = EmailAddress.from("duke", "duke@duke.duke");
 433             var author = EmailAddress.from("author", "author@duke.duke");
 434             var updater = new MailingListUpdater(mailmanList, listAddress, sender, author, false,
 435                                                  MailingListUpdater.Mode.PR_ONLY, Map.of("extra1", "value1"),
 436                                                  Pattern.compile(".*"));
 437             var notifyBot = new NotifyBot(repo, storageFolder, Pattern.compile("master"), tagStorage, branchStorage,
 438                                           prIssuesStorage, List.of(updater), List.of(), Set.of(), Map.of());
 439 
 440             // No mail should be sent on the first run as there is no history
 441             TestBotRunner.runPeriodicItems(notifyBot);
 442             assertThrows(RuntimeException.class, () -> listServer.processIncoming(Duration.ofMillis(1)));
 443 
 444             var editHash = CheckableRepository.appendAndCommit(localRepo, "Another line", "23456789: More fixes");
 445             localRepo.push(editHash, repo.url(), "edit");
 446             var pr = credentials.createPullRequest(repo, "master", "edit", "RFR: My PR");
 447 
 448             // Create a potentially conflicting one
 449             var otherHash = CheckableRepository.appendAndCommit(localRepo, "Another line", "23456789: More fixes");
 450             localRepo.push(otherHash, repo.url(), "other");
 451             var otherPr = credentials.createPullRequest(repo, "master", "other", "RFR: My other PR");
 452 
 453             // PR hasn't been integrated yet, so there should be no mail
 454             TestBotRunner.runPeriodicItems(notifyBot);
 455             assertThrows(RuntimeException.class, () -> listServer.processIncoming(Duration.ofMillis(1)));
 456 
 457             // Simulate an RFR email
 458             var rfr = Email.create(sender, "RFR: My PR", "PR: " + pr.webUrl().toString())
 459                     .recipient(listAddress)
 460                     .build();
 461             mailmanList.post(rfr);
 462             listServer.processIncoming();
 463 
 464             // And an integration
 465             pr.addComment("Pushed as commit " + editHash.hex() + ".");
 466             localRepo.push(editHash, repo.url(), "master");
 467             TestBotRunner.runPeriodicItems(notifyBot);
 468             listServer.processIncoming();
 469 
 470             var conversations = mailmanList.conversations(Duration.ofDays(1));
 471             assertEquals(1, conversations.size());
 472             var first = conversations.get(0).first();
 473             var email = conversations.get(0).replies(first).get(0);
 474             assertEquals(listAddress, email.sender());
 475             assertEquals(author, email.author());
 476             assertEquals(email.recipients(), List.of(listAddress));
 477             assertEquals("Re: [Integrated] RFR: My PR", email.subject());
 478             assertFalse(email.subject().contains("master"));

1025              var tempFolder = new TemporaryDirectory()) {
1026             var repo = credentials.getHostedRepository();
1027             var reviewer = credentials.getHostedRepository();
1028             var repoFolder = tempFolder.path().resolve("repo");

1032

1034             var branchStorage = createBranchStorage(repo);
1035             var prIssuesStorage = createPullRequestIssuesStorage(repo);
1036             var storageFolder = tempFolder.path().resolve("storage");
1037 
1038             var issueProject = credentials.getIssueProject();
1039             var reviewIcon = URI.create("http://www.example.com/review.png");
1040             var updater = new IssueUpdater(issueProject, true, reviewIcon, false, null, false, null);
1041             var notifyBot = new NotifyBot(repo, storageFolder, Pattern.compile("master"), tagStorage, branchStorage,
1042                                           prIssuesStorage, List.of(), List.of(updater), Set.of("rfr"),
1043                                           Map.of(reviewer.forge().currentUser().userName(), Pattern.compile("This is now ready")));
1044 
1045             // Initialize history
1046             TestBotRunner.runPeriodicItems(notifyBot);
1047 
1048             // Create an issue and a pull request to fix it
1049             var issue = issueProject.createIssue("This is an issue", List.of("Indeed"));
1050             var editHash = CheckableRepository.appendAndCommit(localRepo, "Another line", "Fix that issue");
1051             localRepo.push(editHash, repo.url(), "edit", true);
1052             var pr = credentials.createPullRequest(repo, "edit", "master", issue.id() + ": Fix that issue");
1053             pr.setBody("\n\n## Issue\n[" + issue.id() + "](http://www.test.test/): The issue");
1054             TestBotRunner.runPeriodicItems(notifyBot);
1055 
1056             // The issue should not yet contain a link to the PR
1057             var links = issue.links();
1058             assertEquals(0, links.size());
1059 
1060             // Just a label isn't enough
1061             pr.addLabel("rfr");
1062             TestBotRunner.runPeriodicItems(notifyBot);
1063             links = issue.links();
1064             assertEquals(0, links.size());
1065 
1066             // Neither is just a comment
1067             pr.removeLabel("rfr");
1068             var reviewPr = reviewer.pullRequest(pr.id());
1069             reviewPr.addComment("This is now ready");
1070             TestBotRunner.runPeriodicItems(notifyBot);
1071             links = issue.links();
1072             assertEquals(0, links.size());
1073 
1074             // Both are needed
1075             pr.addLabel("rfr");
1076             TestBotRunner.runPeriodicItems(notifyBot);
1077 
1078             // The issue should now contain a link to the PR
1079             links = issue.links();
1080             assertEquals(1, links.size());
1081             assertEquals(pr.webUrl(), links.get(0).uri());
1082             assertEquals(reviewIcon, links.get(0).iconUrl().orElseThrow());
1083 
1084             // Add another issue
1085             var issue2 = issueProject.createIssue("This is another issue", List.of("Yes indeed"));
1086             pr.setBody("\n\n## Issues\n[" + issue.id() + "](http://www.test.test/): The issue\n[" + issue2.id() +
1087                                "](http://www.test2.test/): The second issue");
1088             TestBotRunner.runPeriodicItems(notifyBot);
1089 
1090             // Both issues should contain a link to the PR
1091             var links1 = issue.links();
1092             assertEquals(1, links1.size());
1093             assertEquals(pr.webUrl(), links1.get(0).uri());
1094             var links2 = issue2.links();
1095             assertEquals(1, links2.size());
1096             assertEquals(pr.webUrl(), links2.get(0).uri());
1097 
1098             // Drop the first one
1099             pr.setBody("\n\n## Issue\n[" + issue2.id() + "](http://www.test2.test/): That other issue");
1100             TestBotRunner.runPeriodicItems(notifyBot);
1101 
1102             // Only the second issue should now contain a link to the PR
1103             links1 = issue.links();
1104             assertEquals(0, links1.size());
1105             links2 = issue2.links();
1106             assertEquals(1, links2.size());
1107             assertEquals(pr.webUrl(), links2.get(0).uri());
1108         }
