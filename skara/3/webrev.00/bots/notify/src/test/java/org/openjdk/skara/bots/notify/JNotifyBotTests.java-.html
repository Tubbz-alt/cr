 39 import static org.junit.jupiter.api.Assertions.assertEquals;

 41 class JNotifyBotTests {
 42 
 43     private List<Path> findJsonFiles(Path folder, String partialName) throws IOException {
 44         return Files.walk(folder)
 45                     .filter(path -> path.toString().endsWith(".json"))
 46                     .filter(path -> path.toString().contains(partialName))
 47                     .collect(Collectors.toList());
 48     }
 49 
 50     private StorageBuilder<Tag> createTagStorage(HostedRepository repository) throws IOException {
 51         return new StorageBuilder<Tag>("tags.txt")
 52                 .remoteRepository(repository, "refs/heads/history", "Duke", "duke@openjdk.java.net", "Updated tags");
 53     }
 54 
 55     private StorageBuilder<ResolvedBranch> createBranchStorage(HostedRepository repository) throws IOException {
 56         return new StorageBuilder<ResolvedBranch>("branches.txt")
 57                 .remoteRepository(repository, "refs/heads/history", "Duke", "duke@openjdk.java.net", "Updated branches");
 58     }
 59 
 60     @Test
 61     void testJsonUpdaterBranch(TestInfo testInfo) throws IOException {
 62         try (var credentials = new HostCredentials(testInfo);
 63              var tempFolder = new TemporaryDirectory();
 64              var jsonFolder = new TemporaryDirectory()) {
 65             var repo = credentials.getHostedRepository();
 66             var localRepo = CheckableRepository.init(tempFolder.path(), repo.getRepositoryType());
 67             localRepo.pushAll(repo.getUrl());
 68 
 69             var tagStorage = createTagStorage(repo);
 70             var branchStorage = createBranchStorage(repo);
 71 
 72             var updater = new JsonUpdater(jsonFolder.path(), "12", "team");
 73             var notifyBot = new JNotifyBot(repo, "master", tagStorage, branchStorage, List.of(updater));
 74 
 75             TestBotRunner.runPeriodicItems(notifyBot);
 76             assertEquals(List.of(), findJsonFiles(jsonFolder.path(), ""));
 77 
 78             var editHash = CheckableRepository.appendAndCommit(localRepo, "One more line", "12345678: Fixes");
 79             localRepo.push(editHash, repo.getUrl(), "master");
 80             TestBotRunner.runPeriodicItems(notifyBot);
 81             var jsonFiles = findJsonFiles(jsonFolder.path(), "");
 82             assertEquals(1, jsonFiles.size());
 83             var jsonData = Files.readString(jsonFiles.get(0), StandardCharsets.UTF_8);
 84             var json = JSON.parse(jsonData);
 85             assertEquals(1, json.asArray().size());
 86             assertEquals(repo.getWebUrl(editHash).toString(), json.asArray().get(0).get("url").asString());
 87             assertEquals(List.of("12345678"), json.asArray().get(0).get("issue").asArray().stream()
 88                                                   .map(JSONValue::asString)
 89                                                   .collect(Collectors.toList()));
 90         }
 91     }
 92 
 93     @Test
 94     void testJsonUpdaterTag(TestInfo testInfo) throws IOException {
 95         try (var credentials = new HostCredentials(testInfo);
 96              var tempFolder = new TemporaryDirectory();
 97              var jsonFolder = new TemporaryDirectory()) {
 98             var repo = credentials.getHostedRepository();
 99             var localRepo = CheckableRepository.init(tempFolder.path(), repo.getRepositoryType());
100             var masterHash = localRepo.resolve("master").orElseThrow();
101             localRepo.tag(masterHash, "jdk-12+1", "Added tag 1", "Duke", "duke@openjdk.java.net");
102             localRepo.pushAll(repo.getUrl());
103 
104             var tagStorage = createTagStorage(repo);
105             var branchStorage = createBranchStorage(repo);
106 
107             var updater = new JsonUpdater(jsonFolder.path(), "12", "team");
108             var notifyBot = new JNotifyBot(repo, "master", tagStorage, branchStorage, List.of(updater));
109 
110             TestBotRunner.runPeriodicItems(notifyBot);
111             assertEquals(List.of(), findJsonFiles(jsonFolder.path(), ""));
112 
113             var editHash = CheckableRepository.appendAndCommit(localRepo, "Another line", "23456789: More fixes");
114             localRepo.fetch(repo.getUrl(), "history:history");
115             localRepo.tag(editHash, "jdk-12+2", "Added tag 2", "Duke", "duke@openjdk.java.net");
116             localRepo.pushAll(repo.getUrl());
117 
118             TestBotRunner.runPeriodicItems(notifyBot);
119             var jsonFiles = findJsonFiles(jsonFolder.path(), "");
120             assertEquals(2, jsonFiles.size());
121 
122             for (var file : jsonFiles) {
123                 var jsonData = Files.readString(file, StandardCharsets.UTF_8);
124                 var json = JSON.parse(jsonData);
125                 assertEquals(1, json.asArray().size());
126                 assertEquals(List.of("23456789"), json.asArray().get(0).get("issue").asArray().stream()
127                                                       .map(JSONValue::asString)
128                                                       .collect(Collectors.toList()));
129 
130                 if (json.asArray().get(0).contains("date")) {
131                     assertEquals(repo.getWebUrl(editHash).toString(), json.asArray().get(0).get("url").asString());
132                     assertEquals("team", json.asArray().get(0).get("build").asString());
133                 } else {
134                     assertEquals("b02", json.asArray().get(0).get("build").asString());
135                 }
136             }
137         }
138     }
