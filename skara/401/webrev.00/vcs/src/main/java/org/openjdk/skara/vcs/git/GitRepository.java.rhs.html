  49     private java.lang.Process start(List<String> cmd) throws IOException {
  50         log.fine("Executing " + String.join(" ", cmd));
  51         var pb = new ProcessBuilder(cmd);
  52         pb.directory(dir.toFile());
  53         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  54         return pb.start();
  55     }

  57     private static void stop(java.lang.Process p) throws IOException {
  58         if (p != null && p.isAlive()) {
  59             var stream = p.getInputStream();
  60             var read = 0;
  61             var buf = new byte[128];
  62             while (read != -1) {
  63                 read = stream.read(buf);
  64             }
  65             try {
  66                 p.waitFor();
  67             } catch (InterruptedException e) {
  68                 throw new IOException(e);
  69             }
  70         }
  71     }

  91     private static Execution.Result await(Execution e) throws IOException {
  92         var result = e.await();
  93         if (result.status() != 0) {
  94             throw new IOException("Unexpected exit code\n" + result);
  95         }
  96         return result;
  97     }

  99     private static void await(java.lang.Process p) throws IOException {
 100         try {
 101             var res = p.waitFor();
 102             if (res != 0) {
 103                 throw new IOException("Unexpected exit code: " + res);
 104             }
 105         } catch (InterruptedException e) {
 106             throw new IOException(e);
 107         }
 108     }

 193     public List<CommitMetadata> commitMetadata() throws IOException {
 194         var revisions = "--all";
 195         var p = start("git", "rev-list", "--format=" + GitCommitMetadata.FORMAT, "--no-abbrev", "--reverse", "--no-color", revisions);
 196         var reader = new UnixStreamReader(p.getInputStream());
 197         var result = new ArrayList<CommitMetadata>();
 198 
 199         var line = reader.readLine();
 200         while (line != null) {
 201             if (!line.startsWith("commit")) {
 202                 throw new IOException("Unexpected line: " + line);
 203             }
 204 
 205             result.add(GitCommitMetadata.read(reader));
 206             line = reader.readLine();
 207         }
 208 
 209         await(p);
 210         return result;
 211     }

 213     private List<Hash> refs() throws IOException {
 214         try (var p = capture("git", "show-ref", "--hash", "--abbrev")) {
 215             var res = p.await();
 216             if (res.status() == -1) {
 217                 if (res.stdout().size() != 0) {
 218                     throw new IOException("Unexpected output\n" + res);
 219                 }
 220                 return new ArrayList<>();
 221             } else {
 222                 return res.stdout().stream()
 223                           .map(Hash::new)
 224                           .collect(Collectors.toList());
 225             }
 226         }
 227     }

 229     @Override
 230     public boolean isEmpty() throws IOException {
 231         int numLooseObjects = -1;
 232         int numPackedObjects = -1;
 233 
 234         try (var p = capture("git", "count-objects", "-v")) {
 235             var res = await(p);
 236             var stdout = res.stdout();
 237 
 238             for (var line : stdout) {
 239                 if (line.startsWith("count: ")) {
 240                     try {
 241                         numLooseObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 242                     } catch (NumberFormatException e) {
 243                         throw new IOException("Unexpected 'count' value\n" + res, e);
 244                     }
 245 
 246                 } else if (line.startsWith("in-pack: ")) {
 247                     try {
 248                         numPackedObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 249                     } catch (NumberFormatException e) {
 250                         throw new IOException("Unexpected 'in-pack' value\n" + res, e);
 251                     }
 252                 }
 253             }
 254         }
 255 
 256         return numLooseObjects == 0 && numPackedObjects == 0 && refs().size() == 0;
 257     }

 259     @Override
 260     public boolean isHealthy() throws IOException {
 261         var refs = refs();
 262         if (refs.size() == 0) {
 263             return true;
 264         }
 265 
 266         var name = "health-check";
 267         try (var p = capture("git", "branch", name, refs.get(0).hex())) {
 268             if (p.await().status() != 0) {
 269                 return false;
 270             }
 271         }
 272         try (var p = capture("git", "branch", "-D", name)) {
 273             if (p.await().status() != 0) {
 274                 return false;
 275             }
 276         }
 277 
 278         return true;
 279     }

 441     @Override
 442     public Path root() throws IOException {
 443         if (cachedRoot != null) {
 444             return cachedRoot;
 445         }
 446 
 447         try (var p = capture("git", "rev-parse", "--show-toplevel")) {
 448             var res = p.await();
 449             if (res.status() != 0 || res.stdout().size() != 1) {
 450                 // Perhaps this is a bare repository
 451                 try (var p2 = capture("git", "rev-parse", "--git-dir")) {
 452                     var res2 = await(p2);
 453                     if (res2.stdout().size() != 1) {
 454                         throw new IOException("Unexpected output\n" + res2);
 455                     }
 456                     cachedRoot = dir.resolve(Path.of(res2.stdout().get(0)));
 457                     return cachedRoot;
 458                 }
 459             }
 460 
 461             cachedRoot = Path.of(res.stdout().get(0));
 462             return cachedRoot;
 463         }
 464     }

 473     @FunctionalInterface
 474     private static interface Operation {
 475         void execute(List<Path> args) throws IOException;
 476     }
 477 
 478     private void batch(Operation op, List<Path> args) throws IOException {
 479         var batchSize = 64;
 480         var start = 0;
 481         while (start < args.size()) {
 482             var end = start + batchSize;
 483             if (end > args.size()) {
 484                 end = args.size();
 485             }
 486             op.execute(args.subList(start, end));
 487             start = end;
 488         }
 489     }

 554     @Override
 555     public Hash commit(String message,
 556                        String authorName,
 557                        String authorEmail,
 558                        ZonedDateTime authorDate,
 559                        String committerName,
 560                        String committerEmail,
 561                        ZonedDateTime committerDate) throws IOException {
 562         var cmd = Process.capture("git", "commit", "--message=" + message)
 563                          .workdir(dir)
 564                          .environ("GIT_AUTHOR_NAME", authorName)
 565                          .environ("GIT_AUTHOR_EMAIL", authorEmail)
 566                          .environ("GIT_COMMITTER_NAME", committerName)
 567                          .environ("GIT_COMMITTER_EMAIL", committerEmail);
 568         if (authorDate != null) {
 569             cmd = cmd.environ("GIT_AUTHOR_DATE",
 570                               authorDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 571         }
 572         if (committerDate != null) {
 573             cmd = cmd.environ("GIT_COMMITTER_DATE",
 574                               committerDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 575         }
 576         try (var p = cmd.execute()) {
 577             await(p);
 578             return head();
 579         }
 580     }

 720     private String treeEntry(Path path, Hash hash) throws IOException {
 721         try (var p = Process.capture("git", "ls-tree", hash.hex(), path.toString())
 722                             .workdir(root())
 723                             .execute()) {
 724             var res = await(p);
 725             if (res.stdout().size() == 0) {
 726                 return null;
 727             }
 728             if (res.stdout().size() > 1) {
 729                 throw new IOException("Unexpected output\n" + res);
 730             }

 735     private List<FileEntry> allFiles(Hash hash, List<Path> paths) throws IOException {
 736         var cmd = new ArrayList<String>();
 737         cmd.addAll(List.of("git", "ls-tree", "-r"));
 738         cmd.add(hash.hex());
 739         cmd.addAll(paths.stream().map(Path::toString).collect(Collectors.toList()));
 740         try (var p = Process.capture(cmd.toArray(new String[0]))
 741                             .workdir(root())
 742                             .execute()) {
 743             var res = await(p);
 744             var entries = new ArrayList<FileEntry>();
 745             for (var line : res.stdout()) {
 746                 var parts = line.split("\t");
 747                 var metadata = parts[0].split(" ");
 748                 var filename = parts[1];
 749 
 750                 var entry = new FileEntry(hash,
 751                                           FileType.fromOctal(metadata[0]),
 752                                           new Hash(metadata[2]),
 753                                           Path.of(filename));
 754                 entries.add(entry);
 755             }
 756             return entries;
 757         }

 760     @Override
 761     public List<FileEntry> files(Hash hash, List<Path> paths) throws IOException {
 762         if (paths.isEmpty()) {
 763             return allFiles(hash, paths);
 764         }
 765 
 766         var entries = new ArrayList<FileEntry>();
 767         var batchSize = 64;
 768         var start = 0;
 769         while (start < paths.size()) {
 770             var end = start + batchSize;
 771             if (end > paths.size()) {
 772                 end = paths.size();
 773             }
 774             entries.addAll(allFiles(hash, paths.subList(start, end)));
 775             start = end;
 776         }
 777         return entries;
 778     }

 792     public Optional<byte[]> show(Path path, Hash hash) throws IOException {
 793         var entries = files(hash, path);
 794         if (entries.size() == 0) {
 795             return Optional.empty();
 796         } else if (entries.size() > 1) {
 797             throw new IOException("Multiple files match path " + path.toString() + " in commit " + hash.hex());
 798         }
 799 
 800         var entry = entries.get(0);
 801         var type = entry.type();
 802         if (type.isVCSLink()) {
 803             var content = "Subproject commit " + entry.hash().hex() + " " + entry.path().toString();
 804             return Optional.of(content.getBytes(StandardCharsets.UTF_8));
 805         } else if (type.isRegular()) {
 806             var tmp = unpackFile(entry.hash().hex());
 807             var content = Files.readAllBytes(tmp);
 808             Files.delete(tmp);
 809             return Optional.of(content);
 810         }
 811 
 812         return Optional.empty();
 813     }

1145     @Override
1146     public List<Submodule> submodules() throws IOException {
1147         var gitModules = root().resolve(".gitmodules");
1148         if (!Files.exists(gitModules)) {
1149             return List.of();
1150         }
1151 
1152         var urls = new HashMap<String, String>();
1153         var paths = new HashMap<String, String>();
1154         try (var p = capture("git", "config", "--file", gitModules.toAbsolutePath().toString(),
1155                                               "--list")) {
1156             for (var line : await(p).stdout()) {
1157                 if (line.startsWith("submodule.")) {
1158                     line = line.substring("submodule.".length());
1159                     var parts = line.split("=");
1160                     var nameAndProperty = parts[0].split("\\.");
1161                     var name = nameAndProperty[0];
1162                     var prop = nameAndProperty[1];
1163                     var value = parts[1];
1164                     if (prop.equals("path")) {
1165                         paths.put(name, value);
1166                     } else if (prop.equals("url")) {
1167                         urls.put(name, value);
1168                     } else {
1169                         throw new IOException("Unexpected submodule property: " + prop);
1170                     }
1171                 }
1172             }
1173         }
1174 
1175         var hashes = new HashMap<String, String>();
1176         try (var p = capture("git", "submodule", "status")) {
1177             for (var line : await(p).stdout()) {
1178                 var parts = line.substring(1).split(" ");
1179                 var hash = parts[0];
1180                 var path = parts[1];
1181                 hashes.put(path, hash);
1182             }
1183         }
1184 
1185         var modules = new ArrayList<Submodule>();
1186         for (var name : paths.keySet()) {
1187             var url = urls.get(name);
1188             var path = paths.get(name);
1189             var hash = hashes.get(path);
1190 
1191             modules.add(new Submodule(new Hash(hash), Path.of(path), url));
1192         }
1193 
1194         return modules;
1195     }
