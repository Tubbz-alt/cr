246     private String newBranchSubject(HostedRepository repository, List<Commit> commits, Branch parent, Branch branch) {
247         var subject = new StringBuilder();
248         subject.append(repository.getRepositoryType().shortName());
249         subject.append(": ");
250         subject.append(repository.getName());
251         subject.append(": created branch ");
252         subject.append(branch);
253         subject.append(" based on the branch ");
254         subject.append(parent);
255         subject.append(" containing ");
256         subject.append(commits.size());
257         subject.append(" unique commit");
258         if (commits.size() != 1) {
259             subject.append("s");
260         }
261 
262         return subject.toString();
263     }

266     public void handleNewBranch(HostedRepository repository, List<Commit> commits, Branch parent, Branch branch) {
267         var writer = new StringWriter();
268         var printer = new PrintWriter(writer);
269 
270         if (commits.size() > 0) {
271             printer.println("The following commits are unique to the " + branch.name() + " branch");
272             printer.println("========================================================");
273             for (var commit : commits) {
274                 printer.print(commit.hash().abbreviate());
275                 if (commit.message().size() > 0) {
276                     printer.print(": " + commit.message().get(0));
277                 }
278                 printer.println();
279             }
280         } else {
281             printer.println("The new branch " + branch.name() + " is currently identical to the " + parent.name() + " branch.");
282         }
283 
284         var subject = newBranchSubject(repository, commits, parent, branch);
285         var finalAuthor = author != null ? author : commits.size() > 0 ? commitsToAuthor(commits) : sender;
