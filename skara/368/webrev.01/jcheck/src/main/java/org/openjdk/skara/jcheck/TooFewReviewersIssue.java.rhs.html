23 package org.openjdk.skara.jcheck;
24 
25 public class TooFewReviewersIssue extends CommitIssue {
26     private final int numActual;
27     private final int numRequired;
28     private final String role;
29 
30     TooFewReviewersIssue(int numActual, int numRequired, String role, CommitIssue.Metadata metadata) {
31         super(metadata);
32         this.numActual = numActual;
33         this.numRequired = numRequired;
34         this.role = role;
35     }
36 
37     public int numRequired() {
38         return numRequired;
39     }
40 
41     public int numActual() {
42         return numActual;
43     }
44 
45     public String role() {
46         return role;
47     }
48 
49     @Override
50     public void accept(IssueVisitor v) {
51         v.visit(this);
52     }
53 }
