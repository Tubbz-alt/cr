23 package org.openjdk.skara.jcheck;
24 
25 import org.openjdk.skara.ini.Section;
26 
27 import java.util.List;
28 import java.util.stream.Collectors;
29 
30 public class ReviewersConfiguration {
31     static final ReviewersConfiguration DEFAULT = new ReviewersConfiguration(1, "reviewer", List.of("duke"));
32 
33     private final int minimum;
34     private final String role;



35     private final List<String> ignore;
36 
37     ReviewersConfiguration(int minimum, String role, List<String> ignore) {
38         this.minimum = minimum;
39         this.role = role;



40         this.ignore = ignore;
41     }
42 
43     public int minimum() {
44         return minimum;
45     }
46 
47     public String role() {
48         return role;












49     }
50 
51     public List<String> ignore() {
52         return ignore;
53     }
54 
55     static String name() {
56         return "reviewers";
57     }
58 
59     static ReviewersConfiguration parse(Section s) {
60         if (s == null) {
61             return DEFAULT;
62         }
63 
64         var minimum = s.get("minimum", DEFAULT.minimum());
65         var role = s.get("role", DEFAULT.role());


























66         var ignore = s.get("ignore", DEFAULT.ignore());
67         return new ReviewersConfiguration(minimum, role, ignore);

68     }
69 }
