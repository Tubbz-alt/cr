23 package org.openjdk.skara.jcheck;
24 
25 public class TooFewReviewersIssue extends CommitIssue {
26     private final int numActual;
27     private final int numRequired;

28 
29     TooFewReviewersIssue(int numActual, int numRequired, CommitIssue.Metadata metadata) {
30         super(metadata);
31         this.numActual = numActual;
32         this.numRequired = numRequired;

33     }
34 
35     public int numRequired() {
36         return numRequired;
37     }
38 
39     public int numActual() {
40         return numActual;
41     }
42 




43     @Override
44     public void accept(IssueVisitor v) {
45         v.visit(this);
46     }
47 }
