 55     private static void await(Process p) throws IOException {
 56         try {
 57             var res = p.waitFor();
 58             if (res != 0) {
 59                 throw new IOException("Unexpected exit code " + res);
 60             }
 61         } catch (InterruptedException e) {
 62             throw new IOException(e);
 63         }
 64     }

 66     private static boolean spawnEditor(ReadOnlyRepository repo, Path file) throws IOException {
 67         String editor = null;
 68         var lines = repo.config("core.editor");
 69         if (lines.size() == 1) {
 70             editor = lines.get(0);
 71         }
 72         if (editor == null) {
 73             editor = System.getenv("GIT_EDITOR");
 74         }
 75         if (editor == null) {
 76             editor = System.getenv("EDITOR");
 77         }
 78         if (editor == null) {
 79             editor = System.getenv("VISUAL");
 80         }
 81         if (editor == null) {
 82             editor = "vi";
 83         }
 84 
 85         var pb = new ProcessBuilder(editor, file.toString());
 86         pb.inheritIO();
 87         var p = pb.start();
 88         try {
 89             return p.waitFor() == 0;
 90         } catch (InterruptedException e) {
 91             throw new IOException(e);
 92         }
 93     }

 95     private static String projectName(URI uri) {
 96         var name = uri.getPath().toString().substring(1);
 97         if (name.endsWith(".git")) {
 98             name = name.substring(0, name.length() - ".git".length());
 99         }
100         return name;
101     }

103     private static HostedRepository getHostedRepositoryFor(URI uri, GitCredentials credentials) throws IOException {
104         var host = Forge.from(uri, new Credential(credentials.username(), credentials.password()));
105         if (System.getenv("GIT_TOKEN") == null) {
106             GitCredentials.approve(credentials);
107         }
108         if (host.isEmpty() || !host.get().isValid()) {
109             exit("error: failed to connect to host " + uri);
110         }
111         var remoteRepo = host.get().repository(projectName(uri));
112         var parentRepo = remoteRepo.parent();
113         var targetRepo = parentRepo.isPresent() ? parentRepo.get() : remoteRepo;
114         return targetRepo;
115     }

117     private static PullRequest getPullRequest(URI uri, GitCredentials credentials, Argument prId) throws IOException {
118         if (!prId.isPresent()) {
119             exit("error: missing pull request identifier");
120         }
121 
122         var pr = getHostedRepositoryFor(uri, credentials).pullRequest(prId.asString());
123         if (pr == null) {
124             exit("error: could not fetch PR information");
125         }
126 
127         return pr;
128     }

133     private static void show(String ref, Hash hash, Path dir) throws IOException {
134         var pb = new ProcessBuilder("git", "diff", "--binary",
135                                                    "--patch",
136                                                    "--find-renames=50%",
137                                                    "--find-copies=50%",
138                                                    "--find-copies-harder",
139                                                    "--abbrev",
140                                                    ref + "..." + hash.hex());
141         if (dir != null) {
142             pb.directory(dir.toFile());
143         }
144         pb.inheritIO();
145         await(pb.start());
146     }

160     private static List<String> hgTags() throws IOException, InterruptedException {
161         var pb = new ProcessBuilder("hg", "tags", "--quiet");
162         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
163         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
164         var p = pb.start();
165         var bytes = p.getInputStream().readAllBytes();
166         var exited = p.waitFor(1, TimeUnit.MINUTES);
167         var exitValue = p.exitValue();
168         if (!exited || exitValue != 0) {
169             throw new IOException("'hg tags' exited with value: " + exitValue);
170         }
171 
172         return Arrays.asList(new String(bytes, StandardCharsets.UTF_8).split("\n"));
173     }

175     private static String hgResolve(String ref) throws IOException, InterruptedException {
176         var pb = new ProcessBuilder("hg", "log", "-r", ref, "--template", "{node}");
177         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
178         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
179         var p = pb.start();
180         var bytes = p.getInputStream().readAllBytes();
181         var exited = p.waitFor(1, TimeUnit.MINUTES);
182         var exitValue = p.exitValue();
183         if (!exited || exitValue != 0) {
184             throw new IOException("'hg log' exited with value: " + exitValue);
185         }
186 
187         return new String(bytes, StandardCharsets.UTF_8);
188     }

194     private static Path diff(String ref, Hash hash, Path dir) throws IOException {
195         var patch = Files.createTempFile(hash.hex(), ".patch");
196         var pb = new ProcessBuilder("git", "diff", "--binary",
197                                                    "--patch",
198                                                    "--find-renames=50%",
199                                                    "--find-copies=50%",
200                                                    "--find-copies-harder",
201                                                    "--abbrev",
202                                                    ref + "..." + hash.hex());
203         if (dir != null) {
204             pb.directory(dir.toFile());
205         }
206         pb.redirectOutput(patch.toFile());
207         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
208         await(pb.start());
209         return patch;
210     }

306         var isMercurial = arguments.contains("mercurial");
307         var cwd = Path.of("").toAbsolutePath();
308         var repo = Repository.get(cwd).orElseThrow(() -> new IOException("no git repository found at " + cwd.toString()));
309         var remote = arguments.get("remote").orString(isMercurial ? "default" : "origin");
310         var remotePullPath = repo.pullPath(remote);
311         var username = arguments.contains("username") ? arguments.get("username").asString() : null;
312         var token = isMercurial ? System.getenv("HG_TOKEN") :  System.getenv("GIT_TOKEN");
313         var uri = Remote.toWebURI(remotePullPath);
314         var credentials = GitCredentials.fill(uri.getHost(), uri.getPath(), username, token, uri.getScheme());
315         var host = Forge.from(uri, new Credential(credentials.username(), credentials.password()));

329                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
330                     System.err.println("");
331                     System.exit(1);
332                 }

335                 if (bookmark.equals(new Bookmark("master"))) {
336                     System.err.println("error: you should not create pull requests from the 'master' bookmark");
337                     System.err.println("To create a bookmark and activate it, run:");
338                     System.err.println("");
339                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
340                     System.err.println("");
341                     System.exit(1);
342                 }
343 
344                 var tags = hgTags();
345                 var upstreams = tags.stream()
346                                     .filter(t -> t.endsWith(bookmark.name()))
347                                     .collect(Collectors.toList());
348                 if (upstreams.isEmpty()) {
349                     System.err.println("error: there is no remote branch for the local bookmark '" + bookmark.name() + "'");

351                     System.err.println("To create a remote branch and push the commits for your local branch, run:");
352                     System.err.println("");
353                     System.err.println("    hg push --bookmark " + bookmark.name());
354                     System.err.println("");
355                     System.exit(1);
356                 }
357 
358                 var tagsAndHashes = new HashMap<String, String>();
359                 for (var tag : tags) {
360                     tagsAndHashes.put(tag, hgResolve(tag));
361                 }
362                 var bookmarkHash = hgResolve(bookmark.name());
363                 if (!tagsAndHashes.containsValue(bookmarkHash)) {
364                     System.err.println("error: there are local commits on bookmark '" + bookmark.name() + "' not present in a remote repository");
365                     System.err.println("");
366 
367                     if (upstreams.size() == 1) {
368                         System.err.println("To push the local commits to the remote repository, run:");
369                         System.err.println("");
370                         System.err.println("    hg push --bookmark " + bookmark.name() + " " + upstreams.get(0));
371                         System.err.println("");
372                     } else {
373                         System.err.println("The following paths contains the " + bookmark.name() + " bookmark:");

377                         }
378                         System.err.println("");
379                         System.err.println("To push the local commits to a remote repository, run:");
380                         System.err.println("");
381                         System.err.println("    hg push --bookmark " + bookmark.name() + " <PATH>");
382                         System.err.println("");
383                     }
384                     System.exit(1);
385                 }
386 
387                 var targetBranch = arguments.get("branch").orString("master");
388                 var targetHash = hgResolve(targetBranch);
389                 var commits = repo.commits(targetHash + ".." + bookmarkHash + "-" + targetHash).asList();
390                 if (commits.isEmpty()) {
391                     System.err.println("error: no difference between bookmarks " + targetBranch + " and " + bookmark.name());
392                     System.err.println("       Cannot create an empty pull request, have you committed?");
393                     System.exit(1);
394                 }
395 
396                 var diff = repo.diff(repo.head());
397                 if (!diff.patches().isEmpty()) {
398                     System.err.println("error: there are uncommitted changes in your working tree:");
399                     System.err.println("");
400                     for (var patch : diff.patches()) {
401                         var path = patch.target().path().isPresent() ?
402                             patch.target().path().get() : patch.source().path().get();
403                         System.err.println("    " + patch.status().toString() + " " + path.toString());
404                     }
405                     System.err.println("");
406                     System.err.println("If these changes are meant to be part of the pull request, run:");
407                     System.err.println("");
408                     System.err.println("    hg commit --amend");
409                     System.err.println("    hg git-cleanup");
410                     System.err.println("    hg push --bookmark " + bookmark.name() + " <PATH>");
411                     System.err.println("    hg gimport");
412                     System.err.println("");
413                     System.err.println("If these changes are *not* meant to be part of the pull request, run:");
414                     System.err.println("");
415                     System.err.println("    hg shelve");
416                     System.err.println("");
417                     System.err.println("(You can later restore the changes by running: hg unshelve)");
418                     System.exit(1);
419                 }
420 
421                 var remoteRepo = host.get().repository(projectName(uri));
422                 if (token == null) {
423                     GitCredentials.approve(credentials);
424                 }
425                 var parentRepo = remoteRepo.parent().orElseThrow(() ->
426                         new IOException("error: remote repository " + remotePullPath + " is not a fork of any repository"));
427 
428                 var file = Files.createTempFile("PULL_REQUEST_", ".txt");
429                 if (commits.size() == 1) {
430                     var commit = commits.get(0);
431                     var message = CommitMessageParsers.v1.parse(commit.message());
432                     Files.writeString(file, message.title() + "\n");
433                     if (!message.summaries().isEmpty()) {
434                         Files.write(file, message.summaries(), StandardOpenOption.APPEND);
435                     }
436                     if (!message.additional().isEmpty()) {
437                         Files.write(file, message.additional(), StandardOpenOption.APPEND);
438                     }
439                 } else {
440                     Files.write(file, List.of(""));
441                 }
442                 Files.write(file, List.of(
443                     "# Please enter the pull request message for your changes. Lines starting",
444                     "# with '#' will be ignored, and an empty message aborts the pull request.",
445                     "# The first line will be considered the subject, use a blank line to separate",
446                     "# the subject from the body.",
447                     "#",
448                     "# Commits to be included from branch '" + bookmark.name() + "'"
449                     ),
450                     StandardOpenOption.APPEND
451                 );
452                 for (var commit : commits) {
453                     var desc = commit.hash().abbreviate() + ": " + commit.message().get(0);
454                     Files.writeString(file, "# - " + desc + "\n", StandardOpenOption.APPEND);
455                 }
456                 Files.writeString(file, "#\n", StandardOpenOption.APPEND);
457                 Files.writeString(file, "# Target repository: " + remotePullPath + "\n", StandardOpenOption.APPEND);
458                 Files.writeString(file, "# Target branch: " + targetBranch + "\n", StandardOpenOption.APPEND);

463                 }

465                                  .stream()
466                                  .filter(l -> !l.startsWith("#"))

476                 if (lines.size() > 1) {
477                     body = lines.subList(1, lines.size())
478                                 .stream()
479                                 .dropWhile(String::isEmpty)
480                                 .collect(Collectors.toList());

483                 }

489                                              .map(u -> host.get().user(u))

523                 System.err.println("    git branch --set-upstream " + currentBranch + " " + remote + "/" + currentBranch);
524                 System.err.println("");
525                 System.exit(1);
526             }
527 
528             var upstreamRefName = upstream.get().substring(remote.length() + 1);
529             repo.fetch(uri, upstreamRefName);
530             var branchCommits = repo.commits(upstream.get() + ".." + currentBranch.name()).asList();
531             if (!branchCommits.isEmpty()) {
532                 System.err.println("error: there are local commits on branch '" + currentBranch.name() + "' not present in the remote repository " + remotePullPath);
533                 System.err.println("");
534                 System.err.println("All commits must be present in the remote repository to be part of the pull request");
535                 System.err.println("The following commits are not present in the remote repository:");
536                 System.err.println("");
537                 for (var commit : branchCommits) {
538                     System.err.println("- " + commit.hash().abbreviate() + ": " + commit.message().get(0));
539                 }
540                 System.err.println("");
541                 System.err.println("To push the above local commits to the remote repository, run:");
542                 System.err.println("");
543                 System.err.println("    git push " + remote + " " + currentBranch.name());
544                 System.err.println("");
545                 System.exit(1);
546             }
547 
548             var targetBranch = arguments.get("branch").orString("master");
549             var commits = repo.commits(targetBranch + ".." + currentBranch.name()).asList();
550             if (commits.isEmpty()) {
551                 System.err.println("error: no difference between branches " + targetBranch + " and " + currentBranch.name());
552                 System.err.println("       Cannot create an empty pull request, have you committed?");
553                 System.exit(1);
554             }
555 
556             var diff = repo.diff(repo.head());
557             if (!diff.patches().isEmpty()) {
558                 System.err.println("error: there are uncommitted changes in your working tree:");
559                 System.err.println("");
560                 for (var patch : diff.patches()) {
561                     var path = patch.target().path().isPresent() ?
562                         patch.target().path().get() : patch.source().path().get();
563                     System.err.println("    " + patch.status().toString() + " " + path.toString());
564                 }
565                 System.err.println("");
566                 System.err.println("If these changes are meant to be part of the pull request, run:");
567                 System.err.println("");
568                 System.err.println("    git commit -am 'Forgot to add some changes'");
569                 System.err.println("");
570                 System.err.println("If these changes are *not* meant to be part of the pull request, run:");
571                 System.err.println("");
572                 System.err.println("    git stash");
573                 System.err.println("");
574                 System.err.println("(You can later restore the changes by running: git stash pop)");
575                 System.exit(1);
576             }
577 
578             var remoteRepo = host.get().repository(projectName(uri));
579             if (token == null) {
580                 GitCredentials.approve(credentials);
581             }
582             var parentRepo = remoteRepo.parent().orElseThrow(() ->
583                     new IOException("error: remote repository " + remotePullPath + " is not a fork of any repository"));
584 
585             var file = Files.createTempFile("PULL_REQUEST_", ".txt");
586             if (commits.size() == 1) {
587                 var commit = commits.get(0);
588                 var message = CommitMessageParsers.v1.parse(commit.message());
589                 Files.writeString(file, message.title() + "\n");
590                 if (!message.summaries().isEmpty()) {
591                     Files.write(file, message.summaries(), StandardOpenOption.APPEND);
592                 }
593                 if (!message.additional().isEmpty()) {
594                     Files.write(file, message.additional(), StandardOpenOption.APPEND);
595                 }
596             } else {
597                 Files.write(file, List.of(""));
598             }
599             Files.write(file, List.of(
600                 "# Please enter the pull request message for your changes. Lines starting",
601                 "# with '#' will be ignored, and an empty message aborts the pull request.",
602                 "# The first line will be considered the subject, use a blank line to separate",
603                 "# the subject from the body.",
604                 "#",
605                 "# Commits to be included from branch '" + currentBranch.name() + "'"
606                 ),
607                 StandardOpenOption.APPEND
608             );
609             for (var commit : commits) {
610                 var desc = commit.hash().abbreviate() + ": " + commit.message().get(0);
611                 Files.writeString(file, "# - " + desc + "\n", StandardOpenOption.APPEND);
612             }
613             Files.writeString(file, "#\n", StandardOpenOption.APPEND);
614             Files.writeString(file, "# Target repository: " + remotePullPath + "\n", StandardOpenOption.APPEND);
615             Files.writeString(file, "# Target branch: " + targetBranch + "\n", StandardOpenOption.APPEND);
616             var success = spawnEditor(repo, file);
617             if (!success) {
618                 System.err.println("error: editor exited with non-zero status code, aborting");
619                 System.exit(1);
620             }
621             var lines = Files.readAllLines(file)
622                              .stream()
623                              .filter(l -> !l.startsWith("#"))
624                              .collect(Collectors.toList());
625             var isEmpty = lines.stream().allMatch(String::isEmpty);
626             if (isEmpty) {
627                 System.err.println("error: no message present, aborting");
628                 System.exit(1);
629             }
630 
631             var title = lines.get(0);
632             List<String> body = null;
633             if (lines.size() > 1) {
634                 body = lines.subList(1, lines.size())
635                             .stream()
636                             .dropWhile(String::isEmpty)
637                             .collect(Collectors.toList());
638             } else {
639                 body = Collections.emptyList();
640             }
641 
642             var pr = remoteRepo.createPullRequest(parentRepo, targetBranch, currentBranch.name(), title, body);
643             if (arguments.contains("assignees")) {
644                 var usernames = Arrays.asList(arguments.get("assignees").asString().split(","));
645                 var assignees = usernames.stream()
646                                          .map(u -> host.get().user(u))
647                                          .collect(Collectors.toList());
648                 pr.setAssignees(assignees);
649             }
650             System.out.println(pr.webUrl().toString());
651             Files.deleteIfExists(file);
652         } else if (action.equals("integrate") || action.equals("approve")) {
653             var pr = getPullRequest(uri, credentials, arguments.at(1));
654 
655             if (action.equals("integrate")) {
656                 pr.addComment("/integrate");
657             } else if (action.equals("approve")) {
658                 pr.addReview(Review.Verdict.APPROVED, "Looks good!");
659             } else {
660                 throw new IllegalStateException("unexpected action: " + action);
661             }
662         } else if (action.equals("list")) {
663             var remoteRepo = getHostedRepositoryFor(uri, credentials);
664             var prs = remoteRepo.pullRequests();
665 
666             var ids = new ArrayList<String>();
667             var titles = new ArrayList<String>();
668             var authors = new ArrayList<String>();
669             var assignees = new ArrayList<String>();
670             var labels = new ArrayList<String>();
671 
672             var filterAuthors = arguments.contains("authors") ?
673                 new HashSet<>(Arrays.asList(arguments.get("authors").asString().split(","))) :
674                 Set.of();
675             var filterAssignees = arguments.contains("assignees") ?
676                 Arrays.asList(arguments.get("assignees").asString().split(",")) :
677                 Set.of();
678             var filterLabels = arguments.contains("labels") ?
679                 Arrays.asList(arguments.get("labels").asString().split(",")) :
680                 Set.of();
681 
682             var defaultColumns = List.of("id", "title", "authors", "assignees", "labels");
683             var columnValues = Map.of(defaultColumns.get(0), ids,
684                                       defaultColumns.get(1), titles,
685                                       defaultColumns.get(2), authors,
686                                       defaultColumns.get(3), assignees,
687                                       defaultColumns.get(4), labels);
688             var columns = arguments.contains("columns") ?
689                 Arrays.asList(arguments.get("columns").asString().split(",")) :
690                 defaultColumns;
691             if (columns != defaultColumns) {
692                 for (var column : columns) {
693                     if (!defaultColumns.contains(column)) {
694                         System.err.println("error: unknown column: " + column);
695                         System.err.println("       available columns are: " + String.join(",", defaultColumns));
696                         System.exit(1);
697                     }
698                 }
699             }
700 
701             for (var pr : remoteRepo.pullRequests()) {
702                 var prAuthor = pr.author().userName();
703                 if (!filterAuthors.isEmpty() && !filterAuthors.contains(prAuthor)) {
704                     continue;
705                 }
706 
707                 var prAssignees = pr.assignees().stream()
708                                     .map(HostUser::userName)
709                                     .collect(Collectors.toSet());
710                 if (!filterAssignees.isEmpty() && !filterAssignees.stream().anyMatch(prAssignees::contains)) {
711                     continue;
712                 }
713 
714                 var prLabels = new HashSet<>(pr.labels());
715                 if (!filterLabels.isEmpty() && !filterLabels.stream().anyMatch(prLabels::contains)) {
716                     continue;
717                 }
718 
719                 ids.add(pr.id());
720                 titles.add(pr.title());
721                 authors.add(prAuthor);
722                 assignees.add(String.join(",", prAssignees));
723                 labels.add(String.join(",", prLabels));
724             }
725 
726 
727             String fmt = "";
728             for (var column : columns.subList(0, columns.size() - 1)) {
729                 var values = columnValues.get(column);
730                 var n = Math.max(column.length(), longest(values));
731                 fmt += "%-" + n + "s\t";
732             }
733             fmt += "%s\n";
734 
735             if (!ids.isEmpty() && !arguments.contains("no-decoration")) {
736                 var upperCase = columns.stream()
737                                        .map(String::toUpperCase)
738                                        .collect(Collectors.toList());
739                 System.out.format(fmt, (Object[]) upperCase.toArray(new String[0]));
740             }
741             for (var i = 0; i < ids.size(); i++) {
742                 final int n = i;
743                 var row = columns.stream()
744                                  .map(columnValues::get)
745                                  .map(values -> values.get(n))
746                                  .collect(Collectors.toList());
747                 System.out.format(fmt, (Object[]) row.toArray(new String[0]));
748             }
749         } else if (action.equals("fetch") || action.equals("checkout") || action.equals("show") || action.equals("apply")) {
750             var prId = arguments.at(1);
751             if (!prId.isPresent()) {
752                 exit("error: missing pull request identifier");
753             }
754 
755             var remoteRepo = getHostedRepositoryFor(uri, credentials);
756             var pr = remoteRepo.pullRequest(prId.asString());
757             var repoUrl = remoteRepo.webUrl();
758             var prHeadRef = pr.sourceRef();
759             var isHgGit = isMercurial && Repository.exists(repo.root().resolve(".hg").resolve("git"));
760             if (isHgGit) {
761                 var hgGitRepo = Repository.get(repo.root().resolve(".hg").resolve("git")).get();
762                 var hgGitFetchHead = hgGitRepo.fetch(repoUrl, prHeadRef);
763 
764                 if (action.equals("show") || action.equals("apply")) {
765                     var target = hgGitRepo.fetch(repoUrl, pr.targetRef());
766                     var hgGitMergeBase = hgGitRepo.mergeBase(target, hgGitFetchHead);
767 
768                     if (action.equals("show")) {
769                         show(hgGitMergeBase.hex(), hgGitFetchHead, hgGitRepo.root());
770                     } else {
771                         var patch = diff(hgGitMergeBase.hex(), hgGitFetchHead, hgGitRepo.root());
772                         hgImport(patch);
773                         Files.delete(patch);
774                     }
775                 } else if (action.equals("fetch") || action.equals("checkout")) {
776                     var hgGitRef = prHeadRef.endsWith("/head") ? prHeadRef.replace("/head", "") : prHeadRef;
777                     var hgGitBranches = hgGitRepo.branches();
778                     if (hgGitBranches.contains(new Branch(hgGitRef))) {
779                         hgGitRepo.delete(new Branch(hgGitRef));
780                     }
781                     hgGitRepo.branch(hgGitFetchHead, hgGitRef);
782                     gimport();
783                     var hgFetchHead = repo.resolve(hgGitRef).get();
784 
785                     if (action.equals("fetch") && arguments.contains("branch")) {
786                         repo.branch(hgFetchHead, arguments.get("branch").asString());
787                     } else if (action.equals("checkout")) {
788                         repo.checkout(hgFetchHead);
789                         if (arguments.contains("branch")) {
790                             repo.branch(hgFetchHead, arguments.get("branch").asString());
791                         }
792                     }
793                 } else {
794                     exit("Unexpected action: " + action);
795                 }
796 
797                 return;
798             }
799 
800             var fetchHead = repo.fetch(repoUrl, pr.sourceRef());
801             if (action.equals("fetch")) {
802                 if (arguments.contains("branch")) {
803                     var branchName = arguments.get("branch").asString();
804                     repo.branch(fetchHead, branchName);
805                 } else {
806                     System.out.println(fetchHead.hex());
807                 }
808             } else if (action.equals("checkout")) {
809                 if (arguments.contains("branch")) {
810                     var branchName = arguments.get("branch").asString();
811                     var branch = repo.branch(fetchHead, branchName);
812                     repo.checkout(branch, false);
813                 } else {
814                     repo.checkout(fetchHead, false);
815                 }
816             } else if (action.equals("show")) {
817                 show(pr.targetRef(), fetchHead);
818             } else if (action.equals("apply")) {
819                 var patch = diff(pr.targetRef(), fetchHead);
820                 apply(patch);
821                 Files.deleteIfExists(patch);
822             }
823         } else if (action.equals("close")) {
824             var prId = arguments.at(1);
825             if (!prId.isPresent()) {
826                 exit("error: missing pull request identifier");
827             }
828 
829             var remoteRepo = getHostedRepositoryFor(uri, credentials);
830             var pr = remoteRepo.pullRequest(prId.asString());
831             pr.setState(PullRequest.State.CLOSED);
832         } else if (action.equals("update")) {
833             var prId = arguments.at(1);
834             if (!prId.isPresent()) {
835                 exit("error: missing pull request identifier");
836             }
837 
838             var remoteRepo = getHostedRepositoryFor(uri, credentials);
839             var pr = remoteRepo.pullRequest(prId.asString());
840             if (arguments.contains("assignees")) {
841                 var usernames = Arrays.asList(arguments.get("assignees").asString().split(","));
842                 var assignees = usernames.stream()
843                     .map(u -> host.get().user(u))
844                     .collect(Collectors.toList());
845                 pr.setAssignees(assignees);
846             }
847         } else {
848             exit("error: unexpected action: " + action);
849         }
850     }
