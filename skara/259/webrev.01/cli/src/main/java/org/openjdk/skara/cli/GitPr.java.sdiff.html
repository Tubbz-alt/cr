 84             editor = "vi";
 85         }
 86 
 87         List<String> editorParts = new ArrayList<>();
 88         Matcher em = Pattern.compile("([^\"]\\S*|\".+?\")\\s*").matcher(editor);
 89         while (em.find()) {
 90             editorParts.add(em.group(1));
 91         }
 92         editorParts.add(file.toString());
 93         var pb = new ProcessBuilder(editorParts);
 94         pb.inheritIO();
