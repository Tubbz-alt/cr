23 package org.openjdk.skara.forge;
24 
25 import java.util.Optional;
26 
27 public class CheckAnnotation {
28     public final String path;
29     private final int startLine;
30     private final int endLine;
31     private final Integer startColumn;
32     private final Integer endColumn;
33     private final String title;
34     private final String message;
35     private final CheckAnnotationLevel level;
36 
37     CheckAnnotation(String path, int startLine, int endLine, CheckAnnotationLevel level, Integer startColumn, Integer endColumn, String title, String message) {
38         this.path = path;
39         this.startLine = startLine;
40         this.endLine = endLine;
41         this.startColumn = startColumn;
42         this.endColumn = endColumn;
43         this.title = title;
44         this.message = message;
45         this.level = level;
46     }
47 
48     public String path() {
49         return path;
50     }
51 
52     public int startLine() {
53         return startLine;
54     }
55 
56     public int endLine() {
57         return endLine;
58     }
59 
60     public Optional<Integer> startColumn() {
61         return Optional.ofNullable(startColumn);
62     }
63 
64     public Optional<Integer> endColumn() {
65         return Optional.ofNullable(endColumn);
66     }
67 
68     public Optional<String> title() {
69         return Optional.ofNullable(title);
70     }
71 
72     public String message() {
73         return message;
74     }
75 
76     public CheckAnnotationLevel level() {
77         return level;
78     }
79 }
