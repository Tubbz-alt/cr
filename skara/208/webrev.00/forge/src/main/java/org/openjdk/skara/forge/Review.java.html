23 package org.openjdk.skara.forge;
24 
25 import org.openjdk.skara.host.HostUser;
26 import org.openjdk.skara.vcs.Hash;
27 
28 import java.util.Optional;
29 
30 public class Review {
31     private final HostUser reviewer;
32     private final Verdict verdict;
33     private final Hash hash;
34     private final int id;
35     private final String body;
36 
37     public Review(HostUser reviewer, Verdict verdict, Hash hash, int id, String body) {
38         this.reviewer = reviewer;
39         this.verdict = verdict;
40         this.hash = hash;
41         this.id = id;
42         this.body = body;
43     }
44 
45     public HostUser reviewer() {
46         return reviewer;
47     }
48 
49     public Verdict verdict() {
50         return verdict;
51     }
52 
53     public Hash hash() {
54         return hash;
55     }
56 
57     public int id() {
58         return id;
59     }
60 
61     public Optional<String> body() {
62         return Optional.ofNullable(body);
63     }
64 
65     public enum Verdict {
66         NONE,
67         APPROVED,
68         DISAPPROVED
69     }
70 }
