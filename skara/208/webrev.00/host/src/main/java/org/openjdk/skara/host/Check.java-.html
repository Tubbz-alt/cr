23 package org.openjdk.skara.host;
24 
25 import org.openjdk.skara.vcs.Hash;
26 
27 import java.time.ZonedDateTime;
28 import java.util.*;
29 
30 public class Check {
31     private final ZonedDateTime startedAt;
32     private final ZonedDateTime completedAt;
33     private final CheckStatus status;
34     private final Hash hash;
35     private final String metadata;
36     private final String title;
37     private final String summary;
38     private final List<CheckAnnotation> annotations;
39     private final String name;
40 
41     Check(String name, Hash hash, CheckStatus status, ZonedDateTime startedAt, ZonedDateTime completedAt, String metadata, String title, String summary, List<CheckAnnotation> annotations) {
42         this.name = name;
43         this.hash = hash;
44         this.status = status;
45         this.startedAt = startedAt;
46         this.completedAt = completedAt;
47         this.metadata = metadata;
48         this.title = title;
49         this.summary = summary;
50         this.annotations = annotations;
51     }
52 
53     public String name() {
54         return name;
55     }
56 
57     public Hash hash() {
58         return hash;
59     }
60 
61     public CheckStatus status() {
62         return status;
63     }
64 
65     public ZonedDateTime startedAt() {
66         return startedAt;
67     }
68 
69     public Optional<ZonedDateTime> completedAt() {
70         return Optional.ofNullable(completedAt);
71     }
72 
73     public Optional<String> title() {
74         return Optional.ofNullable(title);
75     }
76 
77     public Optional<String> summary() {
78         return Optional.ofNullable(summary);
79     }
80 
81     public Optional<String> metadata() {
82         return Optional.ofNullable(metadata);
83     }
84 
85     public List<CheckAnnotation> annotations() {
86         return annotations;
87     }
88 }
