23 package org.openjdk.skara.bots.submit;
24 
25 import org.openjdk.skara.host.*;
26 
27 import java.time.*;
28 import java.util.logging.Logger;
29 
30 public class CheckUpdater implements Runnable {
31     private final PullRequest pr;
32     private final CheckBuilder checkBuilder;
33     private Instant lastUpdate;
34     private Duration maxUpdateRate;
35     private final Logger log = Logger.getLogger("org.openjdk.skara.bots.submit");
36 
37     CheckUpdater(PullRequest pr, CheckBuilder checkBuilder) {
38         this.pr = pr;
39         this.checkBuilder = checkBuilder;
40 
41         lastUpdate = Instant.EPOCH;
42         maxUpdateRate = Duration.ofSeconds(10);
43     }
44 
45     void setMaxUpdateRate(Duration maxUpdateRate) {
46         this.maxUpdateRate = maxUpdateRate;
47     }
48 
49     @Override
50     public void run() {
51         if (Instant.now().isAfter(lastUpdate.plus(maxUpdateRate))) {
52             pr.updateCheck(checkBuilder.build());
53             lastUpdate = Instant.now();
54         } else {
55             log.finest("Rate limiting check updates");
56         }
57     }
58 }
