 23 package org.openjdk.skara.issuetracker;
 24 
 25 import org.openjdk.skara.host.HostUser;
 26 
 27 import java.net.URI;
 28 import java.time.ZonedDateTime;
 29 import java.util.List;
 30 
 31 public interface Issue {
 32     /**
 33      * Project containing the issue.
 34      * @return
 35      */
 36     IssueProject project();
 37 
 38     /**
 39      * The repository-specific identifier.
 40      * @return
 41      */
 42     String id();
 43 
 44     /**
 45      * The host-specific author name.
 46      * @return
 47      */
 48     HostUser author();
 49 
 50     /**
 51      * Title of the request.
 52      * @return
 53      */
 54     String title();
 55 
 56     /**
 57      * Update the title of the request.
 58      * @param title
 59      */
 60     void setTitle(String title);
 61 
 62     /**
 63      * The main body of the request.
 64      * @return
 65      */
 66     String body();
 67 
 68     /**
 69      * Update the main body of the request.
 70      * @param body
 71      */
 72     void setBody(String body);
 73 
 74     /**
 75      * All comments on the issue, in ascending creation time order.
 76      * @return
 77      */
 78     List<Comment> comments();
 79 
 80     /**
 81      * Posts a new comment.
 82      * @param body
 83      */
 84     Comment addComment(String body);
 85 
 86     /**
 87      * Updates an existing comment.
 88      * @param id
 89      * @param body
 90      */
 91     Comment updateComment(String id, String body);
 92 
 93     /**
 94      * When the request was created.
 95      * @return
 96      */
 97     ZonedDateTime createdAt();
 98 
 99     /**
100      * When the request was last updated.
101      * @return
102      */
103     ZonedDateTime updatedAt();
104 
105     enum State {
106         OPEN,
107         CLOSED
108     }
109 
110     /**
111      * Set the state.
112      * @param state Desired state
113      */
114     void setState(State state);
115 
116     /**
117      * Adds the given label.
118      * @param label
119      */
120     void addLabel(String label);
121 
122     /**
123      * Removes the given label.
124      * @param label
125      */
126     void removeLabel(String label);
127 
128     /**
129      * Retrieves all the currently set labels.
130      * @return
131      */
132     List<String> labels();
133 
134     /**
135      * Returns a link that will lead to the issue.
136      */
137     URI webUrl();
138 
139     /**
140      * Returns all usernames assigned to the issue.
141      */
142     List<HostUser> assignees();
143 
144     /**
145      * Update the list of assignees.
146      * @param assignees
147      */
148     void setAssignees(List<HostUser> assignees);
149 }
