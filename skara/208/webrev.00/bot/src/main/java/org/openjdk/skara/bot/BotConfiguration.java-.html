23 package org.openjdk.skara.bot;
24 
25 import org.openjdk.skara.host.*;
26 import org.openjdk.skara.json.JSONObject;
27 
28 import java.nio.file.Path;
29 
30 public interface BotConfiguration {
31     /**
32      * Folder that WorkItems may use to store permanent data.
33      * @return
34      */
35     Path storageFolder();
36 
37     /**
38      * Configuration-specific name mapped to a HostedRepository.
39      * @param name
40      * @return
41      */
42     HostedRepository repository(String name);
43 
44     /**
45      * Configuration-specific name mapped to a IssueProject.
46      * @param name
47      * @return
48      */
49     IssueProject issueProject(String name);
50 
51     /**
52      * Retrieves the ref name that optionally follows the configuration-specific repository name.
53      * If not configured, returns the name of the VCS default branch.
54      * @param name
55      * @return
56      */
57     String repositoryRef(String name);
58 
59     /**
60      * Extracts a reasonable short repository name from a full repository specification, e.g. host/org/repo:ref -> repo
61      * @param name
62      * @return
63      */
64     String repositoryName(String name);
65 
66     /**
67      * Additional bot-specific configuration.
68      * @return
69      */
70     JSONObject specific();
71 }
