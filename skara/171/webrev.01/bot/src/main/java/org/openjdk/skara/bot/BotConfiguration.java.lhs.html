23 package org.openjdk.skara.bot;
24 
25 import org.openjdk.skara.host.HostedRepository;
26 import org.openjdk.skara.json.JSONObject;
27 
28 import java.nio.file.Path;
29 
30 public interface BotConfiguration {
31     /**
32      * Folder that WorkItems may use to store permanent data.
33      * @return
34      */
35     Path storageFolder();
36 
37     /**
38      * Configuration-specific name mapped to a HostedRepository.
39      * @param name
40      * @return
41      */
42     HostedRepository repository(String name);
43 







44     /**
45      * Retrieves the ref name that optionally follows the configuration-specific repository name.
46      * If not configured, returns the name of the VCS default branch.
47      * @param name
48      * @return
49      */
50     String repositoryRef(String name);
51 
52     /**
53      * Extracts a reasonable short repository name from a full repository specification, e.g. host/org/repo:ref -> repo
54      * @param name
55      * @return
56      */
57     String repositoryName(String name);
58 
59     /**
60      * Additional bot-specific configuration.
61      * @return
62      */
63     JSONObject specific();
64 }
