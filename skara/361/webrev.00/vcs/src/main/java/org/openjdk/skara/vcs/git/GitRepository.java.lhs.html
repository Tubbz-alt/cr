  49     private java.lang.Process start(List<String> cmd) throws IOException {
  50         log.fine("Executing " + String.join(" ", cmd));
  51         var pb = new ProcessBuilder(cmd);
  52         pb.directory(dir.toFile());
  53         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  54         return pb.start();
  55     }

  57     private static void stop(java.lang.Process p) throws IOException {
  58         if (p != null && p.isAlive()) {
  59             var stream = p.getInputStream();
  60             var read = 0;
  61             var buf = new byte[128];
  62             while (read != -1) {
  63                 read = stream.read(buf);
  64             }
  65             try {
  66                 p.waitFor();
  67             } catch (InterruptedException e) {
  68                 throw new IOException(e);
  69             }
  70         }
  71     }

  91     private static Execution.Result await(Execution e) throws IOException {
  92         var result = e.await();
  93         if (result.status() != 0) {
  94             throw new IOException("Unexpected exit code\n" + result);
  95         }
  96         return result;
  97     }

  99     private static void await(java.lang.Process p) throws IOException {
 100         try {
 101             var res = p.waitFor();
 102             if (res != 0) {
 103                 throw new IOException("Unexpected exit code: " + res);
 104             }
 105         } catch (InterruptedException e) {
 106             throw new IOException(e);
 107         }
 108     }

 193     public List<CommitMetadata> commitMetadata() throws IOException {
 194         var revisions = "--all";
 195         var p = start("git", "rev-list", "--format=" + GitCommitMetadata.FORMAT, "--no-abbrev", "--reverse", "--no-color", revisions);
 196         var reader = new UnixStreamReader(p.getInputStream());
 197         var result = new ArrayList<CommitMetadata>();
 198 
 199         var line = reader.readLine();
 200         while (line != null) {
 201             if (!line.startsWith("commit")) {
 202                 throw new IOException("Unexpected line: " + line);
 203             }
 204 
 205             result.add(GitCommitMetadata.read(reader));
 206             line = reader.readLine();
 207         }
 208 
 209         await(p);
 210         return result;
 211     }

 213     private List<Hash> refs() throws IOException {
 214         try (var p = capture("git", "show-ref", "--hash", "--abbrev")) {
 215             var res = p.await();
 216             if (res.status() == -1) {
 217                 if (res.stdout().size() != 0) {
 218                     throw new IOException("Unexpected output\n" + res);
 219                 }
 220                 return new ArrayList<>();
 221             } else {
 222                 return res.stdout().stream()
 223                           .map(Hash::new)
 224                           .collect(Collectors.toList());
 225             }
 226         }
 227     }

 229     @Override
 230     public boolean isEmpty() throws IOException {
 231         int numLooseObjects = -1;
 232         int numPackedObjects = -1;
 233 
 234         try (var p = capture("git", "count-objects", "-v")) {
 235             var res = await(p);
 236             var stdout = res.stdout();
 237 
 238             for (var line : stdout) {
 239                 if (line.startsWith("count: ")) {
 240                     try {
 241                         numLooseObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 242                     } catch (NumberFormatException e) {
 243                         throw new IOException("Unexpected 'count' value\n" + res, e);
 244                     }
 245 
 246                 } else if (line.startsWith("in-pack: ")) {
 247                     try {
 248                         numPackedObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 249                     } catch (NumberFormatException e) {
 250                         throw new IOException("Unexpected 'in-pack' value\n" + res, e);
 251                     }
 252                 }
 253             }
 254         }
 255 
 256         return numLooseObjects == 0 && numPackedObjects == 0 && refs().size() == 0;
 257     }

 259     @Override
 260     public boolean isHealthy() throws IOException {
 261         var refs = refs();
 262         if (refs.size() == 0) {
 263             return true;
 264         }
 265 
 266         var name = "health-check";
 267         try (var p = capture("git", "branch", name, refs.get(0).hex())) {
 268             if (p.await().status() != 0) {
 269                 return false;
 270             }
 271         }
 272         try (var p = capture("git", "branch", "-D", name)) {
 273             if (p.await().status() != 0) {
 274                 return false;
 275             }
 276         }
 277 
 278         return true;
 279     }

 441     @Override
 442     public Path root() throws IOException {
 443         if (cachedRoot != null) {
 444             return cachedRoot;
 445         }
 446 
 447         try (var p = capture("git", "rev-parse", "--show-toplevel")) {
 448             var res = p.await();
 449             if (res.status() != 0 || res.stdout().size() != 1) {
 450                 // Perhaps this is a bare repository
 451                 try (var p2 = capture("git", "rev-parse", "--git-dir")) {
 452                     var res2 = await(p2);
 453                     if (res2.stdout().size() != 1) {
 454                         throw new IOException("Unexpected output\n" + res2);
 455                     }
 456                     cachedRoot = dir.resolve(Path.of(res2.stdout().get(0)));
 457                     return cachedRoot;
 458                 }
 459             }
 460 
 461             cachedRoot = Path.of(res.stdout().get(0));
 462             return cachedRoot;
 463         }
 464     }

 473     @FunctionalInterface
 474     private static interface Operation {
 475         void execute(List<Path> args) throws IOException;
 476     }

 478     private void batch(Operation op, List<Path> args) throws IOException {
 479         var batchSize = 64;
 480         var start = 0;
 481         while (start < args.size()) {
 482             var end = start + batchSize;
 483             if (end > args.size()) {
 484                 end = args.size();
 485             }
 486             op.execute(args.subList(start, end));
 487             start = end;
 488         }
 489     }

 554     @Override
 555     public Hash commit(String message,
 556                        String authorName,
 557                        String authorEmail,
 558                        ZonedDateTime authorDate,
 559                        String committerName,
 560                        String committerEmail,
 561                        ZonedDateTime committerDate) throws IOException {
 562         var cmd = Process.capture("git", "commit", "--message=" + message)
 563                          .workdir(dir)
 564                          .environ("GIT_AUTHOR_NAME", authorName)
 565                          .environ("GIT_AUTHOR_EMAIL", authorEmail)
 566                          .environ("GIT_COMMITTER_NAME", committerName)
 567                          .environ("GIT_COMMITTER_EMAIL", committerEmail);
 568         if (authorDate != null) {
 569             cmd = cmd.environ("GIT_AUTHOR_DATE",
 570                               authorDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 571         }
 572         if (committerDate != null) {
 573             cmd = cmd.environ("GIT_COMMITTER_DATE",
 574                               committerDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 575         }
 576         try (var p = cmd.execute()) {
 577             await(p);
 578             return head();
 579         }
 580     }

 710     private String treeEntry(Path path, Hash hash) throws IOException {
 711         try (var p = Process.capture("git", "ls-tree", hash.hex(), path.toString())
 712                             .workdir(root())
 713                             .execute()) {
 714             var res = await(p);
 715             if (res.stdout().size() == 0) {
 716                 return null;
 717             }
 718             if (res.stdout().size() > 1) {
 719                 throw new IOException("Unexpected output\n" + res);
 720             }
 721             return res.stdout().get(0);
 722         }
 723     }

 725     private List<FileEntry> allFiles(Hash hash, List<Path> paths) throws IOException {
 726         var cmd = new ArrayList<String>();
 727         cmd.addAll(List.of("git", "ls-tree", "-r"));
 728         cmd.add(hash.hex());
 729         cmd.addAll(paths.stream().map(Path::toString).collect(Collectors.toList()));
 730         try (var p = Process.capture(cmd.toArray(new String[0]))
 731                             .workdir(root())
 732                             .execute()) {
 733             var res = await(p);
 734             var entries = new ArrayList<FileEntry>();
 735             for (var line : res.stdout()) {
 736                 var parts = line.split("\t");
 737                 var metadata = parts[0].split(" ");
 738                 var filename = parts[1];
 739 
 740                 var entry = new FileEntry(hash,
 741                                           FileType.fromOctal(metadata[0]),
 742                                           new Hash(metadata[2]),
 743                                           Path.of(filename));
 744                 entries.add(entry);
 745             }
 746             return entries;
 747         }
 748     }

 751     public List<FileEntry> files(Hash hash, List<Path> paths) throws IOException {
 752         if (paths.isEmpty()) {
 753             return allFiles(hash, paths);
 754         }
 755 
 756         var entries = new ArrayList<FileEntry>();
 757         var batchSize = 64;
 758         var start = 0;
 759         while (start < paths.size()) {
 760             var end = start + batchSize;
 761             if (end > paths.size()) {
 762                 end = paths.size();
 763             }
 764             entries.addAll(allFiles(hash, paths.subList(start, end)));
 765             start = end;
 766         }
 767         return entries;

 781     @Override
 782     public Optional<byte[]> show(Path path, Hash hash) throws IOException {
 783         var entries = files(hash, path);
 784         if (entries.size() == 0) {
 785             return Optional.empty();
 786         } else if (entries.size() > 1) {
 787             throw new IOException("Multiple files match path " + path.toString() + " in commit " + hash.hex());
 788         }
 789 
 790         var entry = entries.get(0);
 791         var type = entry.type();
 792         if (type.isVCSLink()) {
 793             var content = "Subproject commit " + entry.hash().hex() + " " + entry.path().toString();
 794             return Optional.of(content.getBytes(StandardCharsets.UTF_8));
 795         } else if (type.isRegular()) {
 796             var tmp = unpackFile(entry.hash().hex());
 797             var content = Files.readAllBytes(tmp);
 798             Files.delete(tmp);
 799             return Optional.of(content);
 800         }
 801 
 802         return Optional.empty();
 803     }

 844         var cmd = new ArrayList<>(List.of("git", "diff", "--patch",
 845                                                          "--find-renames=99%",

 848                                                          "--binary",

 855             cmd.add(to.hex());

 870         } catch (Throwable t) {
 871             stop(p);
 872             throw t;
 873         }
 874     }

1126     @Override
1127     public List<Submodule> submodules() throws IOException {
1128         var gitModules = root().resolve(".gitmodules");
1129         if (!Files.exists(gitModules)) {
1130             return List.of();
1131         }
1132 
1133         var urls = new HashMap<String, String>();
1134         var paths = new HashMap<String, String>();
1135         try (var p = capture("git", "config", "--file", gitModules.toAbsolutePath().toString(),
1136                                               "--list")) {
1137             for (var line : await(p).stdout()) {
1138                 if (line.startsWith("submodule.")) {
1139                     line = line.substring("submodule.".length());
1140                     var parts = line.split("=");
1141                     var nameAndProperty = parts[0].split("\\.");
1142                     var name = nameAndProperty[0];
1143                     var prop = nameAndProperty[1];
1144                     var value = parts[1];
1145                     if (prop.equals("path")) {
1146                         paths.put(name, value);
1147                     } else if (prop.equals("url")) {
1148                         urls.put(name, value);
1149                     } else {
1150                         throw new IOException("Unexpected submodule property: " + prop);
1151                     }
1152                 }
1153             }
1154         }
1155 
1156         var hashes = new HashMap<String, String>();
1157         try (var p = capture("git", "submodule", "status")) {
1158             for (var line : await(p).stdout()) {
1159                 var parts = line.substring(1).split(" ");
1160                 var hash = parts[0];
1161                 var path = parts[1];
1162                 hashes.put(path, hash);
1163             }
1164         }
1165 
1166         var modules = new ArrayList<Submodule>();
1167         for (var name : paths.keySet()) {
1168             var url = urls.get(name);
1169             var path = paths.get(name);
1170             var hash = hashes.get(path);
1171 
1172             modules.add(new Submodule(new Hash(hash), Path.of(path), url));
1173         }
1174 
1175         return modules;
1176     }
