23 package org.openjdk.skara.issuetracker;
24 
25 import org.openjdk.skara.host.HostUser;
26 
27 import java.time.ZonedDateTime;
28 
29 public class Comment {
30     private final String id;
31     private final String body;
32     private final HostUser author;
33     private final ZonedDateTime createdAt;
34     private final ZonedDateTime updatedAt;
35 
36     public Comment(String id, String body, HostUser author, ZonedDateTime createdAt, ZonedDateTime updatedAt) {
37         this.id = id;
38         this.body = body;
39         this.author = author;
40         this.createdAt = createdAt;
41         this.updatedAt = updatedAt;
42     }
43 
44     public String id() {
45         return id;
46     }
47 
48     public String body() {
49         return body;
50     }
51 
52     public HostUser author() {
53         return author;
54     }
55 
56     public ZonedDateTime createdAt() {
57         return createdAt;
58     }
59 
60     public ZonedDateTime updatedAt() {
61         return updatedAt;
62     }
63 }
