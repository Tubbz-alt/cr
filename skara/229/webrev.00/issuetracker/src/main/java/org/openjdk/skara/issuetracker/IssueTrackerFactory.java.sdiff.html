23 package org.openjdk.skara.issuetracker;
24 
25 import org.openjdk.skara.host.Credential;
26 import org.openjdk.skara.json.JSONObject;
27 
28 import java.net.URI;
29 import java.util.*;
30 import java.util.stream.*;
31 
32 public interface IssueTrackerFactory {
33     /**
34      * A user-friendly name for the given issue tracker, used for configuration section naming. Should be lower case.
35      * @return
36      */
37     String name();
38 
39     /**
40      * Instantiate an instance of this issue tracker.
41      * @return
42      */
43     IssueTracker create(URI uri, Credential credential, JSONObject configuration);
44 
45     static List<IssueTrackerFactory> getIssueTrackerFactories() {
46         return StreamSupport.stream(ServiceLoader.load(IssueTrackerFactory.class).spliterator(), false)
47                             .collect(Collectors.toList());
48     }
49 }
