 49     private java.lang.Process start(List<String> cmd) throws IOException {
 50         log.fine("Executing " + String.join(" ", cmd));
 51         var pb = new ProcessBuilder(cmd);
 52         pb.directory(dir.toFile());
 53         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
 54         return pb.start();
 55     }

 57     private static void stop(java.lang.Process p) throws IOException {
 58         if (p != null && p.isAlive()) {
 59             var stream = p.getInputStream();
 60             var read = 0;
 61             var buf = new byte[128];
 62             while (read != -1) {
 63                 read = stream.read(buf);
 64             }
 65             try {
 66                 p.waitFor();
 67             } catch (InterruptedException e) {
 68                 throw new IOException(e);
 69             }
 70         }
 71     }

 91     private static Execution.Result await(Execution e) throws IOException {
 92         var result = e.await();
 93         if (result.status() != 0) {
 94             throw new IOException("Unexpected exit code\n" + result);
 95         }
 96         return result;
 97     }

 99     private static void await(java.lang.Process p) throws IOException {
100         try {
101             var res = p.waitFor();
102             if (res != 0) {
103                 throw new IOException("Unexpected exit code: " + res);
104             }
105         } catch (InterruptedException e) {
106             throw new IOException(e);
107         }
108     }

193     public List<CommitMetadata> commitMetadata() throws IOException {
194         var revisions = "--all";
195         var p = start("git", "rev-list", "--format=" + GitCommitMetadata.FORMAT, "--no-abbrev", "--reverse", "--no-color", revisions);
196         var reader = new UnixStreamReader(p.getInputStream());
197         var result = new ArrayList<CommitMetadata>();
198 
199         var line = reader.readLine();
200         while (line != null) {
201             if (!line.startsWith("commit")) {
202                 throw new IOException("Unexpected line: " + line);
203             }
204 
205             result.add(GitCommitMetadata.read(reader));
206             line = reader.readLine();
207         }
208 
209         await(p);
210         return result;
211     }

213     @Override
214     public boolean isEmpty() throws IOException {
215         int numLooseObjects = -1;
216         int numPackedObjects = -1;
217         int numRefs = -1;
218 
219         try (var p = capture("git", "count-objects", "-v")) {
220             var res = await(p);
221             var stdout = res.stdout();
222 
223             for (var line : stdout) {
224                 if (line.startsWith("count: ")) {
225                     try {
226                         numLooseObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
227                     } catch (NumberFormatException e) {
228                         throw new IOException("Unexpected 'count' value\n" + res, e);
229                     }
230 
231                 } else if (line.startsWith("in-pack: ")) {
232                     try {
233                         numPackedObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
234                     } catch (NumberFormatException e) {
235                         throw new IOException("Unexpected 'in-pack' value\n" + res, e);
236                     }
237                 }
238             }
239         }
240 
241         try (var p = capture("git", "show-ref", "--hash", "--abbrev")) {
242             var res = p.await();
243             if (res.status() == -1) {
244                 if (res.stdout().size() != 0) {
245                     throw new IOException("Unexpected output\n" + res);
246                 }
247                 numRefs = 0;
248             } else {
249                 numRefs = res.stdout().size();
250             }
251         }
252 
253         return numLooseObjects == 0 && numPackedObjects == 0 && numRefs == 0;
254     }

256     @Override

258         if (isEmpty()) {
259             return true;
260         }
261 
262         var name = "health-check";
263         try (var p = capture("git", "branch", name, "HEAD")) {
264             if (p.await().status() != 0) {
265                 return false;
266             }
267         }
268         try (var p = capture("git", "branch", "-D", name)) {
269             if (p.await().status() != 0) {
270                 return false;
271             }
272         }
273 
274         return true;
275     }

405         try (var p = capture("git", "rev-parse", "--show-toplevel")) {

408                 // Perhaps this is a bare repository
409                 try (var p2 = capture("git", "rev-parse", "--git-dir")) {
410                     var res2 = await(p2);
411                     if (res2.stdout().size() != 1) {
412                         throw new IOException("Unexpected output\n" + res2);

483     @Override
484     public Hash commit(String message,
485                        String authorName,
486                        String authorEmail,
487                        Instant authorDate,
488                        String committerName,
489                        String committerEmail,
490                        Instant committerDate) throws IOException {
491         var cmd = Process.capture("git", "commit", "--message=" + message)
492                          .workdir(dir)
493                          .environ("GIT_AUTHOR_NAME", authorName)
494                          .environ("GIT_AUTHOR_EMAIL", authorEmail)
495                          .environ("GIT_COMMITTER_NAME", committerName)
496                          .environ("GIT_COMMITTER_EMAIL", committerEmail);
497         if (authorDate != null) {
498             var epochSecond = ZonedDateTime.ofInstant(authorDate, ZoneOffset.UTC);
499             cmd = cmd.environ("GIT_AUTHOR_DATE", epochSecond + " +0000");
500         }
501         if (committerDate != null) {
502             var epochSecond = ZonedDateTime.ofInstant(committerDate, ZoneOffset.UTC);
503             cmd = cmd.environ("GIT_COMMITTER_DATE", epochSecond + " +0000");
504         }
505         try (var p = cmd.execute()) {
506             await(p);
507             return head();
508         }
509     }

634     private String treeEntry(Path path, Hash hash) throws IOException {
635         try (var p = Process.capture("git", "ls-tree", hash.hex(), path.toString())
636                             .workdir(root())
637                             .execute()) {
638             var res = await(p);
639             if (res.stdout().size() == 0) {
640                 return null;
641             }
642             if (res.stdout().size() > 1) {
643                 throw new IOException("Unexpected output\n" + res);
644             }
645             return res.stdout().get(0);
646         }
647     }

649     @Override
650     public Optional<byte[]> show(Path path, Hash hash) throws IOException {
651         var entry = treeEntry(path, hash);
652         if (entry == null) {
653             return Optional.empty();
654         }
655 
656         var parts = entry.split(" ");
657         var mode = parts[0];
658         if (mode.equals("160000")) {
659             // submodule
660             var hashAndName = parts[2].split("\t");
661             return Optional.of(("Subproject commit " + hashAndName[0]).getBytes(StandardCharsets.UTF_8));
662         } else if (mode.equals("100644") || mode.equals("100755")) {
663             // blob
664             var blobAndName = parts[2].split("\t");
665             var blob = blobAndName[0];
666             try (var p = capture("git", "unpack-file", blob)) {
667                 var res = await(p);
668                 if (res.stdout().size() != 1) {
669                     throw new IOException("Unexpected output\n" + res);
670                 }
671 
672                 var file = Path.of(root().toString(), res.stdout().get(0));
673                 if (Files.exists(file)) {
674                     var bytes = Files.readAllBytes(file);
675                     Files.delete(file);
676                     return Optional.of(bytes);
677                 }
678             }
679         }
680         return Optional.empty();
681     }

688     @Override
689     public Diff diff(Hash from, Hash to) throws IOException {
690         var cmd = new ArrayList<>(List.of("git", "diff", "--patch",
691                                                          "--find-renames=99%",
692                                                          "--find-copies=99%",
693                                                          "--find-copies-harder",
694                                                          "--binary",
695                                                          "--raw",
696                                                          "--no-abbrev",
697                                                          "--unified=0",
698                                                          "--no-color",
699                                                          from.hex()));
700         if (to != null) {
701             cmd.add(to.hex());
702         }
703 
704         var p = start(cmd);
705         try {
706             var patches = UnifiedDiffParser.parseGitRaw(p.getInputStream());
707             await(p);
708             return new Diff(from, to, patches);
709         } catch (Throwable t) {
710             stop(p);
711             throw t;
712         }
713     }
