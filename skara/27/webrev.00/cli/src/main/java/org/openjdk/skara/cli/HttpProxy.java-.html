23 package org.openjdk.skara.cli;
24 
25 import java.util.List;
26 import java.net.URI;
27 
28 class HttpProxy {
29     static void setup() {
30         for (var key : List.of("http_proxy", "https_proxy")) {
31             var value = System.getenv(key);
32             value = value == null ? System.getenv(key.toUpperCase()) : value;
33             if (value != null) {
34                 var protocol = key.split("_")[0];
35                 var uri = URI.create(value);
36                 System.setProperty(protocol + ".proxyHost", uri.getHost());
37                 System.setProperty(protocol + ".proxyPort", String.valueOf(uri.getPort()));
38             }
39         }
40         var no_proxy = System.getenv("no_proxy");
41         no_proxy = no_proxy == null ? System.getenv("NO_PROXY") : no_proxy;
42         if (no_proxy != null) {
43             var hosts = no_proxy.replace(",", "|")
44                                 .replaceAll("^\\.", "*.")
45                                 .replaceAll("\\|\\.", "|*.");
46             System.setProperty("http.nonProxyHosts", hosts);
47         }
48     }
49 }
