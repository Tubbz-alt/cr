 23 package org.openjdk.skara.cli;
 24 
 25 import org.openjdk.skara.args.*;
 26 import org.openjdk.skara.vcs.*;
 27 
 28 import java.io.*;
 29 import java.net.URI;
 30 import java.nio.file.*;
 31 import java.util.*;
 32 import java.util.logging.*;
 33 
 34 public class GitSync {
 35     private static IOException die(String message) {
 36         System.err.println(message);
 37         System.exit(1);
 38         return new IOException("will never reach here");
 39     }
 40 
 41     public static void main(String[] args) throws IOException {
 42         var flags = List.of(
 43             Option.shortcut("")
 44                   .fullname("branches")
 45                   .describe("BRANCHES")
 46                   .helptext("Comma separated list of branches to sync")
 47                   .optional(),
 48             Switch.shortcut("m")
 49                   .fullname("mercurial")
 50                   .helptext("Force use of mercurial")
 51                   .optional(),
 52             Switch.shortcut("")
 53                   .fullname("verbose")
 54                   .helptext("Turn on verbose output")
 55                   .optional(),
 56             Switch.shortcut("")
 57                   .fullname("debug")
 58                   .helptext("Turn on debugging output")
 59                   .optional(),
 60             Switch.shortcut("v")
 61                   .fullname("version")
 62                   .helptext("Print the version of this tool")
 63                   .optional()
 64         );
 65 
 66         var inputs = List.of(
 67             Input.position(0)
 68                  .describe("REMOTE")
 69                  .singular()
 70                  .required()
 71         );
 72 
 73         var parser = new ArgumentParser("git sync", flags, inputs);
 74         var arguments = parser.parse(args);
 75 
 76         if (arguments.contains("version")) {
 77             System.out.println("git-sync version: " + Version.fromManifest().orElse("unknown"));
 78             System.exit(0);
 79         }
 80 
 81         if (arguments.contains("verbose") || arguments.contains("debug")) {
 82             var level = arguments.contains("debug") ? Level.FINER : Level.FINE;
 83             Logging.setup(level);
 84         }
 85 
 86         var cwd = Paths.get("").toAbsolutePath();
 87         var repo = Repository.get(cwd).orElseThrow(() ->
 88                 die("error: no repository found at " + cwd.toString())
 89         );
 90 
 91         var upstream = arguments.at(0).asString();
 92         var remotes = repo.remotes();
 93         var upstreamPullPath = remotes.contains(upstream) ?
 94             Remote.toURI(repo.pullPath(upstream)) : URI.create(upstream);
 95         var origin = "origin";
 96         var originPushPath = Remote.toURI(repo.pushPath(origin));
 97 
 98         var branches = new HashSet<String>();
 99         if (arguments.contains("branches")) {
100             var requested = arguments.get("branches").asString().split(",");
101             for (var branch : requested) {
102                 branches.add(branch.trim());
103             }
104         }
105 
106         for (var branch : repo.remoteBranches(upstream)) {
107             var name = branch.name();
108             if (!branches.isEmpty() && !branches.contains(name)) {
109                 System.out.println("Skipping branch " + name);
110                 continue;
111             }
112             System.out.print("Syncing " + upstream + "/" + name + " to " + origin + "/" + name + "... ");
113             var fetchHead = repo.fetch(upstreamPullPath, branch.hash().hex());
114             repo.push(fetchHead, originPushPath, name);
115             System.out.println("done");
116         }
117     }
118 }
