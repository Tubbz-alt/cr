23 package org.openjdk.skara.vcs;
24 
25 import java.util.Objects;
26 import java.util.Optional;

27 
28 public class Tag {
































































29     private final String name;
30 
31     public Tag(String name) {
32         this.name = name;
33     }
34 
35     public String name() {
36         return name;
37     }
38 
39     @Override
40     public String toString() {
41         return name;
42     }
43 
44     @Override
45     public int hashCode() {
46         return Objects.hash(name);
47     }
48 
49     @Override
50     public boolean equals(Object o) {
51         if (!(o instanceof Tag)) {
52             return false;
53         }
54 
55         var other = (Tag) o;
56         return Objects.equals(name, other.name);
57     }
58 }
