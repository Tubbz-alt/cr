 23 package org.openjdk.skara.vcs;
 24 
 25 import java.util.Objects;
 26 import java.util.Optional;
 27 import java.time.ZonedDateTime;
 28 
 29 public class Tag {
 30     public static class Annotated {
 31         private final String name;
 32         private final Hash target;
 33         private final Author author;
 34         private final ZonedDateTime date;
 35         private final String message;
 36 
 37         public Annotated(String name, Hash target, Author author, ZonedDateTime date, String message) {
 38             this.name = name;
 39             this.target = target;
 40             this.author = author;
 41             this.date = date;
 42             this.message = message;
 43         }
 44 
 45         public String name() {
 46             return name;
 47         }
 48 
 49         public Hash target() {
 50             return target;
 51         }
 52 
 53         public Author author() {
 54             return author;
 55         }
 56 
 57         public ZonedDateTime date() {
 58             return date;
 59         }
 60 
 61         public String message() {
 62             return message;
 63         }
 64 
 65         @Override
 66         public boolean equals(Object other) {
 67             if (other == this) {
 68                 return true;
 69             }
 70 
 71             if (!(other instanceof Annotated)) {
 72                 return false;
 73             }
 74 
 75             var o = (Annotated) other;
 76             return Objects.equals(name, o.name) &&
 77                    Objects.equals(target, o.target) &&
 78                    Objects.equals(author, o.author) &&
 79                    Objects.equals(date, o.date) &&
 80                    Objects.equals(message, o.message);
 81         }
 82 
 83         @Override
 84         public int hashCode() {
 85             return Objects.hash(name, target, author, date, message);
 86         }
 87 
 88         @Override
 89         public String toString() {
 90             return name + " -> " + target.hex();
 91         }
 92     }
 93 
 94     private final String name;
 95 
 96     public Tag(String name) {
 97         this.name = name;
 98     }
 99 
100     public String name() {
101         return name;
102     }
103 
104     @Override
105     public String toString() {
106         return name;
107     }
108 
109     @Override
110     public int hashCode() {
111         return Objects.hash(name);
112     }
113 
114     @Override
115     public boolean equals(Object o) {
116         if (!(o instanceof Tag)) {
117             return false;
118         }
119 
120         var other = (Tag) o;
121         return Objects.equals(name, other.name);
122     }
123 }
