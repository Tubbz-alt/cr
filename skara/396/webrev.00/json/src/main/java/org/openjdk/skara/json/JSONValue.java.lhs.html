 23 package org.openjdk.skara.json;
 24 
 25 import java.util.stream.Stream;
 26 import java.util.List;
 27 
 28 public interface JSONValue {
 29     default int asInt() {
 30         throw new IllegalStateException("Unsupported conversion to int");
 31     }
 32 
 33     default long asLong() {
 34         throw new IllegalStateException("Unsupported conversion to long");
 35     }
 36 
 37     default double asDouble() {
 38         throw new IllegalStateException("Unsupported conversion to double");
 39     }
 40 
 41     default String asString() {
 42         throw new IllegalStateException("Unsupported conversion to String");
 43     }
 44 
 45     default boolean asBoolean() {
 46         throw new IllegalStateException("Unsupported conversion to boolean");
 47     }
 48 
 49     default JSONArray asArray() {
 50         throw new IllegalStateException("Unsupported conversion to array");
 51     }
 52 
 53     default JSONObject asObject() {
 54         throw new IllegalStateException("Unsupported conversion to object");
 55     }
 56 




























 57     default boolean isNull() {
 58         return false;
 59     }
 60 
 61     default List<JSONObject.Field> fields() {
 62         return asObject().fields();
 63     }
 64 
 65     default boolean contains(String field) {
 66         return asObject().contains(field);
 67     }
 68 
 69     default JSONValue get(String field) {
 70         return asObject().get(field);
 71     }
 72 
 73     default JSONValue get(int i) {
 74         return asArray().get(i);
 75     }
 76 
 77     default Stream<JSONValue> stream() {
 78         return Stream.of(this);
 79     }
 80 
 81     static JSONValue from(int i) {
 82         return new JSONNumber(i);
 83     }
 84 
 85     static JSONValue from(long l) {
 86         return new JSONNumber(l);
 87     }
 88 
 89     static JSONValue from(double d) {
 90         return new JSONDecimal(d);
 91     }
 92 
 93     static JSONValue from(boolean b) {
 94         return new JSONBoolean(b);
 95     }
 96 
 97     static JSONValue from(String s) {
 98         return new JSONString(s);
 99     }
100 
101     static JSONValue fromNull() {
102         return new JSONNull();
103     }
104 }
