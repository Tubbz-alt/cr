23 package org.openjdk.skara.json;
24 
25 class JSONString implements JSONValue {
26     String value;
27 
28     public JSONString(String value) {
29         this.value = value;
30     }
31 





32     @Override
33     public String asString() {
34         return value;
35     }
36 
37     @Override
38     public String toString() {
39         var builder = new StringBuilder();
40         builder.append("\"");
41 
42         for (var i = 0; i < value.length(); i++) {
43             var c = value.charAt(i);
44 
45             switch (c) {
46                 case '"':
47                     builder.append("\\\"");
48                     break;
49                 case '\\':
50                     builder.append("\\\\");
51                     break;
52                 case '/':
53                     builder.append("\\/");
54                     break;
55                 case '\b':
56                     builder.append("\\b");
57                     break;
58                 case '\f':
59                     builder.append("\\f");
60                     break;
61                 case '\n':
62                     builder.append("\\n");
63                     break;
64                 case '\r':
65                     builder.append("\\r");
66                     break;
67                 case '\t':
68                     builder.append("\\t");
69                     break;
70                 default:
71                     builder.append(c);
72                     break;
73             }
74         }
75 
76         builder.append("\"");
77         return builder.toString();
78     }
79 }
