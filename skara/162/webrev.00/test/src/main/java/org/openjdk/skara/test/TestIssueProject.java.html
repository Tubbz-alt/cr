23 package org.openjdk.skara.test;
24 
25 import org.openjdk.skara.host.*;
26 import org.openjdk.skara.host.network.URIBuilder;
27 
28 import java.net.URI;
29 import java.util.*;
30 
31 public class TestIssueProject implements IssueProject {
32     private final String projectName;
33     private final TestHost host;
34 
35     @Override
36     public Host host() {
37         return host;
38     }
39 
40     @Override
41     public URI getWebUrl() {
42         return URIBuilder.base("http://localhost/project/" + projectName).build();
43     }
44 
45     public TestIssueProject(TestHost host, String projectName) {
46         this.host = host;
47         this.projectName = projectName;
48     }
49 
50     @Override
51     public Issue createIssue(String title, List<String> body) {
52         return host.createIssue(this, title, body);
53     }
54 
55     @Override
56     public Issue getIssue(String id) {
57         return host.getIssue(this, id);
58     }
59 
60     @Override
61     public List<Issue> getIssues() {
62         return new ArrayList<>(host.getIssues(this));
63     }
64 }
