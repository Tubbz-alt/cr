  48     private List<Path> findJsonFiles(Path folder, String partialName) throws IOException {
  49         return Files.walk(folder)
  50                     .filter(path -> path.toString().endsWith(".json"))
  51                     .filter(path -> path.toString().contains(partialName))
  52                     .collect(Collectors.toList());
  53     }

  70     private Set<String> fixVersions(Issue issue) {
  71         if (!issue.properties().containsKey("fixVersions")) {
  72             return Set.of();
  73         }
  74         return issue.properties().get("fixVersions").stream()
  75                     .map(JSONValue::asString)
  76                     .collect(Collectors.toSet());
  77     }

 154 
 155             for (var file : jsonFiles) {
 156                 var jsonData = Files.readString(file, StandardCharsets.UTF_8);
 157                 var json = JSON.parse(jsonData);
 158 
 159                 if (json.asArray().get(0).contains("date")) {
 160                     assertEquals(2, json.asArray().size());
 161                     assertEquals(List.of("23456789"), json.asArray().get(0).get("issue").asArray().stream()
 162                                                           .map(JSONValue::asString)
 163                                                           .collect(Collectors.toList()));
 164                     assertEquals(repo.webUrl(editHash).toString(), json.asArray().get(0).get("url").asString());

 166                     assertEquals(List.of("34567890"), json.asArray().get(1).get("issue").asArray().stream()
 167                                                           .map(JSONValue::asString)
 168                                                           .collect(Collectors.toList()));
 169                     assertEquals(repo.webUrl(editHash2).toString(), json.asArray().get(1).get("url").asString());
 170                     assertEquals("team", json.asArray().get(1).get("build").asString());
 171                 } else {
 172                     assertEquals(1, json.asArray().size());
 173                     if (json.asArray().get(0).get("build").asString().equals("b02")) {
 174                         assertEquals(List.of("23456789"), json.asArray().get(0).get("issue").asArray().stream()
 175                                                               .map(JSONValue::asString)
 176                                                               .collect(Collectors.toList()));
 177                     } else {
 178                         assertEquals("b04", json.asArray().get(0).get("build").asString());
 179                         assertEquals(List.of("34567890"), json.asArray().get(0).get("issue").asArray().stream()
 180                                                               .map(JSONValue::asString)
 181                                                               .collect(Collectors.toList()));
 182                     }
 183                 }
 184             }
