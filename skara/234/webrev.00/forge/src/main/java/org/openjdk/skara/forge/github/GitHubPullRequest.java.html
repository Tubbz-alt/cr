 74     @Override
 75     public List<Review> reviews() {
 76         var reviews = request.get("pulls/" + json.get("number").toString() + "/reviews").execute().stream()
 77                              .map(JSONValue::asObject)
 78                              .filter(obj -> !(obj.get("state").asString().equals("COMMENTED") && obj.get("body").asString().isEmpty()))
 79                              .map(obj -> {
 80                                  var reviewer = host.parseUserField(obj);
 81                                  var hash = new Hash(obj.get("commit_id").asString());
 82                                  Review.Verdict verdict;
 83                                  switch (obj.get("state").asString()) {
 84                                      case "APPROVED":
 85                                          verdict = Review.Verdict.APPROVED;
 86                                          break;
 87                                      case "CHANGES_REQUESTED":
 88                                          verdict = Review.Verdict.DISAPPROVED;
 89                                          break;
 90                                      default:
 91                                          verdict = Review.Verdict.NONE;
 92                                          break;
 93                                  }
 94                                  var id = obj.get("id").asInt();
 95                                  var body = obj.get("body").asString();
 96                                  return new Review(reviewer, verdict, hash, id, body);
 97                              })
 98                              .collect(Collectors.toList());
 99         return reviews;
100     }

102     @Override
103     public void addReview(Review.Verdict verdict, String body) {
104         var query = JSON.object();
105         switch (verdict) {
106             case APPROVED:
107                 query.put("event", "APPROVE");
108                 break;
109             case DISAPPROVED:
110                 query.put("event", "REQUEST_CHANGES");
111                 break;
112             case NONE:
113                 query.put("event", "COMMENT");
114                 break;
115         }
116         query.put("body", body);
117         request.post("pulls/" + json.get("number").toString() + "/reviews")
118                .body(query)
119                .execute();
120     }

122     private ReviewComment parseReviewComment(ReviewComment parent, JSONObject json, PositionMapper diff) {
123         var author = host.parseUserField(json);
124         var threadId = parent == null ? json.get("id").toString() : parent.threadId();
125         var comment = new ReviewComment(parent,
126                                         threadId,
127                                         new Hash(json.get("commit_id").asString()),
128                                         json.get("path").asString(),
129                                         diff.positionToLine(json.get("path").asString(), json.get("original_position").asInt()),
130                                         json.get("id").toString(),
131                                         json.get("body").asString(),
132                                         author,
133                                         ZonedDateTime.parse(json.get("created_at").asString()),
134                                         ZonedDateTime.parse(json.get("updated_at").asString()));
135         return comment;
136     }

138     @Override
139     public ReviewComment addReviewComment(Hash base, Hash hash, String path, int line, String body) {
140         try {
141             var rawDiff = request.get("pulls/" + json.get("number").toString())
142                                  .header("Accept", "application/vnd.github.v3.diff")
143                                  .executeUnparsed();
144             var diff = PositionMapper.parse(rawDiff);
145 
146             var query = JSON.object()
147                             .put("body", body)
148                             .put("commit_id", hash.hex())
149                             .put("path", path)
150                             .put("position", diff.lineToPosition(path, line));
151             var response = request.post("pulls/" + json.get("number").toString() + "/comments")
152                                   .body(query)
153                                   .execute();
154             return parseReviewComment(null, response.asObject(), diff);
155         } catch (IOException e) {
156             throw new UncheckedIOException(e);
157         }
158     }

160     @Override
161     public ReviewComment addReviewCommentReply(ReviewComment parent, String body) {
162         try {
163             var rawDiff = request.get("pulls/" + json.get("number").toString())
164                                  .header("Accept", "application/vnd.github.v3.diff")
165                                  .executeUnparsed();
166             var diff = PositionMapper.parse(rawDiff);
167 
168             var query = JSON.object()
169                             .put("body", body)
170                             .put("in_reply_to", Integer.parseInt(parent.threadId()));
171             var response = request.post("pulls/" + json.get("number").toString() + "/comments")
172                                   .body(query)
173                                   .execute();
174             return parseReviewComment(parent, response.asObject(), diff);
175         } catch (IOException e) {
176             throw new UncheckedIOException(e);
177         }
178     }

180     @Override
181     public List<ReviewComment> reviewComments() {
182         try {
183             var rawDiff = request.get("pulls/" + json.get("number").toString())
184                                  .header("Accept", "application/vnd.github.v3.diff")
185                                  .executeUnparsed();
186             var diff = PositionMapper.parse(rawDiff);
187 
188             var ret = new ArrayList<ReviewComment>();
189             var reviewComments = request.get("pulls/" + json.get("number").toString() + "/comments").execute().stream()
190                                         .map(JSONValue::asObject)
191                                         .collect(Collectors.toList());
192             var idToComment = new HashMap<String, ReviewComment>();
193 
194             for (var reviewComment : reviewComments) {
195                 ReviewComment parent = null;
196                 if (reviewComment.contains("in_reply_to_id")) {
197                     parent = idToComment.get(reviewComment.get("in_reply_to_id").toString());
198                 }
199                 var comment = parseReviewComment(parent, reviewComment, diff);
200                 idToComment.put(comment.id(), comment);
201                 ret.add(comment);
202             }
203 
204             return ret;
205         } catch (IOException e) {
206             throw new UncheckedIOException(e);
207         }
208     }

298     @Override
299     public Map<String, Check> checks(Hash hash) {
300         var checks = request.get("commits/" + hash.hex() + "/check-runs").execute();
301 
302         return checks.get("check_runs").stream()
303                 .collect(Collectors.toMap(c -> c.get("name").asString(),
304                         c -> {
305                             var checkBuilder = CheckBuilder.create(c.get("name").asString(), new Hash(c.get("head_sha").asString()));
306                             checkBuilder.startedAt(ZonedDateTime.parse(c.get("started_at").asString()));
307 
308                             var completed = c.get("status").asString().equals("completed");
309                             if (completed) {
310                                 var conclusion = c.get("conclusion").asString();
311                                 var completedAt = ZonedDateTime.parse(c.get("completed_at").asString());
312                                 switch (conclusion) {
313                                     case "cancelled":
314                                         checkBuilder.cancel(completedAt);
315                                         break;
316                                     case "success":
317                                         checkBuilder.complete(true, completedAt);
318                                         break;
319                                     case "failure":
320                                     case "neutral":
321                                         checkBuilder.complete(false, completedAt);
322                                         break;
323                                     default:
324                                         throw new IllegalStateException("Unexpected conclusion: " + conclusion);
325                                 }
326                             }
327                             if (c.contains("external_id")) {
328                                 checkBuilder.metadata(c.get("external_id").asString());
329                             }
330                             if (c.contains("output")) {
331                                 var output = c.get("output").asObject();
332                                 if (output.contains("title")) {
333                                     checkBuilder.title(output.get("title").asString());
334                                 }
335                                 if (output.contains("summary")) {
336                                     checkBuilder.summary(output.get("summary").asString());
337                                 }
338                             }
339 
340                             return checkBuilder.build();
341                         }));
342     }

344     @Override
345     public void createCheck(Check check) {
346         // update and create are currently identical operations, both do an HTTP
347         // POST to the /repos/:owner/:repo/check-runs endpoint. There is an additional
348         // endpoint explicitly for updating check-runs, but that is not currently used.
349         updateCheck(check);
350     }

352     @Override
353     public void updateCheck(Check check) {
354         var completedQuery = JSON.object();
355         completedQuery.put("name", check.name());
356         completedQuery.put("head_branch", json.get("head").get("ref"));
357         completedQuery.put("head_sha", check.hash().hex());
358 
359         if (check.title().isPresent() && check.summary().isPresent()) {
360             var outputQuery = JSON.object();
361             outputQuery.put("title", check.title().get());
362             outputQuery.put("summary", check.summary().get());
363 
364             var annotations = JSON.array();
365             for (var annotation : check.annotations().subList(0, Math.min(check.annotations().size(), 50))) {
366                 var annotationQuery = JSON.object();
367                 annotationQuery.put("path", annotation.path());
368                 annotationQuery.put("start_line", annotation.startLine());
369                 annotationQuery.put("end_line", annotation.endLine());
370                 annotation.startColumn().ifPresent(startColumn -> annotationQuery.put("start_column", startColumn));
371                 annotation.endColumn().ifPresent(endColumn -> annotationQuery.put("end_column", endColumn));
372                 switch (annotation.level()) {
373                     case NOTICE:
374                         annotationQuery.put("annotation_level", "notice");
375                         break;
376                     case WARNING:
377                         annotationQuery.put("annotation_level", "warning");
378                         break;
379                     case FAILURE:
380                         annotationQuery.put("annotation_level", "failure");
381                         break;
382                 }
383 
384                 annotationQuery.put("message", annotation.message());
385                 annotation.title().ifPresent(title -> annotationQuery.put("title", title));
386                 annotations.add(annotationQuery);
387             }
388 
389             outputQuery.put("annotations", annotations);
390             completedQuery.put("output", outputQuery);
391         }
392 
393         if (check.status() == CheckStatus.IN_PROGRESS) {
394             completedQuery.put("status", "in_progress");
395         } else {
396             completedQuery.put("status", "completed");
397             completedQuery.put("conclusion", check.status().name().toLowerCase());
398             completedQuery.put("completed_at", check.completedAt().orElse(ZonedDateTime.now(ZoneOffset.UTC))
399                     .format(DateTimeFormatter.ISO_INSTANT));
400         }
401 
402         completedQuery.put("started_at", check.startedAt().format(DateTimeFormatter.ISO_INSTANT));
403         check.metadata().ifPresent(metadata -> completedQuery.put("external_id", metadata));
404 
405         request.post("check-runs").body(completedQuery).execute();
406     }

438     @Override
439     public void removeLabel(String label) {
440         request.delete("issues/" + json.get("number").toString() + "/labels/" + label)
441                .onError(r -> {
442                    // The GitHub API explicitly states that 404 is the response for deleting labels currently not set
443                    if (r.statusCode() == 404) {
444                        return JSONValue.fromNull();
445                    }
446                    throw new RuntimeException("Invalid response");
447                })
448                .execute();
449     }
