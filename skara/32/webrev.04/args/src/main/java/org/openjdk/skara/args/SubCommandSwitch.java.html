 23 package org.openjdk.skara.args;
 24 
 25 import java.io.PrintStream;
 26 import java.util.Arrays;
 27 import java.util.HashMap;
 28 import java.util.Map;
 29 import java.util.stream.Collectors;
 30 
 31 public class SubCommandSwitch implements Command {
 32 
 33     private static final SubCommandEntry errorCommand = new SubCommandEntry("", "",
 34             args -> System.out.println("error: unknown subcommand: " + args[0]));
 35 
 36     private final String programName;
 37     private final String defaultCommand;
 38     private final Map<String, SubCommandEntry> subCommands;
 39 
 40     private SubCommandSwitch(String programName, String defaultCommand, Map<String, SubCommandEntry> subCommands) {
 41         this.programName = programName;
 42         this.subCommands = subCommands;
 43         this.subCommands.put("help", helpCommand());
 44         this.defaultCommand = defaultCommand;
 45     }
 46 
 47     private SubCommandEntry helpCommand() {
 48         return new SubCommandEntry("help", "print a help message", args -> showUsage());
 49     }
 50 
 51     public static Builder builder(String programName) {
 52         return new Builder(programName);
 53     }
 54 
 55     @Override
 56     public void execute(String[] args) throws Exception {
 57         if (args.length != 0) {
 58             SubCommandEntry p = subCommands.get(args[0]);
 59             if (p != null) {
 60                 String[] forwardedArgs = Arrays.copyOfRange(args, 1, args.length);
 61                 p.command.execute(forwardedArgs);
 62             } else {
 63                 subCommands.getOrDefault(defaultCommand, errorCommand).command.execute(args);
 64             }
 65         } else {
 66             showUsage();
 67         }
 68     }
 69 
 70     private void showUsage() {
 71         showUsage(System.out);
 72     }
 73 
 74     private void showUsage(PrintStream ps) {
 75         ps.print("usage: ");
 76         ps.print(programName);
 77         ps.print(subCommands.keySet().stream().collect(Collectors.joining("|", " <", ">")));
 78         ps.println(" <input>");
 79 
 80         int spacing = subCommands.keySet().stream().mapToInt(String::length).max().orElse(0);
 81         spacing += 8; // some room
 82 
 83         for (var subCommand : subCommands.values()) {
 84             ps.println(String.format("  %-" + spacing + "s%s", subCommand.name, subCommand.description));
 85         }
 86     }
 87 
 88     private static class SubCommandEntry {
 89         private final String name;
 90         private final String description;
 91         private final Command command;
 92 
 93         public SubCommandEntry(String name, String description, Command command) {
 94             this.name = name;
 95             this.description = description;
 96             this.command = command;
 97         }
 98     }
 99 
100     public static class Builder {
101         private final String programName;
102 
103         private String defaultCommand;
104         private final Map<String,  SubCommandEntry> subCommands = new HashMap<>();
105 
106         private Builder(String programName) {
107             this.programName = programName;
108         }
109 
110         public Builder defaultCommand(String command, String description, Command action) {
111             subCommand(command, description, action);
112             this.defaultCommand = command;
113             return this;
114         }
115 
116         public Builder subCommand(String command, String description, Command action) {
117             subCommands.put(command, new SubCommandEntry(command, description, action));
118             return this;
119         }
120 
121         public SubCommandSwitch build() {
122             return new SubCommandSwitch(programName, defaultCommand, subCommands);
123         }
124     }
125 }
