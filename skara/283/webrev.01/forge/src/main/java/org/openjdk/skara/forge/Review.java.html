23 package org.openjdk.skara.forge;
24 
25 import org.openjdk.skara.host.HostUser;
26 import org.openjdk.skara.vcs.Hash;
27 
28 import java.time.ZonedDateTime;
29 import java.util.Optional;
30 
31 public class Review {
32     private final ZonedDateTime createdAt;
33     private final HostUser reviewer;
34     private final Verdict verdict;
35     private final Hash hash;
36     private final int id;
37     private final String body;
38 
39     public Review(ZonedDateTime createdAt, HostUser reviewer, Verdict verdict, Hash hash, int id, String body) {
40         this.createdAt = createdAt;
41         this.reviewer = reviewer;
42         this.verdict = verdict;
43         this.hash = hash;
44         this.id = id;
45         this.body = body;
46     }
47 
48     public ZonedDateTime createdAt() {
49         return createdAt;
50     }
51 
52     public HostUser reviewer() {
53         return reviewer;
54     }
55 
56     public Verdict verdict() {
57         return verdict;
58     }
59 
60     public Hash hash() {
61         return hash;
62     }
63 
64     public int id() {
65         return id;
66     }
67 
68     public Optional<String> body() {
69         return Optional.ofNullable(body);
70     }
71 
72     public enum Verdict {
73         NONE,
74         APPROVED,
75         DISAPPROVED
76     }
77 }
