23 package org.openjdk.skara.forge;
24 
25 import org.openjdk.skara.host.HostUser;
26 import org.openjdk.skara.issuetracker.Comment;
27 import org.openjdk.skara.vcs.Hash;
28 
29 import java.time.ZonedDateTime;
30 import java.util.Optional;
31 
32 public class ReviewComment extends Comment {
33     private final ReviewComment parent;
34     private final String threadId;
35     private final Hash hash;
36     private final String path;
37     private final int line;
38 
39     public ReviewComment(ReviewComment parent, String threadId, Hash hash, String path, int line, String id, String body, HostUser author, ZonedDateTime createdAt, ZonedDateTime updatedAt) {
40         super(id, body, author, createdAt, updatedAt);
41 
42         this.parent = parent;
43         this.threadId = threadId;
44         this.hash = hash;
45         this.path = path;
46         this.line = line;
47     }
48 
49     public Optional<ReviewComment> parent() {
50         return Optional.ofNullable(parent);
51     }
52 
53     public Hash hash() {
54         return hash;
55     }
56 
57     public String path() {
58         return path;
59     }
60 
61     public int line() {
62         return line;
63     }
64 
65     public String threadId() {
66         return threadId;
67     }
68 }
