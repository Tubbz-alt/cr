23 package org.openjdk.skara.bots.mlbridge;
24 
25 import org.junit.jupiter.api.Test;
26 
27 import static org.junit.jupiter.api.Assertions.*;
28 
29 class MarkdownToTextTests {
30     @Test
31     void emoji() {
32         assertEquals("üòÑ", MarkdownToText.removeFormatting(":smile:"));
33         assertEquals("yay üòÑ", MarkdownToText.removeFormatting("yay :smile:"));
34         assertEquals("üòÑ\nüòÑ", MarkdownToText.removeFormatting(":smile:\n:smile:"));
35         assertEquals("üòÑ üôÅ", MarkdownToText.removeFormatting(":smile: :slight_frown:"));
36         assertEquals("üòÑ üôÅ :meh:", MarkdownToText.removeFormatting(":smile: :slight_frown: :meh:"));
37     }
38 
39     @Test
40     void patterns() {
41         for (var emojiMap : EmojiTable.mapping.entrySet()) {
42             var pattern = ":" + emojiMap.getKey() + ":";
43             assertNotEquals(pattern, MarkdownToText.removeFormatting(pattern));
44         }
45     }
46 
47     @Test
48     void code() {
49         assertEquals("Just some text", MarkdownToText.removeFormatting("```\nJust some text\n```"));
50         assertEquals("Multi\nline", MarkdownToText.removeFormatting("```\nMulti\nline\n```"));
51         assertEquals("Script", MarkdownToText.removeFormatting("```bash\nScript\n```"));
52     }
53 
54     @Test
55     void suggestion() {
56         assertEquals("Suggestion:\n\nJust some text", MarkdownToText.removeFormatting("```suggestion\nJust some text\n```"));
57     }










58 }
