  48     private java.lang.Process start(String... cmd) throws IOException {
  49         return start(Arrays.asList(cmd));
  50     }

  52     private java.lang.Process start(List<String> cmd) throws IOException {
  53         log.fine("Executing " + String.join(" ", cmd));
  54         var pb = new ProcessBuilder(cmd);
  55         pb.directory(dir.toFile());
  56         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  57         pb.environment().put("HGRCPATH", "");
  58         pb.environment().put("HGPLAIN", "");
  59         return pb.start();
  60     }

  62     private static void stop(java.lang.Process p) throws IOException {
  63         if (p != null && p.isAlive()) {
  64             var stream = p.getInputStream();
  65             var read = 0;
  66             var buf = new byte[128];
  67             while (read != -1) {
  68                 read = stream.read(buf);
  69             }
  70             try {
  71                 p.waitFor();
  72             } catch (InterruptedException e) {
  73                 throw new IOException(e);
  74             }
  75         }
  76     }

  86     private static Execution capture(Path cwd, List<String> cmd) {
  87         return capture(cwd, cmd.toArray(new String[0]));

  89     private static Execution capture(Path cwd, String... cmd) {
  90         return Process.capture(cmd)
  91                       .environ("HGRCPATH", "")
  92                       .environ("HGPLAIN", "")
  93                       .workdir(cwd)
  94                       .execute();
  95     }

  97     private static Execution.Result await(Execution e) throws IOException {
  98         var result = e.await();
  99         if (result.status() != 0) {
 100             if (result.exception().isPresent()) {
 101                 throw new IOException("Unexpected exit code\n" + result, result.exception().get());
 102             } else {
 103                 throw new IOException("Unexpected exit code\n" + result);
 104             }
 105         }
 106         return result;
 107     }
 108 
 109     private static void await(java.lang.Process p) throws IOException {
 110         try {
 111             var res = p.waitFor();
 112             if (res != 0) {
 113                 throw new IOException("Unexpected exit code: " + res);
 114             }
 115         } catch (InterruptedException e) {
 116             throw new IOException(e);
 117         }
 118     }

 157     private void checkout(String ref, boolean force) throws IOException {
 158         var cmd = new ArrayList<String>();
 159         cmd.addAll(List.of("hg", "update"));
 160         if (!force) {
 161             cmd.add("--check");
 162         }
 163         cmd.add(ref);
 164         try (var p = capture(cmd)) {
 165             await(p);
 166         }

 180     public Optional<Hash> resolve(String ref) throws IOException {
 181         try (var p = capture("hg", "log", "--rev=" + ref, "--template={node}\n")) {
 182             var res = p.await();
 183             if (res.status() == 0 && res.stdout().size() == 1) {
 184                 return Optional.of(new Hash(res.stdout().get(0)));
 185             }
 186             return Optional.empty();

 253     @Override
 254     public List<CommitMetadata> commitMetadata() throws IOException {
 255         var ext = Files.createTempFile("ext", ".py");
 256         copyResource(EXT_PY, ext);
 257 
 258         var p = start("hg", "--config", "extensions.dump=" + ext.toAbsolutePath().toString(), "metadata");
 259         var reader = new UnixStreamReader(p.getInputStream());
 260         var result = new ArrayList<CommitMetadata>();
 261 
 262         var line = reader.readLine();
 263         while (line != null) {
 264             result.add(HgCommitMetadata.read(reader));
 265             line = reader.readLine();
 266         }
 267 
 268         await(p);
 269         return result;
 270     }

 272     @Override
 273     public boolean isEmpty() throws IOException {
 274         var numBranches = branches().size();
 275         var numTags = tags().size();
 276 
 277         if (numBranches > 0 || numTags > 1) {
 278             return false;
 279         }
 280 
 281         var tip = resolve("tip");
 282         return tip.isEmpty() || tip.get().hex().equals("0".repeat(40));
 283     }

 292     @Override
 293     public void clean() throws IOException {
 294         try (var p = capture("hg", "merge", "--abort")) {
 295             p.await();
 296         }
 297 
 298         try (var p = capture("hg", "recover")) {
 299             p.await();
 300         }
 301 
 302         try (var p = capture("hg", "status", "--ignored", "--no-status")) {
 303             var root = root().toString();
 304             for (var filename : await(p).stdout()) {
 305                 Files.delete(Path.of(root, filename));
 306             }
 307         }
 308 
 309         try (var p = capture("hg", "status", "--unknown", "--no-status")) {
 310             var root = root().toString();
 311             for (var filename : await(p).stdout()) {
 312                 Files.delete(Path.of(root, filename));
 313             }
 314         }
 315 
 316         try (var p = capture("hg", "revert", "--no-backup", "--all")) {
 317             await(p);
 318         }
 319     }

 336     @Override
 337     public Hash fetch(URI uri, String refspec) throws IOException {
 338         var oldHeads = new HashSet<Hash>(heads());
 339 
 340         var cmd = new ArrayList<String>();
 341         cmd.add("hg");
 342         cmd.add("pull");
 343         if (refspec != null) {
 344             cmd.add("--rev");
 345             cmd.add(refspec);
 346         }
 347         if (uri != null) {
 348             cmd.add(uri.toString());
 349         }
 350         try (var p = capture(cmd)) {
 351             await(p);
 352         }
 353 
 354         var newHeads = new HashSet<Hash>(heads());
 355         newHeads.removeAll(oldHeads);
 356 
 357         if (newHeads.size() > 1) {
 358             throw new IllegalStateException("fetching multiple heads is not supported");
 359         } else if (newHeads.size() == 0) {
 360             // no new head was fetched, return current head
 361             return head();
 362         }
 363         return newHeads.iterator().next();
 364     }

 366     @Override
 367     public void fetchAll() throws IOException {
 368         var pullPaths = new ArrayList<URI>();
 369         try (var p = capture("hg", "paths")) {
 370             var res = await(p);
 371             for (var line : res.stdout()) {
 372                 var parts = line.split("=");
 373                 var name = parts[0].trim();
 374                 var uri = parts[1].trim();
 375                 if (!name.endsWith("-push")) {
 376                     pullPaths.add(URI.create(uri));
 377                 }
 378             }
 379         }
 380 
 381         for (var uri : pullPaths) {
 382             fetch(uri, null);
 383         }
 384     }

 457     private void export(String revset, Path to) throws IOException {
 458         var cmd = List.of("hg", "export", "--git", "--rev", revset);
 459         log.fine("Executing " + String.join(" ", cmd));
 460         var pb = new ProcessBuilder(cmd);
 461         pb.directory(dir.toFile());
 462         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
 463         pb.redirectOutput(to.toFile());
 464         pb.environment().put("HGRCPATH", "");
 465         pb.environment().put("HGPLAIN", "");
 466         var p = pb.start();
 467         try {
 468             await(p);
 469         } catch (Throwable t) {
 470             if (p.isAlive()) {
 471                 try {
 472                     p.waitFor();
 473                 } catch (InterruptedException e) {
 474                     throw new IOException(e);
 475                 }
 476             }
 477 
 478             throw new IOException(t);
 479         }
 480     }

 482     @Override
 483     public void squash(Hash h) throws IOException {
 484         var revset = ".:" + h.hex() + " and not .";
 485         var patch = Files.createTempFile("squash", ".patch");
 486         export(revset, patch);
 487 
 488         try (var p = capture("hg", "--config", "extensions.mq=", "strip", "--rev", revset)) {
 489             await(p);
 490         }
 491 
 492         try (var p = capture("hg", "import", "--no-commit", patch.toString())) {
 493             await(p);
 494         }
 495     }

 503     @Override
 504     public Hash commit(String message, String authorName, String authorEmail, ZonedDateTime authorDate)  throws IOException {
 505         var user = authorEmail == null ? authorName : authorName + " <" + authorEmail + ">";
 506         var cmd = new ArrayList<String>();
 507         cmd.addAll(List.of("hg", "commit", "--message=" + message, "--user=" + user));
 508         if (authorDate != null) {
 509             var formatter = DateTimeFormatter.ISO_OFFSET_DATE_TIME;
 510             cmd.add("--date=" + authorDate.format(formatter));
 511         }
 512         try (var p = capture(cmd)) {
 513             await(p);
 514         }
 515         return resolve("tip").orElseThrow(() -> new IOException("Could not resolve 'tip'"));
 516     }

 646     @Override
 647     public Optional<byte[]> show(Path path, Hash hash) throws IOException {
 648         var output = Files.createTempFile("hg-cat-rev-" + hash.abbreviate(), ".bin");
 649         try (var p = capture("hg", "cat", "--output=" + output, "--rev=" + hash.hex(), path.toString())) {
 650             var res = p.await();
 651             if (res.status() == 0 && Files.exists(output)) {
 652                 var bytes = Files.readAllBytes(output);
 653                 Files.delete(output);
 654                 return Optional.of(bytes);
 655             }
 656 
 657             if (Files.exists(output)) {
 658                 Files.delete(output);
 659             }
 660             return Optional.empty();
 661         }
 662     }

 664     private List<FileEntry> allFiles(Hash hash, List<Path> paths) throws IOException {
 665         var ext = Files.createTempFile("ext", ".py");
 666         copyResource(EXT_PY, ext);
 667 
 668         var include = new HashSet<>(paths);
 669 
 670         try (var p = capture("hg", "--config", "extensions.ls-tree=" + ext, "ls-tree", hash.hex())) {
 671             var res = await(p);
 672             var entries = new ArrayList<FileEntry>();
 673             for (var line : res.stdout()) {
 674                 var parts = line.split("\t");
 675                 var metadata = parts[0].split(" ");
 676                 var path = Path.of(parts[1]);
 677                 if (include.isEmpty() || include.contains(path)) {
 678                     var entry = new FileEntry(hash,
 679                                               FileType.fromOctal(metadata[0]),
 680                                               new Hash(metadata[2]),
 681                                               path);
 682                     entries.add(entry);
 683                 }
 684             }
 685             return entries;
 686         }
 687     }

 689     @Override
 690     public List<FileEntry> files(Hash hash, List<Path> paths) throws IOException {
 691         if (paths.isEmpty()) {
 692             return allFiles(hash, paths);
 693         }
 694 
 695         var entries = new ArrayList<FileEntry>();
 696         var batchSize = 64;
 697         var start = 0;
 698         while (start < paths.size()) {
 699             var end = start + batchSize;
 700             if (end > paths.size()) {
 701                 end = paths.size();
 702             }
 703             entries.addAll(allFiles(hash, paths.subList(start, end)));
 704             start = end;
 705         }
 706         return entries;
 707     }

 865     @Override
 866     public void setPaths(String remote, String pullPath, String pushPath) throws IOException {
 867         var hgrc = Path.of(root().toString(), ".hg", "hgrc");
 868         if (!Files.exists(hgrc)) {
 869             Files.createFile(hgrc);
 870         }
 871 
 872         var lines = Files.readAllLines(hgrc);
 873         var newLines = new ArrayList<String>();
 874 
 875         var isInPathsSection = false;
 876         var hasPathsSection = false;
 877         for (var line : lines) {
 878             var isSectionHeader = line.startsWith("[") && line.endsWith("]");
 879             if (isSectionHeader && !isInPathsSection) {
 880                 isInPathsSection = line.equals("[paths]");
 881                 if (isInPathsSection) {
 882                     newLines.add(line);
 883                     newLines.add(remote + " = " + (pullPath == null ? "" : pullPath));
 884                     newLines.add(remote + "-push = " + (pushPath == null ? "" : pushPath));
 885                     hasPathsSection = true;
 886                     continue;
 887                 }
 888             }
 889 
 890             if (isInPathsSection && line.startsWith(remote)) {
 891                 if (line.startsWith(remote + "-push")) {
 892                     // skip
 893                 } else if (line.startsWith(remote + ":pushurl")) {
 894                     // skip
 895                 } else if (line.startsWith(remote + " ") || line.startsWith(remote + "=")) {
 896                     // skip
 897                 } else {
 898                     newLines.add(line);
 899                 }
 900             } else {
 901                 newLines.add(line);
 902             }
 903         }
 904 
 905         Files.write(hgrc, newLines, StandardOpenOption.WRITE, StandardOpenOption.TRUNCATE_EXISTING);
 906         if (!hasPathsSection) {
 907             var section = List.of("[paths]",
 908                                   remote + " = " + (pullPath == null ? "" : pullPath),
 909                                   remote + "-push = " + (pushPath == null ? "" : pushPath));
 910             Files.write(hgrc, section, StandardOpenOption.WRITE, StandardOpenOption.APPEND);
 911         }
 912     }

 986     @FunctionalInterface
 987     private static interface Operation {
 988         void execute(List<Path> args) throws IOException;
 989     }
 990 
 991     private void batch(Operation op, List<Path> args) throws IOException {
 992         var batchSize = 64;
 993         var start = 0;
 994         while (start < args.size()) {
 995             var end = start + batchSize;
 996             if (end > args.size()) {
 997                 end = args.size();
 998             }
 999             op.execute(args.subList(start, end));
1000             start = end;
1001         }
1002     }
