23 package org.openjdk.skara.email;
24 
25 import org.junit.jupiter.api.Test;
26 
27 import static org.junit.jupiter.api.Assertions.assertEquals;
28 
29 class MimeTextTests {
30     @Test
31     void encode() {
32         assertEquals("=?UTF-8?B?w6XDpMO2?=", MimeText.encode("åäö"));
33     }
34 
35     @Test
36     void decode() {
37         assertEquals("åäö", MimeText.decode("=?utf-8?b?w6XDpMO2?="));
38     }
39 
40     @Test
41     void decodeIsoQ() {
42         assertEquals("Bä", MimeText.decode("=?iso-8859-1?Q?B=E4?="));
43     }
44 }
