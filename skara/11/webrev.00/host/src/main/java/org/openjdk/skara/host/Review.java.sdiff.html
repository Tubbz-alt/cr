29 public class Review {

39         this.hash = hash;
40         this.id = id;
41         this.body = body;
42     }

48     public Verdict verdict() {

50     }

52     public Hash hash() {
53         return hash;
54     }
55 
56     public int id() {
57         return id;
58     }
59 
60     public Optional<String> body() {
61         return Optional.ofNullable(body);
62     }
63 
64     public enum Verdict {
65         NONE,
66         APPROVED,
67         DISAPPROVED
68     }
