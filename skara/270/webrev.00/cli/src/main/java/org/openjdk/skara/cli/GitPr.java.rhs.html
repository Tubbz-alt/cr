 55     private static void await(Process p) throws IOException {
 56         try {
 57             var res = p.waitFor();
 58             if (res != 0) {
 59                 throw new IOException("Unexpected exit code " + res);
 60             }
 61         } catch (InterruptedException e) {
 62             throw new IOException(e);
 63         }
 64     }

 66     private static boolean spawnEditor(ReadOnlyRepository repo, Path file) throws IOException {
 67         String editor = null;
 68         var lines = repo.config("core.editor");
 69         if (lines.size() == 1) {
 70             editor = lines.get(0);
 71         }
 72         if (editor == null) {
 73             editor = System.getenv("GIT_EDITOR");
 74         }
 75         if (editor == null) {
 76             editor = System.getenv("EDITOR");
 77         }
 78         if (editor == null) {
 79             editor = System.getenv("VISUAL");
 80         }
 81         if (editor == null) {
 82             editor = "vi";
 83         }
 84 
 85         var pb = new ProcessBuilder(editor, file.toString());
 86         pb.inheritIO();
 87         var p = pb.start();
 88         try {
 89             return p.waitFor() == 0;
 90         } catch (InterruptedException e) {
 91             throw new IOException(e);
 92         }
 93     }

 95     private static String projectName(URI uri) {
 96         var name = uri.getPath().toString().substring(1);
 97         if (name.endsWith(".git")) {
 98             name = name.substring(0, name.length() - ".git".length());
 99         }
100         return name;
101     }

103     private static HostedRepository getHostedRepositoryFor(URI uri, Forge host) throws IOException {
104         var remoteRepo = host.repository(projectName(uri)).orElseThrow(() ->







105                 new IOException("Could not find repository at: " + uri.toString())
106         );
107         var parentRepo = remoteRepo.parent();
108         var targetRepo = parentRepo.isPresent() ? parentRepo.get() : remoteRepo;
109         return targetRepo;
110     }

112     private static PullRequest getPullRequest(URI uri, Forge host, Argument prId) throws IOException {
113         if (!prId.isPresent()) {
114             exit("error: missing pull request identifier");
115         }
116 
117         var pr = getHostedRepositoryFor(uri, host).pullRequest(prId.asString());
118         if (pr == null) {
119             exit("error: could not fetch PR information");
120         }
121 
122         return pr;
123     }

128     private static void show(String ref, Hash hash, Path dir) throws IOException {
129         var pb = new ProcessBuilder("git", "diff", "--binary",
130                                                    "--patch",
131                                                    "--find-renames=50%",
132                                                    "--find-copies=50%",
133                                                    "--find-copies-harder",
134                                                    "--abbrev",
135                                                    ref + "..." + hash.hex());
136         if (dir != null) {
137             pb.directory(dir.toFile());
138         }
139         pb.inheritIO();
140         await(pb.start());
141     }

155     private static List<String> hgTags() throws IOException, InterruptedException {
156         var pb = new ProcessBuilder("hg", "tags", "--quiet");
157         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
158         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
159         var p = pb.start();
160         var bytes = p.getInputStream().readAllBytes();
161         var exited = p.waitFor(1, TimeUnit.MINUTES);
162         var exitValue = p.exitValue();
163         if (!exited || exitValue != 0) {
164             throw new IOException("'hg tags' exited with value: " + exitValue);
165         }
166 
167         return Arrays.asList(new String(bytes, StandardCharsets.UTF_8).split("\n"));
168     }

170     private static String hgResolve(String ref) throws IOException, InterruptedException {
171         var pb = new ProcessBuilder("hg", "log", "-r", ref, "--template", "{node}");
172         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
173         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
174         var p = pb.start();
175         var bytes = p.getInputStream().readAllBytes();
176         var exited = p.waitFor(1, TimeUnit.MINUTES);
177         var exitValue = p.exitValue();
178         if (!exited || exitValue != 0) {
179             throw new IOException("'hg log' exited with value: " + exitValue);
180         }
181 
182         return new String(bytes, StandardCharsets.UTF_8);
183     }

189     private static Path diff(String ref, Hash hash, Path dir) throws IOException {
190         var patch = Files.createTempFile(hash.hex(), ".patch");
191         var pb = new ProcessBuilder("git", "diff", "--binary",
192                                                    "--patch",
193                                                    "--find-renames=50%",
194                                                    "--find-copies=50%",
195                                                    "--find-copies-harder",
196                                                    "--abbrev",
197                                                    ref + "..." + hash.hex());
198         if (dir != null) {
199             pb.directory(dir.toFile());
200         }
201         pb.redirectOutput(patch.toFile());
202         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
203         await(pb.start());
204         return patch;
205     }

217     public static void main(String[] args) throws IOException, InterruptedException {
218         var flags = List.of(
219             Option.shortcut("u")
220                   .fullname("username")
221                   .describe("NAME")
222                   .helptext("Username on host")
223                   .optional(),
224             Option.shortcut("r")
225                   .fullname("remote")
226                   .describe("NAME")
227                   .helptext("Name of remote, defaults to 'origin'")
228                   .optional(),

231                   .describe("NAME")
232                   .helptext("Name of target branch, defaults to 'master'")
233                   .optional(),
234             Option.shortcut("")

236                   .describe("LIST")
237                   .helptext("Comma separated list of authors")
238                   .optional(),
239             Option.shortcut("")
240                   .fullname("assignees")
241                   .describe("LIST")
242                   .helptext("Comma separated list of assignees")
243                   .optional(),
244             Option.shortcut("")
245                   .fullname("labels")
246                   .describe("LIST")
247                   .helptext("Comma separated list of labels")
248                   .optional(),
249             Option.shortcut("")
250                   .fullname("columns")

252                   .helptext("Comma separated list of columns to show")
253                   .optional(),
254             Switch.shortcut("")
255                   .fullname("no-decoration")
256                   .helptext("Hide any decorations when listing PRs")
257                   .optional(),
258             Switch.shortcut("")
259                   .fullname("no-token")
260                   .helptext("Do not use a personal access token (PAT). Only works for read-only operations.")
261                   .optional(),
262             Switch.shortcut("")
263                   .fullname("mercurial")
264                   .helptext("Force use of Mercurial (hg)")
265                   .optional(),
266             Switch.shortcut("")
267                   .fullname("verbose")
268                   .helptext("Turn on verbose output")
269                   .optional(),
270             Switch.shortcut("")
271                   .fullname("debug")
272                   .helptext("Turn on debugging output")
273                   .optional(),
274             Switch.shortcut("")
275                   .fullname("version")
276                   .helptext("Print the version of this tool")
277                   .optional());
278 
279         var inputs = List.of(
280             Input.position(0)
281                  .describe("list|fetch|show|checkout|apply|integrate|approve|create|close|update")
282                  .singular()
283                  .required(),
284             Input.position(1)
285                  .describe("ID")
286                  .singular()
287                  .optional()
288         );
289 
290         var parser = new ArgumentParser("git-pr", flags, inputs);
291         var arguments = parser.parse(args);
292 
293         if (arguments.contains("version")) {
294             System.out.println("git-pr version: " + Version.fromManifest().orElse("unknown"));
295             System.exit(0);
296         }
297 
298         if (arguments.contains("verbose") || arguments.contains("debug")) {
299             var level = arguments.contains("debug") ? Level.FINER : Level.FINE;
300             Logging.setup(level);
301         }
302 
303         HttpProxy.setup();
304 
305         var isMercurial = arguments.contains("mercurial");
306         var cwd = Path.of("").toAbsolutePath();
307         var repo = Repository.get(cwd).orElseThrow(() -> new IOException("no git repository found at " + cwd.toString()));
308         var remote = arguments.get("remote").orString(isMercurial ? "default" : "origin");
309         var remotePullPath = repo.pullPath(remote);
310         var username = arguments.contains("username") ? arguments.get("username").asString() : null;
311         var token = isMercurial ? System.getenv("HG_TOKEN") :  System.getenv("GIT_TOKEN");
312         var uri = Remote.toWebURI(remotePullPath);
313         var shouldUseToken = !arguments.contains("no-token");
314         var credentials = !shouldUseToken ?
315             null :
316             GitCredentials.fill(uri.getHost(), uri.getPath(), username, token, uri.getScheme());
317         var forgeURI = URI.create(uri.getScheme() + "://" + uri.getHost());
318         var forge = credentials == null ?
319             Forge.from(forgeURI) :
320             Forge.from(forgeURI, new Credential(credentials.username(), credentials.password()));
321         if (forge.isEmpty() || !forge.get().isValid()) {
322             if (!shouldUseToken) {
323                 if (arguments.contains("verbose")) {
324                     System.err.println("");
325                 }
326                 System.err.println("warning: using git-pr with --no-token may result in rate limiting from " + forgeURI);
327                 if (!arguments.contains("verbose")) {
328                     System.err.println("         Re-run git-pr with --verbose to see if you are being rate limited");
329                     System.err.println("");
330                 }
331             }
332             exit("error: failed to connect to host: " + forgeURI);
333         }
334         var host = forge.get();
335 
336         var action = arguments.at(0).asString();
337         if (!shouldUseToken &&
338             !List.of("list", "fetch", "show", "checkout", "apply").contains(action)) {
339             System.err.println("error: --no-token can only be used with read-only operations");
340             System.exit(1);
341         }
342 
343         if (action.equals("create")) {
344             if (isMercurial) {
345                 var currentBookmark = repo.currentBookmark();
346                 if (!currentBookmark.isPresent()) {
347                     System.err.println("error: no bookmark is active, you must be on an active bookmark");
348                     System.err.println("");
349                     System.err.println("To create a bookmark and activate it, run:");
350                     System.err.println("");
351                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
352                     System.err.println("");
353                     System.exit(1);
354                 }
355 
356                 var bookmark = currentBookmark.get();
357                 if (bookmark.equals(new Bookmark("master"))) {
358                     System.err.println("error: you should not create pull requests from the 'master' bookmark");
359                     System.err.println("To create a bookmark and activate it, run:");
360                     System.err.println("");
361                     System.err.println("    hg bookmark NAME-FOR-YOUR-BOOKMARK");
362                     System.err.println("");
363                     System.exit(1);
364                 }
365 
366                 var tags = hgTags();
367                 var upstreams = tags.stream()
368                                     .filter(t -> t.endsWith(bookmark.name()))
369                                     .collect(Collectors.toList());
370                 if (upstreams.isEmpty()) {
371                     System.err.println("error: there is no remote branch for the local bookmark '" + bookmark.name() + "'");
372                     System.err.println("");
373                     System.err.println("To create a remote branch and push the commits for your local branch, run:");
374                     System.err.println("");
375                     System.err.println("    hg push --bookmark " + bookmark.name());
376                     System.err.println("");
377                     System.exit(1);
378                 }
379 
380                 var tagsAndHashes = new HashMap<String, String>();
381                 for (var tag : tags) {
382                     tagsAndHashes.put(tag, hgResolve(tag));
383                 }
384                 var bookmarkHash = hgResolve(bookmark.name());
385                 if (!tagsAndHashes.containsValue(bookmarkHash)) {
386                     System.err.println("error: there are local commits on bookmark '" + bookmark.name() + "' not present in a remote repository");
387                     System.err.println("");
388 
389                     if (upstreams.size() == 1) {
390                         System.err.println("To push the local commits to the remote repository, run:");
391                         System.err.println("");
392                         System.err.println("    hg push --bookmark " + bookmark.name() + " " + upstreams.get(0));
393                         System.err.println("");
394                     } else {
395                         System.err.println("The following paths contains the " + bookmark.name() + " bookmark:");
396                         System.err.println("");
397                         for (var upstream : upstreams) {
398                             System.err.println("- " + upstream.replace("/" + bookmark.name(), ""));
399                         }
400                         System.err.println("");
401                         System.err.println("To push the local commits to a remote repository, run:");
402                         System.err.println("");
403                         System.err.println("    hg push --bookmark " + bookmark.name() + " <PATH>");
404                         System.err.println("");
405                     }
406                     System.exit(1);
407                 }
408 
409                 var targetBranch = arguments.get("branch").orString("master");
410                 var targetHash = hgResolve(targetBranch);
411                 var commits = repo.commits(targetHash + ".." + bookmarkHash + "-" + targetHash).asList();
412                 if (commits.isEmpty()) {
413                     System.err.println("error: no difference between bookmarks " + targetBranch + " and " + bookmark.name());
414                     System.err.println("       Cannot create an empty pull request, have you committed?");
415                     System.exit(1);
416                 }
417 
418                 var diff = repo.diff(repo.head());
419                 if (!diff.patches().isEmpty()) {
420                     System.err.println("error: there are uncommitted changes in your working tree:");
421                     System.err.println("");
422                     for (var patch : diff.patches()) {
423                         var path = patch.target().path().isPresent() ?
424                             patch.target().path().get() : patch.source().path().get();
425                         System.err.println("    " + patch.status().toString() + " " + path.toString());
426                     }
427                     System.err.println("");
428                     System.err.println("If these changes are meant to be part of the pull request, run:");
429                     System.err.println("");
430                     System.err.println("    hg commit --amend");
431                     System.err.println("    hg git-cleanup");
432                     System.err.println("    hg push --bookmark " + bookmark.name() + " <PATH>");
433                     System.err.println("    hg gimport");
434                     System.err.println("");
435                     System.err.println("If these changes are *not* meant to be part of the pull request, run:");
436                     System.err.println("");
437                     System.err.println("    hg shelve");
438                     System.err.println("");
439                     System.err.println("(You can later restore the changes by running: hg unshelve)");
440                     System.exit(1);
441                 }
442 
443                 var remoteRepo = host.repository(projectName(uri)).orElseThrow(() ->
444                         new IOException("Could not find repository at " + uri.toString())
445                 );
446                 if (token == null) {
447                     GitCredentials.approve(credentials);
448                 }
449                 var parentRepo = remoteRepo.parent().orElseThrow(() ->
450                         new IOException("error: remote repository " + remotePullPath + " is not a fork of any repository"));
451 
452                 var file = Files.createTempFile("PULL_REQUEST_", ".txt");
453                 if (commits.size() == 1) {
454                     var commit = commits.get(0);
455                     var message = CommitMessageParsers.v1.parse(commit.message());
456                     Files.writeString(file, message.title() + "\n");
457                     if (!message.summaries().isEmpty()) {
458                         Files.write(file, message.summaries(), StandardOpenOption.APPEND);
459                     }
460                     if (!message.additional().isEmpty()) {
461                         Files.write(file, message.additional(), StandardOpenOption.APPEND);
462                     }
463                 } else {
464                     Files.write(file, List.of(""));
465                 }
466                 Files.write(file, List.of(
467                     "# Please enter the pull request message for your changes. Lines starting",
468                     "# with '#' will be ignored, and an empty message aborts the pull request.",
469                     "# The first line will be considered the subject, use a blank line to separate",
470                     "# the subject from the body.",
471                     "#",
472                     "# Commits to be included from branch '" + bookmark.name() + "'"
473                     ),
474                     StandardOpenOption.APPEND
475                 );
476                 for (var commit : commits) {
477                     var desc = commit.hash().abbreviate() + ": " + commit.message().get(0);
478                     Files.writeString(file, "# - " + desc + "\n", StandardOpenOption.APPEND);
479                 }
480                 Files.writeString(file, "#\n", StandardOpenOption.APPEND);
481                 Files.writeString(file, "# Target repository: " + remotePullPath + "\n", StandardOpenOption.APPEND);
482                 Files.writeString(file, "# Target branch: " + targetBranch + "\n", StandardOpenOption.APPEND);
483                 var success = spawnEditor(repo, file);
484                 if (!success) {
485                     System.err.println("error: editor exited with non-zero status code, aborting");
486                     System.exit(1);
487                 }
488                 var lines = Files.readAllLines(file)
489                                  .stream()
490                                  .filter(l -> !l.startsWith("#"))
491                                  .collect(Collectors.toList());
492                 var isEmpty = lines.stream().allMatch(String::isEmpty);
493                 if (isEmpty) {
494                     System.err.println("error: no message present, aborting");
495                     System.exit(1);
496                 }
497 
498                 var title = lines.get(0);
499                 List<String> body = null;
500                 if (lines.size() > 1) {
501                     body = lines.subList(1, lines.size())
502                                 .stream()
503                                 .dropWhile(String::isEmpty)
504                                 .collect(Collectors.toList());
505                 } else {
506                     body = Collections.emptyList();
507                 }
508 
509                 var pr = remoteRepo.createPullRequest(parentRepo, targetBranch, bookmark.name(), title, body);
510                 if (arguments.contains("assignees")) {
511                     var usernames = Arrays.asList(arguments.get("assignees").asString().split(","));
512                     var assignees = usernames.stream()
513                                              .map(u -> host.user(u))
514                                              .collect(Collectors.toList());
515                     pr.setAssignees(assignees);
516                 }
517                 System.out.println(pr.webUrl().toString());
518                 Files.deleteIfExists(file);
519 
520                 System.exit(0);
521             }
522             var currentBranch = repo.currentBranch();
523             if (currentBranch.equals(repo.defaultBranch())) {
524                 System.err.println("error: you should not create pull requests from the 'master' branch");
525                 System.err.println("");
526                 System.err.println("To create a local branch for your changes and restore the 'master' branch, run:");
527                 System.err.println("");
528                 System.err.println("    git checkout -b NAME-FOR-YOUR-LOCAL-BRANCH");
529                 System.err.println("    git branch --force master origin/master");
530                 System.err.println("");
531                 System.exit(1);
532             }
533 
534             var upstream = repo.upstreamFor(currentBranch);
535             if (upstream.isEmpty()) {
536                 System.err.println("error: there is no remote branch for the local branch '" + currentBranch.name() + "'");
537                 System.err.println("");
538                 System.err.println("A remote branch must be present at " + remotePullPath + " to create a pull request");
539                 System.err.println("To create a remote branch and push the commits for your local branch, run:");
540                 System.err.println("");
541                 System.err.println("    git push --set-upstream " + remote + " " + currentBranch.name());
542                 System.err.println("");
543                 System.err.println("If you created the remote branch from another client, you must update this repository.");
544                 System.err.println("To update remote information for this repository, run:");
545                 System.err.println("");
546                 System.err.println("    git fetch " + remote);
547                 System.err.println("    git branch --set-upstream " + currentBranch + " " + remote + "/" + currentBranch);
548                 System.err.println("");
549                 System.exit(1);
550             }
551 
552             var upstreamRefName = upstream.get().substring(remote.length() + 1);
553             repo.fetch(uri, upstreamRefName);
554             var branchCommits = repo.commits(upstream.get() + ".." + currentBranch.name()).asList();
555             if (!branchCommits.isEmpty()) {
556                 System.err.println("error: there are local commits on branch '" + currentBranch.name() + "' not present in the remote repository " + remotePullPath);
557                 System.err.println("");
558                 System.err.println("All commits must be present in the remote repository to be part of the pull request");
559                 System.err.println("The following commits are not present in the remote repository:");
560                 System.err.println("");
561                 for (var commit : branchCommits) {
562                     System.err.println("- " + commit.hash().abbreviate() + ": " + commit.message().get(0));
563                 }
564                 System.err.println("");
565                 System.err.println("To push the above local commits to the remote repository, run:");
566                 System.err.println("");
567                 System.err.println("    git push " + remote + " " + currentBranch.name());
568                 System.err.println("");
569                 System.exit(1);
570             }
571 
572             var targetBranch = arguments.get("branch").orString("master");
573             var commits = repo.commits(targetBranch + ".." + currentBranch.name()).asList();
574             if (commits.isEmpty()) {
575                 System.err.println("error: no difference between branches " + targetBranch + " and " + currentBranch.name());
576                 System.err.println("       Cannot create an empty pull request, have you committed?");
577                 System.exit(1);
578             }
579 
580             var diff = repo.diff(repo.head());
581             if (!diff.patches().isEmpty()) {
582                 System.err.println("error: there are uncommitted changes in your working tree:");
583                 System.err.println("");
584                 for (var patch : diff.patches()) {
585                     var path = patch.target().path().isPresent() ?
586                         patch.target().path().get() : patch.source().path().get();
587                     System.err.println("    " + patch.status().toString() + " " + path.toString());
588                 }
589                 System.err.println("");
590                 System.err.println("If these changes are meant to be part of the pull request, run:");
591                 System.err.println("");
592                 System.err.println("    git commit -am 'Forgot to add some changes'");
593                 System.err.println("");
594                 System.err.println("If these changes are *not* meant to be part of the pull request, run:");
595                 System.err.println("");
596                 System.err.println("    git stash");
597                 System.err.println("");
598                 System.err.println("(You can later restore the changes by running: git stash pop)");
599                 System.exit(1);
600             }
601 
602             var remoteRepo = host.repository(projectName(uri)).orElseThrow(() ->
603                     new IOException("Could not find repository at " + uri.toString())
604             );
605             if (token == null) {
606                 GitCredentials.approve(credentials);
607             }
608             var parentRepo = remoteRepo.parent().orElseThrow(() ->
609                     new IOException("error: remote repository " + remotePullPath + " is not a fork of any repository"));
610 
611             var file = Files.createTempFile("PULL_REQUEST_", ".txt");
612             if (commits.size() == 1) {
613                 var commit = commits.get(0);
614                 var message = CommitMessageParsers.v1.parse(commit.message());
615                 Files.writeString(file, message.title() + "\n");
616                 if (!message.summaries().isEmpty()) {
617                     Files.write(file, message.summaries(), StandardOpenOption.APPEND);
618                 }
619                 if (!message.additional().isEmpty()) {
620                     Files.write(file, message.additional(), StandardOpenOption.APPEND);
621                 }
622             } else {
623                 Files.write(file, List.of(""));
624             }
625             Files.write(file, List.of(
626                 "# Please enter the pull request message for your changes. Lines starting",
627                 "# with '#' will be ignored, and an empty message aborts the pull request.",
628                 "# The first line will be considered the subject, use a blank line to separate",
629                 "# the subject from the body.",
630                 "#",
631                 "# Commits to be included from branch '" + currentBranch.name() + "'"
632                 ),
633                 StandardOpenOption.APPEND
634             );
635             for (var commit : commits) {
636                 var desc = commit.hash().abbreviate() + ": " + commit.message().get(0);
637                 Files.writeString(file, "# - " + desc + "\n", StandardOpenOption.APPEND);
638             }
639             Files.writeString(file, "#\n", StandardOpenOption.APPEND);
640             Files.writeString(file, "# Target repository: " + remotePullPath + "\n", StandardOpenOption.APPEND);
641             Files.writeString(file, "# Target branch: " + targetBranch + "\n", StandardOpenOption.APPEND);
642             var success = spawnEditor(repo, file);
643             if (!success) {
644                 System.err.println("error: editor exited with non-zero status code, aborting");
645                 System.exit(1);
646             }
647             var lines = Files.readAllLines(file)
648                              .stream()
649                              .filter(l -> !l.startsWith("#"))
650                              .collect(Collectors.toList());
651             var isEmpty = lines.stream().allMatch(String::isEmpty);
652             if (isEmpty) {
653                 System.err.println("error: no message present, aborting");
654                 System.exit(1);
655             }
656 
657             var title = lines.get(0);
658             List<String> body = null;
659             if (lines.size() > 1) {
660                 body = lines.subList(1, lines.size())
661                             .stream()
662                             .dropWhile(String::isEmpty)
663                             .collect(Collectors.toList());
664             } else {
665                 body = Collections.emptyList();
666             }
667 
668             var pr = remoteRepo.createPullRequest(parentRepo, targetBranch, currentBranch.name(), title, body);
669             if (arguments.contains("assignees")) {
670                 var usernames = Arrays.asList(arguments.get("assignees").asString().split(","));
671                 var assignees = usernames.stream()
672                                          .map(u -> host.user(u))
673                                          .collect(Collectors.toList());
674                 pr.setAssignees(assignees);
675             }
676             System.out.println(pr.webUrl().toString());
677             Files.deleteIfExists(file);
678         } else if (action.equals("integrate") || action.equals("approve")) {
679             var pr = getPullRequest(uri, host, arguments.at(1));
680 
681             if (action.equals("integrate")) {
682                 pr.addComment("/integrate");
683             } else if (action.equals("approve")) {
684                 pr.addReview(Review.Verdict.APPROVED, "Looks good!");
685             } else {
686                 throw new IllegalStateException("unexpected action: " + action);
687             }
688         } else if (action.equals("list")) {
689             var remoteRepo = getHostedRepositoryFor(uri, host);
690             var prs = remoteRepo.pullRequests();
691 
692             var ids = new ArrayList<String>();
693             var titles = new ArrayList<String>();
694             var authors = new ArrayList<String>();
695             var assignees = new ArrayList<String>();
696             var labels = new ArrayList<String>();
697 
698             var filterAuthors = arguments.contains("authors") ?
699                 new HashSet<>(Arrays.asList(arguments.get("authors").asString().split(","))) :
700                 Set.of();
701             var filterAssignees = arguments.contains("assignees") ?
702                 Arrays.asList(arguments.get("assignees").asString().split(",")) :
703                 Set.of();
704             var filterLabels = arguments.contains("labels") ?
705                 Arrays.asList(arguments.get("labels").asString().split(",")) :
706                 Set.of();
707 
708             var defaultColumns = List.of("id", "title", "authors", "assignees", "labels");
709             var columnValues = Map.of(defaultColumns.get(0), ids,
710                                       defaultColumns.get(1), titles,
711                                       defaultColumns.get(2), authors,
712                                       defaultColumns.get(3), assignees,
713                                       defaultColumns.get(4), labels);
714             var columns = arguments.contains("columns") ?
715                 Arrays.asList(arguments.get("columns").asString().split(",")) :
716                 defaultColumns;
717             if (columns != defaultColumns) {
718                 for (var column : columns) {
719                     if (!defaultColumns.contains(column)) {
720                         System.err.println("error: unknown column: " + column);
721                         System.err.println("       available columns are: " + String.join(",", defaultColumns));
722                         System.exit(1);
723                     }
724                 }
725             }
726 
727             for (var pr : remoteRepo.pullRequests()) {
728                 var prAuthor = pr.author().userName();
729                 if (!filterAuthors.isEmpty() && !filterAuthors.contains(prAuthor)) {
730                     continue;
731                 }
732 
733                 var prAssignees = pr.assignees().stream()
734                                     .map(HostUser::userName)
735                                     .collect(Collectors.toSet());
736                 if (!filterAssignees.isEmpty() && !filterAssignees.stream().anyMatch(prAssignees::contains)) {
737                     continue;
738                 }
739 
740                 var prLabels = new HashSet<>(pr.labels());
741                 if (!filterLabels.isEmpty() && !filterLabels.stream().anyMatch(prLabels::contains)) {
742                     continue;
743                 }
744 
745                 ids.add(pr.id());
746                 titles.add(pr.title());
747                 authors.add(prAuthor);
748                 assignees.add(String.join(",", prAssignees));
749                 labels.add(String.join(",", prLabels));
750             }
751 
752 
753             String fmt = "";
754             for (var column : columns.subList(0, columns.size() - 1)) {
755                 var values = columnValues.get(column);
756                 var n = Math.max(column.length(), longest(values));
757                 fmt += "%-" + n + "s\t";
758             }
759             fmt += "%s\n";
760 
761             if (!ids.isEmpty() && !arguments.contains("no-decoration")) {
762                 var upperCase = columns.stream()
763                                        .map(String::toUpperCase)
764                                        .collect(Collectors.toList());
765                 System.out.format(fmt, (Object[]) upperCase.toArray(new String[0]));
766             }
767             for (var i = 0; i < ids.size(); i++) {
768                 final int n = i;
769                 var row = columns.stream()
770                                  .map(columnValues::get)
771                                  .map(values -> values.get(n))
772                                  .collect(Collectors.toList());
773                 System.out.format(fmt, (Object[]) row.toArray(new String[0]));
774             }
775         } else if (action.equals("fetch") || action.equals("checkout") || action.equals("show") || action.equals("apply")) {
776             var prId = arguments.at(1);
777             if (!prId.isPresent()) {
778                 exit("error: missing pull request identifier");
779             }
780 
781             var remoteRepo = getHostedRepositoryFor(uri, host);
782             var pr = remoteRepo.pullRequest(prId.asString());
783             var repoUrl = remoteRepo.webUrl();
784             var prHeadRef = pr.sourceRef();
785             var isHgGit = isMercurial && Repository.exists(repo.root().resolve(".hg").resolve("git"));
786             if (isHgGit) {
787                 var hgGitRepo = Repository.get(repo.root().resolve(".hg").resolve("git")).get();
788                 var hgGitFetchHead = hgGitRepo.fetch(repoUrl, prHeadRef);
789 
790                 if (action.equals("show") || action.equals("apply")) {
791                     var target = hgGitRepo.fetch(repoUrl, pr.targetRef());
792                     var hgGitMergeBase = hgGitRepo.mergeBase(target, hgGitFetchHead);
793 
794                     if (action.equals("show")) {
795                         show(hgGitMergeBase.hex(), hgGitFetchHead, hgGitRepo.root());
796                     } else {
797                         var patch = diff(hgGitMergeBase.hex(), hgGitFetchHead, hgGitRepo.root());
798                         hgImport(patch);
799                         Files.delete(patch);
800                     }
801                 } else if (action.equals("fetch") || action.equals("checkout")) {
802                     var hgGitRef = prHeadRef.endsWith("/head") ? prHeadRef.replace("/head", "") : prHeadRef;
803                     var hgGitBranches = hgGitRepo.branches();
804                     if (hgGitBranches.contains(new Branch(hgGitRef))) {
805                         hgGitRepo.delete(new Branch(hgGitRef));
806                     }
807                     hgGitRepo.branch(hgGitFetchHead, hgGitRef);
808                     gimport();
809                     var hgFetchHead = repo.resolve(hgGitRef).get();
810 
811                     if (action.equals("fetch") && arguments.contains("branch")) {
812                         repo.branch(hgFetchHead, arguments.get("branch").asString());
813                     } else if (action.equals("checkout")) {
814                         repo.checkout(hgFetchHead);
815                         if (arguments.contains("branch")) {
816                             repo.branch(hgFetchHead, arguments.get("branch").asString());
817                         }
818                     }
819                 } else {
820                     exit("Unexpected action: " + action);
821                 }
822 
823                 return;
824             }
825 
826             var fetchHead = repo.fetch(repoUrl, pr.sourceRef());
827             if (action.equals("fetch")) {
828                 if (arguments.contains("branch")) {
829                     var branchName = arguments.get("branch").asString();
830                     repo.branch(fetchHead, branchName);
831                 } else {
832                     System.out.println(fetchHead.hex());
833                 }
834             } else if (action.equals("checkout")) {
835                 if (arguments.contains("branch")) {
836                     var branchName = arguments.get("branch").asString();
837                     var branch = repo.branch(fetchHead, branchName);
838                     repo.checkout(branch, false);
839                 } else {
840                     repo.checkout(fetchHead, false);
841                 }
842             } else if (action.equals("show")) {
843                 show(pr.targetRef(), fetchHead);
844             } else if (action.equals("apply")) {
845                 var patch = diff(pr.targetRef(), fetchHead);
846                 apply(patch);
847                 Files.deleteIfExists(patch);
848             }
849         } else if (action.equals("close")) {
850             var prId = arguments.at(1);
851             if (!prId.isPresent()) {
852                 exit("error: missing pull request identifier");
853             }
854 
855             var remoteRepo = getHostedRepositoryFor(uri, host);
856             var pr = remoteRepo.pullRequest(prId.asString());
857             pr.setState(PullRequest.State.CLOSED);
858         } else if (action.equals("update")) {
859             var prId = arguments.at(1);
860             if (!prId.isPresent()) {
861                 exit("error: missing pull request identifier");
862             }
863 
864             var remoteRepo = getHostedRepositoryFor(uri, host);
865             var pr = remoteRepo.pullRequest(prId.asString());
866             if (arguments.contains("assignees")) {
867                 var usernames = Arrays.asList(arguments.get("assignees").asString().split(","));
868                 var assignees = usernames.stream()
869                     .map(u -> host.user(u))
870                     .collect(Collectors.toList());
871                 pr.setAssignees(assignees);
872             }
873         } else {
874             exit("error: unexpected action: " + action);
875         }
876     }
