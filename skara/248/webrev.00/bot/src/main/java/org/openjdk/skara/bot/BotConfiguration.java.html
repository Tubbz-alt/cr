23 package org.openjdk.skara.bot;
24 
25 import org.openjdk.skara.ci.ContinuousIntegration;
26 import org.openjdk.skara.forge.HostedRepository;
27 import org.openjdk.skara.issuetracker.IssueProject;
28 import org.openjdk.skara.json.JSONObject;
29 
30 import java.nio.file.Path;
31 
32 public interface BotConfiguration {
33     /**
34      * Folder that WorkItems may use to store permanent data.
35      * @return
36      */
37     Path storageFolder();
38 
39     /**
40      * Configuration-specific name mapped to a HostedRepository.
41      * @param name
42      * @return
43      */
44     HostedRepository repository(String name);
45 
46     /**
47      * Configuration-specific name mapped to a IssueProject.
48      * @param name
49      * @return
50      */
51     IssueProject issueProject(String name);
52 
53     /**
54      * Configuration-specific name mapped to a ContinuousIntegration.
55      * @param name
56      * @return
57      */
58     ContinuousIntegration continuousIntegration(String name);
59 
60     /**
61      * Retrieves the ref name that optionally follows the configuration-specific repository name.
62      * If not configured, returns the name of the VCS default branch.
63      * @param name
64      * @return
65      */
66     String repositoryRef(String name);
67 
68     /**
69      * Extracts a reasonable short repository name from a full repository specification, e.g. host/org/repo:ref -> repo
70      * @param name
71      * @return
72      */
73     String repositoryName(String name);
74 
75     /**
76      * Additional bot-specific configuration.
77      * @return
78      */
79     JSONObject specific();
80 }
