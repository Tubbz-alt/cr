23 package org.openjdk.skara.bot;
24 

25 import org.openjdk.skara.forge.HostedRepository;
26 import org.openjdk.skara.issuetracker.IssueProject;
27 import org.openjdk.skara.json.JSONObject;
28 
29 import java.nio.file.Path;
30 
31 public interface BotConfiguration {
32     /**
33      * Folder that WorkItems may use to store permanent data.
34      * @return
35      */
36     Path storageFolder();
37 
38     /**
39      * Configuration-specific name mapped to a HostedRepository.
40      * @param name
41      * @return
42      */
43     HostedRepository repository(String name);
44 
45     /**
46      * Configuration-specific name mapped to a IssueProject.
47      * @param name
48      * @return
49      */
50     IssueProject issueProject(String name);
51 







52     /**
53      * Retrieves the ref name that optionally follows the configuration-specific repository name.
54      * If not configured, returns the name of the VCS default branch.
55      * @param name
56      * @return
57      */
58     String repositoryRef(String name);
59 
60     /**
61      * Extracts a reasonable short repository name from a full repository specification, e.g. host/org/repo:ref -> repo
62      * @param name
63      * @return
64      */
65     String repositoryName(String name);
66 
67     /**
68      * Additional bot-specific configuration.
69      * @return
70      */
71     JSONObject specific();
72 }
