23 package org.openjdk.skara.bots.tester;
24 
25 import org.openjdk.skara.ci.*;
26 
27 import java.util.List;
28 import java.util.ArrayList;
29 
30 class InMemoryJob implements Job {
31     String id = "";
32     List<Build> builds = new ArrayList<>();
33     List<Test> tests = new ArrayList<>();
34     Job.Status status;
35     Job.Result result;
36     Job.State state;
37 
38     @Override
39     public String id() {
40         return id;
41     }
42 
43     @Override
44     public List<Build> builds() {
45         return builds;
46     }
47 
48     @Override
49     public List<Test> tests() {
50         return tests;
51     }
52 
53     @Override
54     public Job.Status status() {
55         return status;
56     }
57 
58     @Override
59     public Job.Result result() {
60         return result;
61     }
62 
63     @Override
64     public Job.State state() {
65         return state;
66     }
67 }
