 23 package org.openjdk.skara.ci;
 24 
 25 import java.util.Objects;
 26 
 27 public class Build {
 28     public static enum OperatingSystem {
 29         WINDOWS,
 30         MACOS,
 31         LINUX,
 32         SOLARIS,
 33         AIX,
 34         FREEBSD,
 35         OPENBSD,
 36         NETBSD,
 37         HPUX,
 38         HAIKU
 39     }
 40 
 41     public static enum CPU {
 42         X86,
 43         X64,
 44         SPARCV9,
 45         AARCH64,
 46         AARCH32,
 47         PPCLE32,
 48         PPCLE64
 49     }
 50 
 51     public static enum DebugLevel {
 52         RELEASE,
 53         FASTDEBUG,
 54         SLOWDEBUG
 55     }
 56 
 57     private final OperatingSystem os;
 58     private final CPU cpu;
 59     private final DebugLevel debugLevel;
 60 
 61     public Build(OperatingSystem os, CPU cpu, DebugLevel debugLevel) {
 62         this.os = os;
 63         this.cpu = cpu;
 64         this.debugLevel = debugLevel;
 65     }
 66 
 67     public OperatingSystem os() {
 68         return os;
 69     }
 70 
 71     public CPU cpu() {
 72         return cpu;
 73     }
 74 
 75     public DebugLevel debugLevel() {
 76         return debugLevel;
 77     }
 78 
 79     @Override
 80     public String toString() {
 81         return os.toString().toLowerCase() + "-" +
 82                cpu.toString().toLowerCase() + "-" +
 83                debugLevel.toString().toLowerCase();
 84     }
 85 
 86     @Override
 87     public int hashCode() {
 88         return Objects.hash(os, cpu, debugLevel);
 89     }
 90 
 91     @Override
 92     public boolean equals(Object other) {
 93         if (other == this) {
 94             return true;
 95         }
 96 
 97         if (!(other instanceof Build)) {
 98             return false;
 99         }
100 
101         var o = (Build) other;
102         return Objects.equals(os, o.os) &&
103                Objects.equals(cpu, o.cpu) &&
104                Objects.equals(debugLevel, o.debugLevel);
105     }
106 }
