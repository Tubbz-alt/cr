23 package org.openjdk.skara.host;
24 
25 import org.openjdk.skara.vcs.Hash;
26 
27 import java.time.ZonedDateTime;
28 import java.util.Optional;
29 
30 public class ReviewComment extends Comment {
31     private final ReviewComment parent;
32     private final String threadId;
33     private final Hash hash;
34     private final String path;
35     private final int line;
36 
37     public ReviewComment(ReviewComment parent, String threadId, Hash hash, String path, int line, String id, String body, HostUser author, ZonedDateTime createdAt, ZonedDateTime updatedAt) {
38         super(id, body, author, createdAt, updatedAt);
39 
40         this.parent = parent;
41         this.threadId = threadId;
42         this.hash = hash;
43         this.path = path;
44         this.line = line;
45     }
46 
47     public Optional<ReviewComment> parent() {
48         return Optional.ofNullable(parent);
49     }
50 
51     public Hash hash() {
52         return hash;
53     }
54 
55     public String path() {
56         return path;
57     }
58 
59     public int line() {
60         return line;
61     }
62 
63     public String threadId() {
64         return threadId;
65     }
66 }
