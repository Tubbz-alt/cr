23 package org.openjdk.skara.host;
24 
25 import java.time.ZonedDateTime;
26 
27 public class Comment {
28 
29     private final String id;
30     private final String body;
31     private final HostUserDetails author;
32     private final ZonedDateTime createdAt;
33     private final ZonedDateTime updatedAt;
34 
35     public Comment(String id, String body, HostUserDetails author, ZonedDateTime createdAt, ZonedDateTime updatedAt) {
36         this.id = id;
37         this.body = body;
38         this.author = author;
39         this.createdAt = createdAt;
40         this.updatedAt = updatedAt;
41     }
42 
43     public String id() {
44         return id;
45     }
46 
47     public String body() {
48         return body;
49     }
50 
51     public HostUserDetails author() {
52         return author;
53     }
54 
55     public ZonedDateTime createdAt() {
56         return createdAt;
57     }
58 
59     public ZonedDateTime updatedAt() {
60         return updatedAt;
61     }
62 }
