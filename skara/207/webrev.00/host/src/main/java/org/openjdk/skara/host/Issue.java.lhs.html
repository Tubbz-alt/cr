 23 package org.openjdk.skara.host;
 24 
 25 import java.net.URI;
 26 import java.time.ZonedDateTime;
 27 import java.util.List;
 28 
 29 public interface Issue {
 30     /**
 31      * Project containing the issue.
 32      * @return
 33      */
 34     IssueProject project();
 35 
 36     /**
 37      * The repository-specific identifier.
 38      * @return
 39      */
 40     String getId();
 41 
 42     /**
 43      * The host-specific author name.
 44      * @return
 45      */
 46     HostUserDetails getAuthor();
 47 
 48     /**
 49      * Title of the request.
 50      * @return
 51      */
 52     String getTitle();
 53 
 54     /**
 55      * Update the title of the request.
 56      * @param title
 57      */
 58     void setTitle(String title);
 59 
 60     /**
 61      * The main body of the request.
 62      * @return
 63      */
 64     String getBody();
 65 
 66     /**
 67      * Update the main body of the request.
 68      * @param body
 69      */
 70     void setBody(String body);
 71 
 72     /**
 73      * All comments on the issue, in ascending creation time order.
 74      * @return
 75      */
 76     List<Comment> getComments();
 77 
 78     /**
 79      * Posts a new comment.
 80      * @param body
 81      */
 82     Comment addComment(String body);
 83 
 84     /**
 85      * Updates an existing comment.
 86      * @param id
 87      * @param body
 88      */
 89     Comment updateComment(String id, String body);
 90 
 91     /**
 92      * When the request was created.
 93      * @return
 94      */
 95     ZonedDateTime getCreated();
 96 
 97     /**
 98      * When the request was last updated.
 99      * @return
100      */
101     ZonedDateTime getUpdated();
102 
103     enum State {
104         OPEN,
105         CLOSED
106     }
107 
108     /**
109      * Set the state.
110      * @param state Desired state
111      */
112     void setState(State state);
113 
114     /**
115      * Adds the given label.
116      * @param label
117      */
118     void addLabel(String label);
119 
120     /**
121      * Removes the given label.
122      * @param label
123      */
124     void removeLabel(String label);
125 
126     /**
127      * Retrieves all the currently set labels.
128      * @return
129      */
130     List<String> getLabels();
131 
132     /**
133      * Returns a link that will lead to the issue.
134      */
135     URI getWebUrl();
136 
137     /**
138      * Returns all usernames assigned to the issue.
139      */
140     List<HostUserDetails> getAssignees();
141 
142     /**
143      * Update the list of assignees.
144      * @param assignees
145      */
146     void setAssignees(List<HostUserDetails> assignees);
147 }
