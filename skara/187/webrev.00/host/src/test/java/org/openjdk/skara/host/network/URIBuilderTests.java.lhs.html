23 package org.openjdk.skara.host.network;
24 
25 import org.junit.jupiter.api.Test;
26 
27 import static org.junit.jupiter.api.Assertions.*;
28 
29 public class URIBuilderTests {
30 
31     private final String validHost = "http://www.test.com";
32 
33     @Test
34     void setPathSimple() {
35         var a = URIBuilder.base(validHost).setPath("/a").build();
36         var b = URIBuilder.base(validHost).setPath("/b").build();
37         var aToB = URIBuilder.base(a).setPath("/b").build();
38 
39         assertEquals("www.test.com", a.getHost());
40         assertEquals("/a", a.getPath());
41         assertEquals("/b", b.getPath());
42         assertEquals("/b", aToB.getPath());
43     }
44 
45     @Test
46     void appendPathSimple() {
47         var a = URIBuilder.base(validHost).setPath("/a").build();
48         var aPlusB = URIBuilder.base(a).appendPath("/b").build();
49 
50         var x = new URIBuilderException();
51 
52         assertEquals("/a", a.getPath());
53         assertEquals("/a/b", aPlusB.getPath());
54     }
55 
56     @Test
57     void invalidBase() {
58         assertThrows(URIBuilderException.class,
59                 () -> URIBuilder.base("x:\\y").build());
60     }
61 
62     @Test
63     void invalidSetPath() {
64         assertThrows(URIBuilderException.class,
65                 () -> URIBuilder.base(validHost).setPath("\\c").build());
66     }
67 
68     @Test
69     void invalidAppendPath() {
70         assertThrows(URIBuilderException.class,
71                 () -> URIBuilder.base(validHost).appendPath("\\c").build());
72     }






73 }
