 23 package org.openjdk.skara.issuetracker;
 24 
 25 import java.net.URI;
 26 import java.util.*;
 27 
 28 public class Link {
 29     private final URI uri;
 30     private final String title;
 31     private final String relationship;
 32     private final String summary;
 33     private final URI iconUrl;
 34     private final String iconTitle;
 35     private final URI statusIconUrl;
 36     private final String statusIconTitle;
 37     private final boolean resolved;
 38 
 39     Link(URI uri, String title, String relationship, String summary, URI iconUrl, String iconTitle, URI statusIconUrl, String statusIconTitle, boolean resolved) {
 40         this.uri = uri;
 41         this.title = title;
 42         this.relationship = relationship;
 43         this.summary = summary;
 44         this.iconUrl = iconUrl;
 45         this.iconTitle = iconTitle;
 46         this.statusIconUrl = statusIconUrl;
 47         this.statusIconTitle = statusIconTitle;
 48         this.resolved = resolved;
 49     }
 50 
 51     public static LinkBuilder create(URI uri, String title) {
 52         return new LinkBuilder(uri, title);
 53     }
 54 
 55     public URI uri() {
 56         return uri;
 57     }
 58 
 59     public String title() {
 60         return title;
 61     }
 62 
 63     public Optional<String> relationship() {
 64         return Optional.ofNullable(relationship);
 65     }
 66 
 67     public Optional<String> summary() {
 68         return Optional.ofNullable(summary);
 69     }
 70 
 71     public Optional<URI> iconUrl() {
 72         return Optional.ofNullable(iconUrl);
 73     }
 74 
 75     public Optional<String> iconTitle() {
 76         return Optional.ofNullable(iconTitle);
 77     }
 78 
 79     public Optional<URI> statusIconUrl() {
 80         return Optional.ofNullable(statusIconUrl);
 81     }
 82 
 83     public Optional<String> statusIconTitle() {
 84         return Optional.ofNullable(statusIconTitle);
 85     }
 86 
 87     public boolean resolved() {
 88         return resolved;
 89     }
 90 
 91     @Override
 92     public boolean equals(Object o) {
 93         if (this == o) {
 94             return true;
 95         }
 96         if (o == null || getClass() != o.getClass()) {
 97             return false;
 98         }
 99         Link link = (Link) o;
100         return resolved == link.resolved &&
101                 uri.equals(link.uri) &&
102                 title.equals(link.title) &&
103                 Objects.equals(relationship, link.relationship) &&
104                 Objects.equals(summary, link.summary) &&
105                 Objects.equals(iconUrl, link.iconUrl) &&
106                 Objects.equals(iconTitle, link.iconTitle) &&
107                 Objects.equals(statusIconUrl, link.statusIconUrl) &&
108                 Objects.equals(statusIconTitle, link.statusIconTitle);
109     }
110 
111     @Override
112     public int hashCode() {
113         return Objects.hash(uri, title, relationship, summary, iconUrl, iconTitle, statusIconUrl, statusIconTitle, resolved);
114     }
115 
116     @Override
117     public String toString() {
118         return "Link{" +
119                 "uri=" + uri +
120                 ", title='" + title + '\'' +
121                 ", relationship='" + relationship + '\'' +
122                 ", summary='" + summary + '\'' +
123                 ", iconUrl=" + iconUrl +
124                 ", iconTitle='" + iconTitle + '\'' +
125                 ", statusIconUrl=" + statusIconUrl +
126                 ", statusIconTitle='" + statusIconTitle + '\'' +
127                 ", resolved=" + resolved +
128                 '}';
129     }
130 }
