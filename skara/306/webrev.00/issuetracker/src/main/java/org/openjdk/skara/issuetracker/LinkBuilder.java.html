23 package org.openjdk.skara.issuetracker;
24 
25 import java.net.URI;
26 
27 public class LinkBuilder {
28     private final URI uri;
29     private final String title;
30 
31     private String relationship;
32     private String summary;
33     private URI iconUrl;
34     private String iconTitle;
35     private URI statusIconUrl;
36     private String statusIconTitle;
37     private boolean resolved;
38 
39     LinkBuilder(URI uri, String title) {
40         this.uri = uri;
41         this.title = title;
42     }
43 
44     public LinkBuilder relationship(String relationship) {
45         this.relationship = relationship;
46         return this;
47     }
48 
49     public LinkBuilder summary(String summary) {
50         this.summary = summary;
51         return this;
52     }
53 
54     public LinkBuilder iconUrl(URI iconUrl) {
55         this.iconUrl = iconUrl;
56         return this;
57     }
58 
59     public LinkBuilder iconTitle(String iconTitle) {
60         this.iconTitle = iconTitle;
61         return this;
62     }
63 
64     public LinkBuilder statusIconUrl(URI statusIconUrl) {
65         this.statusIconUrl = statusIconUrl;
66         return this;
67     }
68 
69     public LinkBuilder statusIconTitle(String statusIconTitle) {
70         this.statusIconTitle = statusIconTitle;
71         return this;
72     }
73 
74     public LinkBuilder resolved(boolean resolved) {
75         this.resolved = resolved;
76         return this;
77     }
78 
79     public Link build() {
80         return new Link(uri, title, relationship, summary, iconUrl, iconTitle, statusIconUrl, statusIconTitle, resolved);
81     }
82 }
