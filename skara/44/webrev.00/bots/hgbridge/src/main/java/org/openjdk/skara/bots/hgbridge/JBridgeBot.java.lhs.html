23 package org.openjdk.skara.bots.hgbridge;
24 
25 import org.openjdk.skara.bot.*;

26 
27 import java.io.*;
28 import java.net.URLEncoder;
29 import java.nio.charset.StandardCharsets;
30 import java.nio.file.*;
31 import java.util.List;
32 import java.util.logging.Logger;
33 
34 public class JBridgeBot implements Bot, WorkItem {
35     private final ExporterConfig exporterConfig;
36     private final Path storage;
37     private final Logger log = Logger.getLogger("org.openjdk.bots.hgbridge");
38 
39     JBridgeBot(ExporterConfig exporterConfig, Path storage) {
40         this.exporterConfig = exporterConfig;
41         this.storage = storage.resolve(URLEncoder.encode(exporterConfig.source().toString(), StandardCharsets.UTF_8));
42     }
43 
44     @Override
45     public String toString() {
46         return "JBridgeBot@" + exporterConfig.source();
47     }
48 
49     @Override
50     public List<WorkItem> getPeriodicItems() {
51         return List.of(this);
52     }
53 
54     @Override
55     public boolean concurrentWith(WorkItem other) {
56         if (other instanceof JBridgeBot) {
57             JBridgeBot otherBridgeBot = (JBridgeBot)other;
58             return !exporterConfig.source().equals(otherBridgeBot.exporterConfig.source());
59         } else {
60             return true;
61         }
62     }
63 




























64     @Override
65     public void run(Path scratchPath) {
66         log.fine("Running export for " + exporterConfig.source().toString());
67 
68         try {
69             var converter = exporterConfig.resolve(scratchPath);
70             var exported = Exporter.export(converter, exporterConfig.source(), storage);
71             IOException lastException = null;
72 


















73             for (var destination : exporterConfig.destinations()) {
74                 var markerBase = destination.getUrl().getHost() + "/" + destination.getName();
75                 var successfulPushMarker = storage.resolve(URLEncoder.encode(markerBase, StandardCharsets.UTF_8) + ".success.txt");
76                 if (exported.isPresent() || !successfulPushMarker.toFile().isFile()) {
77                     var repo = exported.orElse(Exporter.current(storage).orElseThrow());
78                     try {
79                         Files.deleteIfExists(successfulPushMarker);
80                         repo.pushAll(destination.getUrl());
81                         storage.resolve(successfulPushMarker).toFile().createNewFile();
82                     } catch (IOException e) {
83                         log.severe("Failed to push to " + destination.getUrl());
84                         log.throwing("JBridgeBot", "run", e);
85                         lastException = e;
86                     }
87                 } else {
88                     log.fine("No changes detected in " + exporterConfig.source() + " - skipping push to " + destination.getName());
89                 }
90             }
91             if (lastException != null) {
92                 throw new UncheckedIOException(lastException);
93             }
94         } catch (IOException e) {
95             throw new UncheckedIOException(e);
96         }
97     }
98 }
