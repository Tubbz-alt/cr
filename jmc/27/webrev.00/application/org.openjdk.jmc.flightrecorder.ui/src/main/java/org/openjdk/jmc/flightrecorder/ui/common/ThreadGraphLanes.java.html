145 	public IItemFilter getEnabledLanesFilter() {
146 		List<IItemFilter> laneFilters = laneDefs.stream()
147 				.filter((Predicate<? super LaneDefinition>) LaneDefinition::isEnabled).map(ld -> ld.getFilter())

152 	/**
153 	 * Retrieves the set of lane names that are currently enabled.<br>
154 	 * Note: The "Rest lane" is of type ItemFilters$Composite, and cannot be cast to Types, so it
155 	 * gets filtered out of the end result.
156 	 * 
157 	 * @return the enabled lanes independent from the rest lane
158 	 */
159 	public Set<String> getEnabledLanes() {
160 		List<IItemFilter> laneFilters = laneDefs.stream()
161 				.filter((Predicate<? super LaneDefinition>) LaneDefinition::isEnabledAndNotRestLane)
162 				.map(ld -> ld.getFilter()).collect(Collectors.toList());
163 		return ((Types) ItemFilters.or(laneFilters.toArray(new IItemFilter[laneFilters.size()]))).getTypes();
164 	}

178 	/**
179 	 * Introduces a "Quick Filter" to the lane definitions which is controlled by the dropdown lane
180 	 * filter. Initially, the enabled activity lanes will be a copy of the currently enabled lanes.
181 	 * When initially used, the "Quick Filter" will be the only active lane definition in an attempt
182 	 * to preserve the lane activity of the existing lane definitions. The "Quick Filter" is meant
183 	 * for easy viewing of activities, and will not be persisted.
184 	 */
185 	public void useDropdownFilter(LaneDefinition quickFilterDef) {
186 		if (quickFilterExist) {
187 			for (int i = 0; i < laneDefs.size(); i++) {
188 				if (quickFilterDef.getName().equals(laneDefs.get(i).getName())) {
189 					laneDefs.remove(laneDefs.get(i));
190 					laneDefs.add(i, quickFilterDef);
191 				}
192 			}
193 		} else {
194 			for (int i = 0; i < laneDefs.size(); i++) {
195 				setLaneDefinitionEnablement(laneDefs.get(i), i, false);
196 			}
197 			laneDefs.add(0, quickFilterDef);
198 			quickFilterExist = true;
199 		}
200 		buildChart.run();
201 	}

203 	private void setLaneDefinitionEnablement(LaneDefinition oldLd, int laneIndex, boolean isEnabled) {
204 		LaneDefinition newLd = new LaneDefinition(oldLd.getName(), isEnabled, oldLd.getFilter(), oldLd.isRestLane());
205 		laneDefs.set(laneIndex, newLd);
206 	}

208 	public IXDataRenderer buildThreadRenderer(Object thread, IItemCollection items) {
209 		this.resetTooltipTitle();
210 		String threadName = thread == null ? "" : ((IMCThread) thread).getThreadName(); //$NON-NLS-1$
211 		// FIXME: Workaround since this method can be called from super class constructor. Refactor to avoid this.
212 		List<LaneDefinition> laneFilters = this.laneDefs == null ? Collections.emptyList() : this.laneDefs;
213 		List<IXDataRenderer> lanes = new ArrayList<>(laneFilters.size());
214 		laneFilters.stream().filter(ld -> ld.isEnabled()).forEach(lane -> {
215 			IItemCollection laneItems = items.apply(lane.getFilter());
216 			if (laneItems.iterator().hasNext()) {
217 				ISpanSeries<IItem> laneSeries = QuantitySeries.max(laneItems, JfrAttributes.START_TIME,
218 						JfrAttributes.END_TIME);
219 				this.setTooltipTitle(
220 						MessageFormat.format(Messages.ThreadsPage_LANE_TOOLTIP_TITLE, threadName, lane.getName()));
221 				lanes.add(new ItemRow(
222 						SpanRenderer.withBoundaries(laneSeries, DataPageToolkit.ITEM_COLOR, this.getTooltipTitle()),
223 						laneItems));
224 			}
225 		});
226 		IXDataRenderer renderer = !lanes.isEmpty() ? RendererToolkit.uniformRows(lanes)
227 				: new ItemRow(RendererToolkit.empty(), ItemCollectionToolkit.EMPTY);
228 		IItemCollection itemsAndThreadLifespan = addThreadLifeSpanEvents(thread, items);
229 		// If the lane doesn't match a filter, display the Thread name as the tooltip title
230 		if (this.getTooltipTitle() == null) {
231 			this.setTooltipTitle(threadName);
232 		} else {
233 			this.resetTooltipTitle();
234 		}
235 		return new QuantitySpanRenderer(threadRanges(threadName, itemsAndThreadLifespan), renderer, THREAD_BG_COLOR, 10,
236 				threadName, this.getTooltipTitle(), thread);
237 	}

239 	private IItemCollection addThreadLifeSpanEvents(Object thread, final IItemCollection items) {
240 		IItemCollection threadLifeSpan = dataSourceSupplier.get().getItems()
241 				.apply(ItemFilters.and(ItemFilters.equals(JfrAttributes.EVENT_THREAD, (IMCThread) thread),
242 						ItemFilters.type(JdkTypeIDs.JAVA_THREAD_START, JdkTypeIDs.JAVA_THREAD_END)));
243 		IItemCollection itemsAndThreadLifespan = ItemCollectionToolkit.merge(() -> Stream.of(items, threadLifeSpan));
244 		return itemsAndThreadLifespan;
245 	}

247 	private IQuantitySeries<?> threadRanges(String threadName, IItemCollection items) {
248 		IItemCollection startEvents = items.apply(ItemFilters.type(JdkTypeIDs.JAVA_THREAD_START));
249 		IItemCollection endEvents = items.apply(ItemFilters.type(JdkTypeIDs.JAVA_THREAD_END));
250 		Iterator<IQuantity> start = ItemCollectionToolkit.values(startEvents, JfrAttributes.START_TIME).get().sorted()
251 				.iterator();
252 		Iterator<IQuantity> end = ItemCollectionToolkit.values(endEvents, JfrAttributes.END_TIME).get().sorted()
253 				.iterator();
254 
255 		ArrayList<IQuantity> startList = new ArrayList<>();
256 		ArrayList<IQuantity> endList = new ArrayList<>();
257 		IQuantity sq = start.hasNext() ? start.next() : MISSING_START;
258 		IQuantity eq = end.hasNext() ? end.next() : MISSING_END;
259 		if (sq.compareTo(eq) >= 0) {
260 			startList.add(MISSING_START);
261 			endList.add(eq);
262 			eq = end.hasNext() ? end.next() : MISSING_END;
263 		}
264 
265 		while (start.hasNext()) {
266 			startList.add(sq);
267 			endList.add(eq);
268 			sq = start.hasNext() ? start.next() : MISSING_START;
269 			eq = end.hasNext() ? end.next() : MISSING_END;
270 		}
271 		startList.add(sq);
272 		endList.add(eq);
273 
274 		String text = NLS.bind(Messages.JavaApplicationPage_THREAD_LIFESPAN, threadName);
275 		return QuantitySeries.all(startList, endList, new IDisplayable() {
276 
277 			@Override
278 			public String displayUsing(String formatHint) {
279 				return text;
280 			}
281 
282 		});
283 	}

285 	public void saveTo(IWritableState writableState) {
286 		laneDefs.stream().forEach(f -> f.saveTo(writableState.createChild(THREAD_LANE)));
287 		// FIXME: This will change the order from the original lane order, putting all the non applicable lanes last, can we live with that?
288 		naLanes.stream().forEach(f -> f.saveTo(writableState.createChild(THREAD_LANE)));
289 	}

291 	public List<IAction> initializeChartConfiguration(Stream<IState> laneStates) {
292 		laneDefs = new ArrayList<>();
293 		laneStates.map(LaneDefinition::readFrom).forEach(laneDefs::add);
294 		if (laneDefs.isEmpty()) {
295 			laneDefs.add(new LaneDefinition(Messages.JavaApplicationPage_THREAD_LANE_JAVA_LATENCIES, true,
296 					JdkFilters.THREAD_LATENCIES, false));
297 		}
298 		// FIXME: Might be nice to make some sort of model for the whole lane set
299 		LaneEditor.ensureRestLane(laneDefs);
300 		Map<Boolean, List<LaneDefinition>> lanesByApplicability = laneDefs.stream()
301 				.collect(Collectors.partitioningBy(ld -> ld.isRestLane()
302 						|| dataSourceSupplier.get().getItems().apply(ld.getFilter()).iterator().hasNext()));
303 		laneDefs = lanesByApplicability.get(true);
304 		naLanes = lanesByApplicability.get(false);
305 		return Collections.emptyList();
306 	}

308 	//create two action identifiers to handle the chart context menu and the legend context menu
309 	private List<String> chartActionIdentifiers = new ArrayList<>();
310 	private List<String> legendActionIdentifiers = new ArrayList<>();
311 
312 	public void updateContextMenus(MCContextMenuManager[] mms, boolean isLegendMenu) {
313 		if (isLegendMenu) {
314 			for (String id : legendActionIdentifiers) {
315 				for (MCContextMenuManager mm : mms) {
316 					mm.remove(id);
317 				}
318 			}
319 			legendActionIdentifiers.clear();
320 		} else {
321 			for (String id : chartActionIdentifiers) {
322 				for (MCContextMenuManager mm : mms) {
323 					mm.remove(id);
324 				}
325 			}
326 			chartActionIdentifiers.clear();
327 		}
328 		if (mms[0].indexOf(EDIT_LANES) == -1) {
329 			IAction action = ActionToolkit.action(() -> this.openEditLanesDialog(mms, isLegendMenu),
330 					Messages.JavaApplicationPage_EDIT_THREAD_LANES_ACTION,
331 					FlightRecorderUI.getDefault().getMCImageDescriptor(ImageConstants.ICON_LANES_EDIT));
332 			action.setId(EDIT_LANES);
333 			for (MCContextMenuManager mm : mms) {
334 				mm.add(action);
335 				mm.add(new Separator());
336 			}
337 			actions.add(action);
338 		}
339 		laneDefs.stream().forEach(ld -> {
340 			Action checkAction = new Action(ld.getName(), IAction.AS_CHECK_BOX) {
341 				int laneIndex = laneDefs.indexOf(ld);
342 
343 				@Override
344 				public void run() {
345 					setLaneDefinitionEnablement(ld, laneIndex, isChecked());
346 					buildChart.run();
347 				}
348 			};
349 			String identifier = ld.getName() + checkAction.hashCode();
350 			checkAction.setId(identifier);
351 			if (isLegendMenu) {
352 				legendActionIdentifiers.add(identifier);
353 			} else {
354 				chartActionIdentifiers.add(identifier);
355 			}
356 			checkAction.setChecked(ld.isEnabled());
357 			// FIXME: Add a tooltip here
358 			for (MCContextMenuManager mm : mms) {
359 				mm.add(checkAction);
360 			}
361 			actions.add(checkAction);
362 		});
363 	}
