 34 package org.openjdk.jmc.flightrecorder.ui.common;

 36 import java.util.stream.Collectors;
 37 
 38 import org.eclipse.swt.SWT;
 39 import org.eclipse.swt.events.ShellAdapter;
 40 import org.eclipse.swt.events.ShellEvent;
 41 import org.eclipse.swt.graphics.Point;
 42 import org.eclipse.swt.graphics.Rectangle;
 43 import org.eclipse.swt.layout.GridData;
 44 import org.eclipse.swt.layout.GridLayout;
 45 import org.eclipse.swt.widgets.Button;
 46 import org.eclipse.swt.widgets.Composite;
 47 import org.eclipse.swt.widgets.Display;
 48 import org.eclipse.swt.widgets.Event;
 49 import org.eclipse.swt.widgets.Listener;
 50 import org.eclipse.swt.widgets.Shell;
 51 
 52 import org.openjdk.jmc.common.item.ItemFilters;
 53 import org.openjdk.jmc.flightrecorder.ui.common.LaneEditor.LaneDefinition;
 54 import org.openjdk.jmc.flightrecorder.ui.messages.internal.Messages;
 55 import org.openjdk.jmc.ui.handlers.MCContextMenuManager;
 56 import org.openjdk.jmc.ui.misc.PatternFly.Palette;
 57 
 58 public class DropdownLaneFilter extends Composite {
 59 	private static final int EXTRA_SHELL_WIDTH = 100;
 60 	private static final int SHELL_HEIGHT = 400;
 61 	private Button dropdownButton;
 62 	private GridLayout layout;
 63 	private MCContextMenuManager[] mms;
 64 	private Shell shell;
 65 	private ThreadGraphLanes lanes;
 66 	private TypeFilterBuilder filterEditor;
 67 
 68 	public DropdownLaneFilter(Composite parent, ThreadGraphLanes lanes, MCContextMenuManager[] mms) {
 69 		super(parent, SWT.NONE);
 70 		this.lanes = lanes;
 71 		this.mms = mms;
 72 		this.setBackground(Palette.getThreadsPageBackgroundColor());
 73 		GridLayout layout = new GridLayout();
 74 		layout.marginHeight = 0;
 75 		layout.marginWidth = 0;
 76 		this.layout = layout;
 77 		setLayout(layout);
 78 		dropdownButton = new Button(this, SWT.TOGGLE);
 79 		dropdownButton.setLayoutData(new GridData(GridData.FILL_BOTH));
 80 		dropdownButton.setText(Messages.DropdownLaneFilter_THREAD_STATE_SELECTION);
 81 		dropdownButton.addListener(SWT.MouseUp, new Listener() {
 82 			@Override
 83 			public void handleEvent(Event e) {
 84 				if (dropdownButton.getSelection()) {
 85 					displayDropdown();
 86 				}
 87 			}
 88 		});
 89 	}
 90 
 91 	/**
 92 	 * Creates a new shell which is positioned below the dropdown button. This new shell creates the
 93 	 * appearance of a dropdown component, and it's contents will be the TypeFilterBuilder as found
 94 	 * in the Edit Thread Lanes dialog.
 95 	 */
 96 	private void displayDropdown() {
 97 		Point p = dropdownButton.getParent().toDisplay(dropdownButton.getLocation());
 98 		Point size = dropdownButton.getSize();
 99 		Rectangle shellRect = new Rectangle(p.x, p.y + size.y, size.x, 0);
100 
101 		shell = new Shell(DropdownLaneFilter.this.getShell(), SWT.BORDER);
102 		shell.addShellListener(new ShellAdapter() {
103 
104 			public void shellDeactivated(ShellEvent e) {
105 				if (!isCursorOnTopOfButton()) {
106 					// If the shell is closed without clicking the button (i.e., not forcing
107 					// a toggle), then the button must be toggled programmatically.
108 					dropdownButton.setSelection(false);
109 				}
110 				disposeDropdown();
111 			}
112 		});
113 
114 		shell.setLayout(this.layout);
115 		shell.setSize(shellRect.width + EXTRA_SHELL_WIDTH, SHELL_HEIGHT);
116 		shell.setLocation(shellRect.x, shellRect.y);
117 
118 		filterEditor = new TypeFilterBuilder(shell, this::onTypeFilterChange);
119 		filterEditor.setInput(lanes.getTypeTree());
120 		filterEditor.selectTypes(lanes.getEnabledLanes());
121 		filterEditor.getControl().setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
122 		shell.open();
123 	}
124 
125 	private void disposeDropdown() {
126 		if (shell != null && !shell.isDisposed()) {
127 			shell.close();
128 		}
129 	}
130 
131 	/**
132 	 * Determine whether or not the mouse cursor is overlapping the dropdown button. An open
133 	 * dropdown shell should close when the user clicks the button. In Linux, the MouseListener on
134 	 * the button will fire. In Windows, the shell has priority and the MouseListener doesn't get
135 	 * activated. This function is to be used in the ShellAdapter to determine if the user closed
136 	 * the shell by trying to click the button, or by clicking away from the dropdown shell.
137 	 *
138 	 * @return true if the mouse cursor is on top of the button
139 	 */
140 	private boolean isCursorOnTopOfButton() {
141 		Point cursor = Display.getCurrent().getCursorLocation();
142 		Point buttonLoc = dropdownButton.toDisplay(1, 1);
143 		Rectangle buttonRect = new Rectangle(buttonLoc.x, buttonLoc.y, dropdownButton.getSize().x,
144 				dropdownButton.getSize().y);
145 		return buttonRect.contains(cursor);
146 	}
147 
148 	/**
149 	 * Creates a new filter, Quick Filter, to be used by the dropdown lane filter. Sets the quick
150 	 * filter to be active, and update the context menu.
151 	 */
152 	private void onTypeFilterChange() {
153 		LaneDefinition quickLaneDef = new LaneDefinition(Messages.DropdownLaneFilter_QUICK_FILTER, true,
154 				ItemFilters.type(filterEditor.getCheckedTypeIds().collect(Collectors.toSet())), false);
155 		lanes.useDropdownFilter(quickLaneDef);
156 		lanes.updateContextMenus(mms, false);
157 	}
158 }
