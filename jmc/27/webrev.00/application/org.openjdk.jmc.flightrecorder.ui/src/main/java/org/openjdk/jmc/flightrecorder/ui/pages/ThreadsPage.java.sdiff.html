192 		 * Hides a thread from the chart and rebuilds the chart
193 		 */
194 		private void hideThread(Object thread) {
195 			if (this.threadRows != null && this.threadRows.size() > 0 && thread instanceof IMCThread) {
196 				int index = indexOfThread(thread);
197 				if (index != -1) {
198 					this.threadRows.remove(index);
199 					this.reloadThreads = false;
200 					buildChart();
201 					if (!this.isChartModified) {
202 						this.isChartModified = true;
203 						setResetChartActionEnablement(true);
204 					}
205 				}
206 				if (this.threadRows.size() == 0) {
207 					setHideThreadActionEnablement(false);
208 				}
209 			}
210 		}

213 		 * Locates the index of the target Thread in the current selection list
214 		 *
215 		 * @param thread
216 		 *            the thread of interest
217 		 * @return the index of the thread in the current selection, or -1 if not found
218 		 */
219 		private int indexOfThread(Object thread) {
220 			for (int i = 0; i < this.threadRows.size() && thread != null; i++) {
221 				if (this.threadRows.get(i) instanceof QuantitySpanRenderer) {
222 					if (thread.equals(((QuantitySpanRenderer) this.threadRows.get(i)).getData())) {
223 						return i;
224 					}
225 				}
226 			}
227 			return -1;
228 		}

279 		@Override
280 		protected IXDataRenderer getChartRenderer(IItemCollection itemsInTable, HistogramSelection tableSelection) {
281 			List<IXDataRenderer> rows = new ArrayList<>();
282 
283 			IItemCollection selectedItems;
284 			HistogramSelection selection;
285 			if (tableSelection.getRowCount() == 0) {
286 				selectedItems = itemsInTable;
287 				selection = table.getAllRows();
288 			} else {
289 				selectedItems = tableSelection.getItems();
290 				selection = tableSelection;
291 			}
292 			boolean useDefaultSelection = rows.size() > 1;
293 			if (lanes.getLaneDefinitions().stream().anyMatch(a -> a.isEnabled()) && selection.getRowCount() > 0) {
294 				if (this.reloadThreads) {
295 					this.threadRows = selection
296 							.getSelectedRows((object, items) -> lanes.buildThreadRenderer(object, items))
297 							.collect(Collectors.toList());


298 					this.isChartModified = false;
299 					if (this.isChartMenuActionsInit) {
300 						setResetChartActionEnablement(false);
301 						setHideThreadActionEnablement(true);
302 					}
303 				} else {
304 					this.reloadThreads = true;
305 				}
306 
307 				double threadsWeight = Math.sqrt(threadRows.size()) * 0.15;
308 				double otherRowWeight = Math.max(threadsWeight * 0.1, (1 - threadsWeight) / rows.size());
309 				List<Double> weights = Stream
310 						.concat(Stream.generate(() -> otherRowWeight).limit(rows.size()), Stream.of(threadsWeight))
311 						.collect(Collectors.toList());
312 				rows.add(RendererToolkit.uniformRows(this.threadRows));
313 				useDefaultSelection = true;
314 				rows = Arrays.asList(RendererToolkit.weightedRows(rows, weights));
315 			}
316 			IXDataRenderer root = rows.size() == 1 ? rows.get(0) : RendererToolkit.uniformRows(rows);
317 			// We don't use the default selection when there is only one row. This is to get the correct payload.
318 			return useDefaultSelection ? new ItemRow(root, selectedItems.apply(lanes.getEnabledLanesFilter())) : root;
319 		}

220 		/**
221 		 * Hides a thread from the chart and rebuilds the chart
222 		 */
223 		private void hideThread(Object thread) {
224 			if (this.threadRows != null && this.threadRows.size() > 0 && thread instanceof IMCThread) {
225 				int index = indexOfThread(thread);
226 				if (index != -1) {
227 					this.threadRows.remove(index);
228 					this.reloadThreads = false;
229 					buildChart();
230 					if (!this.isChartModified) {
231 						this.isChartModified = true;
232 						setResetChartActionEnablement(true);
233 					}
234 				}
235 				if (this.threadRows.size() == 0) {
236 					setHideThreadActionEnablement(false);
237 				}
238 			}
239 		}

242 		 * Locates the index of the target Thread in the current selection list
243 		 *
244 		 * @param thread
245 		 *            the thread of interest
246 		 * @return the index of the thread in the current selection, or -1 if not found
247 		 */
248 		private int indexOfThread(Object thread) {
249 			for (int i = 0; i < this.threadRows.size() && thread != null; i++) {
250 				if (this.threadRows.get(i) instanceof QuantitySpanRenderer) {
251 					if (thread.equals(((QuantitySpanRenderer) this.threadRows.get(i)).getData())) {
252 						return i;
253 					}
254 				}
255 			}
256 			return -1;
257 		}

320 		@Override
321 		protected IXDataRenderer getChartRenderer(IItemCollection itemsInTable, HistogramSelection tableSelection) {
322 			List<IXDataRenderer> rows = new ArrayList<>();
323 			ItemHistogram histogram = getUndisposedTable();
324 			IItemCollection selectedItems;
325 			HistogramSelection selection;
326 			if (tableSelection.getRowCount() == 0) {
327 				selectedItems = itemsInTable;
328 				selection = histogram.getAllRows();
329 			} else {
330 				selectedItems = tableSelection.getItems();
331 				selection = tableSelection;
332 			}
333 			boolean useDefaultSelection = rows.size() > 1;
334 			if (lanes.getLaneDefinitions().stream().anyMatch(a -> a.isEnabled()) && selection.getRowCount() > 0) {
335 				if (this.reloadThreads) {
336 					this.threadRows = selection
337 							.getSelectedRows((object, items) -> lanes.buildThreadRenderer(object, items))
338 							.collect(Collectors.toList());
339 					chartCanvas.setNumItems(this.threadRows.size());
340 					textCanvas.setNumItems(this.threadRows.size());
341 					this.isChartModified = false;
342 					if (this.isChartMenuActionsInit) {
343 						setResetChartActionEnablement(false);
344 						setHideThreadActionEnablement(true);
345 					}
346 				} else {
347 					this.reloadThreads = true;
348 				}
349 
350 				double threadsWeight = Math.sqrt(threadRows.size()) * 0.15;
351 				double otherRowWeight = Math.max(threadsWeight * 0.1, (1 - threadsWeight) / rows.size());
352 				List<Double> weights = Stream
353 						.concat(Stream.generate(() -> otherRowWeight).limit(rows.size()), Stream.of(threadsWeight))
354 						.collect(Collectors.toList());
355 				rows.add(RendererToolkit.uniformRows(this.threadRows));
356 				useDefaultSelection = true;
357 				rows = Arrays.asList(RendererToolkit.weightedRows(rows, weights));
358 			}
359 			IXDataRenderer root = rows.size() == 1 ? rows.get(0) : RendererToolkit.uniformRows(rows);
360 			// We don't use the default selection when there is only one row. This is to get the correct payload.
361 			return useDefaultSelection ? new ItemRow(root, selectedItems.apply(lanes.getEnabledLanesFilter())) : root;
362 		}
