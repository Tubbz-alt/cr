113 	protected void init(
114 		IItemFilter pageFilter, StreamModel model, Composite parent, FormToolkit toolkit, IPageContainer pageContainer,
115 		IState state, String sectionTitle, IItemFilter tableFilter, Image icon,
116 		FlavorSelectorState flavorSelectorState) {
117 		this.pageFilter = pageFilter;
118 		this.model = model;
119 		this.pageContainer = pageContainer;
120 		form = DataPageToolkit.createForm(parent, toolkit, sectionTitle, icon);
121 
122 		hiddenTableContainer = new Composite(form, SWT.NONE);
123 		hiddenTableContainer.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
124 		hiddenTableContainer.setVisible(false);
125 
126 		hiddenTable = buildHistogram(hiddenTableContainer, state.getChild(TABLE));
127 		hiddenTable.getManager().getViewer().addSelectionChangedListener(e -> buildChart());
128 
129 		tableFilterComponent = FilterComponent.createFilterComponent(hiddenTable.getManager().getViewer().getControl(),
130 				hiddenTable.getManager(), tableFilter, model.getItems().apply(pageFilter),
131 				pageContainer.getSelectionStore()::getSelections, this::onFilterChange);
132 
133 		/**
134 		 * Scrolled Composite Page Container - Contains all page functionality Chart Container (1
135 		 * column gridlayout) - Contains filter bar & graph container Graph Container (2 column
136 		 * gridlayout) - Contains chart and timeline container & display bar Chart and Timeline
137 		 * Container (1 column gridlayout) Contains chart and text container and timeline canvas
138 		 * Timeline and Height Buttons Container (2 column gridlayout) Contains timeline and lane
139 		 * height button controls Zoom-pan and Chart Container (formlayout) - Contains chart and
140 		 * text container contents and zoom-pan overlay Zoom-pan Container (filllayout) - Contains
141 		 * zoom-pan chart overlay Full screen Chart Container (1 column gridlayout) - Contains chart
142 		 * container Chart and Text Container (2 column gridlayout) - Contains scText and
143 		 * textCanvas) & scChart (and chart canvas)
144 		 */
145 
146 		// Scrolled Composite containing all page functionality
147 		ScrolledComposite scPageContainer = new ScrolledComposite(form.getBody(), SWT.H_SCROLL | SWT.V_SCROLL);
148 
149 		chartContainer = toolkit.createComposite(scPageContainer);
150 		chartContainer.setLayout(new GridLayout());
151 		chartContainer.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
152 
153 		scPageContainer.setContent(chartContainer);
154 		scPageContainer.setAlwaysShowScrollBars(false);
155 		scPageContainer.setExpandHorizontal(true);
156 		scPageContainer.setExpandVertical(true);
157 		scPageContainer.addListener(SWT.Resize, new Listener() {
158 			@Override
159 			public void handleEvent(Event event) {
160 				int width = filterBar.computeSize(SWT.DEFAULT, SWT.DEFAULT).x;
161 				int height = filterBar.computeSize(SWT.DEFAULT, SWT.DEFAULT).y
162 						+ displayBar.computeSize(SWT.DEFAULT, SWT.DEFAULT).y;
163 				if (width > 0 && height > 0) {
164 					scPageContainer.setMinSize(scPageContainer.computeSize(width, height));
165 					scPageContainer.removeListener(SWT.Resize, this);
166 				}
167 			}
168 		});
169 
170 		// Filter Controls
171 		Listener resetListener = new Listener() {
172 			@Override
173 			public void handleEvent(Event event) {
174 				onSetRange(false);
175 			}
176 		};
177 		filterBar = new ChartFilterControlBar(chartContainer, resetListener, pageContainer.getRecordingRange());
178 		filterBar.setLayoutData(new GridData(SWT.FILL, SWT.FILL, false, false));
179 
180 		// Container to hold the chart (& timeline) and display toolbar
181 		Composite graphContainer = toolkit.createComposite(chartContainer);
182 		GridLayout gridLayout = new GridLayout(2, false);
183 		gridLayout.marginWidth = 0;
184 		gridLayout.marginHeight = 0;
185 		graphContainer.setLayout(gridLayout);
186 		graphContainer.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
187 
188 		// Container to hold the chart and timeline canvas
189 		Composite chartAndTimelineContainer = toolkit.createComposite(graphContainer);
190 		gridLayout = new GridLayout();
191 		gridLayout.marginWidth = 0;
192 		gridLayout.marginHeight = 0;
193 		chartAndTimelineContainer.setLayout(gridLayout);
194 		chartAndTimelineContainer.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
195 
196 		// Container to hold the chart and a zoom-pan overlay
197 		Composite zoomPanAndChartContainer = toolkit.createComposite(chartAndTimelineContainer);
198 		zoomPanAndChartContainer.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
199 		zoomPanAndChartContainer.setLayout(new FormLayout());
200 
201 		// Container to hold fixed zoom-pan display
202 		Composite zoomPanContainer = toolkit.createComposite(zoomPanAndChartContainer);
203 		zoomPanContainer.setLayout(new FillLayout());
204 		FormData fd = new FormData();
205 		fd.height = 80;
206 		fd.width = 150;
207 		fd.bottom = new FormAttachment(100, -12);
208 		fd.right = new FormAttachment(100, -12);
209 		zoomPanContainer.setLayoutData(fd);
210 
211 		// Container to hold the chart
212 		Composite fullScreenChartContainer = toolkit.createComposite(zoomPanAndChartContainer);
213 		fullScreenChartContainer.setLayout(gridLayout);
214 		fd = new FormData();
215 		fd.right = new FormAttachment(100, -1);
216 		fd.top = new FormAttachment(0, 1);
217 		fd.left = new FormAttachment(0, 1);
218 		fd.bottom = new FormAttachment(100, -1);
219 		fullScreenChartContainer.setLayoutData(fd);
220 
221 		// Container to hold the text and chart canvases
222 		Composite chartAndTextContainer = toolkit.createComposite(fullScreenChartContainer);
223 		gridLayout = new GridLayout(2, false);
224 		gridLayout.horizontalSpacing = 0;
225 		gridLayout.marginWidth = 0;
226 		gridLayout.marginHeight = 0;
227 		chartAndTextContainer.setLayout(gridLayout);
228 		chartAndTextContainer.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
229 
230 		sash = new SashForm(chartAndTextContainer, SWT.VERTICAL);
231 		sash.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
232 		toolkit.adapt(sash);
233 
234 		ScrolledComposite scText = new ScrolledComposite(sash, SWT.BORDER | SWT.H_SCROLL | SWT.V_SCROLL);
235 		GridData scTextGd = new GridData(SWT.FILL, SWT.FILL, false, true);
236 		scTextGd.widthHint = 180;
237 		scText.setLayoutData(scTextGd);
238 		textCanvas = new ChartTextCanvas(scText);
239 		textCanvas.setLayoutData(new GridData(SWT.FILL, SWT.FILL, false, true));
240 
241 		ScrolledComposite scChart = new ScrolledComposite(sash, SWT.BORDER | SWT.V_SCROLL);
242 		scChart.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
243 		chartCanvas = new ChartCanvas(scChart);
244 		chartCanvas.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
245 
246 		chartCanvas.setTextCanvas(textCanvas);
247 		textCanvas.setChartCanvas(chartCanvas);
248 
249 		scChart.setContent(chartCanvas);
250 		scChart.setAlwaysShowScrollBars(true);
251 		scChart.setExpandHorizontal(true);
252 		scChart.setExpandVertical(true);
253 		scText.setContent(textCanvas);
254 		scText.setAlwaysShowScrollBars(false);
255 		scText.setExpandHorizontal(true);
256 		scText.setExpandVertical(true);
257 
258 		Composite timelineAndHeightBtnsContainer = toolkit.createComposite(chartAndTimelineContainer);
259 		gridLayout = new GridLayout(2, false);
260 		gridLayout.horizontalSpacing = 0;
261 		gridLayout.marginWidth = 0;
262 		timelineAndHeightBtnsContainer.setLayout(gridLayout);
263 		timelineAndHeightBtnsContainer.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, false));
264 
265 		heightBtns = new ChartLaneHeightControls(timelineAndHeightBtnsContainer, chartCanvas, textCanvas);
266 		heightBtns.setLayoutData(new GridData(SWT.FILL, SWT.FILL, false, false));
267 
268 		timelineCanvas = new TimelineCanvas(timelineAndHeightBtnsContainer, chartCanvas, sash, Y_SCALE);
269 		GridData gridData = new GridData(SWT.FILL, SWT.DEFAULT, true, false);
270 		gridData.heightHint = (int) (TIMELINE_HEIGHT * Y_SCALE);
271 		timelineCanvas.setLayoutData(gridData);
272 
273 		// add the display bar to the right of the chart scrolled composite
274 		displayBar = new ChartDisplayControlBar(graphContainer);
275 
276 		allChartSeriesActions = initializeChartConfiguration(state);
277 		IState chartState = state.getChild(CHART);
278 		ActionToolkit.loadCheckState(chartState, allChartSeriesActions.stream());
279 		chartLegend = ActionUiToolkit.buildCheckboxViewer(chartContainer, allChartSeriesActions.stream());
280 		gridData = new GridData(SWT.FILL, SWT.FILL, false, true);
281 		gridData.widthHint = 180;
282 		chartLegend.getControl().setLayoutData(gridData);
283 		PersistableSashForm.loadState(sash, state.getChild(SASH));
284 		DataPageToolkit.createChartTimestampTooltip(chartCanvas);
285 
286 		chart = new XYChart(pageContainer.getRecordingRange(), RendererToolkit.empty(), X_OFFSET, Y_OFFSET,
287 				timelineCanvas, filterBar, displayBar);
288 		DataPageToolkit.setChart(chartCanvas, chart, pageContainer::showSelection);
289 		DataPageToolkit.setChart(textCanvas, chart, pageContainer::showSelection);
290 		SelectionStoreActionToolkit.addSelectionStoreRangeActions(pageContainer.getSelectionStore(), chart,
291 				JfrAttributes.LIFETIME, NLS.bind(Messages.ChartAndTableUI_TIMELINE_SELECTION, form.getText()),
292 				chartCanvas.getContextMenu());
293 		SelectionStoreActionToolkit.addSelectionStoreRangeActions(pageContainer.getSelectionStore(), chart,
294 				JfrAttributes.LIFETIME, NLS.bind(Messages.ChartAndTableUI_TIMELINE_SELECTION, form.getText()),
295 				textCanvas.getContextMenu());
296 
297 		// Wire-up the chart & text canvases to the filter and display bars
298 		displayBar.setChart(chart);
299 		displayBar.setChartCanvas(chartCanvas);
300 		displayBar.setTextCanvas(textCanvas);
301 		displayBar.createZoomPan(zoomPanContainer);
302 		chartCanvas.setZoomOnClickListener(mouseDown -> displayBar.zoomOnClick(mouseDown));
303 		chartCanvas.setZoomToSelectionListener(() -> displayBar.zoomToSelection());
304 		timelineCanvas.setChart(chart);
305 
306 		if (chartState != null) {
307 			final String legendSelection = chartState.getAttribute(SELECTED);
308 
309 			if (legendSelection != null) {
310 				allChartSeriesActions.stream().filter(ia -> legendSelection.equals(ia.getId())).findFirst()
311 						.ifPresent(a -> chartLegend.setSelection(new StructuredSelection(a)));
312 			}
313 		}
314 
315 		flavorSelector = FlavorSelector.itemsWithTimerange(form, pageFilter, model.getItems(), pageContainer,
316 				this::onFlavorSelected, this::onSetRange, flavorSelectorState);
317 	}
318 
319 	protected void onFilterChange(IItemFilter filter) {
320 		IItemCollection items = getItems();
321 		if (tableFilterComponent.isVisible()) {
322 			table.show(items.apply(filter));
323 			tableFilterComponent.setColor(table.getAllRows().getRowCount());
324 		} else if (table != null) {
325 			table.show(items);
326 		}
327 	}
328 
329 	@Override
330 	public void saveTo(IWritableState writableState) {
331 		table = getUndisposedTable();
332 		super.saveTo(writableState);
333 	}
334 
335 	private void onSetRange(Boolean useRange) {
336 		IRange<IQuantity> range = useRange ? timeRange : pageContainer.getRecordingRange();
337 		chart.setVisibleRange(range.getStart(), range.getEnd());
338 		chart.resetZoomFactor();
339 		displayBar.resetZoomScale();
340 		buildChart();
341 	}
342 
343 	private void onFlavorSelected(IItemCollection items, IRange<IQuantity> timeRange) {
344 		this.selectionItems = items;
345 		this.timeRange = timeRange;
346 		hiddenTable.show(getItems());
347 		if (selectionItems != null) {
348 			Object[] tableInput = (Object[]) hiddenTable.getManager().getViewer().getInput();
349 			if (tableInput != null) {
350 				hiddenTable.getManager().getViewer().setSelection(new StructuredSelection(tableInput));
351 			} else {
352 				hiddenTable.getManager().getViewer().setSelection(null);
353 			}
354 		}
355 	}
356 
357 	protected void buildChart() {
358 		IXDataRenderer rendererRoot = getChartRenderer(getItems(), getUndisposedTable().getSelection());
359 		chartCanvas.replaceRenderer(rendererRoot);
360 		textCanvas.replaceRenderer(rendererRoot);
361 	}
362 
363 	private IItemCollection getItems() {
364 		return selectionItems != null ? selectionItems.apply(pageFilter) : model.getItems().apply(pageFilter);
365 	}
366 
367 	public void setTimeRange(IRange<IQuantity> range) {
368 		this.timeRange = range;
369 	}
370 
371 	protected ItemHistogram getUndisposedTable() {
372 		return isDisposed(table) ? hiddenTable : table;
373 	}
374 
375 	private boolean isDisposed(ItemHistogram histogram) {
376 		return histogram == null ? true : histogram.getManager().getViewer().getControl().isDisposed();
377 	}
