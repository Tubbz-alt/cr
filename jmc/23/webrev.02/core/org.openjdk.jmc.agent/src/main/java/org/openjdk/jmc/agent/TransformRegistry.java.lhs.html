 33 package org.openjdk.jmc.agent;
 34 
 35 import java.util.List;
 36 
 37 public interface TransformRegistry {
 38 	/**
 39 	 * The named class has transforms that have not been executed yet.
 40 	 *
 41 	 * @param className
 42 	 *            the name of the class to be transformed.
 43 	 * @return true if the class has transforms scheduled. false if not.
 44 	 */
 45 	boolean hasPendingTransforms(String className);
 46 
 47 	/**O
 48 	 * Returns the list of {@link TransformDescriptor}s for the named class.
 49 	 *
 50 	 * @param className
 51 	 *            the class for which to retrieve the transformation metadata.
 52 	 * @return the list of transformation metadata for the named class.
 53 	 */
 54 	List<TransformDescriptor> getTransformData(String className);
 55 
 56 	/**
 57 	 * Modifies class information in the registry according to the xml description.
 58 	 *
 59 	 * @param xmlDescription
 60 	 *           an XML snippet describing the wanted modifications.
 61 	 *
 62 	 * @return a list of {@link TransformDescriptor}s corresponding to the wanted transformations.
 63 	 */
 64 	List<TransformDescriptor> modify(String xmlDescription);
 65 
 66 	/**
 67 	 * Clears all classes and their corresponding transforms in the registry.
 68 	 *
 69 	 * @return the set of class names that were cleared.
 70 	 */
 71 	List<String> clearAllTransformData();
 72 
 73 	/**
 74 	 * Stores the pre instrumentation byte array of a class.
 75 	 * @param className
 76 	 *           the class for which to store the pre instrumentation data.
 77 	 * @param classPreInstrumentation
 78 	 *           the pre instrumentation byte array of the class to store.
 79 	 */
 80 	void storeClassPreInstrumentation(String className, byte[] classPreInstrumentation);
 81 
 82 	/**
 83 	 * Returns a byte array associated with a class pre instrumentation.
 84 	 * @param className
 85 	 *           the name of the class to get pre instrumentation data for.
 86 	 * @return a byte array of a class pre instrumentation.
 87 	 */
 88 	byte[] getClassPreInstrumentation(String className);
 89 
 90 	/**
 91 	 * Signify classes are or are not being reverted to their pre instrumentation versions.
 92 	 * @param shouldRevert
 93 	 *           true if class instrumentation should be reverted, false otherwise.
 94 	 */
 95 	void setRevertInstrumentation(boolean shouldRevert);
 96 
 97 	/**
 98 	 * Determines if classes should be reverted to their pre instrumentation versions.
 99 	 * @return true, if classes should be reverted and false otherwise.
100 	 */
101 	boolean isRevertIntrumentation();
102 
103 }
