 34 package org.openjdk.jmc.flightrecorder.flameview.tree;
 35 
 36 import java.util.ArrayList;
 37 import java.util.List;
 38 
 39 public class TraceNode {
 40 	private final int value;
 41 	private final String name;

 42 	private final List<TraceNode> children = new ArrayList<>();
 43 
 44 	public TraceNode(String name, int value) {
 45 		this.name = name;
 46 		this.value = value;

 47 	}
 48 
 49 	public int getValue() {
 50 		return value;
 51 	}
 52 
 53 	public String getName() {
 54 		return name;
 55 	}




 56 
 57 	public List<TraceNode> getChildren() {
 58 		return children;
 59 	}
 60 
 61 	public void addChild(TraceNode child) {
 62 		children.add(child);
 63 	}
 64 
 65 	@Override
 66 	public int hashCode() {
 67 		final int prime = 31;
 68 		int result = 1;
 69 		result = prime * result + ((children == null) ? 0 : children.hashCode());
 70 		result = prime * result + ((name == null) ? 0 : name.hashCode());

 71 		result = prime * result + value;
 72 		return result;
 73 	}
 74 
 75 	@Override
 76 	public boolean equals(Object obj) {
 77 		if (this == obj)
 78 			return true;
 79 		if (obj == null)
 80 			return false;
 81 		if (getClass() != obj.getClass())
 82 			return false;
 83 		TraceNode other = (TraceNode) obj;
 84 		if (children == null) {
 85 			if (other.children != null)
 86 				return false;
 87 		} else if (!children.equals(other.children))
 88 			return false;
 89 		if (name == null) {
 90 			if (other.name != null)
 91 				return false;
 92 		} else if (!name.equals(other.name))
 93 			return false;






 94 		if (value != other.value)
 95 			return false;
 96 		return true;
 97 	}
 98 
 99 	public String toString() {
100 		return "TraceNode [name: " + name + ", value: " + value + ", children: " + children.size() + "]";
101 	}
102 }
