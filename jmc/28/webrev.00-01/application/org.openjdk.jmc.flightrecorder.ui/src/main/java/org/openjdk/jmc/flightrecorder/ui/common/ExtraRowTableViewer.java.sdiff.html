 44 public class ExtraRowTableViewer extends TableViewer {
 45 
 46 	private String message;
 47 	private TableItem extraRow;
 48 
 49 	// FIXME: Would like to decorate the extra row with a font, but the table we use this for uses ViewerColumns with their own labelproviders
 50 
 51 	public ExtraRowTableViewer(Composite parent) {
 52 		super(parent);
 53 	}
 54 
 55 	public ExtraRowTableViewer(Composite composite, int style) {
 56 		super(composite, style);
 57 	}
 58 
 59 	public void setExtraMessage(String message) {
 60 		this.message = message;
 61 	}
 62 




 63 	@Override
 64 	public void refresh(Object element) {
 65 		if (message == null) {
 66 			super.refresh(element);
 67 		} else {
 68 			removeExtraRow();
 69 			super.refresh(element);
 70 		}
 71 	}
 72 
 73 	@Override
 74 	protected void inputChanged(Object input, Object oldInput) {
 75 		if (message == null) {
 76 			super.inputChanged(input, oldInput);
 77 		} else {
 78 			removeExtraRow();
 79 			super.inputChanged(input, oldInput);
 80 			createExtraRow();
 81 		}
 82 	}
 83 
 84 	@Override
 85 	public void refresh() {
 86 		if (message == null) {
 87 			super.refresh();
 88 		} else {
 89 			removeExtraRow();
 90 			super.refresh();
 91 			createExtraRow();
 92 		}
 93 	}
 94 
 95 	private void createExtraRow() {
 96 		long maxNumRows = FlightRecorderUI.getDefault().getItemListSize().longValue();
 97 		if (getFilteredChildren(getInput()).length < maxNumRows) {
 98 			return;
 99 		}
100 		extraRow = new TableItem(getTable(), SWT.NO_BACKGROUND | SWT.NO_FOCUS);
101 		extraRow.setText(message);
102 	}
103 
104 	private void removeExtraRow() {
105 		if (extraRow != null) {
106 			try {
107 				extraRow.dispose();
108 			} catch (Exception e) {
109 				FlightRecorderUI.getDefault().getLogger().log(Level.WARNING,
110 						"Exception while disposing extra row in table", //$NON-NLS-1$
111 						e);
112 			}
113 		}
114 	}
