33 package org.openjdk.jmc.agent;
34 
35 import org.openjdk.jmc.agent.util.TypeUtils;
36 
37 /**
38  * Metadata for a parameter to be logged by the agent.
39  */
40 public final class Parameter {
41 	public static final int INDEX_INVALID = -2;
42 	public static final int INDEX_RETURN = -1;
43 
44 	private final int index;
45 	private final String name;
46 	private final String fieldName;
47 	private final String description;
48 	private final String contentType;
49 	private final String relationKey;
50 	private final String converterClassName;
51 
52 	public Parameter(int index, String name, String description, String contentType, String relationKey,
53 			String converterClassName) {
54 		this.index = index;
55 		this.name = name;
56 		this.description = description;
57 		this.contentType = contentType;
58 		this.relationKey = relationKey;
59 		this.converterClassName = converterClassName;
60 		this.fieldName = "field" + TypeUtils.deriveIdentifierPart(name); //$NON-NLS-1$
61 	}
62 
63 	public int getIndex() {
64 		return index;
65 	}
66 
67 	public String getName() {
68 		return name;
69 	}
70 
71 	public String getDescription() {
72 		return description;
73 	}
74 
75 	public String getContentType() {
76 		return contentType;
77 	}
78 
79 	public String getRelationKey() {
80 		return relationKey;
81 	}
82 
83 	public String getFieldName() {
84 		return fieldName;
85 	}
86 
87 	public boolean isReturn() {
88 		return index == INDEX_RETURN;
89 	}
90 
91 	public boolean isInvalid() {
92 		return index == INDEX_INVALID;
93 	}
94 
95 	public String getConverterClassName() {
96 		return converterClassName;
97 	}
98 }
