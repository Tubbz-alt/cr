33 package org.openjdk.jmc.rjmx.services.internal;
34 
35 import org.openjdk.jmc.rjmx.services.ICommercialFeaturesService;
36 
37 /**
38  * Used by JVMs with no commercial features, e.g. OpenJDK 8 and JDK 11+ JVMs.
39  */
40 public class NoCommercialFeaturesService implements ICommercialFeaturesService {
41 
42 	@Override
43 	public boolean isCommercialFeaturesEnabled() {
44 		return true;
45 	}
46 
47 	@Override
48 	public void enableCommercialFeatures() throws Exception {
49 		// Noop
50 	}
51 
52 	@Override
53 	public boolean hasCommercialFeatures() {
54 		return false;
55 	}
56 }
