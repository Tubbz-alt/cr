 47 public class TraceTreeUtils {
 48 	public final static String DEFAULT_ROOT_NAME = "__root";
 49 	public final static String DEFAULT_ROOT_PACKAGE_NAME = "";

 50 	public final static FrameSeparator DEFAULT_FRAME_SEPARATOR = new FrameSeparator(FrameCategorization.METHOD, false);

 52 	/**
 53 	 * Traces a TraceTree from a {@link StacktraceModel}.
 54 	 *
 55 	 * @param model
 56 	 *            the model to trace the tree from.
 57 	 * @return the root.
 58 	 */
 59 	public static TraceNode createTree(StacktraceModel model, String rootName) {
 60 		Fork rootFork = model.getRootFork();
 61 		TraceNode root = new TraceNode(rootName == null ? DEFAULT_ROOT_NAME : rootName, rootFork.getItemsInFork(),
 62 				DEFAULT_ROOT_PACKAGE_NAME);
 63 		for (Branch branch : rootFork.getBranches()) {
 64 			addBranch(root, branch);
 65 		}
 66 		return root;
 67 	}

 69 	/**
 70 	 * Traces a tree of stack frames from an {@link IItemCollection}.
 71 	 *
 72 	 * @param items
 73 	 *            the events to aggregate the traces from.
 74 	 * @return the root of the resulting tree.
 75 	 */
 76 	public static TraceNode createTree(
 77 		IItemCollection items, FrameSeparator frameSeparator, boolean threadRootAtTop, String rootName) {
 78 		return createTree(new StacktraceModel(threadRootAtTop, frameSeparator, items), rootName);
 79 	}

 81 	private static void addBranch(TraceNode root, Branch branch) {
 82 		StacktraceFrame firstFrame = branch.getFirstFrame();
 83 		TraceNode currentNode = new TraceNode(format(firstFrame), firstFrame.getItemCount(),
 84 				formatPackageName(firstFrame));
 85 		root.addChild(currentNode);
 86 		for (StacktraceFrame frame : branch.getTailFrames()) {
 87 			TraceNode newNode = new TraceNode(format(frame), frame.getItemCount(), formatPackageName(frame));
 88 			currentNode.addChild(newNode);
 89 			currentNode = newNode;
 90 		}
 91 		addFork(currentNode, branch.getEndFork());
 92 	}
 93 
 94 	private static void addFork(TraceNode node, Fork fork) {
 95 		for (Branch branch : fork.getBranches()) {
 96 			addBranch(node, branch);
 97 		}
 98 	}

100 	private static String format(StacktraceFrame sFrame) {
101 		IMCFrame frame = sFrame.getFrame();
102 		IMCMethod method = frame.getMethod();
103 		return FormatToolkit.getHumanReadable(method, false, false, true, false, true, false);
104 	}
105 
106 	private static String formatPackageName(StacktraceFrame sFrame) {
107 		IMCFrame frame = sFrame.getFrame();
108 		IMCMethod method = frame.getMethod();
109 		return FormatToolkit.getPackage(method.getType().getPackage());
110 	}

112 	public static String printTree(TraceNode node) {
113 		StringBuilder builder = new StringBuilder();
114 		builder.append("=== Tree Printout ===");
115 		builder.append(System.lineSeparator());
116 		printTree(builder, 0, node);
117 		return builder.toString();
118 	}

120 	private static void printTree(StringBuilder builder, int indentation, TraceNode node) {
121 		builder.append(String.format("%s%s - %d%n", indent(indentation), node.getName(), node.getValue()));
122 		for (TraceNode child : node.getChildren()) {
123 			printTree(builder, indentation + 1, child);
124 		}
125 	}
126 
127 	private static String indent(int indentation) {
128 		StringBuilder builder = new StringBuilder();
129 		for (int i = 0; i < indentation; i++) {
130 			builder.append("   ");
131 		}

 52 public class TraceTreeUtils {
 53 	public final static String DEFAULT_ROOT_NAME = "__root";
 54 	public final static String DEFAULT_ROOT_PACKAGE_NAME = "";
 55 	public final static String DEFAULT_TRACE_NODE_DESC = "";
 56 	public final static FrameSeparator DEFAULT_FRAME_SEPARATOR = new FrameSeparator(FrameCategorization.METHOD, false);
 57 
 58 	private static class TraceNodeFactory {
 59 
 60 		private static TraceNode getRootTraceNode(String rootName, Fork rootFork) {
 61 			return new TraceNode(rootName == null ? DEFAULT_ROOT_NAME : rootName, rootFork.getItemsInFork(),
 62 					DEFAULT_ROOT_PACKAGE_NAME, DEFAULT_TRACE_NODE_DESC);
 63 		}
 64 
 65 		private static TraceNode getTraceNodeByStacktraceFrame(StacktraceFrame sFrame) {
 66 			IMCFrame frame = sFrame.getFrame();
 67 			IMCMethod method = frame.getMethod();
 68 			String packageName = FormatToolkit.getPackage(method.getType().getPackage());
 69 			if (frame == StacktraceModel.UNKNOWN_FRAME) {
 70 				return new TraceNode(Messages.getString(STACKTRACE_UNCLASSIFIABLE_FRAME), sFrame.getItemCount(),
 71 						packageName, Flameview_UNCLASSIFIABLE_FRAME_DESC);
 72 			} else {
 73 				String name = FormatToolkit.getHumanReadable(method, false, false, true, false, true, false);
 74 				return new TraceNode(name, sFrame.getItemCount(), packageName, DEFAULT_TRACE_NODE_DESC);
 75 			}
 76 		}
 77 	}

 79 	/**
 80 	 * Traces a TraceTree from a {@link StacktraceModel}.
 81 	 *
 82 	 * @param model
 83 	 *            the model to trace the tree from.
 84 	 * @return the root.
 85 	 */
 86 	public static TraceNode createTree(StacktraceModel model, String rootName) {
 87 		Fork rootFork = model.getRootFork();
 88 		TraceNode root = TraceNodeFactory.getRootTraceNode(rootName, rootFork);

 89 		for (Branch branch : rootFork.getBranches()) {
 90 			addBranch(root, branch);
 91 		}
 92 		return root;
 93 	}

 95 	/**
 96 	 * Traces a tree of stack frames from an {@link IItemCollection}.
 97 	 *
 98 	 * @param items
 99 	 *            the events to aggregate the traces from.
100 	 * @return the root of the resulting tree.
101 	 */
102 	public static TraceNode createTree(
103 		IItemCollection items, FrameSeparator frameSeparator, boolean threadRootAtTop, String rootName) {
104 		return createTree(new StacktraceModel(threadRootAtTop, frameSeparator, items), rootName);
105 	}
106 
107 	private static void addBranch(TraceNode root, Branch branch) {
108 		StacktraceFrame firstFrame = branch.getFirstFrame();
109 		TraceNode currentNode = TraceNodeFactory.getTraceNodeByStacktraceFrame(firstFrame);

110 		root.addChild(currentNode);
111 		for (StacktraceFrame frame : branch.getTailFrames()) {
112 			TraceNode newNode = TraceNodeFactory.getTraceNodeByStacktraceFrame(frame);
113 			currentNode.addChild(newNode);
114 			currentNode = newNode;
115 		}
116 		addFork(currentNode, branch.getEndFork());
117 	}
118 
119 	private static void addFork(TraceNode node, Fork fork) {
120 		for (Branch branch : fork.getBranches()) {
121 			addBranch(node, branch);
122 		}
123 	}
