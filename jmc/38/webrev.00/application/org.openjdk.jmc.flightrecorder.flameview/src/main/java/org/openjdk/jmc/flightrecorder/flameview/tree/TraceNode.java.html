 34 package org.openjdk.jmc.flightrecorder.flameview.tree;
 35 
 36 import java.util.ArrayList;
 37 import java.util.List;
 38 
 39 public class TraceNode {
 40 	private final int value;
 41 	private final String name;
 42 	private final String packageName;
 43 	private final String desc;
 44 	private final List<TraceNode> children = new ArrayList<>();
 45 
 46 	public TraceNode(String name, int value, String packageName, String desc) {
 47 		this.name = name;
 48 		this.value = value;
 49 		this.packageName = packageName;
 50 		this.desc = desc;
 51 	}
 52 
 53 	public int getValue() {
 54 		return value;
 55 	}
 56 
 57 	public String getName() {
 58 		return name;
 59 	}
 60 
 61 	public String getPackageName() {
 62 		return packageName;
 63 	}
 64 
 65 	public String getDesc() {
 66 		return desc;
 67 	}
 68 
 69 	public List<TraceNode> getChildren() {
 70 		return children;
 71 	}
 72 
 73 	public void addChild(TraceNode child) {
 74 		children.add(child);
 75 	}
 76 
 77 	@Override
 78 	public int hashCode() {
 79 		final int prime = 31;
 80 		int result = 1;
 81 		result = prime * result + ((children == null) ? 0 : children.hashCode());
 82 		result = prime * result + ((name == null) ? 0 : name.hashCode());
 83 		result = prime * result + ((packageName == null) ? 0 : packageName.hashCode());
 84 		result = prime * result + ((desc == null) ? 0 : desc.hashCode());
 85 		result = prime * result + value;
 86 		return result;
 87 	}
 88 
 89 	@Override
 90 	public boolean equals(Object obj) {
 91 		if (this == obj)
 92 			return true;
 93 		if (obj == null)
 94 			return false;
 95 		if (getClass() != obj.getClass())
 96 			return false;
 97 		TraceNode other = (TraceNode) obj;
 98 		if (children == null) {
 99 			if (other.children != null)
100 				return false;
101 		} else if (!children.equals(other.children))
102 			return false;
103 		if (name == null) {
104 			if (other.name != null)
105 				return false;
106 		} else if (!name.equals(other.name))
107 			return false;
108 		if (packageName == null) {
109 			if (other.packageName != null) {
110 				return false;
111 			}
112 		} else if (!packageName.equals(other.packageName))
113 			return false;
114 		if (desc == null) {
115 			if (other.desc != null) {
116 				return false;
117 			}
118 		} else if (!desc.equals(other.desc))
119 			return false;
120 		if (value != other.value)
121 			return false;
122 		return true;
123 	}
124 
125 	public String toString() {
126 		return "TraceNode [name: " + name + ", value: " + value + ", packageName: " + packageName + ", desc: " + desc
127 				+ ", children: " + children.size() + "]";
128 	}
129 }
