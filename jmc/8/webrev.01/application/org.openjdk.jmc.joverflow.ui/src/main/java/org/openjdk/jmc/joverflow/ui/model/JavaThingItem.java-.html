 33 package org.openjdk.jmc.joverflow.ui.model;
 34 
 35 import org.openjdk.jmc.joverflow.heap.model.JavaThing;
 36 
 37 /**
 38  * A {@code TreeItem} for a {@code JavaThing}
 39  */
 40 public class JavaThingItem implements TreeItem {
 41 
 42 	private Iterable<JavaThingItem> children;
 43 	private boolean expanded;
 44 	private final int level;
 45 	private final JavaThing content;
 46 	private final String name;
 47 	private final String value;
 48 	private final String size;
 49 
 50 	public JavaThingItem(int level, String name, JavaThing content) {
 51 		this(level, name, content == null ? "null" : content.valueAsString(), content == null ? 0 : content.getSize(), content); //$NON-NLS-1$
 52 	}
 53 
 54 	public JavaThingItem(int level, String name, String value, int size, JavaThing content) {
 55 		this.level = level;
 56 		this.content = content;
 57 		this.name = String.valueOf(name);
 58 		this.value = String.valueOf(value);
 59 		this.size = Integer.toString(size);
 60 	}
 61 
 62 	public String getName() {
 63 		return name;
 64 	}
 65 
 66 	public String getValue() {
 67 		return value;
 68 	}
 69 
 70 	public String getSize() {
 71 		return size;
 72 	}
 73 
 74 	public Iterable<JavaThingItem> getChildItems() {
 75 		return children;
 76 	}
 77 
 78 	public void setChildItems(Iterable<JavaThingItem> children) {
 79 		this.children = children;
 80 	}
 81 
 82 	@Override
 83 	public void setExpended(boolean expanded) {
 84 		this.expanded = expanded;
 85 		if (!expanded && children != null) {
 86 			// Collapse children with parent
 87 			for (TreeItem c : children) {
 88 				c.setExpended(false);
 89 			}
 90 		}
 91 	}
 92 
 93 	@Override
 94 	public boolean isExpanded() {
 95 		return expanded;
 96 	}
 97 
 98 	@Override
 99 	public int getLevel() {
100 		return level;
101 	}
102 
103 	public JavaThing getContent() {
104 		return content;
105 	}
106 }
