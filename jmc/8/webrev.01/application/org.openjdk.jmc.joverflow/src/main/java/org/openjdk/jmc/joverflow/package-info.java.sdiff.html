35 /**
36  * JOverflow is a tool for doing heap waste analysis on hprof heap dumps.
37  * <p>
38  * It is used in the JOverflow plug-in for JMC, but can also be used headless.
39  * <p>
40  * Example:
41  * <p>

42  * <pre>
43  * {@code 
44  *	public static void main(String args[]) throws Exception {
45  *		ReportGenerator rg = ReportGenerator.parseDump(
46  *				"/tmp/heap-dumps/mydump.hprof", new Progress() {
47  *					public void reportProgress(int percent) {
48  *						System.out.println(percent);
49  *						if (percent > 55)
50  *							cancel();
51  *					}
52  *				});
53  *		System.out.println(rg.getReport());
54  *	} 
55  * }
56  * </pre>
