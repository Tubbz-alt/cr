 33 package org.openjdk.jmc.test.jemmy.misc.wrappers;
 34 
 35 import java.util.ArrayList;
 36 import java.util.List;
 37 
 38 import org.eclipse.swt.custom.CCombo;
 39 import org.eclipse.swt.widgets.Shell;
 40 import org.jemmy.control.Wrap;
 41 import org.jemmy.input.SelectionText;
 42 import org.jemmy.interfaces.Keyboard.KeyboardButtons;
 43 import org.jemmy.interfaces.Parent;
 44 import org.jemmy.interfaces.Selectable;
 45 import org.jemmy.resources.StringComparePolicy;
 46 import org.jemmy.swt.lookup.ByItemLookup;
 47 import org.jemmy.swt.lookup.ByName;
 48 
 49 import org.openjdk.jmc.test.jemmy.misc.base.wrappers.MCJemmyBase;
 50 
 51 /**
 52  * The Jemmy wrapper for CCombo objects
 53  */
 54 public class MCCCombo extends MCJemmyBase {
 55 	private static final StringComparePolicy DEFAULT_POLICY = StringComparePolicy.EXACT;
 56 
 57 	private MCCCombo(Wrap<? extends CCombo> ccomboWrap) {
 58 		this.control = ccomboWrap;
 59 	}
 60 
 61 	/**
 62 	 * Returns a {@link MCCCombo} object named according to the supplied name
 63 	 *
 64 	 * @param name
 65 	 *            the name of the CCombo to match
 66 	 * @return a {@link MCCCombo} object that matches the name
 67 	 */
 68 	public static MCCCombo getByName(String name) {
 69 		return getByName(getShell(), name);
 70 	}
 71 
 72 	/**
 73 	 * Returns a {@link MCCCombo} object named according to the supplied name
 74 	 *
 75 	 * @param shell
 76 	 *            the shell from where to start the search for the CCombo object
 77 	 * @param name
 78 	 *            the name of the CCombo to match
 79 	 * @return a {@link MCCCombo} object that matches the name
 80 	 */
 81 	@SuppressWarnings("unchecked")
 82 	public static MCCCombo getByName(Wrap<? extends Shell> shell, String name) {
 83 		return new MCCCombo(shell.as(Parent.class, CCombo.class).lookup(CCombo.class, new ByName<>(name)).wrap());
 84 	}
 85 
 86 	/**
 87 	 * Returns a {@link MCCCombo} object having at least one selectable item that matches the text
 88 	 * provided
 89 	 *
 90 	 * @param text
 91 	 *            the text string of the item to match
 92 	 * @return a {@link MCCCombo} object that matches the text
 93 	 */
 94 	public static MCCCombo getByText(String text) {
 95 		return getByText(getShell(), text);
 96 	}
 97 
 98 	/**
 99 	 * Returns a {@link MCCCombo} object having at least one selectable item that matches the text
100 	 * provided
101 	 *
102 	 * @param shell
103 	 *            the shell from where to start the search for the CCombo object
104 	 * @param text
105 	 *            the text string of the item to match
106 	 * @return a {@link MCCCombo} object that matches the text
107 	 */
108 	public static MCCCombo getByText(Wrap<? extends Shell> shell, String text) {
109 		return getByText(shell, text, DEFAULT_POLICY);
110 	}
111 
112 	/**
113 	 * Returns a {@link MCCCombo} object having at least one selectable item that matches the text
114 	 * provided
115 	 *
116 	 * @param shell
117 	 *            the shell from where to start the search for the CCombo object
118 	 * @param text
119 	 *            the text string of the item to match
120 	 * @param policy
121 	 *            the policy to use when matching
122 	 * @return a {@link MCCCombo} object that matches the text
123 	 */
124 	@SuppressWarnings("unchecked")
125 	public static MCCCombo getByText(Wrap<? extends Shell> shell, String text, StringComparePolicy policy) {
126 		return new MCCCombo(
127 				shell.as(Parent.class, CCombo.class).lookup(CCombo.class, new ByItemLookup<>(text, policy)).wrap());
128 	}
129 
130 	/**
131 	 * Returns all visible {@link MCCCombo} objects underneath the supplied shell
132 	 *
133 	 * @param shell
134 	 *            the shell from where to start the search for the CCombo object
135 	 * @return a {@link List} of {@link MCCCombo} objects
136 	 */
137 	@SuppressWarnings("unchecked")
138 	public static List<MCCCombo> getAll(Wrap<? extends Shell> shell) {
139 		List<Wrap<? extends CCombo>> list = getVisible(shell.as(Parent.class, CCombo.class).lookup(CCombo.class));
140 		List<MCCCombo> ccombos = new ArrayList<>();
141 		for (int i = 0; i < list.size(); i++) {
142 			ccombos.add(new MCCCombo(list.get(i)));
143 		}
144 		return ccombos;
145 	}
146 
147 	/**
148 	 * Returns all visible CCombo objects underneath the supplied {@link MCDialog}
149 	 *
150 	 * @param dialog
151 	 *            the {@link MCDialog} from where to start the search for CCombo objects
152 	 * @return a {@link List} of {@link MCCCombo} objects
153 	 */
154 	public static List<MCCCombo> getAll(MCDialog dialog) {
155 		return getAll(dialog.getDialogShell());
156 	}
157 
158 	/**
159 	 * Returns the first visible {@link MCCCombo} object underneath the supplied shell
160 	 *
161 	 * @param shell
162 	 *            the shell from where to start the search for the CCombo object
163 	 * @return a {@link MCCCombo} object
164 	 */
165 	public static MCCCombo getFirst(Wrap<? extends Shell> shell) {
166 		return getAll(shell).get(0);
167 	}
168 
169 	/**
170 	 * Returns the first visible {@link MCCCombo} object underneath the supplied {@link MCDialog}
171 	 *
172 	 * @param dialog
173 	 *            the {@link MCDialog} from where to start the search for CCombo objects
174 	 * @return a {@link MCCCombo} object
175 	 */
176 	public static MCCCombo getFirst(MCDialog dialog) {
177 		return getFirst(dialog.getDialogShell());
178 	}
179 
180 	/**
181 	 * Returns the first visible {@link MCCCombo} object underneath the Mission Control main shell
182 	 * 
183 	 * @return a {@link MCCCombo} object
184 	 */
185 	public static MCCCombo getFirst() {
186 		return getFirst(getShell());
187 	}
188 
189 	/**
190 	 * Selects the item matching (exactly) the supplied name
191 	 *
192 	 * @param name
193 	 *            the text string of the item to select
194 	 */
195 	@SuppressWarnings("unchecked")
196 	public void select(String name) {
197 		control.as(Selectable.class).selector().select(name);
198 	}
199 
200 	/**
201 	 * @return a list of all items
202 	 */
203 	@SuppressWarnings("unchecked")
204 	public List<String> getStates() {
205 		return control.getProperty(List.class, Selectable.STATES_PROP_NAME);
206 	}
207 
208 	/**
209 	 * @return the currently selected item
210 	 */
211 	public String getState() {
212 		return control.getProperty(String.class, Selectable.STATE_PROP_NAME);
213 	}
214 
215 	/**
216 	 * Sets the text of this {@link MCCCombo}
217 	 *
218 	 * @param text
219 	 *            the text to set
220 	 */
221 	public void setText(String text) {
222 		SelectionText textField = control.as(SelectionText.class);
223 		click();
224 		textField.clear();
225 		textField.type(text);
226 		if (isContentAssistPresent()) {
227 			control.keyboard().pushKey(KeyboardButtons.ESCAPE);
228 		}
229 	}
230 }
