33 package org.openjdk.jmc.console.uitest;
34 
35 import org.junit.Assert;
36 import org.junit.Test;
37 
38 import org.openjdk.jmc.test.jemmy.MCJemmyTestBase;
39 import org.openjdk.jmc.test.jemmy.misc.wrappers.MC;
40 import org.openjdk.jmc.test.jemmy.misc.wrappers.MCMenu;
41 import org.openjdk.jmc.test.jemmy.misc.wrappers.MCTabFolder;
42 
43 /**
44  * UI test which does a very basic check of the Welcome screen. Simply checks that it can be opened
45  * and closed.
46  */
47 public class WelcomePageTest extends MCJemmyTestBase {
48 	
49 	/**
50 	 * Open and close the Welcome page 
51 	 */
52 	@Test
53 	public void testWelcomePage() {
54 		MCMenu.openWelcome();
55 		MCTabFolder welcomePage = MCTabFolder.getByTabName("Welcome");
56 		Assert.assertNotNull("The Welcome Page did not open.", welcomePage);
57 
58 		// Closing the welcome page and ensuring that it doesn't show up any longer
59 		MC.closeWelcome();
60 		welcomePage = MCTabFolder.getByTabName("Welcome");
61 		Assert.assertNull("The Welcome Page wasn't closed.", welcomePage);
62 	}
63 }
