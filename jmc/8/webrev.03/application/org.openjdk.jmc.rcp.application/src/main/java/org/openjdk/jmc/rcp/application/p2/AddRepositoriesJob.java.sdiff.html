43 /**
44  * We're doing repository initialization as a separate job to avoid blocking the start of Mission Control.
45  * 
46  */
47 public final class AddRepositoriesJob extends Job {
48 	private final BundleContext context;
49 
50 	public AddRepositoriesJob(BundleContext context) {
51 		super("Add Update Repositories Job"); //$NON-NLS-1$
52 		this.context = context;
53 	}
54 
55 	@Override
56 	protected IStatus run(IProgressMonitor monitor) {
57 		try {
58 			P2Toolkit.addRepositories(context.getService(context.getServiceReference(IProvisioningAgent.class)));
59 		} catch (Exception e) {
60 			return new Status(IStatus.WARNING, ApplicationPlugin.PLUGIN_ID, "Could not add the P2 repositories!", e); //$NON-NLS-1$
61 		}
62 		return Status.OK_STATUS;
63 	}
64 }
