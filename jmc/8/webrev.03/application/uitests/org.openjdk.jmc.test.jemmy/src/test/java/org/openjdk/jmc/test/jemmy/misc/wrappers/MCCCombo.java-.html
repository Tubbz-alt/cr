 33 package org.openjdk.jmc.test.jemmy.misc.wrappers;
 34 
 35 import java.util.ArrayList;
 36 import java.util.List;
 37 
 38 import org.eclipse.swt.custom.CCombo;
 39 import org.eclipse.swt.widgets.Shell;
 40 import org.jemmy.control.Wrap;
 41 import org.jemmy.input.SelectionText;
 42 import org.jemmy.interfaces.Keyboard.KeyboardButtons;
 43 import org.jemmy.interfaces.Parent;
 44 import org.jemmy.interfaces.Selectable;
 45 import org.jemmy.resources.StringComparePolicy;
 46 import org.jemmy.swt.lookup.ByItemLookup;
 47 import org.jemmy.swt.lookup.ByName;
 48 
 49 import org.openjdk.jmc.test.jemmy.misc.base.wrappers.MCJemmyBase;
 50 
 51 /**
 52  * The Jemmy wrapper for CCombo objects
 53  */
 54 public class MCCCombo extends MCJemmyBase {
 55 	private static final StringComparePolicy DEFAULT_POLICY = StringComparePolicy.EXACT;
 56 
 57 	private MCCCombo(Wrap<? extends CCombo> ccomboWrap) {
 58 		this.control = ccomboWrap;
 59 	}
 60 
 61 	/**
 62 	 * Returns a {@link MCCCombo} object named according to the supplied name
 63 	 *
 64 	 * @param name
 65 	 *            the name of the CCombo to match
 66 	 * @return a {@link MCCCombo} object that matches the name
 67 	 */
 68 	public static MCCCombo getByName(String name) {
 69 		return getByName(getShell(), name);
 70 	}
 71 
 72 	/**
 73 	 * Returns a {@link MCCCombo} object named according to the supplied name
 74 	 *
 75 	 * @param shell
 76 	 *            the shell from where to start the search for the CCombo object
 77 	 * @param name
 78 	 *            the name of the CCombo to match
 79 	 * @return a {@link MCCCombo} object that matches the name
 80 	 */
 81 	@SuppressWarnings("unchecked")
 82 	public static MCCCombo getByName(Wrap<? extends Shell> shell, String name) {
 83 		return new MCCCombo(shell.as(Parent.class, CCombo.class).lookup(CCombo.class, new ByName<>(name)).wrap());
 84 	}
 85 
 86 	/**
 87 	 * Returns a {@link MCCCombo} object having at least one selectable item that matches the text provided
 88 	 *
 89 	 * @param text
 90 	 *            the text string of the item to match
 91 	 * @return a {@link MCCCombo} object that matches the text
 92 	 */
 93 	public static MCCCombo getByText(String text) {
 94 		return getByText(getShell(), text);
 95 	}
 96 
 97 	/**
 98 	 * Returns a {@link MCCCombo} object having at least one selectable item that matches the text provided
 99 	 *
100 	 * @param shell
101 	 *            the shell from where to start the search for the CCombo object
102 	 * @param text
103 	 *            the text string of the item to match
104 	 * @return a {@link MCCCombo} object that matches the text
105 	 */
106 	public static MCCCombo getByText(Wrap<? extends Shell> shell, String text) {
107 		return getByText(shell, text, DEFAULT_POLICY);
108 	}
109 
110 	/**
111 	 * Returns a {@link MCCCombo} object having at least one selectable item that matches the text provided
112 	 *
113 	 * @param shell
114 	 *            the shell from where to start the search for the CCombo object
115 	 * @param text
116 	 *            the text string of the item to match
117 	 * @param policy
118 	 *            the policy to use when matching
119 	 * @return a {@link MCCCombo} object that matches the text
120 	 */
121 	@SuppressWarnings("unchecked")
122 	public static MCCCombo getByText(Wrap<? extends Shell> shell, String text, StringComparePolicy policy) {
123 		return new MCCCombo(
124 				shell.as(Parent.class, CCombo.class).lookup(CCombo.class, new ByItemLookup<>(text, policy)).wrap());
125 	}
126 
127 	/**
128 	 * Returns all visible {@link MCCCombo} objects underneath the supplied shell
129 	 *
130 	 * @param shell
131 	 *            the shell from where to start the search for the CCombo object
132 	 * @return a {@link List} of {@link MCCCombo} objects
133 	 */
134 	@SuppressWarnings("unchecked")
135 	public static List<MCCCombo> getAll(Wrap<? extends Shell> shell) {
136 		List<Wrap<? extends CCombo>> list = getVisible(shell.as(Parent.class, CCombo.class).lookup(CCombo.class));
137 		List<MCCCombo> ccombos = new ArrayList<>();
138 		for (int i = 0; i < list.size(); i++) {
139 			ccombos.add(new MCCCombo(list.get(i)));
140 		}
141 		return ccombos;
142 	}
143 
144 	/**
145 	 * Returns all visible CCombo objects underneath the supplied {@link MCDialog}
146 	 *
147 	 * @param dialog
148 	 *            the {@link MCDialog} from where to start the search for CCombo objects
149 	 * @return a {@link List} of {@link MCCCombo} objects
150 	 */
151 	public static List<MCCCombo> getAll(MCDialog dialog) {
152 		return getAll(dialog.getDialogShell());
153 	}
154 
155 	/**
156 	 * Returns the first visible {@link MCCCombo} object underneath the supplied shell
157 	 *
158 	 * @param shell
159 	 *            the shell from where to start the search for the CCombo object
160 	 * @return a {@link MCCCombo} object
161 	 */
162 	public static MCCCombo getFirst(Wrap<? extends Shell> shell) {
163 		return getAll(shell).get(0);
164 	}
165 
166 	/**
167 	 * Returns the first visible {@link MCCCombo} object underneath the supplied {@link MCDialog}
168 	 *
169 	 * @param dialog
170 	 *            the {@link MCDialog} from where to start the search for CCombo objects
171 	 * @return a {@link MCCCombo} object
172 	 */
173 	public static MCCCombo getFirst(MCDialog dialog) {
174 		return getFirst(dialog.getDialogShell());
175 	}
176 
177 	/**
178 	 * Returns the first visible {@link MCCCombo} object underneath the Mission Control main shell
179 	 * 
180 	 * @return a {@link MCCCombo} object
181 	 */
182 	public static MCCCombo getFirst() {
183 		return getFirst(getShell());
184 	}
185 
186 	/**
187 	 * Selects the item matching (exactly) the supplied name
188 	 *
189 	 * @param name
190 	 *            the text string of the item to select
191 	 */
192 	@SuppressWarnings("unchecked")
193 	public void select(String name) {
194 		control.as(Selectable.class).selector().select(name);
195 	}
196 
197 	/**
198 	 * @return a list of all items
199 	 */
200 	@SuppressWarnings("unchecked")
201 	public List<String> getStates() {
202 		return control.getProperty(List.class, Selectable.STATES_PROP_NAME);
203 	}
204 
205 	/**
206 	 * @return the currently selected item
207 	 */
208 	public String getState() {
209 		return control.getProperty(String.class, Selectable.STATE_PROP_NAME);
210 	}
211 
212 	/**
213 	 * Sets the text of this {@link MCCCombo}
214 	 *
215 	 * @param text
216 	 *            the text to set
217 	 */
218 	public void setText(String text) {
219 		SelectionText textField = control.as(SelectionText.class);
220 		click();
221 		textField.clear();
222 		textField.type(text);
223 		if (isContentAssistPresent()) {
224 			control.keyboard().pushKey(KeyboardButtons.ESCAPE);
225 		}
226 	}
227 }
