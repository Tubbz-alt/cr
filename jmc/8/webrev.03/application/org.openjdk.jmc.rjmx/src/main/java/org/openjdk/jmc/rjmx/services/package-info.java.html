33 /**
34  * This package contains the core RJMX services.
35  * <p>
36  * The following example will print the help for all diagnostic commands:
37  *
38  * <pre>
39  * IConnectionDescriptor descriptor = new ConnectionDescriptorBuilder().hostName("localhost").port(0).build();
40  * IServerHandle server = IServerHandle.create(descriptor);
41  * try {
42  * 	IConnectionHandle connection = server.connect("Run Diagnostic commands");
43  * 	IDiagnosticCommandService dcmd = connection.getServiceOrDummy(IDiagnosticCommandService.class);
44  * 	for (IOperation operation : dcmd.getOperations()) {
45  * 		System.out.println(dcmd.runCtrlBreakHandlerWithResult(String.format("help %s", operation.getName())));
46  * 	}
47  * } finally {
48  * 	server.dispose();
49  * }
50  * </pre>
51  *
52  * Services are normally added through the <tt>org.openjdk.jmc.rjmx.service</tt> extension point.
53  * The extension point requires a factory which will be used to create your service. The factory
54  * must implement the {@link org.openjdk.jmc.rjmx.services.IServiceFactory} interface. The following
55  * example shows how the flight recorder service is added in the <tt>plugin.xml<tt> for the RJMX
56  * plug-in itself:
57  *
58  * <pre>
59 &lt;extension point="org.openjdk.jmc.rjmx.service"&gt;
60 	&lt;service
61 		factory="org.openjdk.jmc.rjmx.services.internal.FlightRecorderServiceFactory"
62 		description="Service for controlling the flight recorder"
63 		name="Flight Recorder"&gt;
64 	&lt;/service&gt;
65 &lt;/extension&gt;
66  * </pre>
67  *
68  * Note that any plug-in can publish services in this manner.
69  */
70 package org.openjdk.jmc.rjmx.services;
