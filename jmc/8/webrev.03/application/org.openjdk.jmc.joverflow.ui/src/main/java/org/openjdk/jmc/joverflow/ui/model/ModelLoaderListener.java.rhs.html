33 package org.openjdk.jmc.joverflow.ui.model;
34 
35 import java.util.Collection;
36 
37 import org.openjdk.jmc.joverflow.heap.model.Snapshot;
38 
39 public interface ModelLoaderListener {
40 	/**
41 	 * Called when loading fails
42 	 * 
43 	 * @param failure
44 	 */
45 	void onModelLoadFailed(Throwable failure);
46 
47 	/**
48 	 * Called when loading is completed successfully. The model is represented by a collection of
49 	 * {@code ReferenceChain} Additional info can be loaded from {@code snapshot}
50 	 * 
51 	 * @param snapshot
52 	 * @param model
53 	 */
54 	void onModelLoaded(Snapshot snapshot, Collection<ReferenceChain> model);
55 
56 	/**
57 	 * Called periodically during loading
58 	 * 
59 	 * @param progress
60 	 */
61 	void onProgressUpdate(double progress);
62 }
