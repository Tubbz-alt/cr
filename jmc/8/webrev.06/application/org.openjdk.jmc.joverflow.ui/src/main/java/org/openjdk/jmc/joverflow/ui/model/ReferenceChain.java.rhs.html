33 package org.openjdk.jmc.joverflow.ui.model;
34 
35 import java.util.ArrayList;
36 import java.util.Iterator;
37 
38 import org.openjdk.jmc.joverflow.support.RefChainElement;
39 
40 /**
41  * Collection of a number of {@code ObjectCluster} sharing the same reference chain held by
42  * {@code RefChainElement}
43  */
44 public class ReferenceChain implements Iterable<ObjectCluster> {
45 
46 	private final ArrayList<ObjectCluster> objectClusters = new ArrayList<>();
47 	private final RefChainElement referenceChain;
48 
49 	ReferenceChain(RefChainElement refChain) {
50 		referenceChain = refChain;
51 	}
52 
53 	void add(ObjectClusterImpl cluster) {
54 		objectClusters.add(cluster);
55 	}
56 
57 	@Override
58 	public Iterator<ObjectCluster> iterator() {
59 		return objectClusters.iterator();
60 	}
61 
62 	public RefChainElement getReferenceChain() {
63 		return referenceChain;
64 	}
65 
66 	void trim() {
67 		objectClusters.trimToSize();
68 	}
69 }
