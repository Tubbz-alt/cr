34 package org.openjdk.jmc.joverflow.ui.util;
35 
36 import org.eclipse.core.runtime.ListenerList;
37 import org.eclipse.swt.graphics.Color;
38 import org.eclipse.swt.widgets.Display;
39 
40 public class BaseArcAttributeProvider implements IArcAttributeProvider {
41 	private final ListenerList<IArcAttributeProviderListener> mListenerList = new ListenerList<>();
42 
43 	private final Color[] COLORS = {new Color(Display.getDefault(), 169, 226, 0),
44 			new Color(Display.getDefault(), 249, 217, 0), new Color(Display.getDefault(), 34, 186, 217),
45 			new Color(Display.getDefault(), 1, 129, 226), new Color(Display.getDefault(), 47, 53, 127),
46 			new Color(Display.getDefault(), 134, 0, 97), new Color(Display.getDefault(), 198, 43, 0),};
47 
48 	private int i = 0;
49 
50 	public BaseArcAttributeProvider() {
51 		addListener((event) -> i = 0);
52 	}
53 
54 	public int getWeight(Object element) {
55 		return 1;
56 	}
57 
58 	public Color getColor(Object element) {
59 		return COLORS[i++ % COLORS.length];
60 	}
61 
62 	public void addListener(IArcAttributeProviderListener listener) {
63 		mListenerList.add(listener);
64 	}
65 
66 	public void dispose() {
67 		for (Color c : COLORS) {
68 			c.dispose();
69 		}
70 	}
71 
72 	public void removeListener(IArcAttributeProviderListener listener) {
73 		mListenerList.remove(listener);
74 	}
75 
76 	public ListenerList<IArcAttributeProviderListener> getListenerList() {
77 		return mListenerList;
78 	}
79 }
