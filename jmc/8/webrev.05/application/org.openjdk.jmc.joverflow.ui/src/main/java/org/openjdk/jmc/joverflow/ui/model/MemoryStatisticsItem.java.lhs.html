33 package org.openjdk.jmc.joverflow.ui.model;
34 
35 /**
36  * Class holding an aggregate of mOverhead/mMemory/mSize for a number of {@code ObjectCluster}. The group is identified by
37  * {@code id}
38  */
39 public class MemoryStatisticsItem {
40 	private final Object id;
41 	private Integer index;
42 
43 	private long mOverhead;
44 	private long mMemory;
45 	private int mSize;
46 
47 	public MemoryStatisticsItem(Object id, long memory, long ovhd, int size) {
48 		this.id = id;
49 	}
50 
51 	public long getOverhead() {
52 		return mOverhead;
53 	}
54 
55 	public long getMemory() {
56 		return mMemory;
57 	}
58 
59 	public int getSize() {
60 		return mSize;
61 	}
62 
63 	public void reset() {
64 		mOverhead = 0;
65 		mMemory = 0;
66 		mSize = 0;
67 	}
68 
69 	public String getName() {
70 		return id == null ? "N/A" : id.toString(); //$NON-NLS-1$
71 	}
72 
73 	public void addObjectCluster(ObjectCluster oc) {
74 		mOverhead += oc.getOverhead();
75 		mMemory += oc.getMemory();
76 		mSize += oc.getObjectCount();
77 	}
78 
79 	public void setIndex(int index) {
80 		this.index = index;
81 	}
82 
83 	public Integer getIndex() {
84 		return index;
85 	}
86 
87 	public Object getId() {
88 		return id;
89 	}
90 
91 	@Override
92 	public String toString() {
93 		return getName();
94 	}
95 }
