 33 package org.openjdk.jmc.joverflow.ui.model;
 34 
 35 import org.openjdk.jmc.joverflow.heap.model.JavaThing;
 36 
 37 /**
 38  * A {@code TreeItem} for a {@code JavaThing}
 39  */
 40 public class JavaThingItem implements TreeItem {
 41 
 42 	private Iterable<JavaThingItem> children;
 43 	private boolean expanded;
 44 	private final int level;
 45 	private final JavaThing content;
 46 	private final String name;
 47 	private final String value;
 48 	private final String size;
 49 
 50 	public JavaThingItem(int level, String name, JavaThing content) {
 51 		this(level, name, content == null ? "null" : content.valueAsString(), content == null ? 0 : content.getSize(), //$NON-NLS-1$
 52 				content);
 53 	}
 54 
 55 	public JavaThingItem(int level, String name, String value, int size, JavaThing content) {
 56 		this.level = level;
 57 		this.content = content;
 58 		this.name = String.valueOf(name);
 59 		this.value = String.valueOf(value);
 60 		this.size = Integer.toString(size);
 61 	}
 62 
 63 	public String getName() {
 64 		return name;
 65 	}
 66 
 67 	public String getValue() {
 68 		return value;
 69 	}
 70 
 71 	public String getSize() {
 72 		return size;
 73 	}
 74 
 75 	public Iterable<JavaThingItem> getChildItems() {
 76 		return children;
 77 	}
 78 
 79 	public void setChildItems(Iterable<JavaThingItem> children) {
 80 		this.children = children;
 81 	}
 82 
 83 	@Override
 84 	public void setExpended(boolean expanded) {
 85 		this.expanded = expanded;
 86 		if (!expanded && children != null) {
 87 			// Collapse children with parent
 88 			for (TreeItem c : children) {
 89 				c.setExpended(false);
 90 			}
 91 		}
 92 	}
 93 
 94 	@Override
 95 	public boolean isExpanded() {
 96 		return expanded;
 97 	}
 98 
 99 	@Override
100 	public int getLevel() {
101 		return level;
102 	}
103 
104 	public JavaThing getContent() {
105 		return content;
106 	}
107 }
