47 @SuppressWarnings("nls")
48 public class EventConfigurationTest {
49 
50 	private IEventConfiguration loadConfig(String jfcName) throws Exception {
51 		InputStream in = EventConfigurationTest.class.getResourceAsStream(jfcName);
52 		XMLModel model = EventConfiguration.createModel(in);
53 		return new EventConfiguration(model);

54 	}
55 
56 	private void validateConfig(String jfcName, SchemaVersion version) throws Exception {
57 		InputStream in = EventConfigurationTest.class.getResourceAsStream(jfcName);
58 		EventConfiguration.validate(in, jfcName, version);

59 	}
60 
61 	@Test
62 	public void testLoadV1() throws Exception {
63 		loadConfig("continous_jfr_1_0.jfc");
64 	}
65 
66 	@Test
67 	public void testLoadV2() throws Exception {
68 		loadConfig("continous_jfr_2_0.jfc");
69 	}
70 
71 	@Test
72 	public void testValidateV1() throws Exception {
73 		validateConfig("continous_jfr_1_0.jfc", V1);
74 	}
75 
76 	@Test
77 	public void testValidateV2() throws Exception {
78 		validateConfig("continous_jfr_2_0.jfc", V2);
79 	}
