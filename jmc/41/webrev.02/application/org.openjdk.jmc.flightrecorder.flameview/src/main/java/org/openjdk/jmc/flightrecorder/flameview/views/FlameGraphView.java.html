130 	private class GroupByAction extends Action {
131 
132 		private final boolean fromThreadRootAction;
133 
134 		GroupByAction(boolean fromRoot) {
135 			super(fromRoot ? Messages.STACKTRACE_VIEW_THREAD_ROOT : Messages.STACKTRACE_VIEW_LAST_FRAME,
136 					IAction.AS_RADIO_BUTTON);
137 			fromThreadRootAction = fromRoot;
138 			setToolTipText(fromRoot ? Messages.STACKTRACE_VIEW_GROUP_TRACES_FROM_ROOT
139 					: Messages.STACKTRACE_VIEW_GROUP_TRACES_FROM_LAST_FRAME);
140 			setImageDescriptor(fromRoot ? CoreImages.THREAD : CoreImages.METHOD_NON_OPTIMIZED);
141 			setChecked(fromRoot == threadRootAtTop);
142 		}
143 
144 		@Override
145 		public void run() {
146 			boolean newValue = isChecked() == fromThreadRootAction;
147 			if (newValue != threadRootAtTop) {
148 				threadRootAtTop = newValue;
149 				rebuildModel(currentItems);
150 			}
151 		}
152 	}

206 	private void setItems(IItemCollection items) {
207 		if (items != null) {
208 			currentItems = items;
209 			rebuildModel(items);
210 		}
211 	}
212 
213 	private void rebuildModel(IItemCollection items) {
214 		// Release old model before building the new
215 		if (currentModelCalculator != null) {
216 			currentModelCalculator.cancel(true);
217 		}
218 		currentModelCalculator = getModelPreparer(items, frameSeparator, true);
219 		currentModelCalculator.thenAcceptAsync(this::setModel, DisplayToolkit.inDisplayThread())
220 				.exceptionally(FlameGraphView::handleModelBuildException);
221 	}
222 
223 	private CompletableFuture<TraceNode> getModelPreparer(
224 		final IItemCollection items, final FrameSeparator separator, final boolean materializeSelectedBranches) {
225 		return CompletableFuture.supplyAsync(() -> {
226 			return TraceTreeUtils.createTree(items, separator, threadRootAtTop, "-- <Root> --");
227 		}, MODEL_EXECUTOR);
228 	}
229 
230 	private void setModel(TraceNode root) {
231 		if (!browser.isDisposed() && !root.equals(currentRoot)) {
232 			currentRoot = root;
233 			setViewerInput(root);
234 		}
235 	}
236 
237 	private void setViewerInput(TraceNode root) {
238 		browser.setText(HTML_PAGE);
239 		browser.addProgressListener(new ProgressAdapter() {
240 			@Override
241 			public void completed(ProgressEvent event) {
242 				browser.removeProgressListener(this);
243 				browser.execute(String.format("processGraph(%s);", toJSon(root)));
244 			}
245 		});
246 	}
247 
248 	private static Void handleModelBuildException(Throwable ex) {
249 		if (!(ex.getCause() instanceof CancellationException)) {
250 			FlightRecorderUI.getDefault().getLogger().log(Level.SEVERE, "Failed to build stacktrace view model", ex); //$NON-NLS-1$
251 		}
252 		return null;
253 	}
254 
255 	private static String toJSon(TraceNode root) {
256 		if (root == null) {
257 			return "\"\"";
258 		}
259 		return render(root);
260 	}
261 
262 	private static String render(TraceNode root) {
263 		StringBuilder builder = new StringBuilder();
264 		render(builder, root);
265 		return builder.toString();
266 	}
267 
268 	private static void render(StringBuilder builder, TraceNode node) {
269 		String start = UNCLASSIFIABLE_FRAME.equals(node.getName()) ? createJsonDescTraceNode(node)
270 				: createJsonTraceNode(node);
271 		builder.append(start);
272 		for (int i = 0; i < node.getChildren().size(); i++) {
273 			render(builder, node.getChildren().get(i));
274 			if (i < node.getChildren().size() - 1) {
275 				builder.append(",");
276 			}
277 		}
278 		builder.append("]}");
279 	}
280 
281 	private static String createJsonTraceNode(TraceNode node) {
282 		return String.format("{%s,%s,%s, \"c\": [ ", toJSonKeyValue("n", node.getName()),
283 				toJSonKeyValue("p", node.getPackageName()), toJSonKeyValue("v", String.valueOf(node.getValue())));
284 	}
285 
286 	private static String createJsonDescTraceNode(TraceNode node) {
287 		return String.format("{%s,%s,%s,%s, \"c\": [ ", toJSonKeyValue("n", node.getName()),
288 				toJSonKeyValue("p", node.getPackageName()), toJSonKeyValue("d", UNCLASSIFIABLE_FRAME_DESC),
289 				toJSonKeyValue("v", String.valueOf(node.getValue())));
290 	}
291 
292 	private static String toJSonKeyValue(String key, String value) {
293 		return "\"" + key + "\": " + "\"" + value + "\"";
294 	}
295 
296 	private static String loadLibraries(String ... libs) {
297 		if (libs == null || libs.length == 0) {
298 			return "";
299 		} else {
300 			return Stream.of(libs).map(FlameGraphView::fileContent).collect(Collectors.joining("\n"));
301 		}
302 	}
303 
304 	private static String fileContent(String fileName) {
305 		try {
306 			return StringToolkit.readString(FlameGraphView.class.getClassLoader().getResourceAsStream(fileName));
307 		} catch (IOException e) {
308 			FlightRecorderUI.getDefault().getLogger().log(Level.WARNING,
309 					MessageFormat.format("Could not load script \"{0}\",\"{1}\"", fileName, e.getMessage())); //$NON-NLS-1$
310 			return "";
311 		}
312 	}

314 	private static String getStacktraceMessage(String key) {
315 		return org.openjdk.jmc.flightrecorder.stacktrace.Messages.getString(key);
