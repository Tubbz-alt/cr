33 package org.openjdk.jmc.jdp.client;
34 
35 /**
36  * A discovery event. These are triggered from the {@link JDPClient} when a {@link Discoverable} is
37  * FOUND, LOST or CHANGED.
38  */
39 public final class DiscoveryEvent {
40 	public enum Kind {
41 		FOUND, LOST, CHANGED
42 	}
43 
44 	private final Kind kind;
45 	private final Discoverable discoverable;
46 
47 	/**
48 	 * Constructor.
49 	 *
50 	 * @param kind
51 	 *            the type of discovery that was made.
52 	 * @param discoverable
53 	 *            what was discovered.
54 	 */
55 	public DiscoveryEvent(Kind kind, Discoverable discoverable) {
56 		this.kind = kind;
57 		this.discoverable = discoverable;
58 	}
59 
60 	/**
61 	 * The kind of discovery. The discoverable was either FOUND, LOST or CHANGED.
62 	 *
63 	 * @return kind of discovery. The discoverable was either FOUND, LOST or CHANGED.
64 	 */
65 	public Kind getKind() {
66 		return kind;
67 	}
68 
69 	/**
70 	 * @return the discoverable for which the status was updated.
71 	 */
72 	public Discoverable getDiscoverable() {
73 		return discoverable;
74 	}
75 
76 	@Override
77 	public String toString() {
78 		return kind + " " + discoverable.getSessionId(); //$NON-NLS-1$
79 	}
80 }
