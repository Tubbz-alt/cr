 33 package org.openjdk.jmc.jdp.client;
 34 
 35 import static org.junit.Assert.assertEquals;
 36 import static org.junit.Assert.assertNotNull;
 37 
 38 import java.util.Properties;
 39 import java.util.concurrent.BlockingQueue;
 40 import java.util.concurrent.LinkedBlockingQueue;
 41 import java.util.concurrent.TimeUnit;
 42 import java.util.logging.Logger;
 43 
 44 import org.junit.Before;
 45 import org.junit.Test;
 46 
 47 import org.openjdk.jmc.jdp.common.Configuration;
 48 import org.openjdk.jmc.jdp.server.JDPServer;
 49 
 50 @SuppressWarnings("nls")
 51 public class JDPClientTest {
 52 	public final static Logger LOGGER = Logger.getLogger("org.openjdk.jmc.jdp.test");
 53 	private final Configuration config = TestToolkit.createConfiguration();
 54 	private volatile BlockingQueue<DiscoveryEvent> eventQueue;
 55 	private String discoverableID;
 56 
 57 	@Before
 58 	public void setUp() throws Exception {
 59 		discoverableID = TestToolkit.generateNewID("JDPClientTest");
 60 		System.out.println("JDP test with discoverable id " + discoverableID);
 61 	}
 62 
 63 	@Test
 64 	public void testJDPClient() throws Exception {
 65 		JDPClient client = createDefaultClient();
 66 		JDPServer server = createDefaultServer();
 67 		client.start();
 68 		server.start();
 69 
 70 		DiscoveryEvent event = eventQueue.poll(config.getBroadcastPeriod() * 2, TimeUnit.MILLISECONDS);
 71 		assertNotNull(event);
 72 		assertEquals(DiscoveryEvent.Kind.FOUND, event.getKind());
 73 
 74 		server.stop();
 75 		long sleepTime = (long) Math.max(Pruner.PRUNING_INTERVAL,
 76 				config.getBroadcastPeriod() * Pruner.HB_MISSED_BEFORE_DOWN);
 77 		event = eventQueue.poll(sleepTime * 8, TimeUnit.MILLISECONDS);
 78 
 79 		assertNotNull(event);
 80 		assertEquals(DiscoveryEvent.Kind.LOST, event.getKind());
 81 		client.stop();
 82 	}
 83 
 84 	@Test
 85 	public void testChangePacket() throws Exception {
 86 		JDPClient client = createDefaultClient();
 87 		JDPServer server = createDefaultServer();
 88 		client.start();
 89 		server.start();
 90 
 91 		DiscoveryEvent event = eventQueue.poll(config.getBroadcastPeriod() * 2, TimeUnit.MILLISECONDS);
 92 		assertNotNull(event);
 93 		assertEquals(event.getKind(), DiscoveryEvent.Kind.FOUND);
 94 		assertEquals(event.getDiscoverable().getPayload().get("apa"), "gorilla");
 95 
 96 		event = null;
 97 		Properties props = createDefaultData();
 98 		props.setProperty("apa", "mongo");
 99 		server.setDiscoveryData(props);
100 
101 		event = eventQueue.poll(config.getBroadcastPeriod() * 2, TimeUnit.MILLISECONDS);
102 		assertNotNull(event);
103 		assertEquals(event.getKind(), DiscoveryEvent.Kind.CHANGED);
104 		assertEquals(event.getDiscoverable().getPayload().get("apa"), "mongo");
105 		client.stop();
106 		server.stop();
107 	}
108 
109 	public JDPClient createDefaultClient() {
110 		eventQueue = new LinkedBlockingQueue<>();
111 		JDPClient client = new JDPClient(config.getMulticastAddress(), config.getMulticastPort());
112 		client.addDiscoveryListener(new DiscoveryListener() {
113 
114 			@Override
115 			public void onDiscovery(DiscoveryEvent event) {
116 				if (discoverableID.equals(event.getDiscoverable().getPayload().get(JDPServer.KEY_DISCOVERABLE_ID))) {
117 					eventQueue.add(event);
118 				}
119 			}
120 		});
121 		return client;
122 	}
123 
124 	public JDPServer createDefaultServer() {
125 		JDPServer server = new JDPServer(discoverableID, config);
126 		TestToolkit.printServerSettings(server);
127 
128 		server.setDiscoveryData(createDefaultData());
129 		return server;
130 	}
131 
132 	private Properties createDefaultData() {
133 		Properties props = new Properties();
134 		props.setProperty("test", "JDPClientTest");
135 		props.setProperty("apa", "gorilla");
136 		return props;
137 	}
138 }
